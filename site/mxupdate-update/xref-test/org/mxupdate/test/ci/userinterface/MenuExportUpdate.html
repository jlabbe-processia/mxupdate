37   * Test cases for the export and update of menus.

42  public class MenuExportUpdate
43      extends AbstractDataExportUpdate<MenuData>
44  {
45      /***
46       * Creates for given <code>_name</code> a new menu instance.
47       *
48       * @param _name     name of the menu instance
49       * @return menu instance
50       */
51      @Override()
52      protected MenuData createNewData(final String _name)
53      {
54          return new MenuData(this, _name);
55      }
56  
57      /***
58       * Data provider for test menus.
59       *
60       * @return object array with all test menus
61       */
62      @DataProvider(name = "menus")
63      public Object[][] getMenus()
64      {
65          return this.prepareData("menu",
66                  new Object[]{
67                          "",
68                          new MenuData(this, "hallo \" test")
69                                  .setValue("label", "command label \" // ' #")
70                                  .setValue("description", "\"//// hallo")
71                                  .setValue("href", "${COMMON_DIR}/emxTree.jsp?mode=insert")
72                                  .setValue("alt", "${COMMON_DIR}/emxTreeAlt.jsp?mode=insert")
73                                  .setSetting("Setting 1", "Setting Value ' 1")
74                                  .addChild(new MenuData(this, "child menu 1"))
75                                  .addChild(new CommandData(this, "child command 1"))});
76      }
77  
78      /***
79       * Cleanup all test commands.
80       *
81       * @throws MatrixException if cleanup failed
82       */
83      @BeforeMethod()
84      @AfterMethod()
85      public void cleanup()
86          throws MatrixException
87      {
88          this.cleanup(AbstractTest.CI.UI_COMMAND);
89          this.cleanup(AbstractTest.CI.UI_MENU);
90      }

92      /***
93       * Tests a 'simple' menu but with a quote in name and with one command and
94       * menu.
95       *
96       * @param _description  description of the test case
97       * @param _menu         menu to test
98       * @throws Exception if test failed
99       */
100     @Test(dataProvider = "menus", description = "test export of a single menu")
101     public void testExportSingleMenu(final String _description,
102                                      final MenuData _menu)
103         throws Exception
104     {
105         _menu.create();
106         _menu.checkExport(_menu.export());
107     }
108 
109     /***
110      * Tests an update of non existing command. The result is tested with by
111      * exporting the command and checking the result.
112      *
113      * @param _description  description of the test case
114      * @param _menu         menu to test
115      * @throws Exception if test failed
116      */
117     @Test(dataProvider = "menus", description = "test update of non existing menu")
118     public void testUpdate(final String _description,
119                            final MenuData _menu)
120         throws Exception
121     {
122         // create referenced property value
123         for (final PropertyDef prop : _menu.getProperties())  {
124             if (prop.getTo() != null)  {
125                 prop.getTo().create();
126             }
127         }
128         // create child menus / commands
129         _menu.createChildren();
130 
131         // first update with original content
132         this.update(_menu);
133         final ExportParser exportParser = _menu.export();
134         _menu.checkExport(exportParser);
135 
136         // second update with delivered content
137         this.update(_menu.getCIFileName(), exportParser.getOrigCode());
138         _menu.checkExport(_menu.export());
139     }
