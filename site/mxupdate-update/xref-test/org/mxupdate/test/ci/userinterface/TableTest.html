150     /***
151      * Tests an update of non existing table. The result is tested with by
152      * exporting the table and checking the result.
153      *
154      * @param _description  description of the test case
155      * @param _table        table to test
156      * @throws Exception if test failed
157      */
158     @Test(dataProvider = "tables",
159           description = "test update of non existing table")
160     public void testUpdate(final String _description,
161                            final TableData _table)
162         throws Exception
163     {
164         // create users
165         for (final FieldData<TableData> field : _table.getFields())  {
166             for (final AbstractUserData<?> user : field.getUsers())  {
167                 user.create();
168             }
169         }
170         // create referenced property value
171         for (final PropertyDef prop : _table.getProperties())  {
172             if (prop.getTo() != null)  {
173                 prop.getTo().create();
174             }
175         }
176 
177         // first update with original content
178         _table.update();
179         final ExportParser exportParser = _table.export();
180         _table.checkExport(exportParser);
181 
182         // second update with delivered content
183         _table.updateWithCode(exportParser.getOrigCode())
184               .checkExport();
185     }
