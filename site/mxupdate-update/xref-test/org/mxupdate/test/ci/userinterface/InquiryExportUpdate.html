43      /***
44       * Data provider for test inquiries.
45       *
46       * @return object array with all test inquiries
47       */
48      @DataProvider(name = "inquires")
49      public Object[][] getInquiries()
50      {
51          final DataCollection data1 = new DataCollection(this);
52          data1.getInquiry("Test1")
53               .setValue("description", "test description")
54               .setValue("pattern", "${ID}|*")
55               .setValue("format", "${ID}")
56               .setArgument("TYPE", "MY_TYPE")
57               .setArgument("NAME", "MY_NAME")
58               .setArgument("REVISION", "MY_REVISION")
59               .setCode("print bus '${TYPE}' '${NAME}' '${REVISION}' select id dump");
60  
61          final DataCollection data2 = new DataCollection(this);
62          data2.getInquiry("Test2")
63               .setValue("description", "test description");
64  
65          final DataCollection data3 = new DataCollection(this);
66          data3.getInquiry("Test3 \" test")
67               .setValue("description", "test description \" '")
68               .setValue("pattern", "patternprefix \" patternsuffix")
69               .setValue("format", "formatprefix \" formatsuffix")
70               .setArgument("ARGUMENT \" ARGSUFFIX", "argumentprefix \" argumentsuffix")
71               .setCode("print bus '${TYPE}' '${NAME}' '${REVISION}' select id dump");
72  
73          return new Object[][]  {
74                  new Object[]{data1, "Test1"},
75                  new Object[]{data2, "Test2"},
76                  new Object[]{data3, "Test3 \" test"},
77          };
78      }

93      /***
94       * Tests a new created inquiry and the related export.
95       *
96       * @param _data     data collection to test
97       * @param _name     name of the inquiry to test
98       * @throws Exception if test failed
99       */
100     @Test(dataProvider = "inquires", description = "test export of new created inquires")
101     public void testExport(final DataCollection _data,
102                            final String _name)
103         throws Exception
104     {
105         _data.create();
106         final InquiryData inquiry = _data.getInquiry(_name);
107         final ExportParser exportParser = inquiry.export();
108         inquiry.checkExport(exportParser);
109     }
110 
111     /***
112      * Tests an update of non existing inquiry. The result is tested by
113      * exporting the inquiry and checking the result.
114      *
115      * @param _data     data collection to test
116      * @param _name     name of the inquiry to test
117      * @throws Exception if test failed
118      */
119     @Test(dataProvider = "inquires", description = "test update of non existing inquiry")
120     public void testUpdate(final DataCollection _data,
121                            final String _name)
122         throws Exception
123     {
124         final InquiryData inquiry = _data.getInquiry(_name);
125         // first update with original content
126         this.update(inquiry.getCIFileName(), inquiry.ciFile());
127         final ExportParser exportParser = inquiry.export();
128         inquiry.checkExport(exportParser);
129         // second update with delivered content
130         this.update(inquiry.getCIFileName(), exportParser.getOrigCode());
131         inquiry.checkExport(inquiry.export());
132     }
