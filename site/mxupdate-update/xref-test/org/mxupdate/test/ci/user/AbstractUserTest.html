43  /***
44   * Prepares test data for users (group, role and person).
45   *
46   * @author The MxUpdate Team
47   * @version $Id: AbstractUserTest.java 797 2010-05-16 22:47:14Z tim.moxter $

53      /***
54       * Prepares the test data.
55       *
56       * @param _logText  text used for the description (logging purpose)
57       * @param _datas    specific test data to append
58       * @return prepared test data
59       */
60      @Override()
61      protected Object[][] prepareData(final String _logText,
62                                       final Object[]... _datas)
63      {
64          final List<Object[]> ret = new ArrayList<Object[]>();
65          ret.addAll(Arrays.asList(_datas));
66  
67          this.appendCue(_logText, ret);
68          this.appendFilter(_logText, ret);
69          this.appendQuery(_logText, ret);
70          this.appendTable(_logText, ret);
71          this.appendTip(_logText, ret);
72          this.appendToolSet(_logText, ret);
73          this.appendView(_logText, ret);
74  
75          return super.prepareData(_logText, ret.toArray(new Object[ret.size()][]));
76      }

758     /***
759      * Prepares the test data for views.
760      *
761      * @param _name                     name of the user data
762      * @param _viewName                 name of the view
763      * @param _activeCueNames           names of active cues
764      * @param _notActiveCueNames        name of inactive cues
765      * @param _activeFilterNames        name of active filters
766      * @param _notActiveFilterNames     name of inactive filters
767      * @param _activeTipNames           name of active tips
768      * @param _notActiveTipNames        name of inactive tips
769      * @param _activeToolSetNames       name of active tool sets
770      * @param _notActiveToolSetNames    name of inactive tool sets
771      * @param _activeTableNames         name of active tables
772      * @param _notActiveTableNames      name of inactive tables
773      * @return user test data used for views
774      */
775     private USER prepareViewTestData(final String _name,
776                                      final String _viewName,
777                                      final String[] _activeCueNames,
778                                      final String[] _notActiveCueNames,
779                                      final String[] _activeFilterNames,
780                                      final String[] _notActiveFilterNames,
781                                      final String[] _activeTipNames,
782                                      final String[] _notActiveTipNames,
783                                      final String[] _activeToolSetNames,
784                                      final String[] _notActiveToolSetNames,
785                                      final String[] _activeTableNames,
786                                      final String[] _notActiveTableNames)
787     {
788         final USER userData = this.createNewData(_name);
789         final ViewData<USER> view = userData.newView(_viewName);
790         if (_activeCueNames != null)  {
791             for (final String name : _activeCueNames)  {
792                 view.addActive(userData.newCue(name));
793             }
794         }
795         if (_notActiveCueNames != null)  {
796             for (final String name : _notActiveCueNames)  {
797                 view.addNotActive(userData.newCue(name));
798             }
799         }
800         if (_activeFilterNames != null)  {
801             for (final String name : _activeFilterNames)  {
802                 view.addActive(userData.newFilter(name));
803             }
804         }
805         if (_notActiveFilterNames != null)  {
806             for (final String name : _notActiveFilterNames)  {
807                 view.addNotActive(userData.newFilter(name));
808             }
809         }
810         if (_activeTipNames != null)  {
811             for (final String name : _activeTipNames)  {
812                 view.addActive(userData.newTip(name));
813             }
814         }
815         if (_notActiveTipNames != null)  {
816             for (final String name : _notActiveTipNames)  {
817                 view.addNotActive(userData.newTip(name));
818             }
819         }
820         if (_activeToolSetNames != null)  {
821             for (final String name : _activeToolSetNames)  {
822                 view.addActive(userData.newToolSet(name));
823             }
824         }
825         if (_notActiveToolSetNames != null)  {
826             for (final String name : _notActiveToolSetNames)  {
827                 view.addNotActive(userData.newToolSet(name));
828             }
829         }
830         if (_activeTableNames != null)  {
831             for (final String name : _activeTableNames)  {
832                 view.addActive(userData.newTable(name));
833             }
834         }
835         if (_notActiveTableNames != null)  {
836             for (final String name : _notActiveTableNames)  {
837                 view.addNotActive(userData.newTable(name));
838             }
839         }
840         return userData;
