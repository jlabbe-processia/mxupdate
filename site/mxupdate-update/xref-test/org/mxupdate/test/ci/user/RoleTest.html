34  /***
35   * Test cases for the export and update of roles.

40  @Test()
41  public class RoleTest
42      extends AbstractUserTest<RoleData>
43  {
44      /***
45       * Creates for given <code>_name</code> a new role instance.
46       *
47       * @param _name     name of the role instance
48       * @return role instance
49       */
50      @Override()
51      protected RoleData createNewData(final String _name)
52      {
53          return new RoleData(this, _name);
54      }
55  
56      /***
57       * Data provider for test roles.
58       *
59       * @return object array with all test roles
60       */
61      @DataProvider(name = "data")
62      public Object[][] getRoles()
63      {
64          return this.prepareData("role",
65                  new Object[]{
66                          "simple role",
67                          new RoleData(this, "hallo \" test")
68                              .setValue("description", "\"//// hallo")},
69                  new Object[]{
70                          "role with two parent roles",
71                          new RoleData(this, "hallo \" test")
72                                  .setValue("description", "\"//// hallo")
73                                  .assignParents(new RoleData(this, "hallo parent1 \" test"))
74                                  .assignParents(new RoleData(this, "hallo parent2 \" test"))},
75                  new Object[]{
76                          "role with assigned site",
77                          new RoleData(this, "hallo \" test")
78                                  .setSite(new SiteData(this, "Test \" Site"))});
79      }
80  
81      /***
82       * Returns the mapping if for given parameter the workspace objects must be
83       * ignored to remove or not.
84       *
85       * @return map between the parameter and the workspace objects are ignored
86       *         to removed
87       */
88      @DataProvider(name = "wsoParameters")
89      public Object[][] getWSOParameters()
90      {
91          return new Object[][]{
92                  new Object[]{"UserIgnoreWSO4Users", true},
93                  new Object[]{"UserIgnoreWSO4Persons", false},
94                  new Object[]{"UserIgnoreWSO4Roles", true},
95                  new Object[]{"UserIgnoreWSO4Groups", false}
96          };
97      }

102      * @throws MatrixException if cleanup failed

104     @BeforeMethod()

111         this.cleanup(AbstractTest.CI.USR_ROLE);
112         this.cleanup(AbstractTest.CI.OTHER_SITE);
113         this.cleanup(AbstractTest.CI.PRG_MQL_PROGRAM);
114     }

116     /***
117      * Test an update of a role where the site is removed.
118      *
119      * @throws Exception if test failed
120      */
121     @Test(description = "update existing existing role with site by removing site")
122     public void checkExistingSiteRemovedWithinUpdate()
123         throws Exception
124     {
125         final RoleData role = new RoleData(this, "hallo \" test")
126                 .setSite(new SiteData(this, "Test \" Site"));
127         role.create();
128         role.setSite(null);
129         role.update();
130 
131         Assert.assertEquals(this.mql("escape print role \"" + AbstractTest.convertMql(role.getName()) + "\" select site dump"),
132                             "",
133                             "check that no site is defined");
134     }
135 
136     /***
137      * Check that the hidden flag is removed within update.
138      *
139      * @throws Exception if test failed
140      */
141     @Test(description = "check that the hidden flag is removed within update")
142     public void checkHiddenFlagRemoveWithinUpdate()
143         throws Exception
144     {
145         final RoleData role = new RoleData(this, "hallo \" test").setHidden(true);
146         role.create();
147         role.setHidden(null);
148         role.update();
149         Assert.assertEquals(this.mql("escape print role \"" + AbstractTest.convertMql(role.getName()) + "\" select hidden dump"),
150                             "FALSE",
151                             "check that role is not hidden");
152     }
