156     /***
157      * Tests an update of non existing role. The result is tested with by
158      * exporting the role and checking the result.
159      *
160      * @param _description  description of the test case
161      * @param _role         role to test
162      * @throws Exception if test failed
163      */
164     @Test(dataProvider = "roles", description = "test update of non existing role")
165     public void simpleUpdate(final String _description,
166                              final RoleData _role)
167         throws Exception
168     {
169         // create referenced property value
170         for (final PropertyDef prop : _role.getProperties())  {
171             if (prop.getTo() != null)  {
172                 prop.getTo().create();
173             }
174         }
175         // create all parent roles
176         for (final RoleData parentRole : _role.getParent())  {
177             parentRole.create();
178         }
179         // create site
180         if (_role.getSite() != null)  {
181             _role.getSite().create();
182         }
183         // create cue properties
184         for (final CueData<RoleData> cue : _role.getCues())  {
185             for (final PropertyDef prop : cue.getProperties())  {
186                 if (prop.getTo() != null)  {
187                     prop.getTo().create();
188                 }
189             }
190         }
191         // create filter properties
192         for (final FilterData<RoleData> filter : _role.getFilters())  {
193             for (final PropertyDef prop : filter.getProperties())  {
194                 if (prop.getTo() != null)  {
195                     prop.getTo().create();
196                 }
197             }
198         }
199         // create query properties
200         for (final QueryData<RoleData> query : _role.getQueries())  {
201             for (final PropertyDef prop : query.getProperties())  {
202                 if (prop.getTo() != null)  {
203                     prop.getTo().create();
204                 }
205             }
206         }
207         // create table properties
208         for (final TableData<RoleData> table : _role.getTables())  {
209             for (final PropertyDef prop : table.getProperties())  {
210                 if (prop.getTo() != null)  {
211                     prop.getTo().create();
212                 }
213             }
214         }
215         // create tip properties
216         for (final TipData<RoleData> tip : _role.getTips())  {
217             for (final PropertyDef prop : tip.getProperties())  {
218                 if (prop.getTo() != null)  {
219                     prop.getTo().create();
220                 }
221             }
222         }
223         // create tool set properties and programs
224         for (final ToolSetData<RoleData> toolSet : _role.getToolSets())  {
225             for (final PropertyDef prop : toolSet.getProperties())  {
226                 if (prop.getTo() != null)  {
227                     prop.getTo().create();
228                 }
229             }
230             for (final AbstractProgramData<?> prog : toolSet.getPrograms())  {
231                 prog.create();
232             }
233         }
234         // create view properties
235         for (final ViewData<RoleData> view : _role.getViews())  {
236             for (final PropertyDef prop : view.getProperties())  {
237                 if (prop.getTo() != null)  {
238                     prop.getTo().create();
239                 }
240             }
241         }
242 
243         // first update with original content
244         this.update(_role.getCIFileName(), _role.ciFile());
245         final ExportParser exportParser = _role.export();
246         _role.checkExport(exportParser);
247 
248         // second update with delivered content
249         this.update(_role.getCIFileName(), exportParser.getOrigCode());
250         _role.checkExport(_role.export());
251     }
