119     /***
120      * Tests an update of non existing command. The result is tested with by
121      * exporting the command and checking the result.
122      *
123      * @param _data         data collection
124      * @param _name         name of type to test
125      * @throws Exception if test failed
126      */
127     @Test(dataProvider = "types", description = "test update of non existing types")
128     public void testUpdate(final DataCollection _data,
129                            final String _name)
130         throws Exception
131     {
132         // create referenced attributes
133         for (final AbstractAttribute<?> attribute : _data.getAttributes())  {
134             attribute.create();
135         }
136         // create referenced methods
137         for (final AbstractProgramData<?> method : _data.getPrograms())  {
138             method.create();
139         }
140 
141         final TypeData type = _data.getType(_name);
142 
143         this.update(type);
144         final ExportParser exportParser = type.export();
145         type.checkExport(exportParser);
146     }
