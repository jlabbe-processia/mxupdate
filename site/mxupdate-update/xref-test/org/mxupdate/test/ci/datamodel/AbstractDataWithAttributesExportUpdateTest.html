35  /***
36   * Class is used to test data with attributes exports and updates.
37   *
38   * @param <DATAWITHATTRIBUTE>     derived data class with attributes
39   * @author The MxUpdate Team
40   * @version $Id: AbstractDataWithAttributesExportUpdateTest.java 797 2010-05-16 22:47:14Z tim.moxter $
41   */
42  public abstract class AbstractDataWithAttributesExportUpdateTest<DATAWITHATTRIBUTE extends AbstractDataWithAttribute<?>>
43      extends AbstractDataExportUpdate<DATAWITHATTRIBUTE>
44  {
45      /***
46       * Prepares the test data.
47       *
48       * @param _logText  text used for the description (logging purpose)
49       * @param _datas    specific test data to append
50       * @return prepared test data
51       */
52      @Override()
53      protected Object[][] prepareData(final String _logText,
54                                       final Object[]... _datas)
55      {
56          final List<Object[]> ret = new ArrayList<Object[]>();
57  
58          ret.add(new Object[]{
59                  _logText + " with one attribute",
60                  this.createNewData("Test \" 1")
61                          .addAttribute(new AttributeStringData(this, "String Attribute"))});
62  
63          ret.add(new Object[]{
64                  _logText + " with two escaped attributes",
65                  this.createNewData("Test \" 1")
66                          .addAttribute(new AttributeStringData(this, "String Attribute \" ' Hello"))
67                          .addAttribute(new AttributeStringData(this, "String Attribute \" { Hello"))});
68  
69          ret.add(new Object[]{
70                  _logText + " with two attributes in braces",
71                  this.createNewData("Test \" 1")
72                          .addAttribute(new AttributeStringData(this, "{String Attribute Hello}"))
73                          .addAttribute(new AttributeStringData(this, "String Attribute {} Hello"))});
74  
75          ret.add(new Object[]{
76                  _logText + " with attributes with two attributes, two ignored attributes and two removed attributes",
77                  this.createNewData("Test \" 1")
78                          .addAttribute(new AttributeStringData(this, "String Attribute \" ' Hello 1"))
79                          .addAttribute(new AttributeStringData(this, "String Attribute \" { Hello 2"))
80                          .addIgnoreAttribute(new AttributeStringData(this, "String Attribute \" ' Hello 3"))
81                          .addIgnoreAttribute(new AttributeStringData(this, "String Attribute \" { Hello 4"))
82                          .addRemoveAttribute(new AttributeStringData(this, "String Attribute \" ' Hello 5"))
83                          .addRemoveAttribute(new AttributeStringData(this, "String Attribute \" { Hello 6"))});
84  
85          ret.addAll(Arrays.asList(_datas));
86          return super.prepareData(_logText, ret.toArray(new Object[ret.size()][]));
87      }

89      /***
90       * Creates a clean data instance used to update an existing data instance.
91       *
92       * @param _original     original data instance
93       * @return new data instance (where all original data is cleaned)
94       */
95      @Override
96      @SuppressWarnings("unchecked")
97      protected DATAWITHATTRIBUTE createCleanNewData(final DATAWITHATTRIBUTE _original)
98      {
99          return (DATAWITHATTRIBUTE) this.createNewData(_original.getName().substring(AbstractTest.PREFIX.length()))
100                                        .addAttribute(_original.getAttributes().toArray(new AbstractAttributeData<?>[_original.getAttributes().size()]));
101     }

146     @SuppressWarnings("unchecked")
147     @Test(description = "test thrown exception of called update with wrong argument for the administration type for TCL procedure 'testAttributes'")

149         throws Exception
150     {
151         final DATAWITHATTRIBUTE data = (DATAWITHATTRIBUTE) this.createNewData("Test").create();
152         if (data.getCI() != AbstractTest.CI.DM_INTERFACE)  {
153             data.failedUpdateWithCode(
154                     "testAttributes -interface ${NAME}",
155                     UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_UKNOWN_PARAMETER);
156         }
157         if (data.getCI() != AbstractTest.CI.DM_RELATIONSHIP)  {
158             data.failedUpdateWithCode(
159                     "testAttributes -relationship ${NAME}",
160                     UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_UKNOWN_PARAMETER);
161         }
162         if (data.getCI() != AbstractTest.CI.DM_TYPE)  {
163             data.failedUpdateWithCode(
164                     "testAttributes -type ${NAME}",
165                     UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_UKNOWN_PARAMETER);
