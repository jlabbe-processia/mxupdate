68       * Makes an update for given <code>_fileName</code> and <code>_code</code>.
69       *
70       * @param _fileName     name of the file to update
71       * @param _code         TCL update code
72       * @param _errorCode    expected error code
73       * @throws Exception if update with failure failed
74       */
75      @Deprecated()
76      private void updateFailure(final String _fileName,
77                                 final String _code,
78                                 final UpdateException_mxJPO.Error _errorCode)
79          throws Exception
80      {
81          final Map<?,?> bck = this.update(_fileName, _code);
82          final Exception ex = (Exception) bck.get("exception");
83           Assert.assertNotNull(ex, "check that action is not allowed");
84           Assert.assertTrue(ex.getMessage().indexOf("UpdateError #" + _errorCode.getCode() + ":") >= 0,
85                             "check for correct error code #" + _errorCode.getCode());
86      }

88      /***
89       * Makes an update for given <code>_code</code>.
90       *
91       * @param _fileName     name of the file to update
92       * @param _code         TCL update code
93       * @param _params       parameters
94       * @return values from the called dispatcher
95       * @throws Exception  if update failed
96       */
97      @Deprecated()
98      private Map<?,?> update(final String _fileName,
99                              final String _code,
100                             final String... _params)
101         throws Exception
102     {
103         final Map<String,String> files = new HashMap<String,String>();
104         files.put(_fileName, _code);
105         final Map<String,String> params = new HashMap<String,String>();
106         if (_params != null)  {
107             for (int idx = 0; idx < _params.length; idx += 2)  {
108                 params.put(_params[idx], _params[idx + 1]);
109             }
110         }
111         final Map<?,?> bck = this.executeEncoded("Update", params, "FileContents", files);
112 
113         return bck;
114     }
