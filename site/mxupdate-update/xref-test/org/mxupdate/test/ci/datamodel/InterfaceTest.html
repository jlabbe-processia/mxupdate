125      * @throws Exception if MQL execution failed

139      * {@inheritDoc}

141      * not be removed.

143     @Override()
144     protected InterfaceData createCleanNewData(final InterfaceData _original)

429      * Update an interface with special characters for all cases.
430      *
431      * @throws Exception if test failed
432      */
433     @Test(description = "update interface with special characters")
434     public void updateWithSpecialCharacters()
435         throws Exception
436     {
437         final InterfaceData parent1 = new InterfaceData(this, "TestInterfaceParent \" 1").create();
438         final InterfaceData parent2 = new InterfaceData(this, "TestInterfaceParent \" 2").create();
439         final AttributeStringData attr1 = new AttributeStringData(this, "Attribute \" 1").create();
440         final AttributeStringData attr2 = new AttributeStringData(this, "Attribute \" 2").create();
441         final TypeData type1 = new TypeData(this, "TestType \" 1").create();
442         final TypeData type2 = new TypeData(this, "TestType \" 2").create();
443         final RelationshipData rel1 = new RelationshipData(this, "TestRel \" 1").create();
444         final RelationshipData rel2 = new RelationshipData(this, "TestRel \" 2").create();
445 
446         final InterfaceData inter = new InterfaceData(this, "TestInterface \"")
447                 .addParent(parent1)
448                 .addParent(parent2)
449                 .addAttribute(attr1)
450                 .addAttribute(attr2)
451                 .addType(type1)
452                 .addType(type2)
453                 .addRelationship(rel1)
454                 .addRelationship(rel2)
455                 .update();
456 
457         final Set<String> resultParent = new HashSet<String>();
458         resultParent.add(parent1.getName());
459         resultParent.add(parent2.getName());
460         Assert.assertEquals(
461                 this.mqlAsSet("print interface '" + inter.getName() + "' select derived dump '\n'"),
462                 resultParent,
463                 "check that all parent interfaces are defined");
464 
465         final Set<String> resultAttrs = new HashSet<String>();
466         resultAttrs.add(attr1.getName());
467         resultAttrs.add(attr2.getName());
468         Assert.assertEquals(
469                 this.mqlAsSet("escape print interface \""
470                                     + AbstractTest.convertMql(inter.getName()) + "\" select attribute dump '\n'"),
471                 resultAttrs,
472                 "check that all types are defined");
473 
474         final Set<String> resultTypes = new HashSet<String>();
475         resultTypes.add(type1.getName());
476         resultTypes.add(type2.getName());
477         Assert.assertEquals(
478                 this.mqlAsSet("escape print interface \""
479                                     + AbstractTest.convertMql(inter.getName()) + "\" select type dump '\n'"),
480                 resultTypes,
481                 "check that all types are defined");
482 
483         final Set<String> resultRels = new HashSet<String>();
484         resultRels.add(rel1.getName());
485         resultRels.add(rel2.getName());
486         Assert.assertEquals(
487                 this.mqlAsSet("escape print interface \""
488                                     + AbstractTest.convertMql(inter.getName()) + "\" select relationship dump '\n'"),
489                 resultRels,
490                 "check that all relationships are defined");
491     }
