151     /***
152      * Tests an update of non existing rule. The result is tested with by
153      * exporting the rule and checking the result.
154      *
155      * @param _description  description of the test case
156      * @param _rule         rule to test
157      * @throws Exception if test failed
158      */
159     @Test(dataProvider = "rules", description = "test update of non existing rule")
160     public void testUpdate(final String _description,
161                            final RuleData _rule)
162         throws Exception
163     {
164         // create referenced property value
165         for (final PropertyDef prop : _rule.getProperties())  {
166             if (prop.getTo() != null)  {
167                 prop.getTo().create();
168             }
169         }
170 
171         // first update with original content
172         _rule.update();
173         final ExportParser exportParser = _rule.export();
174         _rule.checkExport(exportParser);
175 
176         // second update with delivered content
177         _rule.updateWithCode(exportParser.getOrigCode())
178              .checkExport();
179     }
