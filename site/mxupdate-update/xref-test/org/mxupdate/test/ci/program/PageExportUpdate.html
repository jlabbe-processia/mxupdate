96      /***
97       * Checks that the new created MQL progam is exported correctly.
98       *
99       * @param _data     data collection to test
100      * @param _name     name of the inquiry to test
101      * @throws Exception if test failed
102      */
103     @Test(dataProvider = "pages", description = "test export of pages")
104     public void testExport(final DataCollection _data,
105                            final String _name)
106         throws Exception
107     {
108         final PageData page = _data.getPage(_name);
109         page.create();
110         final Export export = this.export(CI.PRG_PAGE, page.getName());
111 
112         // check oath
113         Assert.assertEquals(export.getPath(),
114                             page.getCI().filePath,
115                             "check path is correct");
116 
117         // check file name
118         Assert.assertEquals(export.getFileName(),
119                             page.getCIFileName(),
120                             "check that the correct file name is returned");
121 
122         // check JPO code
123         Assert.assertEquals(export.getCode(),
124                             page.getCode(),
125                             "checks MQL program code");
126     }
127 
128     /***
129      * Tests, if the MQL program within MX is created and registered with the
130      * correct symbolic name.
131      *
132      * @param _data     data collection to test
133      * @param _name     name of the inquiry to test
134      * @throws Exception if test failed
135      */
136     @Test(dataProvider = "pages", description = "test update of non existing pages")
137     public void testUpdate(final DataCollection _data,
138                            final String _name)
139         throws Exception
140     {
141         final PageData page = _data.getPage(_name);
142 
143         // first update with original content
144         this.update(page.getCIFileName(), page.ciFile());
145 
146         Assert.assertTrue(!"".equals(this.mql("list page " + page.getName())),
147                           "check page is created");
148         Assert.assertEquals(this.mql("escape print page \""
149                                     + AbstractTest.convertMql(page.getName()) + "\" select content dump"),
150                             page.getCode(),
151                             "check correct code");
152     }
