40       * Defines the string where the header of symbolic names starts.
41       */
42      private static final String HEADER_SYMBOLIC_NAME = "# SYMBOLIC NAME:\n# ~~~~~~~~~~~~~~\n#";

44      /***
45       * Defines the string which defines the header start and end.
46       */
47      private static final String HEADER_START_END

86      /***
87       *
88       * @param _ci       configuration item type
89       * @param _export   related export instance
90       */
91      public ExportParser(final CI _ci,
92                          final Export _export)
93      {
94          final String complCode = _export.getCode();
95          this.ci = _ci;
96          this.fileName = _export.getFileName();
97          // parse symbolic name
98          final int posSymbolicName = complCode.indexOf(ExportParser.HEADER_SYMBOLIC_NAME);
99          if (posSymbolicName >= 0)  {
100             final int start = posSymbolicName + ExportParser.HEADER_SYMBOLIC_NAME.length();
101             final int end = complCode.indexOf('\n', start);
102             this.symbolicName = complCode.substring(start, end).trim();
103         } else  {
104             this.symbolicName = null;
105         }
106         // parse name in the header
107         final int posName = complCode.indexOf("# " + this.ci.header + ":\n#");
108         if (posName >= 0)  {
109             final int start = posName + (this.ci.header.length() * 2) + 9;
110             final int end = complCode.indexOf('\n', start);
111             this.name = complCode.substring(start, end).trim();
112         } else  {
113             this.name = null;
114         }
115         // extract update code
116         final int posHeaderEnd = complCode.lastIndexOf(ExportParser.HEADER_START_END);
117         if (posHeaderEnd >=0)  {
118             this.code = complCode.substring(posHeaderEnd + ExportParser.HEADER_START_END.length()).trim();
119         } else  {
120             this.code = complCode.trim();
121         }
122         // parse all lines
123         new Line(Arrays.asList(this.code.split("\n")).iterator(), null);
124     }

126     /***

128      *
129      * @param _path     path of the lines
130      * @return list of all found strings
131      */
132     public List<String> getLines(final String _path)
133     {
134         final List<String> ret = new ArrayList<String>();
135         final String[] path = _path.split("/");
136         for (final Line line : this.rootLines)  {
137             line.evalPath(path, 1, ret);
138         }
139         return ret;
140     }

186     /***
187      * <p>The class is used to build a hierarchy of lines depending of the
188      * {@link Line#shifting} where the text of a line begins. The
189      * {@link Line#shifting} for all {@link Line#children} of a line is always
190      * greater.<p>
191      *
192      * <p><b>Example:</b><br/>
193      * <pre>
194      * mql mod form "${NAME}" \
195      *     description "DESCRIPTION" \
196      *     field name "ABC1" \
197      *          add setting "ABC1" "DEF" \
198      *     field name "ABC2" \
199      *          add setting "ABC2" "DEF"
200      * </pre>
201      * This source code builds a structure of lines like (tag names with the
202      * value in braces):
203      * <ul>
204      * <li>mql (mod form "${NAME}")
205      *     <ul>
206      *     <li>description ("DESCRIPTION")</li>
207      *     <li>field (name "ABC1")<ul>
208      *         <li>add (setting "ABC1" "DEF")</li></ul></li>
209      *     <li>field (name "ABC2")<ul>
210      *         <li>add (setting "ABC2" "DEF")</li></ul></li>
211      *     </ul>
212      *     </li>
213      * </ul>
214      * </p>
215      */
216     public final class Line
217     {
218         /***
219          * Complete text of this line.
220          */
221         private final String line;
222 
223         /***
224          * First characters before a space of the trimmed line.
225          */
226         private final String tag;
227 
228         /***
229          * Rest of the line after the tag.
230          */
231         private final String value;
232 
233         /***
234          * Parent line (where the {@link #shifting} is lower than the
235          * {@link #shifting} of this line).
236          */
237         private final Line parent;
238 
239         /***
240          * Children lines (where the {@link #shifting} is greater than the
241          * {@link #shifting} of this line).
242          */
243         private final List<ExportParser.Line> children = new ArrayList<ExportParser.Line>();
244 
245         /***
246          * Count of spaces before first character of the line.
247          */
248         private final int shifting;
249 
250         /***
251          *
252          * @param _lineIter     line iterator
253          * @param _prevLine     previsous line
254          */
255         private Line(final Iterator<String> _lineIter,
256                      final Line _prevLine)
257         {
258             this.line = _lineIter.next();
259             this.tag = this.line.trim().replaceAll(" .*", "");
260             this.value = this.line.trim().substring(this.tag.length()).replaceAll("////$", "").trim();
261             this.shifting = this.line.length() - this.line.trim().length();
262             if (_prevLine == null)  {
263                 this.parent = null;
264                 ExportParser.this.rootLines.add(this);
265             } else if ("".equals(this.line))  {
266                 this.parent = null;
267             } else  {
268                 Line curPar = _prevLine;
269                 while ((curPar != null) && (curPar.shifting >= this.shifting))  {
270                     curPar = curPar.parent;
271                 }
272                 this.parent = curPar;
273                 if (this.parent == null)  {
274                     ExportParser.this.rootLines.add(this);
275                 } else  {
276                     this.parent.children.add(this);
277                 }
278             }
279             // evaluate next line
280             if (_lineIter.hasNext())  {
281                 new Line(_lineIter, this);
282             }
283         }
284 
285         /***
286          *
287          * @param _path     array of path to evaluate
288          * @param _index    current index (level) in the <code>_path</code>
289          * @param _ret      list of all found strings
290          */
291         protected void evalPath(final String[] _path,
292                                 final int _index,
293                                 final List<String> _ret)
294         {
295             final String searchedTag = _path[_index];
296             if (this.tag.equals(searchedTag))  {
297 
298                 if (_index < (_path.length - 2))  {
299                     for (final Line child : this.children)  {
300                         child.evalPath(_path, _index + 1, _ret);
301                     }
302                 } else if (_index == (_path.length - 1))  {
303                     for (final Line child : this.children)  {
304                         _ret.add(child.line.trim());
305                     }
306                 } else if ("@value".equals(_path[_index + 1]))  {
307                     _ret.add(this.value);
308                 }
309             }
310         }
311 
312         /***
313          * String representation of this line including the {@link #tag} and
314          * {@link #value}.
315          *
316          * @return string representation of this line
317          */
318         @Override
319         public String toString()
320         {
321             return "[line " + this.tag + " (" + this.value + ")]";
322         }
323     }
