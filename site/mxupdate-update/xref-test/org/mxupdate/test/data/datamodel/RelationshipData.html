124     /***
125      * Returns the TCL update file of this relationship data instance.
126      *
127      * @return TCL update file content
128      */
129     @Override()
130     public String ciFile()
131     {
132         final StringBuilder cmd = new StringBuilder()
133                 .append("mql escape mod relationship \"${NAME}\"");
134 
135         // append hidden flag
136         if (this.isHidden() != null)  {
137             cmd.append(' ');
138             if (!this.isHidden())  {
139                 cmd.append('!');
140             }
141             cmd.append("hidden");
142         }
143 
144         this.append4CIFileValues(cmd);
145 
146         // prevent duplicates
147         if (this.preventDuplicates != null)  {
148             cmd.append(' ');
149             if (!this.preventDuplicates)  {
150                 cmd.append('!');
151             }
152             cmd.append("preventduplicates");
153         }
154 
155         // append from side
156         this.from.append4CIFile(cmd);
157 
158         // append to types
159         this.to.append4CIFile(cmd);
160 
161         // append attributes
162         this.append4CIAttributes(cmd);
163 
164         return cmd.toString();
165     }

167     /***
168      * Create the related relationship in MX for this relationship data
169      * instance and appends the {@link #attributes}.
170      *
171      * @return this relationship data instance
172      * @throws MatrixException if create failed
173      */
174     @Override()
175     public RelationshipData create()
176         throws MatrixException
177     {
178         if (!this.isCreated())  {
179             this.setCreated(true);
180 
181             this.createDependings();
182 
183             final StringBuilder cmd = new StringBuilder();
184             cmd.append("escape add relationship \"").append(AbstractTest.convertMql(this.getName())).append('\"');
185 
186             // append hidden flag
187             if (this.isHidden() != null)  {
188                 cmd.append(' ');
189                 if (!this.isHidden())  {
190                     cmd.append('!');
191                 }
192                 cmd.append("hidden");
193             }
194 
195             // prevent duplicates
196             if (this.preventDuplicates != null)  {
197                 cmd.append(' ');
198                 if (!this.preventDuplicates)  {
199                     cmd.append('!');
200                 }
201                 cmd.append("preventduplicates");
202             }
203 
204             // append from side
205             this.from.append4Create(cmd);
206 
207             // append to side
208             this.to.append4Create(cmd);
209 
210             this.append4Create(cmd);
211 
212             this.getTest().mql(cmd);
213         }
214 
215         return this;
216     }

218     /***
219      * {@inheritDoc}
220      * Creates all depending types and relationships.
221      *
222      * @see #from
223      * @see #to
224      */
225     @Override()
226     public RelationshipData createDependings()
227         throws MatrixException
228     {
229         super.createDependings();
230 
231         // create from types
232         for (final TypeData type : this.from.getTypes())  {
233             type.create();
234         }
235         // create from relationships
236         for (final RelationshipData relationship : this.from.getRelationships())  {
237             relationship.create();
238         }
239         // create to types
240         for (final TypeData type : this.to.getTypes())  {
241             type.create();
242         }
243         // create to relationships
244         for (final RelationshipData relationship : this.to.getRelationships())  {
245             relationship.create();
246         }
247 
248         return this;
249     }

272 
273         // from side
274         this.from.checkExport(_exportParser);
275 
276         // to side
277         this.to.checkExport(_exportParser);

559         /***
560          * Appends the TCL statements for the CI file for one side.
561          *
562          * @param _cmd      TCL string builder
563          */
564         protected void append4CIFile(final StringBuilder _cmd)
565         {
566             // propagate connection
567             if (this.propagateConnection != null)  {
568                 _cmd.append(' ').append(this.side).append(' ');
569                 if (!this.propagateConnection)  {
570                     _cmd.append('!');
571                 }
572                 _cmd.append("propagateconnection");
573             }
574 
575             // propagate modify
576             if (this.propagateModify != null)  {
577                 _cmd.append(' ').append(this.side).append(' ');
578                 if (!this.propagateModify)  {
579                     _cmd.append('!');
580                 }
581                 _cmd.append("propagatemodify");
582             }
583 
584             // meaning
585             if (this.meaning != null)  {
586                 _cmd.append(" meaning \"").append(StringUtil_mxJPO.convertTcl(this.meaning)).append('\"');
587             }
588 
589             // cardinality
590             if (this.cardinality != null)  {
591                 _cmd.append(" cardinality ").append(this.cardinality.name().toLowerCase());
592             }
593 
594             // clone behavior
595             if (this.clone != null)  {
596                 _cmd.append(" clone ").append(this.clone.name().toLowerCase());
597             }
598 
599             // clone behavior
600             if (this.revision != null)  {
601                 _cmd.append(" revision ").append(this.revision.name().toLowerCase());
602             }
603 
604             // append from types
605             if (this.allTypes)  {
606                 _cmd.append(" " + this.side + " add type all");
607             } else if (!this.types.isEmpty())  {
608                 _cmd.append(" " + this.side);
609                 for (final TypeData type : this.types)  {
610                     _cmd.append(" add type \"").append(AbstractTest.convertTcl(type.getName())).append("\"");
611                 }
612             }
613 
614             // append from relationships
615             if (this.allRelationships)  {
616                 _cmd.append(" " + this.side + " add relationship all");
617             } else if (!this.relationships.isEmpty())  {
618                 _cmd.append(" " + this.side);
619                 for (final RelationshipData relationship : this.relationships)  {
620                     _cmd.append(" add relationship \"").append(AbstractTest.convertTcl(relationship.getName())).append("\"");
621                 }
622             }
623         }
624 
625         /***
626          * Appends the MQL statements for the create for one side.
627          *
628          * @param _cmd      MQL string builder
629          * @throws MatrixException if sub types / relationships could not be
630          *                         created
631          */
632         protected void append4Create(final StringBuilder _cmd)
633             throws MatrixException
634         {
635             _cmd.append(' ').append(this.side);
636 
637             // propagate connection
638             _cmd.append(' ');
639             if ((this.propagateConnection == null) || !this.propagateConnection)  {
640                     _cmd.append('!');
641             }
642             _cmd.append("propagateconnection");
643 
644             // propagate modify
645             _cmd.append(' ');
646             if ((this.propagateModify == null) || !this.propagateModify)  {
647                 _cmd.append('!');
648             }
649             _cmd.append("propagatemodify");
650 
651             // meaning
652             if (this.meaning != null)  {
653                 _cmd.append(" meaning \"").append(StringUtil_mxJPO.convertMql(this.meaning)).append('\"');
654             }
655 
656             // cardinality
657             if (this.cardinality != null)  {
658                 _cmd.append(" cardinality ").append(this.cardinality.name().toLowerCase());
659             } else  {
660                 _cmd.append(" cardinality one");
661             }
662 
663             // clone behavior
664             if (this.clone != null)  {
665                 _cmd.append(" clone ").append(this.clone.name().toLowerCase());
666             }
667 
668             // clone behavior
669             if (this.revision != null)  {
670                 _cmd.append(" revision ").append(this.revision.name().toLowerCase());
671             }
672 
673             // append to types
674             if (this.allTypes)  {
675                 _cmd.append(" type all");
676             } else if (!this.types.isEmpty())  {
677                 _cmd.append(" type ");
678                 boolean first = true;
679                 for (final TypeData type : this.types)  {
680                     if (first)  {
681                         first = false;
682                     } else  {
683                         _cmd.append(',');
684                     }
685                     _cmd.append("\"").append(AbstractTest.convertMql(type.getName())).append("\"");
686                 }
687             }
688 
689             // append to relationships
690             if (this.allRelationships)  {
691                 _cmd.append(" relationship all");
692             } else if (!this.relationships.isEmpty())  {
693                 _cmd.append(" relationship ");
694                 boolean first = true;
695                 for (final RelationshipData relationship : this.relationships)  {
696                     if (first)  {
697                         first = false;
698                     } else  {
699                         _cmd.append(',');
700                     }
701                     _cmd.append("\"").append(AbstractTest.convertMql(relationship.getName())).append("\"");
702                 }
703             }
704         }

706         /***
707          * Checks the export for one side of a relationship.
708          *
709          * @param _exportParser     export parser
710          * @throws MatrixException if check failed
711          */
712         protected void checkExport(final ExportParser _exportParser)
713             throws MatrixException
714         {
715             // propagate modify
716             Assert.assertEquals(
717                     _exportParser.getLines("/mql/" + this.side + "/propagateconnection/@value").size(),
718                     ((this.propagateConnection != null) && this.propagateConnection) ? 1 : 0,
719                     "check propagate connection is defined " + _exportParser.getLines("/mql/" + this.side + "/propagateconnection/@value"));
720             Assert.assertEquals(
721                     _exportParser.getLines("/mql/" + this.side + "/!propagateconnection/@value").size(),
722                     ((this.propagateConnection != null) && this.propagateConnection) ? 0 : 1,
723                     "check propagate connection is defined " + _exportParser.getLines("/mql/" + this.side + "/!propagateconnection/@value"));
724 
725             // propagate modify
726             Assert.assertEquals(
727                     _exportParser.getLines("/mql/" + this.side + "/propagatemodify/@value").size(),
728                     ((this.propagateModify != null) && this.propagateModify) ? 1 : 0,
729                     "check propagate modify is defined " + _exportParser.getLines("/mql/" + this.side + "/propagatemodify/@value"));
730             Assert.assertEquals(
731                     _exportParser.getLines("/mql/" + this.side + "/!propagatemodify/@value").size(),
732                     ((this.propagateModify != null) && this.propagateModify) ? 0 : 1,
733                     "check propagate modify is defined " + _exportParser.getLines("/mql/" + this.side + "/!propagatemodify/@value"));
734 
735             // meaning
736             Assert.assertEquals(
737                     _exportParser.getLines("/mql/" + this.side + "/meaning/@value").get(0),
738                     (this.meaning != null) ? "\"" + StringUtil_mxJPO.convertTcl(this.meaning) + "\"" : "\"\"",
739                     "check for correct meaning");
740 
741             // cardinality
742             Assert.assertEquals(
743                     _exportParser.getLines("/mql/" + this.side + "/cardinality/@value").get(0),
744                     (this.cardinality == null) || (this.cardinality == RelationshipData.Cardinality.ONE) ? "\"One\"" : "\"N\"",
745                     "check for correct cardinality");
746 
747             // clone behavior
748             Assert.assertEquals(
749                     _exportParser.getLines("/mql/" + this.side + "/clone/@value").get(0),
750                     (this.clone != null) ? "\"" + this.clone.name().toLowerCase() + "\"" : "\"none\"",
751                     "check for correct clone behavior");
752 
753             // revision behavior
754             Assert.assertEquals(
755                     _exportParser.getLines("/mql/" + this.side + "/revision/@value").get(0),
756                     (this.revision != null) ? "\"" + this.revision.name().toLowerCase() + "\"" : "\"none\"",
757                     "check for correct revision behavior");
758 
759             // types and relationships
760             final Set<String> paresedFroms = new HashSet<String>(_exportParser.getLines("/mql/" + this.side + "/add/@value"));
761             Assert.assertEquals(
762                     paresedFroms.size(),
763                     ((this.allTypes) ? 1 : this.types.size()) + ((this.allRelationships) ? 1 : this.relationships.size()),
764                     "check that all " + this.side + " types and relationshipos are defined");
765             if (this.allTypes)  {
766                 Assert.assertTrue(
767                         paresedFroms.contains("type \"all\""),
768                         "check that all " + this.side + " types are defined " + paresedFroms);
769             } else  {
770                 for (final TypeData type : this.types)  {
771                     final String line = "type \"" + StringUtil_mxJPO.convertTcl(type.getName()) + "\"";
772                     Assert.assertTrue(
773                             paresedFroms.contains(line),
774                             "check that " + this.side + " " + type.getName() + " is defined");
775                 }
776             }
777             if (this.allRelationships)  {
778                 Assert.assertTrue(
779                         paresedFroms.contains("relationship \"all\""),
780                         "check that all " + this.side + " relationships are defined " + paresedFroms);
