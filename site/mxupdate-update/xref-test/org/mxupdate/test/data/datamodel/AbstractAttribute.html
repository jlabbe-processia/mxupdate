34  /***
35   * The class is used to define all types of attributes, to create them and test
36   * the result.
37   *
38   * @author The MxUpdate Team
39   * @version $Id: AbstractAttribute.java 495 2009-07-27 18:00:10Z tim.moxter $
40   * @param <T>   defines the class which is derived from this class
41   */
42  public abstract class AbstractAttribute<T extends AbstractAttribute<?>>
43      extends AbstractDataWithTrigger<T>
44  {
45      /***
46       * Within export the description and default value must be defined.
47       */
48      private static final Set<String> REQUIRED_EXPORT_VALUES = new HashSet<String>(2);
49      static  {
50          AbstractAttribute.REQUIRED_EXPORT_VALUES.add("description");
51          AbstractAttribute.REQUIRED_EXPORT_VALUES.add("default");
52      }
53  
54      /***
55       * Attribute type of the attribute (string, integer, ....).
56       */
57      private final String attrType;
58  
59      /***
60       * Ranges of this attribute.
61       *
62       * @see #addRange(Range)
63       */
64      private final Set<Range> ranges = new HashSet<Range>();
65  
66      /***
67       *
68       * @param _test     related test instance
69       * @param _ci       related configuration type of this attribute
70       * @param _name     name of this attribute
71       * @param _attrType type of this attribute
72       * @param _filePrefix   prefix for the file name
73       */
74      protected AbstractAttribute(final AbstractTest _test,
75                                  final AbstractTest.CI _ci,
76                                  final String _name,
77                                  final String _attrType,
78                                  final String _filePrefix)
79      {
80          super(_test, _ci, _name, _filePrefix, AbstractAttribute.REQUIRED_EXPORT_VALUES);
81          this.attrType = _attrType;
82      }
83  
84      /***
85       * Defines a new attribute range.
86       *
87       * @param _range    range to assign to this attribute
88       * @return this instance
89       * @see #ranges
90       */
91      @SuppressWarnings("unchecked")
92      public T addRange(final Range _range)
93      {
94          this.ranges.add(_range);
95          return (T) this;
96      }
97  
98      /***
99       * Creates a this attribute with all values and settings.
100      *
101      * @return this attribute instance
102      * @throws MatrixException if create failed
103      */
104     @Override()
105     @SuppressWarnings("unchecked")
106     public T create()
107         throws MatrixException
108     {
109         if (!this.isCreated())  {
110             this.setCreated(true);
111 
112             final StringBuilder cmd = new StringBuilder();
113             cmd.append("escape add attribute \"").append(AbstractTest.convertMql(this.getName()))
114                .append("\" type ").append(this.attrType);
115 
116             this.append4CreateValues(cmd);
117 
118             for (final Range range : this.ranges)  {
119                 range.appendCreate(cmd);
120             }
121 
122             this.getTest().mql(cmd);
123         }
124         return (T) this;
125     }
126 
127     /***
128      * Checks the export of this data piece if all values are correct defined.
129      *
130      * @param _exportParser     parsed export
131      * @throws MatrixException if check failed
132      */
133     @Override()
134     @SuppressWarnings("unchecked")
135     public void checkExport(final ExportParser _exportParser)
136         throws MatrixException
137     {
138         super.checkExport(_exportParser);
139 
140         // check range directly fetched from attribute
141         final AttributeType attr = new AttributeType(this.getName());
142         attr.open(this.getTest().getContext());
143         final StringList list = attr.getChoices(this.getTest().getContext());
144         attr.close(this.getTest().getContext());
145         final Set<String> l = new HashSet<String>();
146         if (list != null)  {
147             l.addAll(list);
148         }
149         for (final Range range : this.ranges)  {
150             Assert.assertTrue(l.contains(range.value),
151                               "check that range value " + range.value + " is defined");
152         }
153         Assert.assertEquals(l.size(), this.ranges.size(), "check that all ranges are defined");
154     }
155 
156     /***
157      * Prepares the configuration item update file depending on the
158      * configuration of this command.
159      *
160      * @return code for the configuration item update file
161      */
162     @Override
163     public String ciFile()
164     {
165         final StringBuilder cmd = new StringBuilder()
166                 .append("mql escape mod attribute \"${NAME}\"");
167         this.append4CIFileValues(cmd);
168 
169         return cmd.toString();
170     }
171 
172     /***
173      * Evaluates all 'adds' for ranges in the configuration item file.
174      *
175      * @param _needAdds     set with add strings used to append the adds for
176      *                      {@link #ranges}
177      */
178     @Override
179     protected void evalAdds4CheckExport(final Set<String> _needAdds)
180     {
181         super.evalAdds4CheckExport(_needAdds);
182         for (final Range range : this.ranges)  {
183             range.evalAdds4CheckExport(_needAdds);
184         }
185     }
186 
187     /***
188      * Used to define ranges with one value.
189      */
190     public static class Range
191     {
192         /***
193          * Comparator of this range.
194          */
195         private final String comparator;
196 
197         /***
198          * Value of this range.
199          */
200         private final String value;
201 
202         /***
203          * Initializes the range values.
204          *
205          * @param _comparator   comparator for the range
206          * @param _value        range value itself
207          */
208         public Range(final String _comparator,
209                      final String _value)
210         {
211             this.comparator = _comparator;
212             this.value = _value;
213         }
214 
215         /***
216          * Appends this range will so that the range is defined while the
217          * attribute is created.
218          *
219          * @param _cmd  string builder where the MQL commands will be appended
220          */
221         protected void appendCreate(final StringBuilder _cmd)
222         {
223             _cmd.append(" range ").append(this.comparator)
224                 .append(" \"").append(AbstractTest.convertMql(this.value))
225                 .append("\"");
226         }
227 
228         /***
229          * Appends the add statement in TCL code for this range.
230          *
231          * @param _needAdds     set with add strings used to append the adds
232          *                      for this range
233          */
234         protected void evalAdds4CheckExport(final Set<String> _needAdds)
235         {
236             final StringBuilder cmd = new StringBuilder()
237                     .append("range ").append(this.comparator)
238                     .append(" \"").append(AbstractTest.convertTclDoubleEscaped(this.value))
239                     .append("\"");
240             _needAdds.add(cmd.toString());
241         }
242     }
