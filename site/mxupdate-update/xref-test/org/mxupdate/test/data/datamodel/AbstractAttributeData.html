96      /***
97       * Creates a this attribute with all values and settings.
98       *
99       * @return this attribute instance
100      * @throws MatrixException if create failed
101      */
102     @Override()
103     @SuppressWarnings("unchecked")
104     public T create()
105         throws MatrixException
106     {
107         if (!this.isCreated())  {
108             this.setCreated(true);
109 
110             this.createDependings();
111 
112             final StringBuilder cmd = new StringBuilder();
113             cmd.append("escape add attribute \"").append(AbstractTest.convertMql(this.getName()))
114                .append("\" type ").append(this.attrType);
115 
116             // append hidden flag
117             if (this.isHidden() != null)  {
118                 cmd.append(' ');
119                 if (!this.isHidden())  {
120                     cmd.append('!');
121                 }
122                 cmd.append("hidden");
123             }
124 
125             this.append4Create(cmd);
126 
127             for (final AbstractRange range : this.ranges)  {
128                 range.appendCreate(cmd);
129             }
130 
131             this.getTest().mql(cmd);
132         }
133         return (T) this;
134     }

136     /***
137      * {@inheritDoc}
138      * Creates depending range programs.
139      */
140     @Override()
141     @SuppressWarnings("unchecked")
142     public T createDependings()
143         throws MatrixException
144     {
145         super.createDependings();
146 
147         // create range programs
148         for (final AbstractRange range : this.ranges)  {
149             if (range instanceof AbstractAttributeData.RangeProgram)  {
150                 ((AbstractAttributeData.RangeProgram) range).program.create();
151             }
152         }
153 
154         return (T) this;
155     }

157     /***
158      * Prepares the configuration item update file depending on the
159      * configuration of this command.
160      *
161      * @return code for the configuration item update file
162      */
163     @Override
164     public String ciFile()
165     {
166         final StringBuilder cmd = new StringBuilder()
167                 .append("mql escape mod attribute \"${NAME}\"");
168 
169         // append hidden flag
170         if (this.isHidden() != null)  {
171             cmd.append(' ');
172             if (!this.isHidden())  {
173                 cmd.append('!');
174             }
175             cmd.append("hidden");
176         }
177 
178         this.append4CIFileValues(cmd);
179 
180         return cmd.toString();
181     }

198     /***
199      * Used to define ranges with one value.
200      */
201     public abstract static class AbstractRange
202     {
203         /***
204          * Comparator of this range.
205          */
206         final String comparator;
207 
208         /***
209          * Value of this range.
210          */
211         final String value;
212 
213         /***
214          * Initializes the range values.
215          *
216          * @param _comparator   comparator for the range
217          * @param _value        range value itself
218          */
219         protected AbstractRange(final String _comparator,
220                                 final String _value)
221         {
222             this.comparator = _comparator;
223             this.value = _value;
224         }
225 
226         /***
227          * Appends this range will so that the range is defined while the
228          * attribute is created.
229          *
230          * @param _cmd  string builder where the MQL commands will be appended
231          */
232         protected void appendCreate(final StringBuilder _cmd)
233         {
234             _cmd.append(" range ").append(this.comparator)
235                 .append(" \"").append(AbstractTest.convertMql(this.value))
236                 .append("\"");
237         }
238 
239         /***
240          * Appends the add statement in TCL code for this range.
241          *
242          * @param _needAdds     set with add strings used to append the adds
243          *                      for this range
244          */
245         protected void evalAdds4CheckExport(final Set<String> _needAdds)
246         {
247             final StringBuilder cmd = new StringBuilder()
248                     .append("range ").append(this.comparator)
249                     .append(" \"").append(AbstractTest.convertTcl(this.value))
250                     .append("\"");
251             _needAdds.add(cmd.toString());
252         }
253     }
