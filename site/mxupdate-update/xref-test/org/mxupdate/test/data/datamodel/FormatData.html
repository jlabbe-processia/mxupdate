160     /***
161      * Returns the TCL update file of this format data instance.
162      *
163      * @return TCL update file content
164      */
165     @Override()
166     public String ciFile()
167     {
168         final StringBuilder cmd = new StringBuilder();
169 
170         this.append4CIFileHeader(cmd);
171 
172         cmd.append("mql escape mod format \"${NAME}\"");
173 
174         this.append4CIFileValues(cmd);
175 
176         if (this.viewProgram != null)  {
177             cmd.append(" view \"").append(AbstractTest.convertTcl(this.viewProgram.getName())).append('\"');;
178         }
179         if (this.editProgram != null)  {
180             cmd.append(" edit \"").append(AbstractTest.convertTcl(this.editProgram.getName())).append('\"');;
181         }
182         if (this.printProgram != null)  {
183             cmd.append(" print \"").append(AbstractTest.convertTcl(this.printProgram.getName())).append('\"');;
184         }
185 
186         return cmd.toString();
187     }

189     /***
190      * Create the related format in MX for this format data instance.
191      *
192      * @return this format data instance
193      * @throws MatrixException if create failed
194      */
195     @Override()
196     public FormatData create()
197         throws MatrixException
198     {
199         if (!this.isCreated())  {
200             this.setCreated(true);
201 
202             final StringBuilder cmd = new StringBuilder();
203             cmd.append("escape add format \"").append(AbstractTest.convertMql(this.getName())).append('\"');
204             this.append4Create(cmd);
205 
206             if (this.viewProgram != null)  {
207                 this.viewProgram.create();
208                 cmd.append(" view \"").append(AbstractTest.convertMql(this.viewProgram.getName())).append('\"');;
209             } else  {
210                 cmd.append(" view \"\"");
211             }
212             if (this.editProgram != null)  {
213                 this.editProgram.create();
214                 cmd.append(" edit \"").append(AbstractTest.convertMql(this.editProgram.getName())).append('\"');;
215             } else  {
216                 cmd.append(" edit \"\"");
217             }
218             if (this.printProgram != null)  {
219                 this.printProgram.create();
220                 cmd.append(" print \"").append(AbstractTest.convertMql(this.printProgram.getName())).append('\"');;
221             } else  {
222                 cmd.append(" print \"\"");
223             }
224 
225             cmd.append(";\n")
226                .append("escape add property ").append(this.getSymbolicName())
227                .append(" on program eServiceSchemaVariableMapping.tcl")
228                .append(" to format \"").append(AbstractTest.convertMql(this.getName())).append("\"");
229 
230             this.getTest().mql(cmd);
231         }
232         return this;
233     }

252     }
