221     /***
222      * Prepares and returns the string of the CI file.
223      *
224      * @return string of the CI file
225      */
226     @Override()
227     public String ciFile()
228     {
229         final StringBuilder strg = new StringBuilder();
230         this.append4CIFileHeader(strg);
231 
232         strg.append("updatePolicy \"${NAME}\" {\n")
233             .append("  hidden \"").append(this.isHidden() != null ? this.isHidden() : false).append("\"\n");
234 
235         // values
236         for (final Map.Entry<String,String> entry : this.getValues().entrySet())  {
237             strg.append(' ').append(entry.getKey()).append(" \"")
238                 .append(AbstractTest.convertTcl(entry.getValue()))
239                 .append('\"');
240         }
241 
242         // type definition
243         if (this.allTypes)  {
244             strg.append(" type all\n");
245         } else if (!this.types.isEmpty()) {
246             final List<String> typeNames = new ArrayList<String>();
247             for (final TypeData type : this.types)  {
248                 typeNames.add(type.getName());
249             }
250             strg.append(" type {").append(StringUtil_mxJPO.joinMql(' ', true, typeNames, null)).append("}\n");
251         }
252 
253         // format definition
254         if (this.allFormats)  {
255             strg.append(" format all\n");
256         } else if (!this.formats.isEmpty()) {
257             final List<String> formatNames = new ArrayList<String>();
258             for (final FormatData format : this.formats)  {
259                 formatNames.add(format.getName());
260             }
261             strg.append(" format {").append(StringUtil_mxJPO.joinMql(' ', true, formatNames, null)).append("}\n");
262         }
263 
264         if ((this.allState != null) && this.allState)  {
265             strg.append("  allstate {\n");
266             this.allStateAccess.append4CIFile(strg);
267             strg.append("  }\n");
268         }
269 
270         strg.append("}");
271 
272         // append properties
273         for (final PropertyDef prop : this.getProperties())  {
274             strg.append('\n').append(prop.getCITCLString(this.getCI()));
275         }
276 
277         return strg.toString();
278     }

280     /***
281      * Create the related policy in MX for this policy data instance.
282      *
283      * @return this policy data instance
284      * @throws MatrixException if create failed
285      */
286     @Override()
287     public PolicyData create()
288         throws MatrixException
289     {
290         if (!this.isCreated())  {
291             this.setCreated(true);
292 
293             final StringBuilder cmd = new StringBuilder();
294             cmd.append("escape add policy \"").append(AbstractTest.convertMql(this.getName())).append('\"');
295 
296             if ((this.isHidden() != null) && this.isHidden())  {
297                 cmd.append(" hidden");
298             }
299 
300             // assign types
301             if (this.allTypes)  {
302                 cmd.append(" type all ");
303             } else if (!this.types.isEmpty()) {
304                 final List<String> typeNames = new ArrayList<String>();
305                 for (final TypeData type : this.types)  {
306                     typeNames.add(type.getName());
307                     type.create();
308                 }
309                 cmd.append(" type ").append(StringUtil_mxJPO.joinMql(',', true, typeNames, null));
310             }
311 
312             // assign formats
313             if (this.allFormats)  {
314                 cmd.append(" format all ");
315             } else if (!this.formats.isEmpty()) {
316                 final List<String> formatNames = new ArrayList<String>();
317                 for (final FormatData format : this.formats)  {
318                     formatNames.add(format.getName());
319                     format.create();
320                 }
321                 cmd.append(" format ").append(StringUtil_mxJPO.joinMql(',', true, formatNames, null));
322             }
323 
324             if ((this.allState != null) && this.allState)  {
325                 cmd.append(" allstate ");
326                 this.allStateAccess.append4Create(cmd);
327             }
328 
329             this.append4Create(cmd);
330 
331             cmd.append(";\n")
332                 .append("escape add property ").append(this.getSymbolicName())
333                 .append(" on program eServiceSchemaVariableMapping.tcl")
334                 .append(" to policy \"").append(AbstractTest.convertMql(this.getName())).append("\"");
335 
336             this.getTest().mql(cmd);
337         }
338         return this;
339     }

341     /***
342      * Checks the export of this data piece if all values are correct defined.
343      *
344      * @param _exportParser     parsed export
345      * @throws MatrixException if check failed
346      */
347     @Override()
348     public void checkExport(final ExportParser _exportParser)
349         throws MatrixException
350     {
351         this.checkExportProperties(_exportParser);
352 
353         // check symbolic name
354         Assert.assertEquals(
355                 _exportParser.getSymbolicName(),
356                 this.getSymbolicName(),
357                 "check symbolic name");
358 
359         // check for all required values
360         for (final String valueName : PolicyData.REQUIRED_EXPORT_VALUES)  {
361             Assert.assertEquals(_exportParser.getLines("/updatePolicy/" + valueName + "/@value").size(),
362                                 1,
363                                 "required check that minimum and maximum one " + valueName + " is defined");
364         }
365         // check for defined values
366         for (final Map.Entry<String,String> entry : this.getValues().entrySet())  {
367             this.checkSingleValue(_exportParser,
368                                   entry.getKey(),
369                                   entry.getKey(),
370                                   "\"" + AbstractTest.convertTcl(entry.getValue()) + "\"");
371         }
372         // check for hidden flag
373         if ((this.isHidden() == null) || !this.isHidden())  {
374             this.checkSingleValue(_exportParser,
375                                   "hidden flag (must be false)",
376                                   "hidden",
377                                    "\"false\"");
378         } else  {
379             this.checkSingleValue(_exportParser,
380                                   "hidden flag (must be true)",
381                                   "hidden",
382                                   "\"true\"");
383         }
384 
385         // check for types
386         if (this.allTypes)  {
387             this.checkSingleValue(_exportParser, "all types", "type", "all");
388         } else if (!this.types.isEmpty()) {
389             final Set<String> typeNames = new TreeSet<String>();
390             for (final TypeData type : this.types)  {
391                 typeNames.add(type.getName());
392             }
393             this.checkSingleValue(_exportParser, "types", "type", "{" + StringUtil_mxJPO.joinMql(' ', true, typeNames, null) + "}");
394         } else  {
395             this.checkSingleValue(_exportParser, "types", "type", "{}");
396         }
397 
398         // check for formats
399         if (this.allFormats)  {
400             this.checkSingleValue(_exportParser, "all formats", "format", "all");
401         } else if (!this.formats.isEmpty()) {
402             final Set<String> formatNames = new TreeSet<String>();
403             for (final FormatData format : this.formats)  {
404                 formatNames.add(format.getName());
405             }
406             this.checkSingleValue(_exportParser, "formats", "format", "{" + StringUtil_mxJPO.joinMql(' ', true, formatNames, null) + "}");
407         } else  {
408             this.checkSingleValue(_exportParser, "formats", "format", "{}");
409         }
410 
411         // check for all state flag
412         if ((this.allState == null) || !this.allState)  {
413             Assert.assertEquals(_exportParser.getLines("/updatePolicy/allstate/@value").size(),
414                                 0,
415                                 "check that not all state access is defined");
416         } else  {
417             Assert.assertEquals(
418                     _exportParser.getLines("/updatePolicy/allstate/@value").size(),
419                     1,
420                     "check that all state access is defined");
421             Assert.assertEquals(
422                     _exportParser.getLines("/updatePolicy/allstate/owner/@value").get(0),
423                     this.allStateAccess.ownerAccess.getCITCLString(),
424                     "check that all state owner access is defined");
425             Assert.assertEquals(
426                     _exportParser.getLines("/updatePolicy/allstate/public/@value").get(0),
427                     this.allStateAccess.publicAccess.getCITCLString(),
428                     "check that all state public access is defined");
429             // owner / public revoke
430             final List<String> revokes = _exportParser.getLines("/updatePolicy/allstate/revoke/@value");
431             int count = 0;
432             if (!this.allStateAccess.ownerRevoke.access.isEmpty() || (this.allStateAccess.ownerRevoke.filter != null))  {
433                 Assert.assertTrue(
434                         revokes.contains("owner " + this.allStateAccess.ownerRevoke.getCITCLString()),
435                         "check that owner revoke is defined");
436                 count++;
437             }
438             if (!this.allStateAccess.publicRevoke.access.isEmpty() || (this.allStateAccess.publicRevoke.filter != null))  {
439                 Assert.assertTrue(
440                         revokes.contains("public " + this.allStateAccess.publicRevoke.getCITCLString()),
441                         "check that public revoke is defined");
442                 count++;
443             }
444             Assert.assertEquals(
445                     revokes.size(),
446                     count,
447                     "check that count of revokes is correct (have " + revokes + ")");
448             // user access
449             final Set<String> mustUsers = new HashSet<String>();
450             for (final PolicyData.UserAccessFilter userAccessFilter : this.allStateAccess.userAccessFilters)  {
451                 mustUsers.add(userAccessFilter.getCITCLString());
452             }
453             for (final String user : _exportParser.getLines("/updatePolicy/allstate/user/@value"))  {
454                 Assert.assertTrue(mustUsers.contains(user), "check that '" + user + "' is original defined");
455                 mustUsers.remove(user);
456             }
457             Assert.assertEquals(mustUsers.size(), 0, "check that all users are defined (have " + mustUsers + ")");
458         }
459     }

461     /***
462      * {@inheritDoc}
463      * The original method is overwritten because for policies another path
464      * exists for the values.
465      *
466      * @param _exportParser     parsed export
467      * @param _kind             kind of the check
468      * @param _tag              tag to check
469      * @param _value            value to check (or <code>null</code> if value
470      *                          is not defined)
471      */
472     @Override()
473     public void checkSingleValue(final ExportParser _exportParser,
474                                  final String _kind,
475                                  final String _tag,
476                                  final String _value)
477     {
478         if (_value != null)  {
479             Assert.assertEquals(_exportParser.getLines("/updatePolicy/" + _tag + "/@value").size(),
480                                 1,
481                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is defined");
482             Assert.assertEquals(_exportParser.getLines("/updatePolicy/" + _tag + "/@value").get(0),
483                                 _value,
484                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is " + _value);
485 
486         } else  {
487             Assert.assertEquals(_exportParser.getLines("/updatePolicy/" + _tag + "/@value").size(),
488                                 0,
489                                 "check " + _kind + " '" + this.getName() + "' that no " + _tag + " is defined");
490         }
491     }

532             this.filter = _filter;
533             return (AF) this;
534         }

536         /***
537          * Prepares the TCL string which are used within the CI file.
538          *
539          * @return TCL string
540          */
541         protected String getCITCLString()
542         {
543             return "{" + StringUtil_mxJPO.joinTcl(' ', false, this.access, "none") +"}"
544                 + ((this.filter != null) ? " filter \"" + StringUtil_mxJPO.convertTcl(this.filter) + "\"" : "");
545         }
546 
547         /***
548          * Returns the MQL string used within create of the policy.
549          *
550          * @return MQL create string

553         {
554             final StringBuilder ret = new StringBuilder();
555             ret.append(StringUtil_mxJPO.joinMql(',', false, this.access, "none"));
556             if (this.filter != null)  {
557                 ret.append(" filter \"").append(AbstractTest.convertTcl(this.filter)).append('\"');
558             }
559             return ret.toString();
560         }
