251     @Override()
252     protected void evalAdds4CheckExport(final Set<String> _needAdds)
253     {
254         super.evalAdds4CheckExport(_needAdds);
255 
256         // append types
257         if (this.allTypes)  {
258             _needAdds.add("type all");
259         } else  {
260             for (final TypeData type : this.types)  {
261                 final StringBuilder cmd = new StringBuilder()
262                         .append("type \"").append(AbstractTest.convertTcl(type.getName())).append("\"");
263                 _needAdds.add(cmd.toString());
264             }
265         }
266 
267         // append relationships
268         if (this.allRelationships)  {
269             _needAdds.add("relationship all");
270         } else  {
271             for (final RelationshipData relationship : this.relationships)  {
272                 final StringBuilder cmd = new StringBuilder()
273                         .append("relationship \"").append(AbstractTest.convertTcl(relationship.getName())).append("\"");
274                 _needAdds.add(cmd.toString());
275             }
276         }
277     }

286     {
287         final StringBuilder cmd = new StringBuilder()
288                 .append("mql escape mod interface \"${NAME}\"");
289         this.append4CIFileValues(cmd);
290 
291         // append attributes
292         cmd.append("\n\ntestAttributes -interface \"${NAME}\" -attributes [list //\n");
293         for (final AbstractAttributeData<?> attribute : this.attributes)  {
294             cmd.append("    \"").append(AbstractTest.convertTcl(attribute.getName())).append("\" //\n");
295         }
296         cmd.append("]\n");
297 
298         // append parent interfaces
299         cmd.append("\n\ntestParents -interface \"${NAME}\" -parents [list //\n");
300         for (final InterfaceData parent : this.parents)  {
301             cmd.append("    \"").append(AbstractTest.convertTcl(parent.getName())).append("\" //\n");
302         }
303         cmd.append("]\n");
304 
305         return cmd.toString();
306     }

308     /***
309      * Create the related interface in MX for this type data instance and
310      * the {@link #types}.
311      *
312      * @return this interface data instance
313      * @throws MatrixException if create failed
314      * @see #types
315      */
316     @Override()
317     public InterfaceData create()
318         throws MatrixException
319     {
320         if (!this.isCreated())  {
321             this.setCreated(true);
322 
323             final StringBuilder cmd = new StringBuilder();
324             cmd.append("escape add interface \"").append(AbstractTest.convertMql(this.getName())).append('\"');
325 
326             // add parent interfaces
327             if (!this.parents.isEmpty())  {
328                 cmd.append(" derived ");
329                 boolean first = true;
330                 for (final InterfaceData parent : this.parents)  {
331                     parent.create();
332                     if (first)  {
333                         first = false;
334                     } else  {
335                         cmd.append(',');
336                     }
337                     cmd.append("\"").append(AbstractTest.convertMql(parent.getName())).append("\"");
338                 }
339             }
340 
341             // add types
342             if (this.allTypes)  {
343                 cmd.append(" type all");
344             } else if (!this.types.isEmpty())  {
345                 cmd.append(" type ");
346                 boolean first = true;
347                 for (final TypeData type : this.types)  {
348                     type.create();
349                     if (first)  {
350                         first = false;
351                     } else  {
352                         cmd.append(',');
353                     }
354                     cmd.append("\"").append(AbstractTest.convertMql(type.getName())).append("\"");
355                 }
356             }
357 
358             // add relationships
359             if (this.allRelationships)  {
360                 cmd.append(" relationship all");
361             } else if (!this.relationships.isEmpty())  {
362                 cmd.append(" relationship ");
363                 boolean first = true;
364                 for (final RelationshipData relationship : this.relationships)  {
365                     relationship.create();
366                     if (first)  {
367                         first = false;
368                     } else  {
369                         cmd.append(',');
370                     }
371                     cmd.append("\"").append(AbstractTest.convertMql(relationship.getName())).append("\"");
372                 }
373             }
374 
375             // append attributes
376             for (final AbstractAttributeData<?> attribute : this.attributes)  {
377                 attribute.create();
378                 cmd.append(" attribute \"").append(AbstractTest.convertMql(attribute.getName())).append("\"");
379             }
380 
381             this.append4Create(cmd);
382 
383             this.getTest().mql(cmd);
384         }
385         return this;
386     }

388     /***
389      * Checks the export of this data piece if all values are correct defined.
390      *
391      * @param _exportParser     parsed export
392      * @throws MatrixException if check failed
393      */
394     @Override()
395     public void checkExport(final ExportParser _exportParser)
396         throws MatrixException
397     {
398         super.checkExport(_exportParser);
399 
400         // check parent interfaces
401         final Set<String> pars = new HashSet<String>(_exportParser.getLines("/testParents/"));
402         for (final InterfaceData parent : this.parents)  {
403             final String parentName = "\"" + AbstractTest.convertTcl(parent.getName()) + "\" //";
404             Assert.assertTrue(pars.contains(parentName),
405                               "check that parent interface '" + parent.getName() + "' is defined");
406         }
407         Assert.assertEquals(pars.size(),
408                             this.parents.size(),
409                             "check all parent interfaces are defined");
410 
411         // check attributes
412         final Set<String> attrs = new HashSet<String>(_exportParser.getLines("/testAttributes/"));
413         for (final AbstractAttributeData<?> attribute : this.attributes)  {
414             final String attrName = "\"" + AbstractTest.convertTcl(attribute.getName()) + "\" //";
415             Assert.assertTrue(attrs.contains(attrName),
416                               "check that attribute '" + attribute.getName() + "' is defined");
417         }
418         Assert.assertEquals(attrs.size(),
419                             this.attributes.size(),
420                             "check all attributes are defined");
421     }
