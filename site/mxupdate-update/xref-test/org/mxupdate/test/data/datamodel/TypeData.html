33  /***
34   * Used to define a type, create them and test the result.
35   *
36   * @author The MxUpdate Team
37   * @version $Id: TypeData.java 503 2009-08-09 19:16:18Z tim.moxter $
38   */
39  public class TypeData
40      extends AbstractDataWithTrigger<TypeData>
41  {
42      /***
43       * Within export the description and default value must be defined.
44       */
45      private static final Set<String> REQUIRED_EXPORT_VALUES = new HashSet<String>(2);
46      static  {
47          TypeData.REQUIRED_EXPORT_VALUES.add("description");
48      }
49  
50      /***
51       * All methods of this type.
52       *
53       * @see #addMethod(AbstractProgram)
54       * @see #create()
55       */
56      private final Set<AbstractProgram<?>> methods = new HashSet<AbstractProgram<?>>();
57  
58      /***
59       * All attributes of this type.
60       */
61      private final Set<AbstractAttribute<?>> attributes = new HashSet<AbstractAttribute<?>>();
62  
63      /***
64       * Initialize this type data with given <code>_name</code>.
65       *
66       * @param _test     related test implementation (where this type is
67       *                  defined)
68       * @param _name     name of the type
69       */
70      public TypeData(final AbstractTest _test,
71                      final String _name)
72      {
73          super(_test, AbstractTest.CI.TYPE, _name, "TYPE_", TypeData.REQUIRED_EXPORT_VALUES);
74      }
75  
76      /***
77       * Assigns the <code>_method</code> to this type.
78       *
79       * @param _method   method to assign
80       * @return this type data instance
81       * @see #methods
82       */
83      public TypeData addMethod(final AbstractProgram<?> _method)
84      {
85          this.methods.add(_method);
86          return this;
87      }
88  
89      /***
90       * Assigns the <code>_attribute</code> to this attribute.
91       *
92       * @param _attribute        attribute to assign
93       * @return this type data instance
94       * @see #attributes
95       */
96      public TypeData addAttribute(final AbstractAttribute<?> _attribute)
97      {
98          this.attributes.add(_attribute);
99          return this;
100     }
101 
102     /***
103      * Returns the TCL update file of this type data instance.
104      *
105      * @return TCL update file content
106      */
107     @Override
108     public String ciFile()
109     {
110         final StringBuilder cmd = new StringBuilder()
111                 .append("mql escape mod type \"${NAME}\"");
112         this.append4CIFileValues(cmd);
113 
114         // append attributes
115         cmd.append("\n\ntestAttributes -type \"${NAME}\" -attributes [list //\n");
116         for (final AbstractAttribute<?> attribute : this.attributes)  {
117             cmd.append("    \"").append(AbstractTest.convertTcl(attribute.getName())).append("\" //\n");
118         }
119         cmd.append("]\n");
120 
121         return cmd.toString();
122     }
123 
124     /***
125      * Create the related type in MX for this type data instance and appends
126      * the {@link #methods} and {@link #attributes}.
127      *
128      * @return this type data instance
129      * @throws MatrixException if create failed
130      * @see #methods
131      * @see #attributes
132      */
133     @Override()
134     public TypeData create()
135         throws MatrixException
136     {
137         if (!this.isCreated())  {
138             this.setCreated(true);
139 
140             final StringBuilder cmd = new StringBuilder();
141             cmd.append("escape add type \"").append(AbstractTest.convertMql(this.getName())).append('\"');
142 
143             // append methods
144             for (final AbstractProgram<?> method : this.methods)  {
145                 method.create();
146                 cmd.append(" method \"").append(AbstractTest.convertMql(method.getName())).append("\"");
147             }
148 
149             // append attributes
150             for (final AbstractAttribute<?> attribute : this.attributes)  {
151                 attribute.create();
152                 cmd.append(" attribute \"").append(AbstractTest.convertMql(attribute.getName())).append("\"");
153             }
154 
155             this.append4CreateValues(cmd);
156 
157             this.getTest().mql(cmd);
158         }
159 
160         return this;
161     }
162 
163     /***
164      * Appends the adds for the {@link #methods} and {@link #attributes}.
165      *
166      * @param _needAdds     set with add strings used to append the adds
167      * @see #methods
168      * @see #attributes
169      */
170     @Override
171     protected void evalAdds4CheckExport(final Set<String> _needAdds)
172     {
173         super.evalAdds4CheckExport(_needAdds);
174 
175         // append methods
176         for (final AbstractProgram<?> method : this.methods)  {
177             final StringBuilder cmd = new StringBuilder()
178                     .append("method \"").append(AbstractTest.convertTcl(method.getName())).append("\"");
179             _needAdds.add(cmd.toString());
180         }
181     }
182 
183     /***
184      * Checks the export of this data piece if all values are correct defined.
185      *
186      * @param _exportParser     parsed export
187      * @throws MatrixException if check failed
188      */
189     @Override
190     public void checkExport(final ExportParser _exportParser)
191         throws MatrixException
192     {
193         super.checkExport(_exportParser);
194 
195         // check attributes
196         final Set<String> attrs = new HashSet<String>(_exportParser.getLines("/testAttributes/"));
197         for (final AbstractAttribute<?> attribute : this.attributes)  {
198             final String attrName = "\"" + AbstractTest.convertTcl(attribute.getName()) + "\" //";
199             Assert.assertTrue(attrs.contains(attrName),
200                               "check that attribute '" + attribute.getName() + "' is defined");
201         }
202         Assert.assertEquals(attrs.size(),
203                             this.attributes.size(),
204                             "check all attributes are defined");
205     }
