45       * Within export the description and label must be defined.

100     /***
101      * Prepares the configuration item update file depending on the
102      * configuration of this portal.
103      *
104      * @return code for the configuration item update file
105      */
106     @Override
107     public String ciFile()
108     {
109         final StringBuilder cmd = new StringBuilder();
110         this.append4CIFileHeader(cmd);
111         cmd.append("mql escape mod portal \"${NAME}\"");
112 
113         // append channels
114         for (final ChannelData channel : this.channels)  {
115             cmd.append(" //\n  place \"").append(AbstractTest.convertTcl(channel.getName())).append("\" after \"\"");
116         }
117 
118         this.append4CIFileValues(cmd);
119 
120         return cmd.toString();
121     }
122 
123     /***
124      * Creates a this portal with all values and settings.
125      *
126      * @return this portal instance
127      * @throws MatrixException if create failed
128      */
129     @Override
130     public PortalData create()
131         throws MatrixException
132     {
133         if (!this.isCreated())  {
134             this.setCreated(true);
135 
136             final StringBuilder cmd = new StringBuilder()
137                     .append("escape add portal \"" + AbstractTest.convertMql(this.getName()) + "\"");
138             // append all channels
139             if (!this.channels.isEmpty())  {
140                 final List<String> names = new ArrayList<String>();
141                 for (final ChannelData channel : this.channels)  {
142                     channel.create();
143                     names.add(channel.getName());
144                 }
145                 cmd.append(" channel ").append(StringUtil_mxJPO.joinMql(',', true, names, null));
146             }
147             this.append4Create(cmd);
148 
149             cmd.append(";\n")
150                .append("escape add property ").append(this.getSymbolicName())
151                .append(" on program eServiceSchemaVariableMapping.tcl")
152                .append(" to portal \"").append(AbstractTest.convertMql(this.getName())).append("\"");
153 
154             this.getTest().mql(cmd);
155         }
156         return this;
157     }

159     /***
160      * {@inheritDoc}
161      * Also the channels which are defined as place are checked.
162      */
163     @Override()
164     public void checkExport(final ExportParser _exportParser)
165         throws MatrixException
166     {
167         super.checkExport(_exportParser);
168         // check for channels (they are not add's!)
169         final Set<String> needPlaces = new HashSet<String>();
170         for (final ChannelData channel : this.channels)  {
171             needPlaces.add("\"" + AbstractTest.convertTcl(channel.getName()) + "\" after \"\"");
172         }
173         final List<String> foundPlaces = _exportParser.getLines("/mql/place/@value");
174         Assert.assertEquals(foundPlaces.size(), needPlaces.size(), "all adds defined (found places = " + foundPlaces + "; need places = " + needPlaces + ")");
175         for (final String foundPlace : foundPlaces)  {
176             Assert.assertTrue(needPlaces.contains(foundPlace), "check that place '" + foundPlace + "' is defined");
177         }
178     }
