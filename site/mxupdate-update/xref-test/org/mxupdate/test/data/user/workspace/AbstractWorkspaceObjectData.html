38  /***
39   * The class is used to define all common things for workspace objects related
40   * to users used to create / update and to export.

42   * @author The MxUpdate Team
43   * @version $Id: AbstractWorkspaceObjectData.java 803 2010-05-24 23:56:50Z tim.moxter $

127     /***
128      * Creates this workspace object.
129      *
130      * @return this workspace object instance
131      * @throws MatrixException if create failed
132      */
133     @SuppressWarnings("unchecked")
134     @Override()
135     public DATA create()
136         throws MatrixException
137     {
138         if (!this.isCreated())  {
139             this.setCreated(true);
140 
141             final StringBuilder cmd = new StringBuilder()
142                     .append("escape add ").append(this.mxAdminType).append(" \"")
143                     .append(AbstractTest.convertMql(this.getName()))
144                     .append("\" user \"")
145                     .append(AbstractTest.convertMql(this.user.getName()))
146                     .append('\"');
147 
148             // append values
149             this.append4Create(cmd);
150 
151             this.getTest().mql(cmd);
152         }
153         return (DATA) this;
154     }

164     @SuppressWarnings("unchecked")
165     public DATA createDependings()

177     /***
178      * Appends the MQL commands to define the hidden flag and the
179      * {@link #visible users} within a create.
180      *
181      * @param _cmd  string builder used to append MQL commands
182      * @throws MatrixException if used object could not be created
183      * @see #values
184      */
185     @Override()
186     protected void append4Create(final StringBuilder _cmd)
187         throws MatrixException
188     {
189         super.append4Create(_cmd);
190 
191         // hidden flag
192         if (this.isHidden())  {
193             _cmd.append(" hidden");
194         } else  {
195             _cmd.append(" !hidden");
196         }
197 
198         // visible users
199         if (!this.visible.isEmpty())  {
200             _cmd.append(" visible ");
201             boolean first = true;
202             for (final AbstractUserData<?> user : this.visible)  {
203                 if (first)  {
204                     first = false;
205                 } else  {
206                     _cmd.append(',');
207                 }
208                 _cmd.append('\"').append(AbstractTest.convertMql(user.getName())).append('\"');
209             }
210         }
211     }

213     /***
214      * Returns the part of the CI file to create this workspace object of an
215      * user.
216      *
217      * @return part of the CI file to create this workspace object of an user
218      */
219     @Override()
220     public String ciFile()
221     {
222         final StringBuilder cmd = new StringBuilder()
223                 .append("mql escape add ").append(this.mxAdminType).append(" \"")
224                 .append(AbstractTest.convertTcl(this.getName()))
225                 .append("\" //\n    user \"${NAME}\"");
226 
227         // hidden flag
228         if (this.isHidden())  {
229             cmd.append(" //\n    hidden");
230         } else  {
231             cmd.append(" //\n    !hidden");
232         }
233 
234         // visible users
235         if (!this.visible.isEmpty())  {
236             for (final AbstractUserData<?> user : this.visible)  {
237                 cmd.append(" //\n    visible \"")
238                    .append(AbstractTest.convertTcl(user.getName()))
239                    .append('\"');
240             }
241         }
242 
243         this.append4CIFileValues(cmd);
244         return cmd.toString();
245     }

247     /***
248      * Checks the export of this data piece if all values are correct defined.
249      * The hidden flag and the {@link #visible visible users} are checked.
250      *
251      * @param _exportParser     parsed export
252      * @throws MatrixException if check failed
253      */
254     @Override()
255     public void checkExport(final ExportParser _exportParser)
256         throws MatrixException
257     {
258         super.checkExport(_exportParser);
259 
260         // check visible users
261         final List<String> foundVisible = _exportParser.getLines("/mql/visible/@value");
262         for (final String user : foundVisible)  {
263             final String tmpUser = user.replaceAll("^\"", "").replaceAll("\"$", "");
264             boolean found = false;
265             for (final AbstractUserData<?> userAbstr : this.visible)  {
266                 if (tmpUser.equals(userAbstr.getName()))  {
267                     found = true;
268                     break;
269                 }
270             }
271             Assert.assertTrue(found, "check that '" + tmpUser + "' was defined visible");
272         }
273         Assert.assertEquals(foundVisible.size(),
274                             this.visible.size(),
275                             "check that all visible users are defined");
276 
277         // check properties
278         final Set<String> foundProps = new HashSet<String>(_exportParser.getLines("/mql/property/@value"));
279         for (final PropertyDef property : this.getProperties())  {
280             final StringBuilder key = new StringBuilder()
281                     .append("\"").append(AbstractTest.convertTcl(property.getName())).append("\"");
282             if (property.getTo() != null)  {
283                 key.append(" to ").append(property.getTo().getCI().getMxType()).append(" \"")
284                     .append(AbstractTest.convertTcl(property.getTo().getName())).append("\"");
285             }
286             if (property.getValue() != null)  {
287                 key.append(" value \"").append(AbstractTest.convertTcl(property.getValue())).append("\"");
288             }
289             foundProps.remove(key.toString());
290         }
291         Assert.assertTrue(foundProps.isEmpty(), "check that all properties are defined");
292     }
