255     /***
256      * Appends the person specific values to the TCL update code
257      * <code>_cmd</code> of the configuration item file. This includes:
258      * <ul>
259      * <li>{@link #access}</li>
260      * <li>{@link #adminAccess administration access}</li>
261      * <li>{@link #types}</li>
262      * <li>{@link #groups}<li>
263      * <li>{@link #roles}</li>
264      * <li>{@link #passwordNeverExpires password never expires flag}</li>
265      * <li>{@link #wantsEmail wants email flag}</li>
266      * <li>{@link #wantsIconMail wants icon mail flag}</li>
267      * </ul>
268      *
269      * @param _cmd  string builder with the TCL commands of the configuration
270      *              item file
271      * @see #values
272      */
273     @Override()
274     protected void append4CIFileValues(final StringBuilder _cmd)
275     {
276         super.append4CIFileValues(_cmd);
277         _cmd.append(" access ").append(StringUtil_mxJPO.joinTcl(',', false, this.access, "none"))
278             .append(" admin ").append(StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none"));
279         if (!this.types.isEmpty())  {
280             _cmd.append(" type ").append(StringUtil_mxJPO.joinTcl(',', false, this.types, null)).append(";\n");
281         }
282         // groups
283         for (final GroupData group : this.groups)  {
284             _cmd.append(" assign group \"").append(AbstractTest.convertTcl(group.getName())).append("\"");
285         }
286         // roles
287         for (final RoleData role : this.roles)  {
288             _cmd.append(" assign role \"").append(AbstractTest.convertTcl(role.getName())).append("\"");
289         }
290         // password never expires flag
291         if (this.passwordNeverExpires != null)  {
292             _cmd.append(' ');
293             if (!this.passwordNeverExpires)  {
294                 _cmd.append('!');
295             }
296             _cmd.append("neverexpires");
297         }
298         // wants email
299         if (this.wantsEmail != null)  {
300             _cmd.append(' ')
301                 .append(this.wantsEmail ? "enable" : "disable")
302                 .append(" email");
303         }
304         // wants icon mail
305         if (this.wantsIconMail != null)  {
306             _cmd.append(' ')
307                 .append(this.wantsIconMail ? "enable" : "disable")
308                 .append(" iconmail");
309         }
310     }

404     /***
405      * Checks the export of this person if all values are correct defined. It
406      * is checked that
407      * <ul>
408      * <li>all {@link #access} is correct defined</li>
409      * <li>all {@link #adminAccess administration access} is correct defined
410      *     </li>
411      * <li>all {@link #groups} are correct defined</li>
412      * <li>all {@link #roles} are correct defined</li>
413      * <li>{@link #passwordNeverExpires password never expires flag} is defined
414      *     if set to true</li>
415      * <li>{@link #wantsEmail wants email flag} is correct defined</li>
416      * <li>{@link #wantsIconMail wants icon mail flag} is correct defined</li>
417      * </ul>
418      *
419      * @param _exportParser     parsed export
420      * @throws MatrixException if check failed
421      */
422     @Override()
423     public void checkExport(final ExportParser _exportParser)
424         throws MatrixException
425     {
426         super.checkExport(_exportParser);
427 
428         // check access
429         this.checkSingleValue(_exportParser,
430                               "person",
431                               "access",
432                               "\"" + StringUtil_mxJPO.joinTcl(',', false, this.access, "none") + "\"");
433 
434         // check administration access
435         this.checkSingleValue(_exportParser,
436                               "person",
437                               "admin",
438                               "\"" + StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none") + "\"");
439 
440         // prepare list of all assignments
441         final Set<String> assigns = new HashSet<String>();
442         for (final GroupData group : this.groups)  {
443             assigns.add("group \"" + AbstractTest.convertTcl(group.getName()) + "\"");
444         }
445         for (final RoleData role : this.roles)  {
446             assigns.add("role \"" + AbstractTest.convertTcl(role.getName()) + "\"");
447         }
448 
449         // remove all assigned groups / roles in the TCL update file
450         for (final String assign : _exportParser.getLines("/mql/assign/@value"))  {
451             if (assigns.contains(assign))  {
452                 assigns.remove(assign);
453             } else  {
454                 Assert.assertNotNull(assign, "assign " + assign + "not known");
455             }
456         }
457         Assert.assertEquals(assigns.size(), 0, "check that all assignments for groups / roles are defined");
458 
459         // password never expires flag
460         this.checkValueExists(_exportParser, "person", "neverexpires", (this.passwordNeverExpires != null) && this.passwordNeverExpires);
461         this.checkValueExists(_exportParser, "person", "!neverexpires", (this.passwordNeverExpires == null) || !this.passwordNeverExpires);
462 
463         // email / icon mail flags
464         final Set<String> enabled = new HashSet<String>(_exportParser.getLines("/mql/enable/@value"));
465         final Set<String> disabled = new HashSet<String>(_exportParser.getLines("/mql/disable/@value"));
466         if ((this.wantsEmail != null) && this.wantsEmail)  {
467             Assert.assertTrue(enabled.contains("email"), "check email enabled");
468             Assert.assertFalse(disabled.contains("email"), "check email not disabled");
469         } else  {
470             Assert.assertFalse(enabled.contains("email"), "check email not enabled");
471             Assert.assertTrue(disabled.contains("email"), "check email disabled");
472         }
473         if ((this.wantsIconMail == null) || this.wantsIconMail)  {
474             Assert.assertTrue(enabled.contains("iconmail"), "check icon mail enabled");
475             Assert.assertFalse(disabled.contains("iconmail"), "check icon mail not disabled");
476         } else  {
477             Assert.assertFalse(enabled.contains("iconmail"), "check icon mail not enabled");
478             Assert.assertTrue(disabled.contains("iconmail"), "check icon mail disabled");
479         }
480     }
