384     /***
385      * Tests that for given <code>_mxType</code> all related values of
386      * <code>_datas</code> are defined.
387      *
388      * @param _exportParser     parsed export
389      * @param _mxType           related MX administration type
390      * @param _datas            set of visual workspace objects to test
391      */
392     private void check(final ExportParser _exportParser,
393                        final String _mxType,
394                        final Set<? extends AbstractWorkspaceObjectData<?,USER>>... _datas)
395     {
396         // prepare target values
397         final Set<String> targets = new HashSet<String>();
398         for (final Set<? extends AbstractWorkspaceObjectData<?,USER>> oneDatas : _datas)  {
399             for (final AbstractWorkspaceObjectData<?,USER> data : oneDatas)  {
400                 targets.add(data.getMxAdminType() + " \"" + AbstractTest.convertTcl(data.getName()) + "\"");
401             }
402         }
403         // check against parsed data
404         for (final String current : _exportParser.getLines("/mql/" + _mxType + "/@value"))  {
405             final String test = current.replaceAll("////$", "").trim();
406             Assert.assertTrue(targets.contains(test), "check that " + _mxType + " " + test + " must be defined");
407             targets.remove(test);
408         }
409         Assert.assertTrue(targets.isEmpty(), "check that all " + _mxType + "'s are defined (not found " + targets + ")");
410     }
