26  import matrix.util.MatrixException;

28  import org.mxupdate.test.AbstractTest;
29  import org.mxupdate.test.ExportParser;
30  import org.mxupdate.test.data.user.AbstractUserData;
31  import org.testng.Assert;

33  /***
34   * The class is used to define all common things for workspace objects related
35   * to users used to create / update and to export.
36   *
37   * @author The MxUpdate Team
38   * @version $Id: AbstractVisualWorkspaceObjectData.java 664 2010-01-17 20:08:58Z tim.moxter $
39   * @param <DATA> workspace object class
40   * @param <USER> for which user class the workspace object class is defined
41   */
42  public class AbstractVisualWorkspaceObjectData<DATA extends AbstractVisualWorkspaceObjectData<?,USER>, USER extends AbstractUserData<?>>
43          extends AbstractWorkspaceObjectData<DATA,USER>
44  {
45      /***
46       * Is this visual workspace object active?
47       *
48       * @see #setActive(boolean)
49       * @see #ciFile()
50       * @see #append4Create(StringBuilder)
51       * @see #checkExport(ExportParser)
52       */
53      private boolean active = true;
54  
55      /***
56       * Default constructor.
57       *
58       * @param _test                 related test case
59       * @param _mxAdminType          MX administration type of the visual query
60       *                              workspace object
61       * @param _user                 user for which this visual query workspace
62       *                              object is defined
63       * @param _name                 name of the visual query workspace object
64       * @param _requiredExportValues defines the required values of the
65       *                              export within the configuration item
66       *                              file
67       */
68      AbstractVisualWorkspaceObjectData(final AbstractTest _test,
69                                        final String _mxAdminType,
70                                        final USER _user,
71                                        final String _name,
72                                        final Set<String> _requiredExportValues)
73      {
74          super(_test, _mxAdminType, _user, _name, _requiredExportValues);
75      }
76  
77      /***
78       * Defines if this visual workspace object object is active or not.
79       *
80       * @param _active   <i>true</i> if active; otherwise <i>false</i>
81       * @return this visual workspace object instance
82       */
83      @SuppressWarnings("unchecked")
84      public DATA setActive(final boolean _active)
85      {
86          this.active = _active;
87          return (DATA) this;
88      }
89  
90      /***
91       * Returns the part of the CI file to create this visual workspace object
92       * of an user.
93       *
94       * @return part of the CI file to create this visual workspace object of an
95       *         user
96       */
97      @Override()
98      public String ciFile()
99      {
100         final StringBuilder cmd = new StringBuilder().append(super.ciFile());
101 
102         // active flag
103         if (this.active)  {
104             cmd.append(" //\n    active");
105         } else  {
106             cmd.append(" //\n    !active");
107         }
108 
109         return cmd.toString();
110     }
111 
112 
113     /***
114      * Appends the MQL commands to define the {@link #active flag} within a
115      * create.
116      *
117      * @param _cmd  string builder used to append MQL commands
118      * @throws MatrixException if used object could not be created
119      * @see #active
120      */
121     @Override()
122     protected void append4Create(final StringBuilder _cmd)
123         throws MatrixException
124     {
125         super.append4Create(_cmd);
126 
127         // active flag
128         if (this.active)  {
129             _cmd.append(" active");
130         } else  {
131             _cmd.append(" !active");
132         }
133     }
134 
135     /***
136      * Checks the export of this tool set data if all values are correct
137      * defined. The {@link #active active flag} is checked.
138      *
139      * @param _exportParser     parsed export
140      * @throws MatrixException if check failed
141      */
142     @Override()
143     public void checkExport(final ExportParser _exportParser)
144         throws MatrixException
145     {
146         super.checkExport(_exportParser);
147 
148         // check active flag
149         final Set<String> main = new HashSet<String>(_exportParser.getLines("/mql/"));
150         if (this.active)  {
151             Assert.assertTrue(main.contains("active") || main.contains("active //"),
152                               "check that tool set '" + this.getName() + "' is active");
153         } else  {
154             Assert.assertTrue(main.contains("!active") || main.contains("!active //"),
155                               "check that tool set '" + this.getName() + "' is not active");
156         }
157     }
