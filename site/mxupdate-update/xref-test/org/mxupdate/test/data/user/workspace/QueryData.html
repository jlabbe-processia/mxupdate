35  /***
36   * The class is used to define all query objects related to users used to
37   * create / update and to export.
38   *
39   * @author The MxUpdate Team

43  public class QueryData<USER extends AbstractUserData<?>>
44      extends AbstractQueryWorkspaceObjectData<QueryData<USER>,USER>
45  {
46      /***
47       * Within export the description must be defined.
48       */
49      private static final Map<String,String> REQUIRED_EXPORT_VALUES = new HashMap<String,String>();
50      static  {
51          QueryData.REQUIRED_EXPORT_VALUES.put("user", "");

88      /***
89       *
90       * Appends the type, name and revision defined in the values as
91       * business object pattern to the <code>_cmd</code> string builder for
92       * the create.
93       *
94       * @param _cmd  string builder with the MQL commands for the create
95       * @throws MatrixException if failed
96       */
97      @Override()
98      protected void append4Create(final StringBuilder _cmd)
99          throws MatrixException
100     {
101         // cache type, name and revision
102         final String type = this.getValue("type");
103         final String name = this.getValue("name");
104         final String revision = this.getValue("revision");
105 
106         // remove type, name and revision from values
107         this.getValues().remove("type");
108         this.getValues().remove("name");
109         this.getValues().remove("revision");
110 
111         super.append4Create(_cmd);
112 
113         // and append again type name revision
114         this.setValue("type", type);
115         this.setValue("name", name);
116         this.setValue("revision", revision);
117 
118         _cmd.append(" businessobject")
119             .append(" \"").append(AbstractTest.convertMql(type)).append("\"")
120             .append(" \"").append(AbstractTest.convertMql(name)).append("\"")
121             .append(" \"").append(AbstractTest.convertMql(revision)).append("\"");
122 
123         // expand type flag
124         if (this.expandType)  {
125             _cmd.append(" expandtype");
126         } else  {
127             _cmd.append(" !expandtype");
128         }
129     }

131     /***
132      * Appends the type, name and revision defined in the values as
133      * business object pattern to the <code>_cmd</code> string builder for
134      * the configuration item file.
135      *
136      * @param _cmd  string builder with the TCL commands of the configuration
137      *              item file
138      */
139     @Override()
140     protected void append4CIFileValues(final StringBuilder _cmd)
141     {
142         // cache type, name and revision
143         final String type = this.getValue("type");
144         final String name = this.getValue("name");
145         final String revision = this.getValue("revision");
146 
147         // cache type, name and revision
148         this.getValues().remove("type");
149         this.getValues().remove("name");
150         this.getValues().remove("revision");
151 
152         super.append4CIFileValues(_cmd);
153 
154         // and append again type name revision
155         this.setValue("type", type);
156         this.setValue("name", name);
157         this.setValue("revision", revision);
158 
159         // business object pattern
160         _cmd.append(" //\n    businessobject")
161             .append(" \"").append(AbstractTest.convertTcl(type)).append("\"")
162             .append(" \"").append(AbstractTest.convertTcl(name)).append("\"")
163             .append(" \"").append(AbstractTest.convertTcl(revision)).append("\"");
164 
165         // expand type flag
166         if (this.expandType)  {
167             _cmd.append(" //\n    expandtype");
168         } else  {
169             _cmd.append(" //\n    !expandtype");
170         }
171     }

173     /***
174      * Checks the export of this data piece if all values are correct defined.
175      *
176      * @param _exportParser     parsed export
177      * @throws MatrixException if check failed
178      */
179     @Override
180     public void checkExport(final ExportParser _exportParser)
181         throws MatrixException
182     {
183         // cache type, name and revision
184         final String type = this.getValue("type");
185         final String name = this.getValue("name");
186         final String revision = this.getValue("revision");
187 
188         // cache type, name and revision
189         this.getValues().remove("type");
190         this.getValues().remove("name");
191         this.getValues().remove("revision");
192 
193         super.checkExport(_exportParser);
194 
195         // and append again type name revision
196         this.setValue("type", type);
197         this.setValue("name", name);
198         this.setValue("revision", revision);
199 
200         // check busines object pattern
201         Assert.assertEquals(_exportParser.getLines("/mql/businessobject/@value").size(),
202                             1,
203                             "check that minimum and maximum one business object pattern is defined");
204         Assert.assertEquals(_exportParser.getLines("/mql/businessobject/@value").get(0),
205                             "\"" + AbstractTest.convertTcl(type) + "\" \"" + AbstractTest.convertTcl(name) + "\" \""
206                                         + AbstractTest.convertTcl(revision) + "\"",
207                             "check that business object pattern is correct defined");
208 
209         // check expand type flag
210         final Set<String> main = new HashSet<String>(_exportParser.getLines("/mql/"));
211         if (this.expandType)  {
212             Assert.assertTrue(main.contains("expandtype") || main.contains("expandtype //"),
213                               "check that query has expand type");
214         } else  {
215             Assert.assertTrue(main.contains("!expandtype") || main.contains("!expandtype //"),
216                               "check that has no expand type");
217         }
218     }
