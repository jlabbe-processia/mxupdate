368     /***
369      * <p>Prepares the configuration item update file depending on the
370      * configuration of this user.</p>
371      * This includes:
372      * <ul>
373      * <li>{@link #cues}</li>
374      * <li>{@link #filters}</li>
375      * <li>{@link #queries}</li>
376      * <li>{@link #tables}</li>
377      * <li>{@link #tips}</li>
378      * <li>{@link #toolSets tool sets}</li>
379      * <li>{@link #views}</li>
380      * </ul>
381      *
382      * @return code for the configuration item update file
383      */
384     @Override()
385     public String ciFile()
386     {
387         final StringBuilder cmd = new StringBuilder()
388                 .append("mql escape mod " + this.getCI().getMxType() + " \"${NAME}\"");
389         // hidden flag
390         if (this.isHidden() != null)  {
391             if (this.isHidden())  {
392                 cmd.append(" //\n    hidden");
393             } else  {
394                 cmd.append(" //\n    !hidden");
395             }
396         }
397 
398         this.append4CIFileValues(cmd);
399 
400         cmd.append(";\n");
401 
402         // cues
403         for (final CueData<DATA> cue : this.cues)  {
404             cmd.append(cue.ciFile()).append('\n');
405         }
406 
407         // filters
408         for (final FilterData<DATA> filter : this.filters)  {
409             cmd.append(filter.ciFile()).append('\n');
410         }
411 
412         // queries
413         for (final QueryData<DATA> query : this.queries)  {
414             cmd.append(query.ciFile()).append('\n');
415         }
416 
417         // tables
418         for (final TableData<DATA> table : this.tables)  {
419             cmd.append(table.ciFile()).append('\n');
420         }
421 
422         // tips
423         for (final TipData<DATA> tip : this.tips)  {
424             cmd.append(tip.ciFile()).append('\n');
425         }
426 
427         // tool sets
428         for (final ToolSetData<DATA> toolSet : this.toolSets)  {
429             cmd.append(toolSet.ciFile()).append('\n');
430         }
431 
432         // views
433         for (final ViewData<DATA> view : this.views)  {
434             cmd.append(view.ciFile()).append('\n');
435         }
436 
437         return cmd.toString();
438     }

440     /***
441      * Creates this user. This includes:
442      * <ul>
443      * <li>{@link #cues}</li>
444      * <li>{@link #filters}</li>
445      * <li>{@link #queries}</li>
446      * <li>{@link #tables}</li>
447      * <li>{@link #tips}</li>
448      * <li>{@link #toolSets tool sets}</li>
449      * <li>{@link #views}</li>
450      * </ul>
451      *
452      * @return this collection user data instance
453      * @throws MatrixException if create failed
454      */
455     @SuppressWarnings("unchecked")
456     @Override()
457     public DATA create()
458         throws MatrixException
459     {
460         if (!this.isCreated())  {
461             this.setCreated(true);
462 
463             this.createDependings();
464 
465             final StringBuilder cmd = new StringBuilder()
466                     .append("escape add ")
467                     .append(this.getCI().getMxType())
468                     .append(" \"").append(AbstractTest.convertMql(this.getName())).append("\"");
469             // hidden flag
470             if (this.isHidden() != null)  {
471                 if (this.isHidden())  {
472                     cmd.append(" hidden");
473                 } else  {
474                     cmd.append(" !hidden");
475                 }
476             }
477             // if site assigned, the site must be created
478             if (this.site != null)  {
479                 this.site.create();
480             }
481             this.append4Create(cmd);
482             cmd.append(";\n");
483             this.getTest().mql(cmd);
484 
485             // cues
486             for (final CueData<DATA> cue : this.cues)  {
487                 cue.create();
488             }
489 
490             // filters
491             for (final FilterData<DATA> filter : this.filters)  {
492                 filter.create();
493             }
494 
495             // queries
496             for (final QueryData<DATA> query : this.queries)  {
497                 query.create();
498             }
499 
500             // tables
501             for (final TableData<DATA> table : this.tables)  {
502                 table.create();
503             }
504 
505             // tips
506             for (final TipData<DATA> tip : this.tips)  {
507                 tip.create();
508             }
509 
510             // tool sets
511             for (final ToolSetData<DATA> toolSet : this.toolSets)  {
512                 toolSet.create();
513             }
514 
515             // views
516             for (final ViewData<DATA> view : this.views)  {
517                 view.create();
518             }
519         }
520         return (DATA) this;
521     }

523     /***
524      * {@inheritDoc}
525      * Creates depending {@link #site} and properties for {@link #cues},
526      * {@link #filters}, {@link #queries}, {@link #tables}, {@link #tips},
527      * {@link #toolSets} and {@link #views}.
528      *
529      * @see #site
530      * @see #cues
531      * @see #filters
532      * @see #queries
533      * @see #tables
534      * @see #tips
535      * @see #toolSets
536      * @see #views
537      */
538     @Override()
539     @SuppressWarnings("unchecked")
540     public DATA createDependings()
541         throws MatrixException
542     {
543         super.createDependings();
544 
545         // create site
546         if (this.site != null)  {
547             this.site.create();
548         }
549         // create cue properties
550         for (final CueData<DATA> cue : this.cues)  {
551             cue.createDependings();
552         }
553         // create filter properties
554         for (final FilterData<DATA> filter : this.filters)  {
555             filter.createDependings();
556         }
557         // create query properties
558         for (final QueryData<DATA> query : this.queries)  {
559             query.createDependings();
560         }
561         // create table properties
562         for (final TableData<DATA> table : this.tables)  {
563             table.createDependings();
564         }
565         // create tip properties
566         for (final TipData<DATA> tip : this.tips)  {
567             tip.createDependings();
568         }
569         // create tool set properties and programs
570         for (final ToolSetData<DATA> toolSet : this.toolSets)  {
571             toolSet.createDependings();
572         }
573         // create view properties
574         for (final ViewData<DATA> view : this.views)  {
575             view.createDependings();
576         }
577 
578         return (DATA) this;
579     }

581     /***
582      * Checks the export of this data piece if all values are correct defined.
583      * Following workspace objects for this user are checked that they are
584      * defined:
585      * <ul>
586      * <li>hidden flag</li>
587      * <li>{@link #site}</li>
588      * <li>{@link #cues}</li>
589      * <li>{@link #filters}</li>
590      * <li>{@link #queries}</li>
591      * <li>{@link #tables}</li>
592      * <li>{@link #tips}</li>
593      * <li>{@link #toolSets tool sets}</li>
594      * <li>{@link #views}</li>
595      * </ul>
596      *
597      * @param _exportParser     parsed export
598      * @throws MatrixException if check failed
599      */
600     @Override()
601     public void checkExport(final ExportParser _exportParser)
602         throws MatrixException
603     {
604         // site
605         this.checkSingleValue(_exportParser,
606                               this.getCI().getMxType(),
607                               "site",
608                               (this.site != null)
609                                       ? "\"" + AbstractTest.convertTcl(this.site.getName()) + "\""
610                                       : null);
611 
612         // all workspace objects
613         final Set<CueData<DATA>> tmpCues            = new HashSet<CueData<DATA>>(this.cues);
614         final Set<FilterData<DATA>> tmpFilters      = new HashSet<FilterData<DATA>>(this.filters);
615         final Set<QueryData<DATA>> tmpQueries       = new HashSet<QueryData<DATA>>(this.queries);
616         final Set<TableData<DATA>> tmpTables        = new HashSet<TableData<DATA>>(this.tables);
617         final Set<TipData<DATA>> tmpTips            = new HashSet<TipData<DATA>>(this.tips);
618         final Set<ToolSetData<DATA>> tmpToolSets    = new HashSet<ToolSetData<DATA>>(this.toolSets);
619         final Set<ViewData<DATA>> tmpViews          = new HashSet<ViewData<DATA>>(this.views);
620 
621         final List<ExportParser.Line> lines = new ArrayList<ExportParser.Line>();
622         for (final ExportParser.Line rootLine : _exportParser.getRootLines())  {
623             // cues
624             if (rootLine.getValue().startsWith("escape add cue "))  {
625                 for (final CueData<DATA> cue : this.cues)  {
626                     final String key = new StringBuilder()
627                             .append("escape add cue \"")
628                             .append(AbstractTest.convertTcl(cue.getName()))
629                             .append("\"").toString();
630                     if (key.equals(rootLine.getValue()))  {
631                         tmpCues.remove(cue);
632                         cue.checkExport(new ExportParser(cue.getName(), rootLine));
633                         break;
634                     }
635                 }
636             // filters
637             } else if (rootLine.getValue().startsWith("escape add filter "))  {
638                 for (final FilterData<DATA> filter : this.filters)  {
639                     final String key = new StringBuilder()
640                             .append("escape add filter \"")
641                             .append(AbstractTest.convertTcl(filter.getName()))
642                             .append("\"").toString();
643                     if (key.equals(rootLine.getValue()))  {
644                         tmpFilters.remove(filter);
645                         filter.checkExport(new ExportParser(filter.getName(), rootLine));
646                         break;
647                     }
648                 }
649             // queries
650             } else if (rootLine.getValue().startsWith("escape add query "))  {
651                 for (final QueryData<DATA> query : this.queries)  {
652                     final String key = new StringBuilder()
653                             .append("escape add query \"")
654                             .append(AbstractTest.convertTcl(query.getName()))
655                             .append("\"").toString();
656                     if (key.equals(rootLine.getValue()))  {
657                         tmpQueries.remove(query);
658                         query.checkExport(new ExportParser(query.getName(), rootLine));
659                         break;
660                     }
661                 }
662             // tables
663             } else if (rootLine.getValue().startsWith("escape add table "))  {
664                 for (final TableData<DATA> table : this.tables)  {
665                     final String key = new StringBuilder()
666                             .append("escape add table \"")
667                             .append(AbstractTest.convertTcl(table.getName()))
668                             .append("\"").toString();
669                     if (key.equals(rootLine.getValue()))  {
670                         tmpTables.remove(table);
671                         table.checkExport(new ExportParser(table.getName(), rootLine));
672                         break;
673                     }
674                 }
675             // tips
676             } else if (rootLine.getValue().startsWith("escape add tip "))  {
677                 for (final TipData<DATA> tip : this.tips)  {
678                     final String key = new StringBuilder()
679                             .append("escape add tip \"")
680                             .append(AbstractTest.convertTcl(tip.getName()))
681                             .append("\"").toString();
682                     if (key.equals(rootLine.getValue()))  {
683                         tmpTips.remove(tip);
684                         tip.checkExport(new ExportParser(tip.getName(), rootLine));
685                         break;
686                     }
687                 }
688             // tool sets
689             } else if (rootLine.getValue().startsWith("escape add toolset "))  {
690                 for (final ToolSetData<DATA> toolSet : this.toolSets)  {
691                     final String key = new StringBuilder()
692                             .append("escape add toolset \"")
693                             .append(AbstractTest.convertTcl(toolSet.getName()))
694                             .append("\"").toString();
695                     if (key.equals(rootLine.getValue()))  {
696                         tmpToolSets.remove(toolSet);
697                         toolSet.checkExport(new ExportParser(toolSet.getName(), rootLine));
698                         break;
699                     }
700                 }
701             // views
702             } else if (rootLine.getValue().startsWith("escape add view "))  {
703                 for (final ViewData<DATA> view : this.views)  {
704                     final String key = new StringBuilder()
705                             .append("escape add view \"")
706                             .append(AbstractTest.convertTcl(view.getName()))
707                             .append("\"").toString();
708                     if (key.equals(rootLine.getValue()))  {
709                         tmpViews.remove(view);
710                         view.checkExport(new ExportParser(view.getName(), rootLine));
711                         break;
712                     }
713                 }
714             // all other
715             } else  {
716                 lines.add(rootLine);
717             }
718         }
719 
720         // and check all others
721         super.checkExport(new ExportParser(
722                 _exportParser.getName(),
723                 _exportParser.getSymbolicName(),
724                 lines.toArray(new ExportParser.Line[lines.size()])));
725
