33  /***
34   * The class is used to define all filter objects related to users used to
35   * create / update and to export.
36   *
37   * @author The MxUpdate Team
38   * @version $Id: FilterData.java 664 2010-01-17 20:08:58Z tim.moxter $
39   * @param <USER> class of the related user for which this filter is defined
40   */
41  public class FilterData<USER extends AbstractUserData<?>>
42      extends AbstractVisualQueryWorkspaceObjectData<FilterData<USER>,USER>
43  {
44      /***
45       * Within export the description must be defined.
46       */
47      private static final Set<String> REQUIRED_EXPORT_VALUES = new HashSet<String>(3);
48      static  {
49          FilterData.REQUIRED_EXPORT_VALUES.add("user");
50          FilterData.REQUIRED_EXPORT_VALUES.add("appliesto");
51          FilterData.REQUIRED_EXPORT_VALUES.add("type");
52          FilterData.REQUIRED_EXPORT_VALUES.add("name");
53          FilterData.REQUIRED_EXPORT_VALUES.add("revision");
54          FilterData.REQUIRED_EXPORT_VALUES.add("vault");
55          FilterData.REQUIRED_EXPORT_VALUES.add("owner");
56      }
57  
58      /***
59       * Defines the direction of the connections to which the filter applies.
60       */
61      private Direction direction = Direction.BOTH;
62  
63      /***
64       * Default constructor.
65       *
66       * @param _test     related test case
67       * @param _user     user for which this filter is defined
68       * @param _name     name of the filter
69       */
70      public FilterData(final AbstractTest _test,
71                        final USER _user,
72                        final String _name)
73      {
74          super(_test, "filter", _user, _name, FilterData.REQUIRED_EXPORT_VALUES);
75          this.setValue("appliesto", "relationship");
76      }
77  
78      /***
79       * Defines the direction of the connections this filter applies.
80       *
81       * @param _direction    direction of the connections
82       * @return this filter data instance
83       * @see #direction
84       */
85      public FilterData<USER> setDirection(final Direction _direction)
86      {
87          this.direction = _direction;
88          return this;
89      }
90  
91      /***
92       * Returns the part of the CI file to create this filter of an user.
93       *
94       * @return part of the CI file to create this filter of an user
95       */
96      @Override()
97      public String ciFile()
98      {
99          final StringBuilder cmd = new StringBuilder().append(super.ciFile());
100 
101         // active flag
102         if (this.direction != null)  {
103             cmd.append(" //\n    ").append(this.direction.toString().toLowerCase());
104         }
105 
106         return cmd.toString();
107     }
108 
109     /***
110      * Appends the MQL commands to define the {@link #direction} within a
111      * create.
112      *
113      * @param _cmd  string builder used to append MQL commands
114      * @throws MatrixException if used object could not be created
115      * @see #values
116      */
117     @Override()
118     protected void append4Create(final StringBuilder _cmd)
119         throws MatrixException
120     {
121         super.append4Create(_cmd);
122 
123         // direction
124         if (this.direction != null)  {
125             _cmd.append(" ").append(this.direction.toString().toLowerCase());
126         }
127     }

129     /***
130      * Checks the export of this data piece if all values are correct defined.
131      * The {@link #active active flag} is checked.
132      *
133      * @param _exportParser     parsed export
134      * @throws MatrixException if check failed
135      */
136     @Override()
137     public void checkExport(final ExportParser _exportParser)
138         throws MatrixException
139     {
140         super.checkExport(_exportParser);
141 
142         // check direction
143         final Set<String> main = new HashSet<String>(_exportParser.getLines("/mql/"));
144         switch (this.direction)  {
145             case TO:
146                 Assert.assertTrue(main.contains("to") || main.contains("to //"),        "check to direction");
147                 Assert.assertFalse(main.contains("from") || main.contains("from //"),   "check no from direction");
148                 Assert.assertFalse(main.contains("both") || main.contains("both //"),   "check no both direction");
149                 break;
150             case FROM:
151                 Assert.assertFalse(main.contains("to") || main.contains("to //"),       "check no to direction");
152                 Assert.assertTrue(main.contains("from") || main.contains("from //"),    "check from direction");
153                 Assert.assertFalse(main.contains("both") || main.contains("both //"),   "check no both direction");
154                 break;
155             case BOTH:
156             default:
157                 Assert.assertFalse(main.contains("to") || main.contains("to //"),       "check no to direction");
158                 Assert.assertFalse(main.contains("from") || main.contains("from //"),   "check no from direction");
159                 Assert.assertTrue(main.contains("both") || main.contains("both //"),    "check both direction");
160         }
161     }

179          * Both (to and from) directions.

181         BOTH;
