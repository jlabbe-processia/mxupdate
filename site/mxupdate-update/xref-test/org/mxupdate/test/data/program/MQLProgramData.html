30  /***
31   * The class is used to define all MQL program objects used to create / update
32   * and to export.
33   *
34   * @author The MxUpdate Team
35   * @version $Id: MQLProgramData.java 799 2010-05-24 21:08:55Z tim.moxter $
36   */
37  @Test()
38  public class MQLProgramData
39      extends AbstractProgramData<MQLProgramData>
40  {
41      /***
42       * Start marker for the update code.
43       */
44      private static final String MARK_START
45              = "################################################################################\n"
46              + "# START NEEDED MQL UPDATE FOR THIS PROGRAM                                     #\n"
47              + "################################################################################";
48  
49      /***
50       * End marker for the update code.
51       */
52      private static final String MARK_END
53              = "################################################################################\n"
54              + "# END NEEDED MQL UPDATE FOR THIS PROGRAM                                       #\n"
55              + "################################################################################";
56  
57      /***
58       * Initializes this MQL program.
59       *
60       * @param _test     related test instance
61       * @param _name     name of the MQL program
62       */
63      public MQLProgramData(final AbstractTest _test,
64                            final String _name)
65      {
66          super(_test, AbstractTest.CI.PRG_MQL_PROGRAM, _name);
67      }
68  
69      /***
70       * Creates this MQL program within MX.
71       *
72       * @return this MQL program instance
73       * @throws MatrixException if create of the MQL program failed
74       */
75      @Override()
76      public MQLProgramData create()
77          throws MatrixException
78      {
79          if (!this.isCreated())  {
80              final StringBuilder cmd = new StringBuilder()
81                      .append("escape add program \"").append(AbstractTest.convertMql(this.getName()))
82                      .append("\" mql ")
83                      .append("code \"").append(AbstractTest.convertMql(this.getCode())).append("\"");
84              this.append4Create(cmd);
85              cmd.append(";\n")
86                 .append("escape add property ").append(this.getSymbolicName())
87                 .append(" on program eServiceSchemaVariableMapping.tcl")
88                 .append(" to program \"").append(AbstractTest.convertMql(this.getName())).append("\"");
89  
90              this.getTest().mql(cmd);
91              this.setCreated(true);
92          }
93          return this;
94      }
95  
96      /***
97       * Returns the configuration item file name of this MQL program. The
98       * configuration item file name of a MQL program excludes the
99       * <code>.tcl</code> extension.
100      *
101      * @return file name of a JPO
102      */
103     @Override()
104     public String getCIFileName()
105     {
106         final String ciFileName = super.getCIFileName();
107         return ciFileName.replaceAll("//.tcl$", "");
108     }
109 
110     /***
111      * The related configuration item file is the {@link #code} of the program.
112      *
113      *  @return {@link #code} of the program
114      *  @see #code
115      */
116     @Override()
117     public String ciFile()
118     {
119         final StringBuilder cmd = new StringBuilder()
120            .append(MQLProgramData.MARK_START)
121            .append("\nmql escape mod program ${NAME} ");
122         this.append4CIFileValues(cmd);
123         cmd.append("\n")
124            .append(MQLProgramData.MARK_END)
125            .append("\n");
126 
127         final StringBuilder ciFile = new StringBuilder();
128         for (final String line : cmd.toString().split("\n"))  {
129             if (this.getName().endsWith(".tcl"))  {
130                 ciFile.append('#');
131             }
132             ciFile.append(line).append('\n');
133         }
134         ciFile.append('\n').append(this.getCode());
135 
136         return ciFile.toString();
137     }
138 
139     /***
140      * {@inheritDoc}
141      * The source code of the MQL program is checked. Because it could be that
142      * some properties are defined, the MQL update code within the source code
143      * will be removed for the test.
144      */
145     @Override()
146     public void checkExport(final ExportParser _exportParser)
147         throws MatrixException
148     {
149         super.checkExport(_exportParser);
150 
151         // check MQL code
152         final String markEnd;
153         if (this.getName().endsWith(".tcl"))  {
154             final StringBuilder markEndBuilder = new StringBuilder();
155             for (final String line : MQLProgramData.MARK_END.split("\n"))  {
156                 markEndBuilder.append('#').append(line).append('\n');
157             }
158             markEnd = markEndBuilder.toString();
159         } else  {
160             markEnd = MQLProgramData.MARK_END;
161         }
162         final int index = _exportParser.getOrigCode().indexOf(markEnd);
163         Assert.assertEquals(
164                 ((index > 0) ? _exportParser.getOrigCode().substring(index + markEnd.length()).trim() : _exportParser.getOrigCode()).trim(),
165                 this.getCode(),
166                 "checks MQL program code");
167     }
