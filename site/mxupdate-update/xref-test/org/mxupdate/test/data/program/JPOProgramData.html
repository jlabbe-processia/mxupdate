21  package org.mxupdate.test.data.program;
22  
23  import matrix.util.MatrixException;
24  
25  import org.mxupdate.test.AbstractTest;
26  
27  /***
28   * The class is used to define all JPO program objects used to create / update
29   * and to export.
30   *
31   * @author The MxUpdate Team
32   * @version $Id: JPOProgramData.java 594 2009-10-29 00:37:31Z tim.moxter $
33   */
34  public class JPOProgramData
35      extends AbstractProgramData<JPOProgramData>
36  {
37      /***
38       * Start marker for the update code.
39       */
40      private static final String MARK_START
41              = "################################################################################\n"
42              + "# START NEEDED MQL UPDATE FOR THIS JPO PROGRAM                                 #\n"
43              + "################################################################################";
44  
45      /***
46       * End marker for the update code.
47       */
48      private static final String MARK_END
49              = "################################################################################\n"
50              + "# END NEEDED MQL UPDATE FOR THIS JPO PROGRAM                                   #\n"
51              + "################################################################################";
52  
53      /***
54       * Initializes this JPO program.
55       *
56       * @param _test     related test instance
57       * @param _name     name of the JPO program
58       */
59      public JPOProgramData(final AbstractTest _test,
60                            final String _name)
61      {
62          super(_test, AbstractTest.CI.PRG_JPO, _name);
63      }
64  
65      /***
66       * Returns the configuration item file name of this JPO. The configuration
67       * item file name of a JPO excludes the package name and has the suffix of
68       * <code>_mxJPO.java</code>.
69       *
70       * @return file name of a JPO
71       */
72      @Override()
73      public String getCIFileName()
74      {
75          final String ciFileName = super.getCIFileName();
76  
77          return ciFileName.replaceAll("//.tcl$", "")
78                           .replaceAll(".*//.", "")
79                          + "_mxJPO.java";
80      }
81  
82      /***
83       * The related configuration item file is the {@link #code} of the program.
84       *
85       *  @return {@link #code} of the program
86       *  @see #code
87       */
88      @Override()
89      public String ciFile()
90      {
91          final StringBuilder cmd = new StringBuilder()
92             .append(JPOProgramData.MARK_START)
93             .append("\nmql escape mod program ${NAME} ");
94          this.append4CIFileValues(cmd);
95          cmd.append("\n")
96             .append(JPOProgramData.MARK_END)
97             .append("\n");
98  
99          final StringBuilder ciFile = new StringBuilder();
100         for (final String line : cmd.toString().split("\n"))  {
101             ciFile.append("//").append(line).append('\n');
102         }
103         ciFile.append('\n').append(this.getCode());
104 
105         return ciFile.toString();
106     }
107 
108     /***
109      * Creates this JPO program within MX.
110      *
111      * @return this JPO program instance
112      * @throws MatrixException if create of the JPO program failed
113      */
114     @Override()
115     public JPOProgramData create() throws MatrixException
116     {
117         if (!this.isCreated())  {
118             final StringBuilder cmd = new StringBuilder()
119                 .append("escape add program \"").append(AbstractTest.convertMql(this.getName()))
120                 .append("\" java ")
121                 .append("code \"").append(AbstractTest.convertMql(this.getCode().replaceAll("^package .*;", ""))).append("\"");
122             this.append4Create(cmd);
123             cmd.append(";\n")
124                .append("escape add property ").append(this.getSymbolicName())
125                .append(" on program eServiceSchemaVariableMapping.tcl")
126                .append(" to program \"").append(AbstractTest.convertMql(this.getName())).append("\"");
127 
128             this.setCreated(true);
129 
130             this.getTest().mql(cmd);
131         }
132         return this;
133     }
