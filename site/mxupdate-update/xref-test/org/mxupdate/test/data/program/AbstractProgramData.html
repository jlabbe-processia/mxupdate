263     /***
264      * Appends the program related MQL commands to create this program. This
265      * includes:
266      * <ul>
267      * <li>{@link #user}</li>
268      * <li>{@link #isHidden() hidden flag}</li>
269      * <li>{@link #deferred}</li>
270      * <li>{@link #needsBusinessObject needs business object}</li>
271      * <li>{@link #downloadable} flag</li>
272      * <li>{@link #pipe} flag</li>
273      * <li>{@link #pooled} flag</li>
274      * </ul>
275      *
276      * @param _cmd  string builder used to append MQL commands
277      * @throws MatrixException if used object could not be created
278      */
279     @Override()
280     protected void append4Create(final StringBuilder _cmd)
281         throws MatrixException
282     {
283         super.append4Create(_cmd);
284 
285         // user
286         if (this.getUser() != null)  {
287             this.getUser().create();
288             _cmd.append(" execute user \"").append(AbstractTest.convertMql(this.getUser().getName())).append("\"");
289         }
290         // hidden flag
291         if ((this.isHidden() != null) && this.isHidden())  {
292             _cmd.append(" hidden");
293         }
294         // deferred flag
295         if (this.deferred)  {
296             _cmd.append(" execute deferred");
297         }
298         // needs business object
299         if (this.needsBusinessObject)  {
300             _cmd.append(" needsbusinessobject");
301         }
302         // downloable flag
303         if (this.downloadable)  {
304             _cmd.append(" downloadable");
305         }
306         // pipe flag
307         if (this.pipe)  {
308             _cmd.append(" pipe");
309         }
310         // pooled flag
311         if (this.pooled)  {
312             _cmd.append(" pooled");
313         }
314     }

366     /***
367      * Checks the export of this data piece if all values are correct defined.
368      * Because the handling of a program is different (the TCL update code
369      * is embedded), the values must be checked directly via MQL commands.
370      *
371      * @param _exportParser     parsed export
372      * @throws MatrixException if check failed
373      */
374     @Override()
375     public void checkExport(final ExportParser _exportParser)
376         throws MatrixException
377     {
378         Assert.assertTrue(!"".equals(this.getTest().mql("escape list program \"" + AbstractTest.convertMql(this.getName()) + "\"")),
379                           "check program is created");
380         Assert.assertTrue(!"".equals(this.getTest().mql("escape list property to program \"" + AbstractTest.convertMql(this.getName()) + "\"")),
381                           "check that the program is registered");
382         Assert.assertEquals(this.getTest().mql("escape list property to program \"" + AbstractTest.convertMql(this.getName()) + "\""),
383                             this.getSymbolicName()
384                                     + " on program eServiceSchemaVariableMapping.tcl to program "
385                                     + this.getName(),
386                             "check that the MQL program is registered with correct symbolic name");
387         // check hidden
388         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
389                                         + AbstractTest.convertMql(this.getName()) + "\" select hidden dump")).booleanValue(),
390                             (this.isHidden() != null) ? this.isHidden().booleanValue() : false,
391                             "hidden flag correct defined");
392         // check description
393         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select description dump"),
394                             (this.getValue("description") != null) ? this.getValue("description") : "",
395                             "correct description defined");
396         // check user
397         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select user dump"),
398                             (this.getUser() != null) ? this.getUser().getName() : "",
399                             "correct user assigned");
400         // execution is deferred
401         Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName()) + "\" select execute dump"),
402                             (this.deferred) ? "deferred" : "immediate",
403                             "check execution");
404         // needs business object
405         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
406                                     + AbstractTest.convertMql(this.getName()) + "\" select doesneedcontext dump")).booleanValue(),
407                             this.needsBusinessObject,
408                             "needs business object flag");
409         // downloadable
410         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
411                                     + AbstractTest.convertMql(this.getName()) + "\" select downloadable dump")).booleanValue(),
412                             this.downloadable,
413                             "downloadable flag");
414         // pipe
415         Assert.assertEquals(Boolean.valueOf(this.getTest().mql("escape print program \""
416                                     + AbstractTest.convertMql(this.getName()) + "\" select ispipedprogram dump")).booleanValue(),
417                             this.pipe,
418                             "pipe flag");
419 
420         // check properties
421         for (final PropertyDef prop : this.getProperties())  {
422             Assert.assertTrue(!"".equals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
423                                             + "\" select property[" + prop.getName() + "] dump")),
424                               "property '" + prop.getName() + "' is defined");
425             Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
426                                             + "\" select property[" + prop.getName() + "].value dump"),
427                                 (prop.getValue() != null) ? prop.getValue() : "",
428                                 "value of property '" + prop.getName() + "' is defined");
429             if (prop.getTo() != null)  {
430                 Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
431                                     + "\" select property[" + prop.getName() + "].to.name dump"),
432                                     prop.getTo().getName(),
433                                     "to of property '" + prop.getName() + "' is defined");
434             } else  {
435                 Assert.assertEquals(this.getTest().mql("escape print program \"" + AbstractTest.convertMql(this.getName())
436                                             + "\" select property[" + prop.getName() + "].to dump"),
437                                     "",
438                                     "to of property '" + prop.getName() + "' is not defined");
439             }
440         }
441     }
