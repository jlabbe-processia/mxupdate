170     /***
171      * Returns the name of the configuration item file.
172      *
173      * @return name of the CI file
174      */
175     public String getCIFileName()
176     {
177         final char[] charName = this.name.toCharArray();
178         final StringBuilder fileName = new StringBuilder().append(this.getCI().filePrefix);
179         for (int idx = 0; idx < charName.length; idx++)  {
180             final char ch = charName[idx];
181             if (ch == '@')  {
182                 fileName.append("@@");
183             } else if ((ch < '(' || ch > ')')
184                     && (ch < '+' || ch > '.')
185                     && (ch < '0' || ch > '9')
186                     && (ch < 'A' || ch > 'Z')
187                     && (ch < 'a' || ch > 'z')
188                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
189 
190                 final String hex = String.valueOf(Integer.toHexString(ch));
191                 fileName.append('@');
192                 switch (hex.length())  {
193                     case 1:
194                         fileName.append('0').append(hex);
195                         break;
196                     case 3:
197                         fileName.append("u0").append(hex);
198                         break;
199                     case 4:
200                         fileName.append('u').append(hex);
201                         break;
202                     default:
203                         fileName.append(hex);
204                         break;
205                 }
206             } else  {
207                 fileName.append(ch);
208             }
209         }
210         fileName.append(".tcl");
211 
212         return fileName.toString();
213     }

294     /***
295      * Makes an update for given <code>_code</code>.
296      *
297      * @param _code         TCL update code
298      * @param _params       parameters
299      * @return this data instance
300      * @throws Exception  if update failed
301      */
302     @SuppressWarnings("unchecked")
303     public DATA updateWithCode(final String _code,
304                                final String... _params)
305         throws Exception
306     {
307         final Map<String,String> files = new HashMap<String,String>();
308         files.put(this.getCIFileName(), _code);
309         final Map<String,String> params = new HashMap<String,String>();
310         if (_params != null)  {
311             for (int idx = 0; idx < _params.length; idx += 2)  {
312                 params.put(_params[idx], _params[idx + 1]);
313             }
314         }
315         final Map<?,?> bck = this.getTest().executeEncoded("Update", params, "FileContents", files);
316         if (bck.get("exception") != null)  {
317             throw (Exception) bck.get("exception");
318         }
319         return (DATA) this;
320     }

360     /***
361      * Exports this data piece from MX. The returned values from the export are
362      * checked for:
363      * <ul>
364      * <li>value is returned</li>
365      * <li>returned export value includes exact one export for given
366      *     {@link #ci configuration item type}</li>
367      * <li>name of the exported object is equal {@link #name}</li>
368      * <li>path of the exported object is equal {@link #ciFile()}</li>
369      * <li>file name of the export object is equal returned value from
370      *     {@link #getCIFileName()}</li>
371      * </ul>
372      *
373      * @param _params       parameters
374      * @return parsed export
375      * @throws Exception if export failed
376      */
377     public ExportParser export(final String... _params)
378         throws Exception
379     {
380         final Map<String,Collection<String>> files = new HashMap<String,Collection<String>>(1);
381         files.put(this.ci.updateType, Arrays.asList(new String[]{this.getName()}));
382 
383         final Map<String,String> params = new HashMap<String,String>();
384         if (_params != null)  {
385             for (int idx = 0; idx < _params.length; idx += 2)  {
386                 params.put(_params[idx], _params[idx + 1]);
387             }
388         }
389 
390         final Map<?,?> bck = this.test.executeEncoded("Export",
391                                                       params,
392                                                       "TypeDef", this.ci.updateType,
393                                                       "Name", this.getName());
394 
395         // check for no exception
396         if (bck.get("exception") != null)  {
397             throw new MatrixException((Exception) bck.get("exception"));
398         }
399 
400         // extract values
401         final Map<?,?> values = (Map<?,?>) bck.get("values");
402 
403         // check existence and element is defined
404         Assert.assertNotNull(values);
405         Assert.assertEquals((String) values.get("TypeDef"),
406                             this.ci.updateType,
407                             "check correct type definition");
408 
409         // parse first element
410         final ExportParser ret = this.parseExport(this.ci, (String) values.get("Code"));
411 
412         // check returned configuration item name
413         Assert.assertEquals((String) values.get("Name"),
414                             this.name,
415                             "returned name is equal to given name");
416         if (this.ci != AbstractTest.CI.PRG_JPO)  {
417             // check path of the configuration item update file
418             Assert.assertEquals((String) values.get("FilePath"),
419                                 this.getCI().filePath,
420                                 "check path where the configuration item update file is located is correct");
421         } else  {
422             // check path for JPOs (because of packages...)
423             final String path;
424             if (this.name.indexOf('.') < 0)  {
425                 path = this.ci.filePath;
426             } else  {
427                 path = this.ci.filePath + "/" + this.name.replaceAll("//.[^.]*$", "").replaceAll("//.", "/");
428             }
429             Assert.assertEquals((String) values.get("FilePath"),
430                                 path,
431                                 "check path where the configuration item update file is located is correct");
432         }
433         // check file name of the configuration item update file
434         Assert.assertEquals((String) values.get("FileName"),
435                             this.getCIFileNameFromExport(),
436                             "check that the correct configuration item file name is returned");
437 
438         return ret;
439     }

483     /***
484      * Checks for given <code>_tag</code> in the <code>_exportParser</code>

486      * <code>_value</code> is <code>null</code>, it is checked that no value is

494      */

497                                  final String _tag,
498                                  final String _value)
499     {
500         if (_value != null)  {
501             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
502                                 1,
503                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is defined");
504             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").get(0),
505                                 _value,
506                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is " + _value);
507 
508         } else  {
509             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
510                                 0,
511                                 "check " + _kind + " '" + this.getName() + "' that no " + _tag + " is defined");
512         }
