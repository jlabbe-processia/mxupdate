167     /***
168      * Returns the name of the configuration item file.
169      *
170      * @return name of the CI file
171      */
172     public String getCIFileName()
173     {
174         final char[] charName = this.name.toCharArray();
175         final StringBuilder fileName = new StringBuilder().append(this.getCI().filePrefix);
176         for (int idx = 0; idx < charName.length; idx++)  {
177             final char ch = charName[idx];
178             if (ch == '@')  {
179                 fileName.append("@@");
180             } else if ((ch < '(' || ch > ')')
181                     && (ch < '+' || ch > '.')
182                     && (ch < '0' || ch > '9')
183                     && (ch < 'A' || ch > 'Z')
184                     && (ch < 'a' || ch > 'z')
185                     && (ch != ' ') && (ch != '=') && (ch != '_'))  {
186 
187                 final String hex = String.valueOf(Integer.toHexString(ch));
188                 fileName.append('@');
189                 switch (hex.length())  {
190                     case 1:
191                         fileName.append('0').append(hex);
192                         break;
193                     case 3:
194                         fileName.append("u0").append(hex);
195                         break;
196                     case 4:
197                         fileName.append('u').append(hex);
198                         break;
199                     default:
200                         fileName.append(hex);
201                         break;
202                 }
203             } else  {
204                 fileName.append(ch);
205             }
206         }
207         fileName.append(".tcl");
208 
209         return fileName.toString();
210     }

250     /***
251      * Exports this data piece from MX. The returned values from the export are
252      * checked for:
253      * <ul>
254      * <li>value is returned</li>
255      * <li>returned export value includes exact one export for given
256      *     {@link #ci configuration item type}</li>
257      * <li>name of the exported object is equal {@link #name}</li>
258      * <li>path of the exported object is equal {@link #ciPath}</li>
259      * <li>file name of the export object is equal returned value from
260      *     {@link #getCIFileName()}</li>
261      * </ul>
262      *
263      * @param _params       parameters
264      * @return parsed export
265      * @throws IOException      if the parameter could not be encoded
266      * @throws MatrixException  if MQL calls failed
267      */
268     public ExportParser export(final String... _params)
269         throws IOException, MatrixException
270     {
271         final Map<String,Collection<String>> files = new HashMap<String,Collection<String>>(1);
272         files.put(this.ci.updateType, Arrays.asList(new String[]{this.getName()}));
273 
274         final Map<String,String> params = new HashMap<String,String>();
275         if (_params != null)  {
276             for (int idx = 0; idx < _params.length; idx += 2)  {
277                 params.put(_params[idx], _params[idx + 1]);
278             }
279         }
280 
281         final JPOReturn<Map<String,Collection<Map<String,String>>>> jpoReturn =
282                 this.test.<Map<String,Collection<Map<String,String>>>>jpoInvoke("org.mxupdate.plugin.Export",
283                                                                                 "exportByName",
284                                                                                 files,
285                                                                                 params);
286 
287         // check for no exception
288         if (jpoReturn.getException() != null)  {
289             throw new MatrixException(jpoReturn.getException());
290         }
291 
292         // extract values
293         final Map<String,Collection<Map<String,String>>> values = jpoReturn.getValues();
294 
295         // check existence and element is defined
296         Assert.assertNotNull(values);
297         Assert.assertTrue(values.containsKey(this.ci.updateType));
298         Assert.assertEquals(values.get(this.ci.updateType).size(), 1, "one element is returned");
299 
300         // parse first element
301         final Map<String,String> exportDesc = values.get(this.getCI().updateType).iterator().next();
302         final ExportParser ret = this.parseExport(this.ci, exportDesc.get("code"));
303 
304         // check returned configuration item name
305         Assert.assertEquals(exportDesc.get("name"),
306                             this.name,
307                             "returned name is equal to given name");
308         if (this.ci != AbstractTest.CI.PRG_JPO)  {
309             // check path of the configuration item update file
310             Assert.assertEquals(exportDesc.get("path"),
311                                 this.getCI().filePath,
312                                 "check path where the configuration item update file is located is correct");
313         } else  {
314             // check path for JPOs (because of packages...)
315             final String path;
316             if (this.name.indexOf('.') < 0)  {
317                 path = this.ci.filePath;
318             } else  {
319                 path = this.ci.filePath + "/" + this.name.replaceAll("//.[^.]*$", "").replaceAll("//.", "/");
320             }
321             Assert.assertEquals(exportDesc.get("path"),
322                                 path,
323                                 "check path where the configuration item update file is located is correct");
324         }
325         // check file name of the configuration item update file
326         Assert.assertEquals(exportDesc.get("filename"),
327                             this.getCIFileName(),
328                             "check that the correct configuration item file name is returned");
329 
330         return ret;
331     }

358     /***
359      * Checks for given <code>_tag</code> in the <code>_exportParser</code>
360      * if the <code>_value</code> exists and is correct defined. If
361      * <code>_value</code> is <code>null</code>, it is checked that no value is
362      * defined.
363      *
364      * @param _exportParser     parsed export
365      * @param _kind             kind of the check
366      * @param _tag              tag to check
367      * @param _value            value to check (or <code>null</code> if value
368      *                          is not defined)
369      */
370     public void checkSingleValue(final ExportParser _exportParser,
371                                  final String _kind,
372                                  final String _tag,
373                                  final String _value)
374     {
375         if (_value != null)  {
376             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
377                                 1,
378                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is defined");
379             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").get(0),
380                                 _value,
381                                 "check " + _kind + " for '" + this.getName() + "' that " + _tag + " is " + _value);
382 
383         } else  {
384             Assert.assertEquals(_exportParser.getLines("/mql/" + _tag + "/@value").size(),
385                                 0,
386                                 "check " + _kind + " '" + this.getName() + "' that no " + _tag + " is defined");
387         }
388     }
