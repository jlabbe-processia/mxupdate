453     /***
454      * Checks that the new created or updated dimension {@link #DIMENSION_NAME}
455      * is equal to the defined update code {@link #DIMENSION_CODE}.
456      *
457      * @throws MatrixException if MQL select statements failed
458      */
459     private void checkAll()
460         throws MatrixException
461     {
462         Assert.assertTrue(!"".equals(this.mql("list dimension " + Dimension.DIMENSION_NAME)),
463                           "dimension was not created!");
464         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
465                                     + " select description dump"),
466                             "a simple description",
467                             "Check that description is set");
468         Assert.assertEquals(this.mql("list property to dimension " + Dimension.DIMENSION_NAME),
469                             "dimension_" + Dimension.DIMENSION_NAME
470                                     + " on program eServiceSchemaVariableMapping.tcl to dimension "
471                                     + Dimension.DIMENSION_NAME,
472                             "Check for symbolic name registration.");
473 
474         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
475                                             + " select unit dump @")
476                                     .replaceAll("name1", "")
477                                     .replaceAll("name2", "")
478                                     .replaceAll("name 3", ""),
479                             "@@",
480                             "check that exact three units are defined");
481 
482         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
483                                     + " select unit[name1] dump"),
484                             "TRUE",
485                             "check that unit 'name1' exists");
486         this.checkUnit("name1", "default", "TRUE");
487         this.checkUnit("name1", "label", "label1");
488         this.checkUnit("name1", "description", "description 1");
489         this.checkUnit("name1", "multiplier", "1.0");
490         this.checkUnit("name1", "offset", "0.0");
491         this.checkUnit("name1", "setting[SettingKey2]", "TRUE");
492         this.checkUnit("name1", "setting[SettingKey2].value", "SettingValue2");
493         this.checkUnit("name1", "setting[to]", "TRUE");
494         this.checkUnit("name1", "setting[to].value", "SettingValue");
495         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
496                                             + " select unit[name1].setting dump @")
497                                     .replaceAll("to", "")
498                                     .replaceAll("SettingKey2", ""),
499                             "@",
500                             "Check that only two settings are defined");
501         this.checkUnit("name1", "property[Zest].value", "Zest");
502         this.checkUnit("name1", "property[Zest].to", "");
503         this.checkUnit("name1", "property[_test3].value", "Test3");
504         this.checkUnit("name1", "property[_test3].to", "");
505         this.checkUnit("name1", "property[abc].value", "abc");
506         this.checkUnit("name1", "property[abc].to", "");
507         this.checkUnit("name1", "property[abc2].value", "abc2");
508         this.checkUnit("name1", "property[abc2].to", "");
509         this.checkUnit("name1", "property[test2].value", "Test2");
510         this.checkUnit("name1", "property[test2].to", "");
511         this.checkUnit("name1", "property[test].value", "Test");
512         this.checkUnit("name1", "property[test].to", "type Part");
513         this.checkUnit("name1", "systemunit", "");
514 
515         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
516                                     + " select unit[name2] dump"),
517                             "TRUE",
518                             "Check that unit 'name2' exists");
519         this.checkUnit("name2", "default", "FALSE");
520         this.checkUnit("name2", "label", "label2");
521         this.checkUnit("name2", "description", "description 2");
522         this.checkUnit("name2", "multiplier", "10.0");
523         this.checkUnit("name2", "offset", "20.0");
524         this.checkUnit("name2", "property", "");
525         this.checkUnit("name2", "systemunit", "name2");
526         this.checkUnit("name2", "systemunit[Duration Units]", "name2");
527 
528         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
529                                     + " select unit[name 3] dump"),
530                             "TRUE",
531                             "Check that unit 'name 3' exists");
532         this.checkUnit("name 3", "default", "FALSE");
533         this.checkUnit("name 3", "label", "label 3");
534         this.checkUnit("name 3", "description", "description3");
535         this.checkUnit("name 3", "multiplier", "30.0");
536         this.checkUnit("name 3", "offset", "40.0");
537         this.checkUnit("name 3", "property", "");
538         Assert.assertEquals(this.mql("print dimension " + Dimension.DIMENSION_NAME
539                                             + " select 'unit[name 3].systemunit[]' dump @")
540                                     .replaceAll("name2", "")
541                                     .replaceAll("name 3", ""),
542                             "@",
543                             "Check that exact two system infos are defined");
544         this.checkUnit("name 3", "systemunit[Duration Units]", "name2");
545         this.checkUnit("name 3", "systemunit[Test]", "name 3");
546     }
