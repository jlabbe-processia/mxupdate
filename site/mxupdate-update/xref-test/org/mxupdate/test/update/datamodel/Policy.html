173     /***
174      * Checks that states where no symbolic name is defined in the TCL update
175      * does not define a symbolic name (with empty name).
176      *
177      * @throws Exception if MQL execution failed
178      */
179     public void testProperties()
180         throws Exception
181     {
182         final Map<String,String> params = new HashMap<String,String>();
183         params.put("POLICY_" + Policy.POLICY_NAME + ".tcl",
184                    Policy.POLICY_UPDATE_CODE);
185         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
186 
187         Assert.assertTrue(!"".equals(this.mql("list policy MxUpdate_Test")),
188                           "policy was not created!");
189 
190         // check that only 11 properties are defined....
191         // 4 state properties + installer + installed date + original name +
192         // file date + author + version + application
193         final Set<String> propNames = new HashSet<String>();
194         final String propNamesStr = this.mql("print policy MxUpdate_Test select property.name dump '\n'");
195         for (final String propName : propNamesStr.split("\n"))  {
196             propNames.add(propName);
197         }
198         Assert.assertTrue(propNames.size() == 11, "check that all properties are defined");
199         Assert.assertTrue(this.mql("print policy MxUpdate_Test").indexOf("property  value ") < 0,
200                           "Update did set an empty property!");
201     }

203     /***
204      * Test for for an update for existing policy with empty property and state
205      * names without symbolic names. The empty property must be removed after
206      * the update and for the state without symbolic name no registration must
207      * be done.
208      *
209      * @throws Exception if MQL execution failed
210      */
211     public void testNullProperties()
212         throws Exception
213     {
214         this.mql("add policy MxUpdate_Test state Pending property \"\" value Test");
215 
216         final Map<String,String> params = new HashMap<String,String>();
217         params.put("POLICY_" + Policy.POLICY_NAME + ".tcl",
218                    Policy.POLICY_UPDATE_CODE);
219         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
220 
221         // check that only 11 properties are defined....
222         // 4 state properties + installer + installed date + original name +
223         // file date + author + version + application
224         final Set<String> propNames = new HashSet<String>();
225         final String propNamesStr = this.mql("print policy MxUpdate_Test select property.name dump '\n'");
226         for (final String propName : propNamesStr.split("\n"))  {
227             propNames.add(propName);
228         }
229         Assert.assertTrue(!propNames.contains(""), "Update did not remove empty property!");
230         Assert.assertTrue(this.mql("print policy MxUpdate_Test").indexOf("property  value Test") < 0,
231                           "Update did not remove empty property!");
232         Assert.assertTrue(propNames.size() == 11, "check that all properties are defined");
233     }

237      *
238      * @throws Exception if test failed

240     public void testMultipleStateSymbolicNames()
241         throws Exception
242     {

245                    "updatePolicy \"${NAME}\" {"

247                     + "    registeredName \"state_Exists\""
248                     + "    registeredName \"state_Pending\""
249                     + "  }"
250                     + "}");
251         this.jpoInvoke("org.mxupdate.plugin.Update", "updateByContent", params);
252 
253         Assert.assertEquals(this.mql("print policy " + Policy.POLICY_NAME
254                                 + " select property[state_Exists].value dump"),
255                             "Pending",
256                             "test that symbolic name 'state_Exists' is correct registered");
257         Assert.assertEquals(this.mql("print policy " + Policy.POLICY_NAME
258                                 + " select property[state_Pending].value dump"),
259                             "Pending",
260                             "test that symbolic name 'state_Pending' is correct registered");
261     }
