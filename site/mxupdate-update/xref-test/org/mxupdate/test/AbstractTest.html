242     /***
243      * Executes given MQL command and splits the returned value by new line
244      * '\n' and returns this values as a set.
245      *
246      * @param _cmd                  MQL command to execute
247      * @return set of found values
248      * @throws MatrixException if MQL execution failed
249      */
250     public Set<String> mqlAsSet(final CharSequence _cmd)
251         throws MatrixException
252     {
253         final String bck = this.mql(_cmd);
254         final Set<String> ret;
255         if ("".equals(bck))  {
256             ret = new HashSet<String>(0);
257         } else  {
258             ret = new HashSet<String>(Arrays.asList(bck.split("\n")));
259         }
260         return ret;

277     /***
278      * Export given configuration item <code>_type</code> with
279      * <code>_name</code>. The returned values from the export are checked for:
280      * <ul>
281      * <li>value is returned</li>
282      * <li>the returned export value includes exact one export for given
283      *     <code>_type</code></li>
284      * <li>the name of the exported object is equal <code>_name</code></li>
285      * </ul>
286      *
287      * @param _type     type to export
288      * @param _name     name to export
289      * @return map with the exported object
290      * @throws IOException      if the parameter could not be encoded
291      * @throws MatrixException  if MQL calls failed
292      */
293     protected Export export(final CI _type,
294                             final String _name)
295         throws IOException, MatrixException
296     {
297         final Map<String,Collection<String>> params = new HashMap<String,Collection<String>>(1);
298         params.put(_type.updateType, Arrays.asList(new String[]{_name}));
299         final Map<String,Collection<Map<String,String>>> bck =
300                 this.<Map<String,Collection<Map<String,String>>>>jpoInvoke("org.mxupdate.plugin.Export",
301                                                                            "exportByName",
302                                                                            params)
303                     .getValues();
304 
305         Assert.assertNotNull(bck);
306         Assert.assertTrue(bck.containsKey(_type.updateType));
307         Assert.assertEquals(bck.get(_type.updateType).size(), 1, "one element is returned");
308 
309         final Export ret =  new Export(bck.get(_type.updateType).iterator().next());
310 
311         Assert.assertEquals(ret.getName(), _name, "returned name is equal to given name");
312 
313         return ret;
314     }

366     /***
367      * Cleanups given configuration <code>_type</code>. This means that all
368      * existing administration object which starts with {@link #PREFIX} are
369      * deleted. Only if {@link CI#wildcardSearch} is <i>true</i>, a search via
370      * wild card is done; otherwise a list of all administration object is
371      * used.
372      *
373      * @param _type     type of configuration item
374      * @throws MatrixException if cleanup for given configuration item
375      *                         <code>_type</code> failed
376      */
377     protected void cleanup(final CI _type)
378         throws MatrixException
379     {
380         final Set<String> elements;
381         if (_type.wildcardSearch)  {
382             elements = this.mqlAsSet("escape list " + _type.mxType + " \"" + AbstractTest.PREFIX + "*\"");
383         } else  {
384             elements = this.mqlAsSet("escape list " + _type.mxType);
385         }
386         for (final String element : elements)  {
387             if (element.startsWith(AbstractTest.PREFIX))  {
388                 this.mql("escape delete " + _type.mxType + " \"" + AbstractTest.convertMql(element) + "\"");
389             }
390         }
391     }

393     /***
394      * Calls given <code>_method</code> in <code>_jpo</code>. The MX context
395      * {@link #mxContext} is connected to the database if not already done.
396      *
397      * @param <T>           class which is returned
398      * @param _jpo          name of JPO to call
399      * @param _method       method of the called <code>_jpo</code>
400      * @param _parameters   list of all parameters for the <code>_jpo</code>
401      *                      which are automatically encoded encoded
402      * @return returned value from the called <code>_jpo</code>
403      * @throws IOException      if the parameter could not be encoded
404      * @throws MatrixException  if the called <code>_jpo</code> throws an
405      *                          exception
406      * @see #context
407      * @see JPOReturn
408      */
409     protected <T> JPOReturn<T> jpoInvoke(final String _jpo,
410                                          final String _method,
411                                          final Object... _parameters)
412         throws IOException, MatrixException
413     {
414         // encode parameters
415         final String[] paramStrings = new String[_parameters.length];
416         int idx = 0;
417         for (final Object parameter : _parameters)  {
418             final ByteArrayOutputStream out = new ByteArrayOutputStream();
419             final ObjectOutputStream oos = new ObjectOutputStream(out);
420             oos.writeObject(parameter);
421             oos.close();
422             paramStrings[idx++] = new String(Base64.encodeBase64(out.toByteArray()));
423         }
424 
425         return new JPOReturn<T>(JPO.invoke(this.context,
426                                            _jpo,
427                                            null,
428                                            _method,
429                                            paramStrings,
430                                            Object.class));
431     }

484     /***
485      * Class holding the returned values from the JPO.
486      *
487      * @param <T>   type of the returned values
488      * @see AbstractTest#jpoInvoke(String, String, Object...)
489      */
490     protected final class JPOReturn<T>
491     {
492         /***
493          * Returns map from the JPO invoke.
494          */
495         private final Map<String,?> jpoReturn;
496 
497         /***
498          * Constructor to hold the return map from the JPO invoke.
499          *
500          * @param _jpoReturn    return map from the JPO invoke
501          */
502         @SuppressWarnings("unchecked")
503         private JPOReturn(final Object _jpoReturn)
504         {
505             if (_jpoReturn instanceof String)  {
506                 this.jpoReturn = new HashMap<String,Object>();
507                 ((Map<String,Object>) this.jpoReturn).put("values", _jpoReturn);
508             } else  {
509                 this.jpoReturn = (Map<String,?>) _jpoReturn;
510             }
511         }
512 
513         /***
514          * Returns the stored values from the JPO return.
515          *
516          * @return stored values from the JPO return
517          * @see #jpoReturn
518          */
519         @SuppressWarnings("unchecked")
520         public T getValues()
521         {
522             return (T) this.jpoReturn.get("values");
523         }
524     }
