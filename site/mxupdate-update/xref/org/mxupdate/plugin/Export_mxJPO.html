31  /***
32   * The JPO class is the plug-in to export configuration items (administration
33   * objects) or in other words with the class the TCL update file could be
34   * created depending on the configuration item type (administration type).
35   *
36   * @author The MxUpdate Team
37   * @version $Id: Export_mxJPO.java 780 2010-04-06 11:22:55Z tim.moxter $
38   */
39  class Export_mxJPO
40      extends AbstractPlugin_mxJPO
41  {
42      /***
43       * Argument key for the items.
44       */
45      private static final String ARGUMENT_KEY_NAME = "Name"; //$NON-NLS-1$
46  
47      /***
48       * Argument key for the type definition.
49       */
50      private static final String ARGUMENT_KEY_TYPEDEF = "TypeDef"; //$NON-NLS-1$
51  
52      /***
53       * Argument key for the file name.
54       */
55      private static final String ARGUMENT_KEY_FILENAME = "FileName"; //$NON-NLS-1$
56  
57      /***
58       * Name of the key in the returned mapped for the &quot;Name&quot; of the
59       * administration object (which administration object is exported).
60       */
61      private static final String RETURN_KEY_NAME = "Name"; //$NON-NLS-1$
62  
63      /***
64       * Name of the key in the returned mapped for the &quot;Code&quot; of the
65       * administration object (the TCL update code itself).
66       */
67      private static final String RETURN_KEY_CODE = "Code"; //$NON-NLS-1$
68  
69      /***
70       * Name of the key in the returned mapped for the &quot;FilePath&quot; of
71       * the administration object (where is the TCL update code located in the
72       * file system).
73       */
74      private static final String RETURN_KEY_PATH = "FilePath"; //$NON-NLS-1$
75  
76      /***
77       * Name of the key in the returned mapped for the &quot;FileName&quot; of
78       * the administration object (what is the name of the file).
79       */
80      private static final String RETURN_KEY_FILENAME = "FileName"; //$NON-NLS-1$
81  
82      /***
83       * Key in the map for the name of the type definition of found one
84       * configuration item..

86      private static final String RETURN_KEY_TYPEDEF = "TypeDef"; //$NON-NLS-1$

88      /***
89       * <p>All given administration objects are exported and the related update
90       * file is returned. The administration objects to export in
91       * <code>_args</code> are identified by the administration type (see
92       * {@link TypeDef_mxJPO}) and the MX names of the administration objects.
93       * </p>
94       * <p>The <code>_arguments</code> differs between two different possible
95       * modes. The first possibility depends on the file name with the argument
96       * key
97       * <ul>
98       * <li>{@link #ARGUMENT_KEY_FILENAME}: name of the file to export</li>
99       * </ul>
100      * The second possibility depends on the type definition and the name of
101      * the configuration item with the argument keys
102      * <ul>
103      * <li>{@link #ARGUMENT_KEY_TYPEDEF}: type definition</li>
104      * <li>{@link #ARGUMENT_KEY_NAME}: name of the configuration item</li>
105      * </ul>
106      * </p>
107      * <p>The returned packed values (packed with method
108      * {@link AbstractPlugin_mxJPO#prepareReturn(String, String, Exception, Object)})
109      * includes the logging messages itself and for configuration item type
110      * a map with exporting information. The exporting information map defines
111      * four keys:
112      * <ul>
113      * <li>{@link #RETURN_KEY_TYPEDEF}:     type definition of the
114      *                                      configuration item</li>
115      * <li>{@link #RETURN_KEY_NAME}:        name of the configuration item</li>
116      * <li>{@link #RETURN_KEY_PATH}:        path within the file directory of
117      *                                      the configuration item (without
118      *                                      file name)</li>
119      * <li>{@link #RETURN_KEY_CODE}:        TCL update source code of the
120      *                                      configuration item</li>
121      * <li>{@link #RETURN_KEY_FILENAME}:    file name of the configuration
122      *                                      item</li>
123      * </ul></p>
124      * <p>If an exception is thrown the exception is packed in the returned
125      * map.</p>
126      *
127      * @param _paramCache   parameter cache with the MX context
128      * @param _arguments    map with all search arguments
129      * @return packed return values in maps
130      * @throws Exception if export failed
131      * @see AbstractPlugin_mxJPO#prepareReturn(String, String, Exception, Object)
132      */
133     Map<String,String> execute(final ParameterCache_mxJPO _paramCache,
134                                final Map<String,Object> _arguments)
135         throws Exception
136     {
137         final String fileName = this.getArgument(_arguments, Export_mxJPO.ARGUMENT_KEY_FILENAME, null);
138 
139         AbstractObject_mxJPO instance = null;
140 
141         if (fileName != null)  {
142             final File file = new File(fileName);
143 
144             // first found related type definition
145             for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefsSorted())  {
146                 if (!typeDef.isFileMatchLast())  {
147                     instance = typeDef.newTypeInstance(null);
148                     final String mxName = instance.extractMxName(_paramCache, file);
149                     if (mxName != null)  {
150                         instance = typeDef.newTypeInstance(mxName);
151                         break;
152                     } else  {
153                         instance = null;
154                     }
155                 }
156             }
157             if (instance == null)  {
158                 for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefsSorted())  {
159                     if (typeDef.isFileMatchLast())  {
160                         instance = typeDef.newTypeInstance(null);
161                         final String mxName = instance.extractMxName(_paramCache, file);
162                         if (mxName != null)  {
163                             instance = typeDef.newTypeInstance(mxName);
164                             break;
165                         } else  {
166                             instance = null;
167                         }
168                     }
169                 }
170             }
171         } else  {
172             // initialize arguments
173             final String typeDefName = this.getArgument(_arguments, Export_mxJPO.ARGUMENT_KEY_TYPEDEF, null);
174             final String item = this.getArgument(_arguments, Export_mxJPO.ARGUMENT_KEY_NAME, null);
175 
176             // export all objects depending on the type definitions
177             final TypeDef_mxJPO typeDef = _paramCache.getMapping().getTypeDef(typeDefName);
178             instance = typeDef.newTypeInstance(item);
179         }
180 
181         // export code
182         final Map<String,String> ret;
183         if (instance == null)  {
184             ret = null;
185         } else  {
186             final StringBuilder code = new StringBuilder();
187             instance.export(_paramCache, code);
188 
189             ret = new HashMap<String,String>();
190             ret.put(Export_mxJPO.RETURN_KEY_TYPEDEF,    instance.getTypeDef().getName());
191             ret.put(Export_mxJPO.RETURN_KEY_NAME,       instance.getName());
192             ret.put(Export_mxJPO.RETURN_KEY_CODE,       code.toString());
193             ret.put(Export_mxJPO.RETURN_KEY_PATH,       instance.getPath());
194             ret.put(Export_mxJPO.RETURN_KEY_FILENAME,   instance.getFileName());
195         }
196 
197         return ret;
198     }
