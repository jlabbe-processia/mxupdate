23  import java.io.ByteArrayOutputStream;
24  import java.io.IOException;
25  import java.io.ObjectOutputStream;
26  import java.util.Map;
27  
28  import matrix.db.Context;
29  import matrix.db.MatrixWriter;
30  import matrix.util.Mime64;
31  
32  import org.mxupdate.update.util.ParameterCache_mxJPO;

34  /***
35   * Abstract class which defines common methods to extract called parameters
36   * from the Eclipse plug-in.
37   *
38   * @author The MxUpdate Team
39   * @version $Id: Dispatcher_mxJPO.java 713 2010-02-06 23:28:00Z tim.moxter $
40   */
41  public class Dispatcher_mxJPO
42      extends AbstractPlugin_mxJPO
43  {
44      /***
45       * Name of the &quot;Export&quot; method within the parameters map.
46       */
47      private static final String METHOD_EXPORT = "Export";
48  
49      /***
50       * Name of the &quot;Search&quot; method within the parameters map.
51       */
52      private static final String METHOD_SEARCH = "Search";
53  
54      /***
55       * Name of the &quot;TypeDefTreeList&quot; method within the parameters
56       * map.
57       */
58      private static final String METHOD_TYPEDEFTREELIST = "TypeDefTreeList";
59  
60      /***
61       * Name of the &quot;Update&quot; method within the parameters map.
62       */
63      private static final String METHOD_UPDATE = "Update";
64  
65      /***
66       * Main method to dispatch between different plug-in methods.
67       *
68       * @param _context  MX context
69       * @param _args     packed arguments
70       * @throws IOException if returned values could not be written
71       */
72      public void mxMain(final Context _context,
73                         final String... _args)
74          throws IOException
75      {
76          Map<String,?> ret = null;
77          try {
78              final Map<String,String> params     = this.<Map<String,String>>decode(_args, 0, null);
79              final String method                 = this.<String>decode(_args, 1, null);
80              final Map<String,Object> arguments  = this.<Map<String,Object>>decode(_args, 2, null);
81  
82              // initialize mapping
83              final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, true, params);
84  
85              if (Dispatcher_mxJPO.METHOD_EXPORT.equals(method))  {
86                  ret = new Export_mxJPO().execute(paramCache, arguments);
87              } else if (Dispatcher_mxJPO.METHOD_SEARCH.equals(method))  {
88                  ret = new Search_mxJPO().execute(paramCache, arguments);
89              } else if (Dispatcher_mxJPO.METHOD_TYPEDEFTREELIST.equals(method))  {
90                  ret = new TypeDefTreeList_mxJPO().execute(paramCache, arguments);
91              } else if (Dispatcher_mxJPO.METHOD_UPDATE.equals(method))  {
92                  ret = new Update_mxJPO().execute(paramCache, arguments);
93              } else  {
94                  throw new Exception("unknown plug-in method '" + method + "'");
95              }
96          } catch (final Exception exception)  {
97              ret = this.prepareReturn(null, null, exception, null);
98          }
99  
100         // and write return values to the matrix writer
101         final MatrixWriter writer = new MatrixWriter(_context);
102         writer.write(this.encode(ret));
103         writer.flush();
104         writer.close();
105     }
106 
107     /***
108      * Packs the parameters into a map
109      * (with {@link #prepareReturn(String, String, Exception, Object)}) and
110      * encodes this map as string (in Base64).
111      *
112      * @param _values       values itself
113      * @return packed return values including log etc. as string encoded
114      * @throws IOException if encoding failed
115      */
116     protected final String encode(final Map<String,?> _values)
117         throws IOException
118     {
119         final ByteArrayOutputStream out = new ByteArrayOutputStream();
120         final ObjectOutputStream oos = new ObjectOutputStream(out);
121         oos.writeObject(_values);
122         oos.close();
123 
124         return Mime64.encode(out.toByteArray());
125     }
