46      /***
47       * TCL procedure used to order fields of form, because Matrix has a bug
48       * that sometimes the fields of a form are not in the correct order.
49       *
50       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
51       */
52      private static final String ORDER_PROC
53              = "proc orderFields {_name _fields}  {\n"
54                  + "foreach offset [list 100000 1] {\n"
55                      + "foreach field $_fields {"
56                          + "mql mod form \"${_name}\" "
57                              + "field modify name \"$field\" "
58                              + "order ${offset}\n"
59                          + "incr offset\n"
60                      + "}\n"
61                  + "}"
62              + "}";

83      @Override
84      protected void parse(final String _url,
85                           final String _content)
86      {
87          if (_url.startsWith("/footer"))  {
88              // to be ignored ...
89          } else if (_url.startsWith("/header"))  {
90              // to be ignored ...
91          } else if (_url.startsWith("/height"))  {
92              // to be ignored ...
93          } else if (_url.startsWith("/leftMargin"))  {
94              // to be ignored ...
95          } else if (_url.startsWith("/rightMargin"))  {
96              // to be ignored ...
97          } else if (_url.startsWith("/webform"))  {
98              // to be ignored ...
99          } else if (_url.startsWith("/width"))  {
100             // to be ignored ...
101         } else  {
102             super.parse(_url, _content);
103         }
104     }

106     /***
107      * Writes all field of the web form to the TCL update file.
108      *
109      * @param _paramCache   parameter cache
110      * @param _out          appendable instance to the TCL update file
111      * @throws IOException if the TCL update code for the fields could not be
112      *                     written
113      * @see AbstractUIWithFields_mxJPO.Field#write(Appendable)
114      */
115     @Override
116     protected void writeObject(final ParameterCache_mxJPO _paramCache,
117                                final Appendable _out)
118             throws IOException
119     {
120         for (final Field field : this.getFields())  {
121             _out.append(" //\n    field");
122             field.write(_out);
123         }
124     }

126     /***
127      * At the end of the TCL update file a call to a procedure must be included
128      * to order all web form fields correctly.
129      *
130      * @param _paramCache   parameter cache
131      * @param _out          appendable instance to the TCL update file
132      * @throws IOException if the extension could not be written
133      */
134     @Override
135     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
136                             final Appendable _out)
137             throws IOException
138     {
139         _out.append("\n\norderFields \"${NAME}\" [list //\n");
140         for (final Field field : this.getFields())  {
141             _out.append("    \"").append(convertTcl(field.getName())).append("\" //\n");
142         }
143         _out.append("]");
144     }

146     /***
147      * Creates given web form object with given name. Because the MQL add
148      * command for a web form must include the string &quot;web&quot; a
149      * specific create method must be written.
150      *
151      * @param _paramCache   parameter cache
152      * @throws Exception if the web form could not be created within MX
153      */
154     @Override
155     public void create(final ParameterCache_mxJPO _paramCache)
156             throws Exception
157     {
158         final StringBuilder cmd = new StringBuilder()
159                 .append("add ").append(this.getTypeDef().getMxAdminName())
160                         .append(" \"").append(this.getName()).append("\" web;");
161         execMql(_paramCache.getContext(), cmd);
162     }

164     /***
165      * The method overwrites the original method to append the MQL statements
166      * in the <code>_preMQLCode</code> to reset this form. Following steps are
167      * done:
168      * <ul>
169      * <li>remove all fields of the web form</li>
170      * <li>set to not hidden</li>
171      * </ul>
172      * The update of web forms works sometimes not correctly for the correct
173      * order of fields. Because of that, the TCL update code is includes a
174      * procedure to order the form fields.
175      *
176      * @param _paramCache       parameter cache
177      * @param _preMQLCode       MQL statements which must be called before the
178      *                          TCL code is executed
179      * @param _postMQLCode      MQL statements which must be called after the
180      *                          TCL code is executed
181      * @param _preTCLCode       TCL code which is defined before the source
182      *                          file is sourced
183      * @param _tclVariables     map of all TCL variables where the key is the
184      *                          name and the value is value of the TCL variable
185      *                          (the value is automatically converted to TCL
186      *                          syntax!)
187      * @param _sourceFile       souce file with the TCL code to update
188      * @throws Exception if the update from derived class failed
189      * @see #ORDER_PROC
190      */
191     @Override
192     protected void update(final ParameterCache_mxJPO _paramCache,
193                           final CharSequence _preMQLCode,
194                           final CharSequence _postMQLCode,
195                           final CharSequence _preTCLCode,
196                           final Map<String,String> _tclVariables,
197                           final File _sourceFile)
198             throws Exception
199     {
200         // reset HRef, description, alt and label
201         final StringBuilder preMQLCode = new StringBuilder()
202                 .append("mod ").append(this.getTypeDef().getMxAdminName())
203                 .append(" \"").append(this.getName()).append('\"')
204                 .append(" !hidden");
205 
206         // remove all fields
207         for (final Field field : this.getFields())  {
208             preMQLCode.append(" field delete name \"").append(field.getName()).append('\"');
209         }
210 
211         // append already existing pre MQL code
212         preMQLCode.append(";\n")
213                   .append(_preMQLCode);
214 
215         // append procedure to order fields of the form
216         final StringBuilder tclCode = new StringBuilder()
217                 .append(ORDER_PROC)
218                 .append('\n')
219                 .append(_preTCLCode);
220 
221         super.update(_paramCache, preMQLCode, _postMQLCode, tclCode, _tclVariables, _sourceFile);
222     }
