35  public class Table_mxJPO
36          extends AbstractUIWithFields_mxJPO
37  {
38      /***
39       * Defines the serialize version unique identifier.
40       */
41      private static final long serialVersionUID = -518184934631890227L;
42  
43      /***
44       * Constructor used to initialize the type definition enumeration.
45       *
46       * @param _typeDef  defines the related type definition enumeration
47       * @param _mxName   MX name of the administration object
48       */
49      public Table_mxJPO(final TypeDef_mxJPO _typeDef,
50                         final String _mxName)
51      {
52          super(_typeDef, _mxName);
53      }
54  
55      /***
56       * Writes each column to the appendable instance.
57       *
58       * @param _paramCache   parameter cache
59       * @param _out          appendable instance to the TCL update file
60       * @throws IOException if the TCL update code could not be written
61       * @see AbstractUIWithFields_mxJPO.Field#write(Appendable)
62       */
63      @Override
64      protected void writeObject(final ParameterCache_mxJPO _paramCache,
65                                 final Appendable _out)
66              throws IOException
67      {
68          for (final Field column : this.getFields())  {
69              _out.append(" //\n    column");
70              column.write(_out);
71          }
72      }
73  
74      /***
75       * The method overwrites the original method to append the MQL statements
76       * in the <code>_preMQLCode</code> to reset this web table. Following steps
77       * are done:
78       * <ul>
79       * <li>remove all columns of the web table</li>
80       * <li>set to not hidden</li>
81       * </ul>
82       *
83       * @param _paramCache       parameter cache
84       * @param _preMQLCode       MQL statements which must be called before the
85       *                          TCL code is executed
86       * @param _postMQLCode      MQL statements which must be called after the
87       *                          TCL code is executed
88       * @param _preTCLCode       TCL code which is defined before the source
89       *                          file is sourced
90       * @param _tclVariables     map of all TCL variables where the key is the
91       *                          name and the value is value of the TCL variable
92       *                          (the value is automatically converted to TCL
93       *                          syntax!)
94       * @param _sourceFile       souce file with the TCL code to update
95       * @throws Exception if the update from derived class failed
96       */
97      @Override
98      protected void update(final ParameterCache_mxJPO _paramCache,
99                            final CharSequence _preMQLCode,
100                           final CharSequence _postMQLCode,
101                           final CharSequence _preTCLCode,
102                           final Map<String,String> _tclVariables,
103                           final File _sourceFile)
104             throws Exception
105     {
106         // set to not hidden
107         final StringBuilder preMQLCode = new StringBuilder()
108                 .append("mod ").append(this.getTypeDef().getMxAdminName())
109                 .append(" \"").append(this.getName()).append('\"')
110                 .append(' ').append(this.getTypeDef().getMxAdminSuffix())
111                 .append(" !hidden");
112 
113         // remove all columns
114         for (final Field column : this.getFields())  {
115             preMQLCode.append(" column delete name \"").append(column.getName()).append('\"');
116         }
117 
118         // append already existing pre MQL code
119         preMQLCode.append(";\n")
120                   .append(_preMQLCode);
121 
122         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
123     }
