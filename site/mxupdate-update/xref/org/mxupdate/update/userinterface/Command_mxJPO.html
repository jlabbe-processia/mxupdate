36  /***
37   * The class is used to export and import / update command configuration items.

42  public class Command_mxJPO
43          extends AbstractAdminObject_mxJPO

45      /***
46       * Defines the serialize version unique identifier.

48      private static final long serialVersionUID = -6311434999021971324L;

50      /***
51       * Set of all ignored URLs from the XML definition for commands.
52       *
53       * @see #parse(String, String)
54       */
55      private static final Set<String> IGNORED_URLS = new HashSet<String>();
56      static  {
57          Command_mxJPO.IGNORED_URLS.add("/code");
58          Command_mxJPO.IGNORED_URLS.add("/input");
59          Command_mxJPO.IGNORED_URLS.add("/userRefList");
60      }

63       * Alt label of the command.
64       *

68      private String alt;

70      /***
71       * Label of the command.
72       *
73       * @see #parse(String, String)
74       * @see #writeObject(ParameterCache_mxJPO, Appendable)
75       */
76      private String label;

79       * HRef of the command.
80       *

106     /***
107      * Parses all command specific values. This includes:
108      * <ul>
109      * <li>{@link #alt}</li>
110      * <li>{@link #href}</li>
111      * <li>user references in {@link #users}</li>
112      * </ul>
113      *
114      * @param _url      URL to parse
115      * @param _content  related content of the URL to parse
116      */
117     @Override
118     protected void parse(final String _url,
119                          final String _content)
120     {
121         if (!Command_mxJPO.IGNORED_URLS.contains(_url))  {
122             if ("/alt".equals(_url))  {
123                 this.alt = _content;
124             } else if ("/href".equals(_url))  {
125                 this.href = _content;
126             } else if ("/label".equals(_url))  {
127                 this.label = _content;
128             } else if ("/userRefList/userRef".equals(_url))  {
129                 this.users.add(_content);
130             } else  {
131                 super.parse(_url, _content);
132             }
133         }
134     }
135 
136     /***
137      * @param _paramCache   parameter cache
138      * @param _out          appendable instance to the TCL update file
139      * @throws IOException if the TCL update code for the command could not be
140      *                     written
141      */
142     @Override
143     protected void writeObject(final ParameterCache_mxJPO _paramCache,
144                                final Appendable _out)
145             throws IOException
146     {
147         _out.append(" //\n    label \"").append(StringUtil_mxJPO.convertTcl(this.label)).append("\"")
148             .append(" //\n    href \"").append(StringUtil_mxJPO.convertTcl(this.href)).append("\"")
149             .append(" //\n    alt \"").append(StringUtil_mxJPO.convertTcl(this.alt)).append("\"");
150         for (final String user : this.users)  {
151             _out.append(" //\n    add user \"").append(StringUtil_mxJPO.convertTcl(user)).append("\"");
152         }
153         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
154             if (prop.isSetting())  {
155                 _out.append(" //\n    add setting \"")
156                     .append(StringUtil_mxJPO.convertTcl(prop.getName().substring(1))).append("\"")
157                     .append(" \"").append(StringUtil_mxJPO.convertTcl(prop.getValue())).append("\"");
158             }
159         }
160     }

162     /***
163      * The method overwrites the original method to append the MQL statements
164      * in the <code>_preMQLCode</code> to reset this command. Following steps
165      * are done:
166      * <ul>
167      * <li>HRef, description, alt and label is set to empty string</li>
168      * <li>all settings and users are removed</li>
169      * </ul>
170      *
171      * @param _paramCache       parameter cache
172      * @param _preMQLCode       MQL statements which must be called before the
173      *                          TCL code is executed
174      * @param _postMQLCode      MQL statements which must be called after the
175      *                          TCL code is executed
176      * @param _preTCLCode       TCL code which is defined before the source
177      *                          file is sourced
178      * @param _tclVariables     map of all TCL variables where the key is the
179      *                          name and the value is value of the TCL variable
180      *                          (the value is automatically converted to TCL
181      *                          syntax!)
182      * @param _sourceFile       souce file with the TCL code to update
183      * @throws Exception if the update from derived class failed
184      */
185     @Override
186     protected void update(final ParameterCache_mxJPO _paramCache,
187                           final CharSequence _preMQLCode,
188                           final CharSequence _postMQLCode,
189                           final CharSequence _preTCLCode,
190                           final Map<String,String> _tclVariables,
191                           final File _sourceFile)
192             throws Exception
193     {
194         // reset HRef, description, alt and label
195         final StringBuilder preMQLCode = new StringBuilder()
196                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
197                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
198                 .append(" href \"\" description \"\" alt \"\" label \"\"");
199 
200         // reset settings
201         for (final AdminProperty_mxJPO prop : this.getPropertiesMap().values())  {
202             if (prop.isSetting())  {
203                 preMQLCode.append(" remove setting \"")
204                           .append(StringUtil_mxJPO.convertMql(prop.getName().substring(1)))
205                           .append('\"');
206             }
207         }
208 
209         // reset users
210         for (final String user : this.users)  {
211             preMQLCode.append(" remove user \"").append(StringUtil_mxJPO.convertMql(user)).append('\"');
212         }
213 
214         // append already existing pre MQL code
215         preMQLCode.append(";\n")
216                   .append(_preMQLCode);
217 
218         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
219     }
