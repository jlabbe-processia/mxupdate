237     /***
238      * <p>Parses all common user specific URL values. This includes:
239      * <ul>
240      * <li>{@link #site}</li>
241      * <li>{@link #cues}</li>
242      * <li>{@link #filters}</li>
243      * <li>{@link #queries}</li>
244      * <li>{@link #tables}</li>
245      * <li>{@link #tips}</li>
246      * <li>{@link #toolSets tool sets}</li>
247      * <li>{@link #views}</li>
248      * </ul></p>
249      * <p>If an <code>_url</code> is included in {@link #IGNORED_URLS}, this
250      * URL is ignored.</p>
251      *
252      * @param _url      URL to parse
253      * @param _content  content of the URL to parse
254      * @see #IGNORED_URLS
255      */
256     @Override()
257     protected void parse(final String _url,
258                          final String _content)
259     {
260         if (!AbstractUser_mxJPO.IGNORED_URLS.contains(_url))  {
261             if ("/homeSite/siteRef".equals(_url))  {
262                 this.site = _content;
263 
264             } else if ("/cueList/cue".equals(_url))  {
265                 this.currentCue = new Cue_mxJPO(this);
266             } else if ("/cueList/cue/name".equals(_url))  {
267                 this.cues.put(_content, this.currentCue);
268                 this.currentCue.parse(_url.substring(12), _content);
269             } else if (_url.startsWith("/cueList/cue/"))  {
270                 this.currentCue.parse(_url.substring(12), _content);
271 
272             } else if ("/filterList/filter".equals(_url))  {
273                 this.currentFilter = new Filter_mxJPO(this);
274             } else if ("/filterList/filter/name".equals(_url))  {
275                 this.filters.put(_content, this.currentFilter);
276                 this.currentFilter.parse(_url.substring(18), _content);
277             } else if (_url.startsWith("/filterList/filter/"))  {
278                 this.currentFilter.parse(_url.substring(18), _content);
279 
280             } else if ("/queryList/query".equals(_url))  {
281                 this.currentQuery = new Query_mxJPO(this);
282             } else if ("/queryList/query/name".equals(_url))  {
283                 this.queries.put(_content, this.currentQuery);
284                 this.currentQuery.parse(_url.substring(16), _content);
285             } else if (_url.startsWith("/queryList/query/"))  {
286                 this.currentQuery.parse(_url.substring(16), _content);
287 
288             } else if ("/tableList/table".equals(_url))  {
289                 this.currentTable = new Table_mxJPO(this);
290             } else if ("/tableList/table/adminProperties/name".equals(_url))  {
291                 this.tables.put(_content, this.currentTable);
292                 this.currentTable.parse(_url.substring(32), _content);
293             } else if (_url.startsWith("/tableList/table/adminProperties/"))  {
294                 this.currentTable.parse(_url.substring(32), _content);
295             } else if (_url.startsWith("/tableList/table/"))  {
296                 this.currentTable.parse(_url.substring(16), _content);
297 
298             } else if ("/tipList/tip".equals(_url))  {
299                 this.currentTip = new Tip_mxJPO(this);
300             } else if ("/tipList/tip/name".equals(_url))  {
301                 this.tips.put(_content, this.currentTip);
302                 this.currentTip.parse(_url.substring(12), _content);
303             } else if (_url.startsWith("/tipList/tip/"))  {
304                 this.currentTip.parse(_url.substring(12), _content);
305 
306             } else if ("/toolsetList/toolset".equals(_url))  {
307                 this.currentToolSet = new ToolSet_mxJPO(this);
308             } else if ("/toolsetList/toolset/name".equals(_url))  {
309                 this.toolSets.put(_content, this.currentToolSet);
310                 this.currentToolSet.parse(_url.substring(20), _content);
311             } else if (_url.startsWith("/toolsetList/toolset/"))  {
312                 this.currentToolSet.parse(_url.substring(20), _content);
313 
314             } else if ("/viewList/view".equals(_url))  {
315                 this.currentView = new View_mxJPO(this);
316             } else if ("/viewList/view/name".equals(_url))  {
317                 this.views.put(_content, this.currentView);
318                 this.currentView.parse(_url.substring(14), _content);
319             } else if (_url.startsWith("/viewList/view/"))  {
320                 this.currentView.parse(_url.substring(14), _content);
321 
322             } else  {
323                 super.parse(_url, _content);
324             }
325         }
326     }

386     /***
387      * Writes specific information about the cached user to the given
388      * writer instance. This includes:
389      * <ul>
390      * <li>{@link #isHidden() hidden flag}</li>
391      * <li>{@link #site assigned site}</li>
392      * </ul>
393      *
394      * @param _paramCache   parameter cache
395      * @param _out          appendable instance to the TCL update file
396      * @throws IOException if the TCL update code for the group could not be
397      *                     written
398      */
399     @Override()
400     protected void writeObject(final ParameterCache_mxJPO _paramCache,
401                                final Appendable _out)
402             throws IOException
403     {
404         _out.append(" //\n    ").append(this.isHidden() ? "hidden" : "!hidden");
405 
406         // site
407         if (this.site != null)  {
408             _out.append(" //\n    site \"").append(StringUtil_mxJPO.convertTcl(this.site)).append('\"');
409         }
410     }

412     /***
413      * <p>Writes specific information about the cached role to the given
414      * writer instance. The included information is:
415      * <ul>
416      * <li>{@link #cues}</li>
417      * <li>{@link #filters}</li>
418      * <li>{@link #queries}</li>
419      * <li>{@link #tables}</li>
420      * <li>{@link #tips}</li>
421      * <li>{@link #toolSets tool sets}</li>
422      * <li>{@link #views}</li>
423      * </ul></p>
424      * <p>The workspace specific objects are only written to the TCL update
425      * file if the name of this user object does not match to the matches
426      * defined with the parameter {@link #PARAM_IGNORE_WSO_USERS}.</p>
427      *
428      * @param _paramCache   parameter cache
429      * @param _out          appendable instance to the TCL update file
430      * @throws IOException if the TCL update code could not written
431      * @see #ignoreWorkspaceObjects(ParameterCache_mxJPO)
432      */
433     @Override()
434     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
435                             final Appendable _out)
436         throws IOException
437     {
438 
439         if (!this.ignoreWorkspaceObjects(_paramCache))  {
440             // cues
441             for (final Cue_mxJPO cue : this.cues.values())  {
442                 cue.write(_paramCache, _out);
443             }
444             // filters
445             for (final Filter_mxJPO filter : this.filters.values())  {
446                 filter.write(_paramCache, _out);
447             }
448             // queries
449             for (final Query_mxJPO query : this.queries.values())  {
450                 query.write(_paramCache, _out);
451             }
452             // tables
453             for (final Table_mxJPO table : this.tables.values())  {
454                 table.write(_paramCache, _out);
455             }
456             // tips
457             for (final Tip_mxJPO tip : this.tips.values())  {
458                 tip.write(_paramCache, _out);
459             }
460             // tool sets
461             for (final ToolSet_mxJPO toolSet : this.toolSets.values())  {
462                 toolSet.write(_paramCache, _out);
463             }
464             // views
465             for (final View_mxJPO view : this.views.values())  {
466                 view.write(_paramCache, _out);
467             }
468         }
469     }

471     /***
472      * <p>The method overwrites the original method to append the MQL
473      * statements in the <code>_preMQLCode</code> to reset this user. Following
474      * steps are done:
475      * <ul>
476      * <li>remove assigned {@link #site}</li>
477      * <li>remove all {@link #cues}</li>
478      * <li>remove all {@link #filters}</li>
479      * <li>remove all {@link #queries}</li>
480      * <li>remove all {@link #tables}</li>
481      * <li>remove all {@link #tips}</li>
482      * <li>remove all {@link #toolSets tool sets}</li>
483      * <li>remove all {@link #views}</li>
484      * </ul></p>
485      * <p>The workspace specific objects are only removed if the name of this
486      * user object does not match to the matches defined with the parameter
487      * {@link #PARAM_IGNORE_WSO_USERS}.</p>
488      *
489      * @param _paramCache       parameter cache
490      * @param _preMQLCode       MQL statements which must be called before the
491      *                          TCL code is executed
492      * @param _postMQLCode      MQL statements which must be called after the
493      *                          TCL code is executed
494      * @param _preTCLCode       TCL code which is defined before the source
495      *                          file is sourced
496      * @param _tclVariables     map of all TCL variables where the key is the
497      *                          name and the value is value of the TCL variable
498      *                          (the value is automatically converted to TCL
499      *                          syntax!)
500      * @param _sourceFile       souce file with the TCL code to update
501      * @throws Exception if the update from derived class failed
502      * @see #ignoreWorkspaceObjects(ParameterCache_mxJPO)
503      */
504     @Override()
505     protected void update(final ParameterCache_mxJPO _paramCache,
506                           final CharSequence _preMQLCode,
507                           final CharSequence _postMQLCode,
508                           final CharSequence _preTCLCode,
509                           final Map<String,String> _tclVariables,
510                           final File _sourceFile)
511         throws Exception
512     {
513         final StringBuilder preMQLCode = new StringBuilder();
514 
515         // remove hidden flag
516         if (this.isHidden())  {
517             preMQLCode.append("escape mod ").append(this.getTypeDef().getMxAdminName())
518                       .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
519                       .append(" !hidden;\n");
520         }
521 
522         // remove site...
523         if (this.site != null)  {
524             preMQLCode.append("escape mod ").append(this.getTypeDef().getMxAdminName())
525                       .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
526                       .append(" site \"\";\n");
527         }
528 
529         if (!this.ignoreWorkspaceObjects(_paramCache))  {
530             // remove all assigned cues
531             for (final Cue_mxJPO cue : this.cues.values())  {
532                 preMQLCode.append("escape delete cue \"").append(StringUtil_mxJPO.convertMql(cue.getName()))
533                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
534             }
535             // remove all assigned filters
536             for (final Filter_mxJPO filter : this.filters.values())  {
537                 preMQLCode.append("escape delete filter \"").append(StringUtil_mxJPO.convertMql(filter.getName()))
538                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
539             }
540             // remove all assigned queries
541             for (final Query_mxJPO query : this.queries.values())  {
542                 preMQLCode.append("escape delete query \"").append(StringUtil_mxJPO.convertMql(query.getName()))
543                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
544             }
545             // remove all assigned tables
546             for (final Table_mxJPO table : this.tables.values())  {
547                 preMQLCode.append("escape delete table \"").append(StringUtil_mxJPO.convertMql(table.getName()))
548                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
549             }
550             // remove all assigned tips
551             for (final Tip_mxJPO tip : this.tips.values())  {
552                 preMQLCode.append("escape delete tip \"").append(StringUtil_mxJPO.convertMql(tip.getName()))
553                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
554             }
555             // remove all assigned tool sets
556             for (final ToolSet_mxJPO toolSet : this.toolSets.values())  {
557                 preMQLCode.append("escape delete toolset \"").append(StringUtil_mxJPO.convertMql(toolSet.getName()))
558                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
559             }
560             // remove all assigned views
561             for (final View_mxJPO view : this.views.values())  {
562                 preMQLCode.append("escape delete view \"").append(StringUtil_mxJPO.convertMql(view.getName()))
563                           .append("\" user \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\";\n");
564             }
565         }
566 
567         // append already existing pre MQL code
568         preMQLCode.append(_preMQLCode);
569 
570         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
571     }

573     /***
574      * Calculates if workspace objects for this user are not handled. This is
575      * done by checking if the name of this user matches one of the match lists
576      * defined with parameter {@link #PARAM_IGNORE_WSO_USERS}. In this case the
577      * the workspace objects are ignored for this user.
578      *
579      * @param _paramCache       parameter cache
580      * @return <i>true</i> if the handling of workspace objects for this user
581      *         is ignored
582      * @see #PARAM_IGNORE_WSO_USERS
583      */
584     protected boolean ignoreWorkspaceObjects(final ParameterCache_mxJPO _paramCache)
585     {
586         final Collection<String> ignoreMatches = _paramCache.getValueList(AbstractUser_mxJPO.PARAM_IGNORE_WSO_USERS);
587         boolean ignore = false;
