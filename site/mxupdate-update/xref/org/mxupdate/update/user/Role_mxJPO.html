42  public class Role_mxJPO

51       * Name of the parameter to define if current MX version supports role

56      private static final String PARAM_SUPPORT_ROLE_TYPES = "UserRoleSupportRoleType";

59       * Set to hold all parent roles.

61      private final Set<String> parentRoles = new TreeSet<String>();

64       * Stores the information about the role type.

68      private RoleType roleType = RoleType.ROLE;

82      /***
83       * Parses all role specific URLs.
84       *
85       * @param _url      URL to parse
86       * @param _content  content of the URL to parse
87       */
88      @Override
89      protected void parse(final String _url,
90                           final String _content)
91      {
92          if ("/parentRole".equals(_url))  {
93              // to be ignored ...
94          } else if ("/parentRole/roleRef".equals(_url))  {
95              this.parentRoles.add(_content);
96          } else if ("/roleType".equals(_url))  {
97              // to be ignored, because read within prepare method
98  
99          } else  {
100             super.parse(_url, _content);
101         }
102     }

104     /***
105      * Prepares the internal information after the XML export was parsed by
106      * adding the information about the role type (because the role type
107      * information was numbers within the XML string).
108      *
109      * @param _paramCache   parameter cache with the MX context
110      * @throws MatrixException if the information about the role type could not
111      *                         be fetched or if the exception was thrown from
112      *                         called super method
113      * @see #roleType
114      * @see #PARAM_SUPPORT_ROLE_TYPES
115      */
116     @Override
117     protected void prepare(final ParameterCache_mxJPO _paramCache)
118             throws MatrixException
119     {
120         // must the role type evaluated?
121         if (_paramCache.getValueBoolean(Role_mxJPO.PARAM_SUPPORT_ROLE_TYPES))  {
122             final String testRoleType = MqlUtil_mxJPO.execMql(_paramCache.getContext(), new StringBuilder()
123                     .append("escape print role \"").append(StringUtil_mxJPO.convertMql(this.getName()))
124                     .append("\" select isanorg isaproject dump"));
125             if ("FALSE,TRUE".equals(testRoleType))  {
126                 this.roleType = RoleType.PROJECT;
127             } else if ("TRUE,FALSE".equals(testRoleType)) {
128                 this.roleType = RoleType.ORGANIZATION;
129             }
130         }
131 
132         super.prepare(_paramCache);
133     }

135     /***
136      * Writes specific information about the cached role to the given
137      * writer instance.
138      *
139      * @param _paramCache   parameter cache
140      * @param _out          appendable instance to the TCL update file
141      * @throws IOException if the TCL update code could not written
142      */
143     @Override
144     protected void writeObject(final ParameterCache_mxJPO _paramCache,
145                                final Appendable _out)
146             throws IOException
147     {
148         _out.append(" //\n    ").append(this.isHidden() ? "hidden" : "!hidden");
149         // role type
150         switch (this.roleType)  {
151             case PROJECT:
152                 _out.append(" //\n    asaproject");
153                 break;
154             case ORGANIZATION:
155                 _out.append(" //\n    asanorg");
156                 break;
157         }
158         // parent roles
159         for (final String role : this.parentRoles)  {
160             _out.append("\nmql mod role \"")
161                 .append(StringUtil_mxJPO.convertTcl(role))
162                 .append("\" child \"${NAME}\"");
163         }
164     }

166     /***
167      * The method overwrites the original method to append the MQL statements
168      * in the <code>_preMQLCode</code> to reset this role. Following steps are
169      * done:
170      * <ul>
171      * <li>reset description</li>
172      * <li>remove all parent groups</li>
173      * <li>define as &quot;normal&quot; role</li>
174      * </ul>
175      *
176      * @param _paramCache       parameter cache
177      * @param _preMQLCode       MQL statements which must be called before the
178      *                          TCL code is executed
179      * @param _postMQLCode      MQL statements which must be called after the
180      *                          TCL code is executed
181      * @param _preTCLCode       TCL code which is defined before the source
182      *                          file is sourced
183      * @param _tclVariables     map of all TCL variables where the key is the
184      *                          name and the value is value of the TCL variable
185      *                          (the value is automatically converted to TCL
186      *                          syntax!)
187      * @param _sourceFile       souce file with the TCL code to update
188      * @throws Exception if the update from derived class failed
189      */
190     @Override
191     protected void update(final ParameterCache_mxJPO _paramCache,
192                           final CharSequence _preMQLCode,
193                           final CharSequence _postMQLCode,
194                           final CharSequence _preTCLCode,
195                           final Map<String,String> _tclVariables,
196                           final File _sourceFile)
197             throws Exception
198     {
199         // description and all parents
200         final StringBuilder preMQLCode = new StringBuilder()
201                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
202                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
203                 .append(" description \"\"")
204                 .append(" remove parent all");
205         // define normal role type
206         if (this.roleType != RoleType.ROLE)  {
207             preMQLCode.append(" asarole");
208         }
209 
210         // append already existing pre MQL code
211         preMQLCode.append(";\n")
212                   .append(_preMQLCode);
213 
214         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
215     }
