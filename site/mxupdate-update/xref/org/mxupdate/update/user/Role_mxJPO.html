52      /***
53       * Set of all ignored URLs from the XML definition for roles.

57      private static final Set<String> IGNORED_URLS = new HashSet<String>();
58      static  {
59          Role_mxJPO.IGNORED_URLS.add("/parentRole");

61          Role_mxJPO.IGNORED_URLS.add("/roleType");

78      private static final String PARAM_IGNORE_WSO_ROLES = "UserIgnoreWSO4Roles";

108     /***
109      * <p>Parses all role specific URL values. This includes:
110      * <ul>
111      * <li>{@link #parentRoles parent roles}</li>
112      * </ul></p>
113      * <p>If an <code>_url</code> is included in {@link #IGNORED_URLS}, this
114      * URL is ignored.</p>
115      *
116      * @param _url      URL to parse
117      * @param _content  content of the URL to parse
118      * @see #IGNORED_URLS
119      * @see #prepare(ParameterCache_mxJPO)
120      */
121     @Override()
122     protected void parse(final String _url,
123                          final String _content)
124     {
125         if (!Role_mxJPO.IGNORED_URLS.contains(_url))  {
126             if ("/parentRole/roleRef".equals(_url))  {
127                 this.parentRoles.add(_content);
128             } else  {
129                 super.parse(_url, _content);
130             }
131         }
132     }

134     /***
135      * Prepares the internal information after the XML export was parsed by
136      * adding the information about the role type (because the role type
137      * information was numbers within the XML string).
138      *
139      * @param _paramCache   parameter cache with the MX context
140      * @throws MatrixException if the information about the role type could not
141      *                         be fetched or if the exception was thrown from
142      *                         called super method
143      * @see #roleType
144      * @see #PARAM_SUPPORT_ROLE_TYPES
145      */
146     @Override()
147     protected void prepare(final ParameterCache_mxJPO _paramCache)
148         throws MatrixException
149     {
150         // must the role type evaluated?
151         if (_paramCache.getValueBoolean(Role_mxJPO.PARAM_SUPPORT_ROLE_TYPES))  {
152             final String testRoleType = MqlUtil_mxJPO.execMql(_paramCache, new StringBuilder()
153                     .append("escape print role \"").append(StringUtil_mxJPO.convertMql(this.getName()))
154                     .append("\" select isanorg isaproject dump"));
155             if ("FALSE,TRUE".equals(testRoleType))  {
156                 this.roleType = RoleType.PROJECT;
157             } else if ("TRUE,FALSE".equals(testRoleType)) {
158                 this.roleType = RoleType.ORGANIZATION;
159             }
160         }
161 
162         super.prepare(_paramCache);
163     }

165     /***
166      * Writes specific information about the cached role to the given
167      * writer instance.
168      *
169      * @param _paramCache   parameter cache
170      * @param _out          appendable instance to the TCL update file
171      * @throws IOException if the TCL update code could not written
172      */
173     @Override()
174     protected void writeObject(final ParameterCache_mxJPO _paramCache,
175                                final Appendable _out)
176         throws IOException
177     {
178         super.writeObject(_paramCache, _out);
179 
180         // role type
181         switch (this.roleType)  {
182             case PROJECT:
183                 _out.append(" //\n    asaproject");
184                 break;
185             case ORGANIZATION:
186                 _out.append(" //\n    asanorg");
187                 break;
188             case ROLE:
189             default:
190                 // the role is the default value
191         }
192         // parent roles
193         for (final String role : this.parentRoles)  {
194             _out.append("\nmql escape mod role \"")
195                 .append(StringUtil_mxJPO.convertTcl(role))
196                 .append("\" child \"${NAME}\"");
197         }
198     }

200     /***
201      * The method overwrites the original method to append the MQL statements
202      * in the <code>_preMQLCode</code> to reset this role. Following steps are
203      * done:
204      * <ul>
205      * <li>reset description</li>
206      * <li>remove all parent groups</li>
207      * <li>define as &quot;normal&quot; role</li>
208      * </ul>
209      *
210      * @param _paramCache       parameter cache
211      * @param _preMQLCode       MQL statements which must be called before the
212      *                          TCL code is executed
213      * @param _postMQLCode      MQL statements which must be called after the
214      *                          TCL code is executed
215      * @param _preTCLCode       TCL code which is defined before the source
216      *                          file is sourced
217      * @param _tclVariables     map of all TCL variables where the key is the
218      *                          name and the value is value of the TCL variable
219      *                          (the value is automatically converted to TCL
220      *                          syntax!)
221      * @param _sourceFile       souce file with the TCL code to update
222      * @throws Exception if the update from derived class failed
223      */
224     @Override()
225     protected void update(final ParameterCache_mxJPO _paramCache,
226                           final CharSequence _preMQLCode,
227                           final CharSequence _postMQLCode,
228                           final CharSequence _preTCLCode,
229                           final Map<String,String> _tclVariables,
230                           final File _sourceFile)
231         throws Exception
232     {
233         // description and all parents
234         final StringBuilder preMQLCode = new StringBuilder()
235                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
236                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
237                 .append(" description \"\"")
238                 .append(" remove parent all");
239         // define normal role type
240         if (this.roleType != RoleType.ROLE)  {
241             preMQLCode.append(" asarole");
242         }
243 
244         // append already existing pre MQL code
245         preMQLCode.append(";\n")
246                   .append(_preMQLCode);
247 
248         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
249     }

251     /***
252      * <p>Calculates if workspace objects for this role are not handled. This
253      * is done by checking if the name of this role matches one of the match
254      * lists defined with parameter {@link #PARAM_IGNORE_WSO_ROLES}. In this
255      * case the the workspace objects are ignored for this role.</p>
256      *
257      * @param _paramCache       parameter cache
258      * @return <i>true</i> if the handling of workspace objects for this role
259      *         is ignored
260      * @see #PARAM_IGNORE_WSO_ROLES
261      */
262     @Override()
263     protected boolean ignoreWorkspaceObjects(final ParameterCache_mxJPO _paramCache)
264     {
265         boolean ignore = super.ignoreWorkspaceObjects(_paramCache);
266         if (!ignore)  {
267             final Collection<String> ignoreMatches = _paramCache.getValueList(Role_mxJPO.PARAM_IGNORE_WSO_ROLES);
268             if (ignoreMatches != null)  {
269                 ignore = StringUtil_mxJPO.match(this.getName(), ignoreMatches);
270             }
271         }
272         return ignore;
273     }
