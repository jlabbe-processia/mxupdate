166     /***
167      * Returns a set of found person names. First for all administration
168      * persons are searched. Then, all persons are removed from the matching
169      * set, for whom a related business person object exists. This new set is
170      * returned.
171      *
172      * @param _paramCache   parameter cache
173      * @return set of person names for which no person business object exists
174      * @throws MatrixException if the query for person objects failed
175      */
176     @Override
177     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
178             throws MatrixException
179     {
180         final Set<String> persons = super.getMxNames(_paramCache);
181 
182         // evaluate for all business person object names
183         final StringList selects = new StringList();
184         selects.addElement("name");
185         final Query query = new Query();
186         query.open(_paramCache.getContext());
187         query.setBusinessObjectType(this.getTypeDef().getMxBusType());
188         final BusinessObjectWithSelectList list = query.select(_paramCache.getContext(), selects);
189         query.close(_paramCache.getContext());
190 
191         // remove the found persons which related business object
192         for (final Object mapObj : list)  {
193             final BusinessObjectWithSelect map = (BusinessObjectWithSelect) mapObj;
194             final String busName = (String) map.getSelectDataList("name").get(0);
195             persons.remove(busName);
196         }
197 
198         return persons;
199     }

201     /***

203      * @param _content  content depending on the URL
204      */
205     @Override
206     protected void parse(final String _url,
207                          final String _content)
208     {
209         if ("/assignmentList".equals(_url))  {
210             // to be ignored ...
211         } else if ("/assignmentList/assignment".equals(_url))  {
212             // to be ignored ...
213         } else if ("/assignmentList/assignment/groupRef".equals(_url))  {
214             this.groups.add(_content);
215         } else if ("/assignmentList/assignment/roleRef".equals(_url))  {
216             this.roles.add(_content);
217 
218         } else if ("/access".equals(_url))  {
219             // to be ignored ...
220         } else if (_url.startsWith("/access"))  {
221             this.access.add(_url.substring(8).replaceAll("Access$", ""));
222 
223         } else if ("/adminAccess".equals(_url))  {
224             // to be ignored ...
225         } else if (_url.startsWith("/adminAccess"))  {
226             this.adminAccess.add(_url.substring(13).replaceAll("Access$", ""));
227 
228         } else if ("/defaultApplication".equals(_url))  {
229             // to be ignored
230         } else if ("/defaultApplication/applicationRef".equals(_url))  {
231             this.defaultApplication = _content;
232 
233         } else if ("/inactive".equals(_url))  {
234             this.isInactive = true;
235         } else if ("/applicationsOnly".equals(_url))  {
236             this.isApplicationUser = true;
237         } else if ("/fullUser".equals(_url))  {
238             this.isFullUser = true;
239         } else if ("/businessAdministrator".equals(_url))  {
240             this.isBusinessAdministrator = true;
241         } else if ("/systemAdministrator".equals(_url))  {
242             this.isSystemAdministrator = true;
243         } else if ("/trusted".equals(_url))  {
244             this.isTrusted = true;
245 
246         } else if ("/wantsEmail".equals(_url))  {
247             this.wantsEmail = true;
248         } else if ("/wantsIconMail".equals(_url))  {
249             this.wantsIconMail = true;
250 
251         } else if ("/address".equals(_url))  {
252             this.address = _content;
253         } else if ("/email".equals(_url))  {
254             this.email = _content;
255         } else if ("/fax".equals(_url))  {
256             this.fax = _content;
257         } else if ("/fullName".equals(_url))  {
258             this.fullName = _content;
259         } else if ("/homeVault".equals(_url))  {
260             // to be ignored ...

263         } else if ("/phone".equals(_url))  {

271         // password must be ignored...
272         } else if (_url.startsWith("/password"))  {
273

275         } else if (_url.startsWith("/cueList"))  {
276         } else if (_url.startsWith("/filterList"))  {
277         } else if (_url.startsWith("/queryList"))  {
278         } else if (_url.startsWith("/tableList"))  {

323             .append(" //\n    vault \"").append(StringUtil_mxJPO.convertTcl(this.vault)).append("\"");
324         if (this.site != null)  {
325             _out.append(" //\n    site \"").append(StringUtil_mxJPO.convertTcl(this.site)).append("\"");
326         }
327         if (this.defaultApplication != null)  {
328             _out.append(" //\n    application \"")
329                 .append(StringUtil_mxJPO.convertTcl(this.defaultApplication)).append("\"");
330         }
331         for (final String group : this.groups)  {
332             _out.append(" //\n    assign group \"")
333                 .append(StringUtil_mxJPO.convertTcl(group))
334                 .append("\"");
335         }
336         for (final String role : this.roles)  {
337             _out.append(" //\n    assign role \"")
338                 .append(StringUtil_mxJPO.convertTcl(role))
339                 .append("\"");

343     /***
344      * Appends at the end of the TCL update file the change of the person
345      * &quot;type&quot;. The person &quot;type&quot; defines, if the person is
346      * e.g. active, or trusted etc..
347      *
348      * @param _paramCache   parameter cache
349      * @param _out          appendable instance to the TCL update file
350      * @throws IOException if the extension could not be written
351      */
352     @Override
353     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
354                             final Appendable _out)
355             throws IOException
356     {
357         _out.append("\nmql mod person \"${NAME}\" type ");
358         if (!this.isApplicationUser)  {
359             _out.append("not");
360         }
361         _out.append("application,");
362         if (!this.isFullUser)  {
363             _out.append("not");
364         }
365         _out.append("full,");
366         if (!this.isBusinessAdministrator)  {
367             _out.append("not");
368         }
369         _out.append("business,");
370         if (!this.isInactive)  {
371             _out.append("not");
372         }
373         _out.append("inactive,");
374         if (!this.isTrusted)  {
375             _out.append("not");
376         }
377         _out.append("trusted,");
378         if (!this.isSystemAdministrator)  {
379             _out.append("not");
380         }
381         _out.append("system");
382     }

384     /***
385      * The method overwrites the original method to
386      * <ul>
387      * <li>reset the description</li>
388      * <li>set the version and author attribute</li>
389      * <li>reset all not ignored attributes</li>
390      * <li>define the TCL variable &quot;OBJECTID&quot; with the object id of
391      *     the represented business object</li>
392      * </ul>
393      * The original method of the super class if called surrounded with a
394      * history off, because if the update itself is done the modified basic
395      * attribute and the version attribute of the business object is updated.
396      * <br/>
397      * The new generated MQL code is set in the front of the already defined
398      * MQL code in <code>_preMQLCode</code> and appended to the MQL statements
399      * in <code>_postMQLCode</code>.
400      *
401      * @param _paramCache       parameter cache
402      * @param _preMQLCode       MQL statements which must be called before the
403      *                          TCL code is executed
404      * @param _postMQLCode      MQL statements which must be called after the
405      *                          TCL code is executed
406      * @param _preTCLCode       TCL code which is defined before the source
407      *                          file is sourced
408      * @param _tclVariables     map of all TCL variables where the key is the
409      *                          name and the value is value of the TCL variable
410      *                          (the value is automatically converted to TCL
411      *                          syntax!)
412      * @param _sourceFile       souce file with the TCL code to update
413      * @throws Exception if the update from derived class failed
414      */
415     @Override
416     protected void update(final ParameterCache_mxJPO _paramCache,
417                           final CharSequence _preMQLCode,
418                           final CharSequence _postMQLCode,
419                           final CharSequence _preTCLCode,
420                           final Map<String,String> _tclVariables,
421                           final File _sourceFile)
422             throws Exception
423     {
424         // append other pre MQL code
425         final StringBuilder preMQLCode = new StringBuilder()
426                 .append(_preMQLCode)
427                 .append("escape mod person \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
428                         .append("access none ")
429                         .append("admin none ")
430                         .append("address '' ")
431                         .append("email '' ")
432                         .append("fax '' ")
433                         .append("fullname '' ")
434                         .append("phone '' ")
435                         .append("remove assign all");
436         if (this.defaultApplication != null)  {
437             preMQLCode.append(" application \"\"");
438         }
439         preMQLCode.append(";\n");
440 
441         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
442     }
