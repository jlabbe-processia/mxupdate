284     /***
285      * Returns a set of found person names. First for all administration
286      * persons are searched. Then, all persons are removed from the matching
287      * set, for whom a related business person object exists. This new set is
288      * returned.
289      *
290      * @param _paramCache   parameter cache
291      * @return set of person names for which no person business object exists
292      * @throws MatrixException if the query for person objects failed
293      */
294     @Override()
295     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
296             throws MatrixException
297     {
298         final Set<String> persons = super.getMxNames(_paramCache);
299 
300         // evaluate for all business person object names
301         final StringList selects = new StringList();
302         selects.addElement("name");
303         final Query query = new Query();
304         query.open(_paramCache.getContext());
305         query.setBusinessObjectType(this.getTypeDef().getMxBusType());
306         final BusinessObjectWithSelectList list = query.select(_paramCache.getContext(), selects);
307         query.close(_paramCache.getContext());
308 
309         // remove the found persons which related business object
310         for (final Object mapObj : list)  {
311             final BusinessObjectWithSelect map = (BusinessObjectWithSelect) mapObj;
312             final String busName = (String) map.getSelectDataList("name").get(0);
313             persons.remove(busName);
314         }
315 
316         return persons;
317     }

319     /***
320      * <p>Parses all person specific URL values. This includes:
321      * <ul>
322      * <li>{@link #groups assigned groups}</li>
323      * <li>{@link #roles assigned roles}</li>
324      * <li>{@link #access}</li>
325      * <li>{@link #adminAccess business administration access}</li>
326      * <li>is person {@link #isInactive inactive}</li>
327      * <li>flag if the person is an {@link #isApplicationUser application user}
328      *     </li>
329      * <li>flag if the person is a {@link #isFullUser full user}</li>
330      * <li>flag if the person is a
331      *     {@link #isBusinessAdministrator business administrator}</li>
332      * <li>flag if the person is a
333      *     {@link #isSystemAdministrator system administrator}</li>
334      * <li>flag if the person is a {@link #isTrusted trusted} person</li>
335      * <li>person wants {@link #wantsEmail email}</li>
336      * <li>person wants {@link #wantsIconMail icon mail}</li>
337      * <li>{@link #address}</li>
338      * <li>{@link #email email address}</li>
339      * <li>{@link #fax fax number}</li>
340      * <li>{@link #fullName full name}</li>
341      * <li>{@link #vault default vault}</li>
342      * <li>{@link #phone phone number}</li>
343      * <li>{@link #passwordNeverExpires password never expires flag}</li>
344      * </ul></p>
345      * <p>If an <code>_url</code> is included in {@link #IGNORED_URLS}, this
346      * URL is ignored.</p>
347      *
348      * @param _url      URL to parse
349      * @param _content  content depending on the URL
350      * @see #IGNORED_URLS
351      */
352     @Override()
353     protected void parse(final String _url,
354                          final String _content)
355     {
356         if (!PersonAdmin_mxJPO.IGNORED_URLS.contains(_url))  {
357             if ("/assignmentList/assignment/groupRef".equals(_url))  {
358                 this.groups.add(_content);
359             } else if ("/assignmentList/assignment/roleRef".equals(_url))  {
360                 this.roles.add(_content);
361 
362             } else if (_url.startsWith("/access"))  {
363                 this.access.add(_url.substring(8).replaceAll("Access$", ""));
364 
365             } else if (_url.startsWith("/adminAccess"))  {
366                 this.adminAccess.add(_url.substring(13).replaceAll("Access$", ""));
367 
368             } else if ("/defaultApplication/applicationRef".equals(_url))  {
369                 this.defaultApplication = _content;
370 
371             } else if ("/inactive".equals(_url))  {
372                 this.isInactive = true;
373             } else if ("/applicationsOnly".equals(_url))  {
374                 this.isApplicationUser = true;
375             } else if ("/fullUser".equals(_url))  {
376                 this.isFullUser = true;
377             } else if ("/businessAdministrator".equals(_url))  {
378                 this.isBusinessAdministrator = true;
379             } else if ("/systemAdministrator".equals(_url))  {
380                 this.isSystemAdministrator = true;
381             } else if ("/trusted".equals(_url))  {
382                 this.isTrusted = true;
383 
384             } else if ("/wantsEmail".equals(_url))  {
385                 this.wantsEmail = true;
386             } else if ("/wantsIconMail".equals(_url))  {
387                 this.wantsIconMail = true;
388 
389             } else if ("/address".equals(_url))  {
390                 this.address = _content;
391             } else if ("/email".equals(_url))  {
392                 this.email = _content;
393             } else if ("/fax".equals(_url))  {
394                 this.fax = _content;
395             } else if ("/fullName".equals(_url))  {
396                 this.fullName = _content;
397             } else if ("/homeVault/vaultRef".equals(_url))  {
398                 this.vault = _content;
399             } else if ("/phone".equals(_url))  {
400                 this.phone = _content;
401 
402             } else if ("/passwordNeverExpires".equals(_url))  {
403                 this.passwordNeverExpires = true;
404 
405             } else  {
406                 super.parse(_url, _content);
407             }
408         }
409     }

427     /***
428      * Writes specific information about the cached administration person to
429      * the given TCL update file <code>_out</code>. The included information is
430      * <ul>
431      * <li>{@link #passwordNeverExpires password never expires flag} (if
432      *     parameter {@link #PARAM_IGNORE_PSWD_NEVER_EXPIRES} is not set)</li>
433      * <li>{@link #access}</li>
434      * <li>{@link #adminAccess business administration access}</li>
435      * <li>person wants {@link #wantsEmail email} (if parameter
436      *     {@link #PARAM_IGNORE_WANTS_EMAIL} is not set)</li>
437      * <li>person wants {@link #wantsIconMail icon mail} (if parameter
438      *     {@link #PARAM_IGNORE_WANTS_ICON_MAIL} is not set)</li>
439      * <li>{@link #address}</li>
440      * <li>{@link #email email address}</li>
441      * <li>{@link #fax fax number}</li>
442      * <li>{@link #fullName full name}</li>
443      * <li>{@link #phone phone number}</li>
444      * <li>{@link #vault default vault}</li>
445      * <li>{@link #defaultApplication default application} (if defined)</li>
446      * <li>assigned {@link #groups}</li>
447      * <li>assigned {@link #roles}</li>
448      * </ul>
449      *
450      * @param _paramCache   parameter cache
451      * @param _out          appendable instance to the TCL update file
452      * @throws IOException if the TCL update code could not written
453      * @see #PARAM_IGNORE_PSWD_NEVER_EXPIRES
454      * @see #PARAM_IGNORE_WANTS_EMAIL
455      */
456     @Override()
457     protected void writeObject(final ParameterCache_mxJPO _paramCache,
458                                final Appendable _out)
459         throws IOException
460     {
461         super.writeObject(_paramCache, _out);
462 
463         // password never expires flag if not matched
464         final Collection<String> ignorePswdNeverExpires = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_PSWD_NEVER_EXPIRES);
465         if ((ignorePswdNeverExpires == null) || !StringUtil_mxJPO.match(this.getName(), ignorePswdNeverExpires))  {
466             _out.append(" //\n    ").append(this.passwordNeverExpires ? "" : "!").append("neverexpires");
467         }
468         _out.append(" //\n    access \"").append(StringUtil_mxJPO.joinTcl(',', false, this.access, "none")).append("\"")
469             .append(" //\n    admin \"").append(StringUtil_mxJPO.joinTcl(',', false, this.adminAccess, "none")).append("\"");
470         // wants email only if not matched
471         final Collection<String> ignoreWantsEmail = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_WANTS_EMAIL);
472         if ((ignoreWantsEmail == null) || !StringUtil_mxJPO.match(this.getName(), ignoreWantsEmail))  {
473             _out.append(" //\n    ").append(this.wantsEmail ? "enable" : "disable").append(" email");
474         }
475         // wants icon mail only if not matched
476         final Collection<String> ignoreWantsIconMail = _paramCache.getValueList(PersonAdmin_mxJPO.PARAM_IGNORE_WANTS_ICON_MAIL);
477         if ((ignoreWantsIconMail == null) || !StringUtil_mxJPO.match(this.getName(), ignoreWantsIconMail))  {
478             _out.append(" //\n    ").append(this.wantsIconMail ? "enable" : "disable").append(" iconmail");
479         }
480         _out.append(" //\n    address \"").append(StringUtil_mxJPO.convertTcl(this.address)).append("\"")
481             .append(" //\n    email \"").append(StringUtil_mxJPO.convertTcl(this.email)).append("\"")
482             .append(" //\n    fax \"").append(StringUtil_mxJPO.convertTcl(this.fax)).append("\"")
483             .append(" //\n    fullname \"").append(StringUtil_mxJPO.convertTcl(this.fullName)).append("\"")
484             .append(" //\n    phone \"").append(StringUtil_mxJPO.convertTcl(this.phone)).append("\"")
485             .append(" //\n    vault \"").append(StringUtil_mxJPO.convertTcl(this.vault)).append("\"");
486         if (this.defaultApplication != null)  {
487             _out.append(" //\n    application \"")
488                 .append(StringUtil_mxJPO.convertTcl(this.defaultApplication)).append("\"");
489         }
490         for (final String group : this.groups)  {
491             _out.append(" //\n    assign group \"")
492                 .append(StringUtil_mxJPO.convertTcl(group))
493                 .append("\"");
494         }
495         for (final String role : this.roles)  {
496             _out.append(" //\n    assign role \"")
497                 .append(StringUtil_mxJPO.convertTcl(role))
498                 .append("\"");
499         }
500     }

502     /***
503      * Appends at the end of the TCL update file the change of the person
504      * &quot;type&quot;. The person &quot;type&quot; defines, if the person is
505      * e.g. active, or trusted etc..
506      *
507      * @param _paramCache   parameter cache
508      * @param _out          appendable instance to the TCL update file
509      * @throws IOException if the extension could not be written
510      */
511     @Override()
512     protected void writeEnd(final ParameterCache_mxJPO _paramCache,
513                             final Appendable _out)
514             throws IOException
515     {
516         super.writeEnd(_paramCache, _out);
517         _out.append("\nmql mod person \"${NAME}\" type ");
518         if (!this.isApplicationUser)  {
519             _out.append("not");
520         }
521         _out.append("application,");
522         if (!this.isFullUser)  {
523             _out.append("not");
524         }
525         _out.append("full,");
526         if (!this.isBusinessAdministrator)  {
527             _out.append("not");
528         }
529         _out.append("business,");
530         if (!this.isInactive)  {
531             _out.append("not");
532         }
533         _out.append("inactive,");
534         if (!this.isTrusted)  {
535             _out.append("not");
536         }
537         _out.append("trusted,");
538         if (!this.isSystemAdministrator)  {
539             _out.append("not");
540         }
541         _out.append("system");
542     }

544     /***
545      * The method overwrites the original method to
546      * <ul>
547      * <li>reset the comment (description)</li>
548      * <li>set the version and author attribute</li>
549      * <li>reset all not ignored attributes</li>
550      * <li>define the TCL variable &quot;OBJECTID&quot; with the object id of
551      *     the represented business object</li>
552      * <li>sets the {@link #passwordNeverExpires password never expires flag}
553      *     to <i>false</i> (means that the password expires); depends on
554      *     parameter {@link #PARAM_IGNORE_PSWD_NEVER_EXPIRES}</li>
555      * <li>disables that the person wants email; depends on
556      *     parameter {@link #PARAM_IGNORE_WANTS_EMAIL}</li>
557      * <li>enables that the person wants icon mail; depends on
558      *     parameter {@link #PARAM_IGNORE_WANTS_ICON_MAIL}</li>
559      * </ul>
560      * The original method of the super class if called surrounded with a
561      * history off, because if the update itself is done the modified basic
562      * attribute and the version attribute of the business object is updated.
563      * <br/>
564      * The new generated MQL code is set in the front of the already defined
565      * MQL code in <code>_preMQLCode</code> and appended to the MQL statements
566      * in <code>_postMQLCode</code>.
567      *
568      * @param _paramCache       parameter cache
569      * @param _preMQLCode       MQL statements which must be called before the
570      *                          TCL code is executed
571      * @param _postMQLCode      MQL statements which must be called after the
572      *                          TCL code is executed
573      * @param _preTCLCode       TCL code which is defined before the source
574      *                          file is sourced
575      * @param _tclVariables     map of all TCL variables where the key is the
576      *                          name and the value is value of the TCL variable
577      *                          (the value is automatically converted to TCL
578      *                          syntax!)
579      * @param _sourceFile       souce file with the TCL code to update
580      * @throws Exception if the update from derived class failed
581      * @see #PARAM_IGNORE_PSWD_NEVER_EXPIRES
582      * @see #PARAM_IGNORE_WANTS_EMAIL
583      * @see #PARAM_IGNORE_WANTS_ICON_MAIL
584      */
585     @Override()
586     protected void update(final ParameterCache_mxJPO _paramCache,
587                           final CharSequence _preMQLCode,
588                           final CharSequence _postMQLCode,
589                           final CharSequence _preTCLCode,
590                           final Map<String,String> _tclVariables,
591                           final File _sourceFile)
592         throws Exception
593     {
594         // append other pre MQL code
595         final StringBuilder preMQLCode = new StringBuilder()
596                 .append(_preMQLCode)
597                 .append("escape mod person \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
598                         .append("comment '' ")
599                         .append("access none ")
600                         .append("admin none ")
601                         .append("address '' ")
602                         .append("email '' ")
603                         .append("fax '' ")
604                         .append("fullname '' ")
605                         .append("phone '' ")
606                         .append("remove assign all");
