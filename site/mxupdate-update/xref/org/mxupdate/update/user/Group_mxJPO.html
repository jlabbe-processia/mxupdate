40  public class Group_mxJPO
41          extends AbstractAdminObject_mxJPO
42  {
43      /***
44       * Defines the serialize version unique identifier.
45       */
46      private static final long serialVersionUID = -4819262751735809709L;
47  
48      /***
49       * Set to hold all parent groups.
50       */
51      final Set<String> parentGroups = new TreeSet<String>();
52  
53      /***
54       * Constructor used to initialize the type definition enumeration.
55       *
56       * @param _typeDef  defines the related type definition enumeration
57       * @param _mxName   MX name of the group object
58       */
59      public Group_mxJPO(final TypeDef_mxJPO _typeDef,
60                         final String _mxName)
61      {
62          super(_typeDef, _mxName);
63      }
64  
65      /***
66       * Parses all group specific URLs.
67       *
68       * @param _url      URL to parse
69       * @param _content  content of the URL to parse
70       */
71      @Override
72      protected void parse(final String _url,
73                        final String _content)
74      {
75          if ("/parentGroup".equals(_url))  {
76              // to be ignored ...
77          } else if ("/parentGroup/groupRef".equals(_url))  {
78              this.parentGroups.add(_content);
79  
80          } else  {
81              super.parse(_url, _content);
82          }
83      }
84  
85      /***
86       * Writes specific information about the cached group to the given
87       * writer instance.
88       *
89       * @param _paramCache   parameter cache
90       * @param _out          appendable instance to the TCL update file
91       * @throws IOException if the TCL update code for the group could not be
92       *                     written
93       */
94      @Override
95      protected void writeObject(final ParameterCache_mxJPO _paramCache,
96                                 final Appendable _out)
97              throws IOException
98      {
99          _out.append(" //\n    ").append(isHidden() ? "hidden" : "!hidden");
100         for (final String group : this.parentGroups)  {
101             _out.append("\nmql mod group \"")
102                 .append(convertTcl(group))
103                 .append("\" child \"${NAME}\"");
104         }
105     }
106 
107     /***
108      * The method overwrites the original method to append the MQL statements
109      * in the <code>_preMQLCode</code> to reset this group. Following steps
110      * are done:
111      * <ul>
112      * <li>reset description</li>
113      * <li>remove all parent groups</li>
114      * </ul>
115      *
116      * @param _paramCache       parameter cache
117      * @param _preMQLCode       MQL statements which must be called before the
118      *                          TCL code is executed
119      * @param _postMQLCode      MQL statements which must be called after the
120      *                          TCL code is executed
121      * @param _preTCLCode       TCL code which is defined before the source
122      *                          file is sourced
123      * @param _tclVariables     map of all TCL variables where the key is the
124      *                          name and the value is value of the TCL variable
125      *                          (the value is automatically converted to TCL
126      *                          syntax!)
127      * @param _sourceFile       souce file with the TCL code to update
128      * @throws Exception if the update from derived class failed
129      */
130     @Override
131     protected void update(final ParameterCache_mxJPO _paramCache,
132                           final CharSequence _preMQLCode,
133                           final CharSequence _postMQLCode,
134                           final CharSequence _preTCLCode,
135                           final Map<String,String> _tclVariables,
136                           final File _sourceFile)
137             throws Exception
138     {
139         // description and all parents
140         final StringBuilder preMQLCode = new StringBuilder()
141                 .append("mod ").append(this.getTypeDef().getMxAdminName())
142                 .append(" \"").append(this.getName()).append('\"')
143                 .append(" description \"\"")
144                 .append(" remove parent all;\n");
145 
146         // append already existing pre MQL code
147         preMQLCode.append(_preMQLCode);
148 
149         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
150     }
