160     /***
161      *
162      * @param _paramCache   parameter cache
163      * @param _out          writer instance
164      * @throws IOException if the header could not be written to the TCL update
165      *                     file
166      * @see #HEADER_APPLICATION
167      * @see #HEADER_AUTHOR
168      * @see #HEADER_INSTALLER
169      * @see #HEADER_ORIGINALNAME
170      * @see #HEADER_SYMBOLIC_NAME
171      * @see #HEADER_VERSION
172      * @see #PARAM_EXPORTAPPLICATION
173      * @see #PARAM_EXPORTAUTHOR
174      * @see #PARAM_EXPORTINSTALLER
175      * @see #PARAM_EXPORTORIGINALNAME
176      * @see #PARAM_EXPORTVERSION
177      */
178     protected void writeHeader(final ParameterCache_mxJPO _paramCache,
179                                final Appendable _out)
180             throws IOException
181     {
182         final String headerText = this.getTypeDef().getTitle();
183         _out.append("################################################################################\n")
184             .append("# ").append(headerText).append(":\n")
185             .append("# ~");
186         for (int i = 0; i < headerText.length(); i++)  {
187             _out.append("~");
188         }
189         _out.append("\n")
190             .append("# ").append(this.getName()).append("\n");
191         // symbolic name only if defined
192         if (this.getSymbolicNames() != null)  {
193             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_SYMBOLIC_NAME).append(' ');
194             if (this.getSymbolicNames().isEmpty())  {
195                 final String symbName = this.calcDefaultSymbolicName(_paramCache);
196                 _paramCache.logWarning("    - No symbolic name defined for '" + this.getName()
197                         + "'! Symbolic name '" + symbName + "' will be used!");
198                 _out.append(symbName);
199             } else if (this.getSymbolicNames().size() > 1)  {
200                 final String symbName = this.getSymbolicNames().iterator().next();
201                 _paramCache.logError("    - Found " + this.getSymbolicNames().size()
202                         + " symbolic names! Only one is allowed! '"
203                         + symbName + "' will be used!");
204                 _paramCache.logTrace("    - Following symbolic names found:");
205                 for (final String origSymb : this.getSymbolicNames())  {
206                     _paramCache.logTrace("      * '" + origSymb + "'");
207                 }
208                 _out.append(symbName);
209             } else  {
210                 _out.append(this.getSymbolicNames().iterator().next());
211             }
212             _out.append('\n');
213         }
214         // original name
215         // (only if an administration type and related parameter is defined)
216         if ((this.getTypeDef().getMxAdminName() != null)
217                 && _paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTORIGINALNAME))  {
218             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_ORIGINALNAME);
219             if ((this.getOriginalName() != null) && !"".equals(this.getOriginalName()))  {
220                 _out.append(" ").append(this.getOriginalName()).append('\n');
221             } else {
222                 _out.append("\n");
223             }
224         }
225         // description
226         _out.append("#\n# DESCRIPTION:\n")
227             .append("# ~~~~~~~~~~~~\n");
228         if ((this.getDescription() != null) && !"".equals(this.getDescription()))  {
229             for (final String partDesc : this.getDescription().split("\n"))  {
230                 _out.append('#');
231                 int length = 0;
232                 for (final String desc : partDesc.split(" "))  {
233                     if (!"".equals(desc))  {
234                         length += desc.length() + 1;
235                         if (length > 79)  {
236                             _out.append("\n#");
237                             length = desc.length() + 1;
238                         }
239                         _out.append(' ').append(desc);
240                     }
241                 }
242                 _out.append("\n");
243             }
244         } else  {
245             _out.append("#\n");
246         }
247         // write header
248         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTAUTHOR))  {
249             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_AUTHOR);
250             if ((this.getAuthor() != null) && !"".equals(this.getAuthor()))  {
251                 _out.append(" ").append(this.getAuthor()).append('\n');
252             } else {
253                 _out.append("\n");
254             }
255         }
256         // write installer
257         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTINSTALLER))  {
258             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_INSTALLER);
259             if ((this.getInstaller() != null) && !"".equals(this.getInstaller()))  {
260                 _out.append(" ").append(this.getInstaller()).append('\n');
261             } else {
262                 _out.append("\n");
263             }
264         }
265         // write application
266         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTAPPLICATION))  {
267             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_APPLICATION);
268             if ((this.getApplication() != null) && !"".equals(this.getApplication()))  {
269                 _out.append(" ").append(this.getApplication()).append('\n');
270             } else {
271                 _out.append("\n");
272             }
273         }
274         // write version
275         if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_EXPORTVERSION))  {
276             _out.append('#').append(AbstractPropertyObject_mxJPO.HEADER_VERSION);
277             if ((this.getVersion() != null) && !"".equals(this.getVersion()))  {
278                 _out.append(" ").append(this.getVersion()).append('\n');
279             } else {
280                 _out.append("\n");
281             }
282         }
283         _out.append("################################################################################\n\n");
284     }

287     /***
288      * Updates this administration (business) object if the stored information
289      * about the version is not the same as the file date. If an update is
290      * required, the file is read and the object is updated with
291      * {@link #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)}.
292      *
293      * @param _paramCache       parameter cache
294      * @param _file             file with TCL update code
295      * @param _newVersion       new version which must be set within the update
296      *                          (or <code>null</code> if the version must not
297      *                          be set).
298      * @throws Exception if the update from the derived class failed
299      * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
300      * @see #extractFromCode(StringBuilder, String, String)
301      * @see #extractSymbolicNameFromCode(ParameterCache_mxJPO,StringBuilder)
302      * @see #HEADER_APPLICATION
303      * @see #HEADER_AUTHOR
304      * @see #HEADER_INSTALLER
305      * @see #HEADER_ORIGINALNAME
306      * @see #HEADER_VERSION
307      */
308     @Override()
309     public void update(final ParameterCache_mxJPO _paramCache,
310                        final File _file,
311                        final String _newVersion)
312         throws Exception
313     {
314         // parse objects
315         this.parse(_paramCache);
316 
317         // read code
318         final StringBuilder code = this.getCode(_file);
319 
320         // defines the version in the TCL variables
321         final Map<String,String> tclVariables = new HashMap<String,String>();
322         if (_newVersion == null)  {
323             tclVariables.put(PropertyDef_mxJPO.VERSION.name(),
324                               this.extractFromCode(code, AbstractPropertyObject_mxJPO.HEADER_VERSION, ""));
325         } else  {
326             tclVariables.put(PropertyDef_mxJPO.VERSION.name(), _newVersion);
327         }
328 
329         // define author
330         final String author;
331         if (_paramCache.contains(ParameterCache_mxJPO.KEY_AUTHOR))  {
332             author = _paramCache.getValueString(ParameterCache_mxJPO.KEY_AUTHOR);
333         } else  {
334             author = this.extractFromCode(code,
335                                           AbstractPropertyObject_mxJPO.HEADER_AUTHOR,
336                                           _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAUTHOR));
337         }
338         tclVariables.put(PropertyDef_mxJPO.AUTHOR.name(), author);
339 
340         // define application
341         String appl = null;
342         if (_paramCache.contains(ParameterCache_mxJPO.KEY_APPLICATION))  {
343             appl = _paramCache.getValueString(ParameterCache_mxJPO.KEY_APPLICATION);
344         }
345         if ((appl == null) || "".equals(appl))  {
346             appl = this.extractFromCode(code,
347                                         AbstractPropertyObject_mxJPO.HEADER_APPLICATION,
348                                         _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTAPPLICATION));
349         }
350         if (appl == null)  {
351             appl = "";
352         }
353         tclVariables.put(PropertyDef_mxJPO.APPLICATION.name(), appl);
354 
355         // define installer
356         final String installer;
357         if (_paramCache.contains(ParameterCache_mxJPO.KEY_INSTALLER))  {
358             installer = _paramCache.getValueString(ParameterCache_mxJPO.KEY_INSTALLER);
359         } else  {
360             installer = this.extractFromCode(code,
361                                              AbstractPropertyObject_mxJPO.HEADER_INSTALLER,
362                                              _paramCache.getValueString(ParameterCache_mxJPO.KEY_DEFAULTINSTALLER));
363         }
364         tclVariables.put(PropertyDef_mxJPO.INSTALLER.name(), installer);
365 
366         if (this.getTypeDef().getMxAdminName() != null)  {
367             // define symbolic name
368             tclVariables.put("SYMBOLICNAME", this.extractSymbolicNameFromCode(_paramCache, code));
369 
370             // define original name
371             final String origName;
372             if ((this.getOriginalName() != null) && !"".equals(this.getOriginalName()))  {
373                 origName = this.getOriginalName();
374             } else  {
375                 origName = this.getName();
376             }
377             tclVariables.put(PropertyDef_mxJPO.ORIGINALNAME.name(), origName);
378         }
379 
380         // define file date
381         tclVariables.put(PropertyDef_mxJPO.FILEDATE.name(),
382                          StringUtil_mxJPO.formatFileDate(_paramCache, new Date(_file.lastModified())));
383 
384         this.update(_paramCache, "", "", "", tclVariables, _file);
385     }

402     /***
403      * Extracts for given header text the related value from the source code.
404      * If no value in the update file is defined, the default value from is
405      * used.
406      *
407      * @param _code             TCL update source code
408      * @param _headerText       text used to identify the value
409      * @param _default          default value if not defined within header
410      * @return extracted string from source code
411      */
412     protected String extractFromCode(final StringBuilder _code,
413                                      final String _headerText,
414                                      final String _default)
415     {
416         final int length = _headerText.length();
417         final int start = _code.indexOf(_headerText) + length;
418         final String value;
419         if ((start > length) && (_code.charAt(start) == ' '))  {
420             final int end = _code.indexOf("\n", start);
421             if (end > 0)  {
422                 final String tmp = _code.substring(start, end).trim();
423                 if ("".equals(tmp))  {
424                     value = _default;
425                 } else  {
426                     value = tmp;
427                 }
428             } else  {
429                 value = _default;
430             }
431         } else  {
432             value = _default;
433         }
434         return value;
435     }

437     /***
438      * <p>Returns the symbolic name for current property object. If parameter
439      * {@link #PARAM_CALCSYMBOLICNAMES} is defined, the symbolic name is only
440      * calculated. Otherwise the header is checked if there is a defined
441      * symbolic name (and by used if exists). A check is done if the symbolic
442      * name starts correctly with the administration type name (in lower
443      * case!). If not, the calculated symbolic name is used. A warning is shown
444      * if the symbolic name defined in the header is not equal to the
445      * calculated symbolic name.</p>
446      * <p>The symbolic name is calculated by removing all spaces and slashes
447      * within the name of the object and add as prefix the administration type
448      * name.</p>
449      *
450      * @param _paramCache   parameter cache
451      * @param _code         TCL update source code
452      * @return extracted symbolic name from the source code header
453      * @see #PARAM_CALCSYMBOLICNAMES
454      */
455     private String extractSymbolicNameFromCode(final ParameterCache_mxJPO _paramCache,
456                                                final StringBuilder _code)
457     {
458         String codeSymbName = null;
459 
460         if (this.getTypeDef().getMxAdminName() != null)  {
461             final String symbName = this.calcDefaultSymbolicName(_paramCache);
462             if (_paramCache.getValueBoolean(AbstractPropertyObject_mxJPO.PARAM_CALCSYMBOLICNAMES))  {
463                 codeSymbName = symbName;
464                 _paramCache.logDebug("    - using calculated symbolic name '" + symbName + "'");
465             } else  {
466                 codeSymbName = this.extractFromCode(_code, AbstractPropertyObject_mxJPO.HEADER_SYMBOLIC_NAME, null);
467                 if (codeSymbName == null)  {
468                     _paramCache.logError("No symbolic name defined! So '" + symbName + "' will be used.");
469                     codeSymbName = symbName;
470                 } else if (!codeSymbName.startsWith(this.getTypeDef().getMxAdminName()))  {
471                     _paramCache.logError("Symbolic name does not start correctly! So '"
472                             + symbName + "' will be used (instead of '" + codeSymbName + "').");
473                     codeSymbName = symbName;
474                 } else if (!codeSymbName.equals(symbName))  {
475                     _paramCache.logWarning("Symbolic name '" + symbName
476                             + "' should be used! But defined is '" + codeSymbName + "'.");
477                 }
478             }
479         }
480 
481         return codeSymbName;
482     }

484     /***
485      * The method updates this administration (business) object. First all MQL
486      * commands are concatenated:
487      * <ul>
488      * <li>pre MQL commands (from parameter <code>_preMQLCode</code>)</li>
489      * <li>change to TCL mode</li>
490      * <li>set all required TCL variables (from parameter
491      *     <code>_tclVariables</code>)</li>
492      * <li>append TCL update code from file (from parameter
493      *     <code>_tclCode</code>)</li>
494      * <li>change back to MQL mode</li>
495      * <li>append post MQL statements (from parameter
496      *     <code>_postMQLCode</code>)</li>
497      * </ul>
498      * At the end of the MQL statements, {@link #TEST_EXECUTED} is printed via
499      * MQL command &quot;<code>output</code> so that this output is returned
500      * from the MQL execution. The return of the MQL execution is tested if
501      * this output is returned. If not an exception is thrown because the
502      * update failed.
503      *
504      * @param _paramCache       parameter cache
505      * @param _preMQLCode       MQL statements which must be called before the
506      *                          TCL code is executed
507      * @param _postMQLCode      MQL statements which must be called after the
508      *                          TCL code is executed
509      * @param _preTCLCode       TCL code which is defined before the source
510      *                          file is sourced
511      * @param _tclVariables     map of all TCL variables where the key is the
512      *                          name and the value is value of the TCL variable
513      *                          (the value is automatically converted to TCL
514      *                          syntax!)
515      * @param _sourceFile       souce file with the TCL code to update; if
516      *                          <code>null</code> file is not called (sourced)
517      * @throws Exception if update failed
518      */
519     protected void update(final ParameterCache_mxJPO _paramCache,
520                           final CharSequence _preMQLCode,
521                           final CharSequence _postMQLCode,
522                           final CharSequence _preTCLCode,
523                           final Map<String,String> _tclVariables,
524                           final File _sourceFile)
525         throws Exception
526     {
527         final StringBuilder cmd = new StringBuilder().append(_preMQLCode);
528 
529         // append TCL mode
530         cmd.append("tcl;\n")
531            .append("eval  {\n");
532 
533         // define all TCL variables
534         for (final Map.Entry<String, String> entry : _tclVariables.entrySet())  {
535             cmd.append("set ").append(entry.getKey())
536                .append(" \"").append(StringUtil_mxJPO.convertTcl(entry.getValue())).append("\"\n");
537         }
538         // append TCL code, end of TCL mode and post MQL statements
539         // (source with the file must be replace for windows ...)
540         cmd.append(_preTCLCode);
541         if (_sourceFile != null)  {
542             cmd.append("\nsource \"").append(_sourceFile.toString().replaceAll("////", "/")).append("\"");
543         }
544         cmd.append("\n}\nexit;\n")
545            .append(_postMQLCode)
546            .append("output '';output '").append(AbstractPropertyObject_mxJPO.TEST_EXECUTED).append("';");
547 
548         // execute update
549         JPOCaller_mxJPO.defineInstance(_paramCache, this);
550         try  {
551             final String[] ret = MqlUtil_mxJPO.execMql(_paramCache, cmd).split("\n");
552             if (!AbstractPropertyObject_mxJPO.TEST_EXECUTED.equals(ret[ret.length - 1]))  {
553                 throw new Exception("Execution of the update was not complete! Update Failed!");
554             }
555         } finally  {
556             JPOCaller_mxJPO.undefineInstance(_paramCache);
557         }
558     }
