226     /***
227      * Searches for all attribute objects depending on the attribute type.
228      *
229      * @param _paramCache   parameter cache
230      * @return set of MX names of all attributes of attribute type
231      *         {@link #attrTypeList}
232      * @throws MatrixException if the query for attribute objects failed
233      * @see #SELECT_ATTRS
234      * @see #attrTypeList
235      */
236     @Override()
237     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
238         throws MatrixException
239     {
240         final Set<String> ret = new TreeSet<String>();
241         final int length = this.attrTypeList.length();
242         for (final String name : MqlUtil_mxJPO.execMql(_paramCache,
243                                                        AbstractAttribute_mxJPO.SELECT_ATTRS).split("\n"))  {
244             if (!"".equals(name) && name.startsWith(this.attrTypeList))  {
245                 ret.add(name.substring(length));
246             }
247         }
248         return ret;
249     }

251     /***
252      * The method parses the attribute specific XML urls. This includes
253      * information about:
254      * <ul>
255      * <li>assigned rules (see {@link #rules})</li>
256      * <li>default value (see {@link #defaultValue})</li>
257      * <li>is the attribute multiline (see {@link #multiline})</li>
258      * <li>attribute type (see {@link #type})</li>
259      * <li>defined ranges (see {@link #ranges}, {@link #rangeProgramRef} and
260      *     {@link #rangeProgramInputArguments}</li>
261      * </ul>
262      *
263      * @param _url      URL to parse
264      * @param _content  content of the URL to parse
265      * @see #IGNORED_URLS
266      */
267     @Override()
268     protected void parse(final String _url,
269                          final String _content)
270     {
271         if (!AbstractAttribute_mxJPO.IGNORED_URLS.contains(_url))  {
272             if ("/accessRuleRef".equals(_url))  {
273                 this.rules.add(_content);
274             } else if ("/defaultValue".equals(_url))  {
275                 this.defaultValue = _content;
276             } else if ("/dimensionRef".equals(_url))  {
277                 this.dimension = _content;
278             } else if ("/multiline".equals(_url))  {
279                 this.multiline = true;
280             } else if ("/primitiveType".equals(_url))  {
281                 this.type = _content;
282 
283             } else if ("/rangeList/range".equals(_url))  {
284                 this.ranges.add(new Range());
285             } else if ("/rangeList/range/rangeType".equals(_url))  {
286                 this.ranges.peek().type = AbstractAttribute_mxJPO.RANGE_COMP.get(_content);
287                 if (this.ranges.peek().type == null)  {
288                     throw new Error("unknown range comparator " + _content);
289                 }
290             } else if ("/rangeList/range/rangeValue".equals(_url))  {
291                 this.ranges.peek().value1 = _content;
292             } else if ("/rangeList/range/includingValue".equals(_url))  {
293                 this.ranges.peek().include1 = true;
294             } else if ("/rangeList/range/rangeSecondValue".equals(_url))  {
295                 this.ranges.peek().value2 = _content;
296             } else if ("/rangeList/range/includingSecondValue".equals(_url))  {
297                 this.ranges.peek().include2 = true;
298 
299             } else if ("/rangeProgram/programRef".equals(_url))  {
300                 this.rangeProgramRef = _content;
301             } else if ("/rangeProgram/inputArguments".equals(_url))  {
302                 this.rangeProgramInputArguments = _content;
303 
304             } else  {
305                 super.parse(_url, _content);
306             }
307         }
308     }

326     /***
327      * Writes specific information about the cached attribute to the given
328      * writer instance.
329      *
330      * @param _paramCache   parameter cache
331      * @param _out          appendable instance to the TCL update file
332      * @throws IOException if the TCL update code could not be written to the
333      *                     writer instance
334      */
335     @Override()
336     protected void writeObject(final ParameterCache_mxJPO _paramCache,
337                                final Appendable _out)
338         throws IOException
339     {
340         _out.append(" //\n    ").append(this.isHidden() ? "hidden" : "!hidden");
341         if ("string".equalsIgnoreCase(this.type))  {
342             _out.append(" //\n    ").append(this.multiline ? "" : "!").append("multiline");
343         }
344         for (final String rule : this.rules)  {
345             _out.append(" //\n    add rule \"").append(StringUtil_mxJPO.convertTcl(rule)).append("\"");
346         }
347         if (this.defaultValue != null)  {
348             _out.append(" //\n    default \"").append(StringUtil_mxJPO.convertTcl(this.defaultValue)).append("\"");
349         } else  {
350             _out.append(" //\n    default \"\"");
351         }
352         // append triggers
353         this.writeTriggers(_out);
354         // append ranges
355         for (final Range range : this.rangesSorted)  {
356             range.write(_out);
357         }
358     }

403     /***
404      * The method overwrites the original method to append the MQL statements
405      * in the <code>_preMQLCode</code> to reset this attribute. This includes:
406      * <ul>
407      * <li>set to not hidden</li>
408      * <li>reset description and default value</li>
409      * <li>remove all ranges</li>
410      * <li>define the TCL procedure {@link #TCL_PROCEDURE} to set the
411      *     dimension</li>
412      * <li>if the dimension is not updated within the TCL update file (tested
413      *     with {@link #dimensionUpdated}) method
414      *     {@link #updateDimension(ParameterCache_mxJPO, String)} is called
415      *     </li>
416      * </ul>
417      *
418      * @param _paramCache       parameter cache
419      * @param _preMQLCode       MQL statements which must be called before the
420      *                          TCL code is executed
421      * @param _postMQLCode      MQL statements which must be called after the
422      *                          TCL code is executed
423      * @param _preTCLCode       TCL code which is defined before the source
424      *                          file is sourced
425      * @param _tclVariables     map of all TCL variables where the key is the
426      *                          name and the value is value of the TCL variable
427      *                          (the value is automatically converted to TCL
428      *                          syntax!)
429      * @param _sourceFile       souce file with the TCL code to update
430      * @throws Exception if the update from derived class failed
431      * @see Range#remove4Update(Appendable)
432      * @see #updateDimension(ParameterCache_mxJPO, String)
433      * @see #dimensionUpdated
434      * @see #TCL_PROCEDURE
435      */
436     @Override()
437     protected void update(final ParameterCache_mxJPO _paramCache,
438                           final CharSequence _preMQLCode,
439                           final CharSequence _postMQLCode,
440                           final CharSequence _preTCLCode,
441                           final Map<String,String> _tclVariables,
442                           final File _sourceFile)
443         throws Exception
444     {
445         // remove all properties
446         final StringBuilder preMQLCode = new StringBuilder()
447                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
448                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
449                 .append(" !hidden description \"\" default \"\"");
450         // remove rules
451         for (final String rule : this.rules)  {
452             preMQLCode.append(" remove rule \"").append(StringUtil_mxJPO.convertMql(rule)).append('\"');
453         }
454         // remove ranges
455         for (final Range range : this.rangesSorted)  {
456             range.remove4Update(preMQLCode);
457         }
458         // append already existing pre MQL code
459         preMQLCode.append(";\n")
460                   .append(_preMQLCode);
461         // add TCL code for the procedure
462         final StringBuilder preTCLCode = new StringBuilder()
463                 .append(AbstractAttribute_mxJPO.TCL_PROCEDURE)
464                 .append(_preTCLCode);
465 
466         super.update(_paramCache, preMQLCode, _postMQLCode, preTCLCode, _tclVariables, _sourceFile);
467 
468         // is the dimension already updated?
469         if (!this.dimensionUpdated)  {
470             this.updateDimension(_paramCache, "");
471         }
472     }

474     /***
475      * The method is called from the TCL update code to define the dimension
476      * for this attribute. If the correct use case is defined method
477      * {@link #updateDimension(ParameterCache_mxJPO, String)} is called.
478      *
479      * @param _paramCache   parameter cache
480      * @param _args         first index defines the use case (must be
481      *                      &quot;defineAttrDimension&quot; that the dimension
482      *                      of the attribute is updated); second index the name
483      *                      of the attribute to update; third index the new
484      *                      dimension to set
485      * @throws Exception if the update of the dimension failed or for all other
486      *                   use cases from super JPO call
487      * @see #updateDimension(ParameterCache_mxJPO, String)
488      * @see #dimensionUpdated
489      */
490     @Override()
491     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
492                                final String... _args)
493         throws Exception
494     {
495         // check if dimension is defined
496         if ((_args.length == 0) || !AbstractAttribute_mxJPO.JPO_CALLER_KEY.equals(_args[0]))  {
497             super.jpoCallExecute(_paramCache, _args);
498         } else  {
499             // check that attribute names are equal
500             if (!this.getName().equals(_args[1]))  {
501                 throw new Exception("dimension for wrong attribute '"
502                         + _args[1] + "' is set (attribute '" + this.getName()
503                         + "' is updated!)");
504             }
505 
506             this.updateDimension(_paramCache, _args[2]);
507             this.dimensionUpdated = true;
508         }
509     }
510 
511     /***
512      * Updates the dimension of this attribute. If a dimension is already
513      * defined for this attribute and it is not the same attribute, an
514      * exception is thrown (because information could be lost).
515      *
516      * @param _paramCache   parameter cache
517      * @param _dimension    new dimension to set
518      * @throws Exception if update failed or the dimension of an update should
519      *                   be changed (and some information could be lost)
520      * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
521      * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
522      */
523     protected void updateDimension(final ParameterCache_mxJPO _paramCache,
524                                    final String _dimension)
525         throws Exception
526     {
527         if ((this.dimension == null) || "".equals(this.dimension))  {
528             if (!"".equals(_dimension))  {
529                 _paramCache.logDebug("    - set dimension '" + _dimension + "'");
530                 MqlUtil_mxJPO.execMql(_paramCache,
531                         new StringBuilder().append("escape mod attribute \"")
532                         .append(StringUtil_mxJPO.convertMql(this.getName()))
533                         .append("\" add dimension \"")
534                         .append(StringUtil_mxJPO.convertMql(_dimension))
535                         .append('\"'));
536             }
537         } else if (!this.dimension.equals(_dimension))  {
538             throw new Exception("If dimension '" + this.dimension
539                     + "' is changed to new dimension '" + _dimension
540                     + "' some information could be lost!");
541         }
542     }

544     /***
545      * Class holding range values of this attribute.
546      */
547     private class Range
548         implements Comparable<AbstractAttribute_mxJPO.Range>
549     {
550         /***
551          * Holds the range type. Typically following range types are known:
552          * <ul>
553          * <li>between<li>
554          * <li>equal<li>
555          * <li>greaterthan<li>
556          * <li>greaterthanequal<li>
557          * <li>lessthan<li>
558          * <li>lessthanequal<li>
559          * <li>match<li>
560          * <li>notequal<li>
561          * <li>notmatch<li>
562          * <li>notsmatch<li>
563          * <li>smatch<li>
564          * </ul>
565          */
566         String type = null;
567 
568         /***
569          * Hold the first range value.
570          */
571         String value1 = null;
572 
573         /***
574          * Holds the second range value.
575          */
576         String value2 = null;
577 
578         /***
579          * Include first value (used for range type &quot;between&quot;).
580          */
581         Boolean include1 = false;
582 
583         /***
584          * Include second value (used for range type &quot;between&quot;).
585          */
586         Boolean include2 = false;
587 
588         /***
589          * Write this range value to the writer instance.
590          * If the range type is a program, the name of the program and the
591          * input arguments are defined directly on the attribute in
592          * {@link AbstractAttribute_mxJPO#rangeProgramRef} and
593          * {@link AbstractAttribute_mxJPO#rangeProgramInputArguments}.
594          *
595          * @param _out  writer instance
596          * @throws IOException if write to the writer instance is not possible
597          */
598         private void write(final Appendable _out)
599                 throws IOException
600         {
601             _out.append(" //\n    add range ").append(this.type);
602             // if the range is a program it is a 'global' attribute info
603             if ("program".equals(this.type))  {
604                 _out.append(" \"")
605                     .append(StringUtil_mxJPO.convertTcl(AbstractAttribute_mxJPO.this.rangeProgramRef))
606                     .append('\"');
607                 if (AbstractAttribute_mxJPO.this.rangeProgramInputArguments != null)  {
608                     _out.append(" input \"")
609                         .append(StringUtil_mxJPO.convertTcl(AbstractAttribute_mxJPO.this.rangeProgramInputArguments))
610                         .append('\"');
611                 }
612             } else  {
613                 _out.append(" \"").append(StringUtil_mxJPO.convertTcl(this.value1)).append("\"");
614                 if ("between".equals(this.type))  {
615                     if (this.include1)  {
616                         _out.append(" inclusive");
617                     } else  {
618                         _out.append(" exclusive");
619                     }
620                     _out.append(" \"").append(StringUtil_mxJPO.convertTcl(this.value2)).append("\"");
621                     if (this.include2)  {
622                         _out.append(" inclusive");
623                     } else  {
624                         _out.append(" exclusive");
625                     }
626                 }
627             }
628         }
629 
630         /***
631          * Appends the MQL code to remove this range depending on the type of
632          * range in {@link #type}. If it is a program the related name of
633          * program is defined as attribute variable in
634          * {@link AbstractAttribute_mxJPO#rangeProgramRef}.
635          *
636          * @param _out      appendable instance where the MQL code is appended
637          *                  to remove this range
638          * @throws IOException if the MQL code could not appended
639          */
640         private void remove4Update(final Appendable _out)
