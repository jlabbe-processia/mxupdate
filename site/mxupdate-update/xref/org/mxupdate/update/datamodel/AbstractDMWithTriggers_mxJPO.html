39  /***
40   * Abstract class for all data model administration objects with triggers.
41   *
42   * @author The MxUpdate Team
43   * @version $Id: AbstractDMWithTriggers_mxJPO.java 344 2009-05-24 21:52:37Z tim.moxter $
44   */
45  public abstract class AbstractDMWithTriggers_mxJPO
46          extends AbstractAdminObject_mxJPO
47  {
48      /***
49       * Defines the serialize version unique identifier.
50       */
51      private static final long serialVersionUID = -3691181822025363195L;
52  
53      /***
54       * Stack with all triggers for this type.
55       */
56      private final Stack<Trigger> triggersStack = new Stack<Trigger>();
57  
58      /***
59       * Map with all triggers for this type. The key is the name of the trigger.
60       *
61       * @see #prepare(ParameterCache_mxJPO)
62       * @see #triggersStack              stack with trigger from parsing method
63       * @see #writeTriggers(Appendable)
64       */
65      private final Map<String,Trigger> triggers = new TreeMap<String,Trigger>();
66  
67      /***
68       * Constructor used to initialize the type definition enumeration.
69       *
70       * @param _typeDef  defines the related type definition enumeration
71       * @param _mxName   MX name of the administration object
72       */
73      public AbstractDMWithTriggers_mxJPO(final TypeDef_mxJPO _typeDef,
74                                          final String _mxName)
75      {
76          super(_typeDef, _mxName);
77      }
78  
79      /***
80       * @param _url      URL to parse
81       * @param _content  content of the URL to parse
82       */
83      @Override
84      protected void parse(final String _url,
85                           final String _content)
86      {
87          if ("/triggerList".equals(_url))  {
88              // to be ignored ...
89          } else if ("/triggerList/trigger".equals(_url))  {
90              this.triggersStack.add(new Trigger());
91          } else if ("/triggerList/trigger/triggerName".equals(_url))  {
92              this.triggersStack.peek().name = _content;
93          } else if ("/triggerList/trigger/programRef".equals(_url))  {
94              this.triggersStack.peek().program = _content;
95          } else if ("/triggerList/trigger/inputArguments".equals(_url))  {
96              this.triggersStack.peek().arguments = _content;
97  
98          } else  {
99              super.parse(_url, _content);
100         }
101     }
102 
103     /***
104      * After the type XML file is parsed, the triggers must be sorted.
105      *
106      * @param _paramCache   parameter cache
107      * @throws MatrixException if the preparation from derived class failed
108      * @see #triggersStack
109      */
110     @Override
111     protected void prepare(final ParameterCache_mxJPO _paramCache)
112             throws MatrixException
113     {
114         // sort all triggers
115         for (final Trigger trigger : this.triggersStack)  {
116             this.triggers.put(trigger.name, trigger);
117         }
118 
119         super.prepare(_paramCache);
120     }
121 
122     /***
123      * Writes the trigger information to the writer instance.
124      *
125      * @param _out      writer instance
126      * @throws IOException if write failed
127      * @see #triggers
128      */
129     protected void writeTriggers(final Appendable _out)
130             throws IOException
131     {
132         // output of triggers, but sorted!
133         for (final Trigger trigger : this.triggers.values())  {
134             trigger.write(_out);
135         }
136     }

138     /***
139      * The method overwrites the original method to append the MQL statements
140      * in the <code>_preMQLCode</code> to remove all current assigned triggers.
141      * Then the update method of the super class is called.
142      *
143      * @param _paramCache       parameter cache
144      * @param _preMQLCode       MQL statements which must be called before the
145      *                          TCL code is executed
146      * @param _postMQLCode      MQL statements which must be called after the
147      *                          TCL code is executed
148      * @param _preTCLCode       TCL code which is defined before the source
149      *                          file is sourced
150      * @param _tclVariables     map of all TCL variables where the key is the
151      *                          name and the value is value of the TCL variable
152      *                          (the value is automatically converted to TCL
153      *                          syntax!)
154      * @param _sourceFile       souce file with the TCL code to update
155      * @throws Exception if the update from derived class failed
156      */
157     @Override
158     protected void update(final ParameterCache_mxJPO _paramCache,
159                           final CharSequence _preMQLCode,
160                           final CharSequence _postMQLCode,
161                           final CharSequence _preTCLCode,
162                           final Map<String,String> _tclVariables,
163                           final File _sourceFile)
164             throws Exception
165     {
166         // remove all properties
167         final StringBuilder preMQLCode = new StringBuilder()
168                 .append("mod ").append(this.getTypeDef().getMxAdminName())
169                 .append(" \"").append(this.getName()).append("\" ")
170                 .append(this.getTypeDef().getMxAdminSuffix());
171         for (final Trigger trigger : this.triggers.values())  {
172             trigger.appendResetMQLStatement(preMQLCode);
173         }
174         preMQLCode.append(";\n");
175 
176         // append already existing pre MQL code
177         preMQLCode.append(_preMQLCode);
178 
179         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
180     }

182     /***
183      * Class used to store informations about triggers.
184      */
185     public static class Trigger
186             implements Serializable
187     {
188         /***
189          * Defines the serialize version unique identifier.
190          */
191         private static final long serialVersionUID = -241775149089053095L;
192 
193         /***
194          * Used to parse the event type of the trigger from the {@link #name}.
195          *
196          * @see #getEventType()
197          */
198         private static final Pattern PATTERN_EVENTTYPE = Pattern.compile("^.*(?=((action)|(check)|(override))$)");
199 
200         /***
201          * Used to parse the event kind (&quot;Action&quot;, &quot;Check&quot;
202          * or &quot;Override&quot;).
203          *
204          * @see #getKind()
205          */
206         private static final Pattern PATTERN_KIND = Pattern.compile("((action)|(check)|(override))$");
207 
208         /***
209          * Name of the trigger (like ChangeVaultAction etc...).
210          */
211         String name;
212 
213         /***
214          * Name of the referenced program.
215          */
216         String program;
217 
218         /***
219          * Arguments of the called program.
220          */
221         String arguments = "";
222 
223         /***
224          * Evaluates from the {@link #name} the related event type of the
225          * trigger.
226          *
227          * @return event type of the trigger  (always in lower case)
228          * @see #name
229          * @see #PATTERN_EVENTTYPE
230          */
231         protected String getEventType()
232         {
233             final Matcher matchEventType
234                     = AbstractDMWithTriggers_mxJPO.Trigger.PATTERN_EVENTTYPE.matcher(this.name.toLowerCase());
235             matchEventType.find();
236             return matchEventType.group();
237         }
238 
239         /***
240          * Evaluates from the {@link #name} the related king of the trigger.
241          *
242          * @return kind of the trigger (always in lower case)
243          * @see #name
244          * @see #PATTERN_KIND
245          */
246         protected String getKind()
247         {
248             final Matcher matchKind
249                     = AbstractDMWithTriggers_mxJPO.Trigger.PATTERN_KIND.matcher(this.name.toLowerCase());
250             matchKind.find();
251             return matchKind.group();
252         }
253 
254         /***
255          * Writes this trigger TCL source code. The trigger event type and kind
256          * is parsed by using regular expression {@link #PATTERN_EVENTTYPE} and
257          * {@link #PATTERN_KIND} to extract the event type and kind.
258          *
259          * @param _out  writer instance
260          * @throws IOException if write failed
261          */
262         protected void write(final Appendable _out)
263                 throws IOException
264         {
265             _out.append(" //\n    add trigger ").append(this.getEventType()).append(' ').append(this.getKind())
266                 .append(" \"").append(StringUtil_mxJPO.convertTcl(this.program)).append("\"")
267                 .append(" input \"").append(StringUtil_mxJPO.convertTcl(this.arguments)).append("\"");
268         }
269 
270         /***
271          * Appends the MQL statement to remove all triggers.
272          *
273          * @param _cmd          string builder with current MQL statement
274          */
275         protected void appendResetMQLStatement(final StringBuilder _cmd)
276         {
277             _cmd.append(" remove trigger ").append(this.getEventType()).append(' ').append(this.getKind());
278         }
279     }
