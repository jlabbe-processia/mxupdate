84      /***
85       * Called TCL procedure within the TCL update to parse the new policy
86       * definition. The TCL procedure calls method
87       * {@link #jpoCallExecute(ParameterCache_mxJPO, String...)} with the new
88       * policy definition. All quot's are replaced by <code>@0@0@</code> and all
89       * apostroph's are replaced by <code>@1@1@</code>.
90       *
91       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
92       * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
93       */
94      private static final String TCL_PROCEDURE
95              = "proc updatePolicy {_sPolicy _lsArgs}  {\n"
96                  + "regsub -all {'} $_lsArgs {@0@0@} sArg\n"
97                  + "regsub -all {//\"} $sArg {@1@1@} sArg\n"
98                  + "regsub -all {//////[} $sArg {[} sArg\n"
99                  + "regsub -all {//////]} $sArg {]} sArg\n"
100                 + "mql exec prog org.mxupdate.update.util.JPOCaller policy $_sPolicy \"${sArg}\"\n"
101             + "}\n";

285     /***
286      * Calls the state prepare method. All policies properties starting with
287      * <code>state_</code> are checked if they are defining a symbolic name of
288      * a state. If yes the related symbolic name of the state is updated and
289      * removed from the property list.
290      *
291      * @param _paramCache   parameter cache
292      * @throws MatrixException if the prepare within super class failed
293      */
294     @Override()
295     protected void prepare(final ParameterCache_mxJPO _paramCache)
296         throws MatrixException
297     {
298         for (final State state : this.states)  {
299             state.prepare();
300         }
301         super.prepare(_paramCache);
302         for (final AdminProperty_mxJPO property : new HashSet<AdminProperty_mxJPO>(this.getPropertiesMap().values()))  {
303             if ((property.getName() != null) && property.getName().startsWith("state_"))  {
304                 for (final State state : this.states)  {
305                     if (state.name.equals(property.getValue()))  {
306                         state.symbolicNames.add(property.getName());
307                         this.getPropertiesMap().remove(property.getName());
308                     }
309                 }
310             }
311         }
312     }

314     /***
315      * Writes the update script for this policy.
316      *
317      * @param _paramCache   parameter cache
318      * @param _out          writer instance
319      * @throws IOException if the TCL update code could not be written
320      */
321     @Override()
322     protected void write(final ParameterCache_mxJPO _paramCache,
323                          final Appendable _out)
324         throws IOException
325     {
326         this.writeHeader(_paramCache, _out);
327         _out.append("updatePolicy \"${NAME}\" {");
328         final String suffix = this.getTypeDef().getMxAdminSuffix();
329         if (!"".equals(suffix))  {
330             _out.append(" ").append(suffix);
331         }
332         _out.append("\n  description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
333         // types
334         if (this.allTypes)  {
335             _out.append("\n  type all");
336         } else  {
337             _out.append("\n  type {")
338                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.types, null))
339                 .append("}");
340         }
341         // formats
342         if (this.allFormats)  {
343             _out.append("\n  format all");
344         } else  {
345             _out.append("\n  format {")
346                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.formats, null))
347                 .append("}");
348         }
349         _out.append("\n  defaultformat \"").append(StringUtil_mxJPO.convertTcl(this.defaultFormat)).append('\"')
350             .append("\n  sequence \"").append(StringUtil_mxJPO.convertTcl(this.sequence)).append('\"')
351             .append("\n  store \"").append(StringUtil_mxJPO.convertTcl(this.store)).append('\"')
352             .append("\n  hidden \"").append(Boolean.toString(this.isHidden())).append("\"");
353         // all states
354         for (final State state : this.states)  {
355             state.writeObject(_out);
356         }
357         _out.append("\n}");
358         this.writeProperties(_paramCache, _out);
359     }

411     /***
412      * The method is called within the update of an administration object. The
413      * method is called directly within the update.
414      * <ul>
415      * <li>All <code>@0@0@</code> are replaced by quot's and all
416      *     <code>@1@1@</code> are replaced by apostroph's.</li>
417      * <li>The new policy definition is parsed.</li>
418      * <li>A delta MQL script generated to update the policy to the new target
419      *     definition.</li>
420      * <li>All symbolic names for states are defined (as property on the
421      *     policy).</li>
422      * <li>The delta MQL script is executed.</li>
423      * </ul>
424      *
425      * @param _paramCache   parameter cache
426      * @param _args         arguments from the TCL procedure
427      * @throws Exception if a state is not defined anymore or the policy could
428      *                   not be updated
429      * @see #TCL_PROCEDURE
430      */
431     @Override()
432     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
433                                final String... _args)
434         throws Exception
435     {
436         final String code = _args[2].replaceAll("@0@0@", "'")
437                                     .replaceAll("@1@1@", "//\"");
438 
439         final PolicyDefParser_mxJPO parser = new PolicyDefParser_mxJPO(new StringReader(code));
440         final Policy_mxJPO policy = parser.policy(_paramCache, this.getTypeDef(), _args[1]);
441 
442         final StringBuilder cmd = new StringBuilder()
443                 .append("mod policy \"").append(this.getName()).append("\" ");
444 
445         // basic information
446         this.calcValueDelta(cmd, "description", policy.getDescription(), this.getDescription());
447         this.calcListDelta(cmd, "type", policy.types, this.types);
448         this.calcListDelta(cmd, "format", policy.formats, this.formats);
449         this.calcValueDelta(cmd, "defaultformat", policy.defaultFormat, this.defaultFormat);
450         this.calcValueDelta(cmd, "sequence", policy.sequence, this.sequence);
451         // hidden flag, because hidden flag must be set with special syntax
452         if (this.isHidden() != policy.isHidden())  {
453             if (!policy.isHidden())  {
454                 cmd.append('!');
455             }
456             cmd.append("hidden ");
457         }
458         // because the store of a policy could not be removed....
459         if ((policy.store != null) && !"".equals(policy.store))  {
460             this.calcValueDelta(cmd, "store", policy.store, this.store);
461         }
462 
463         // states....
464         // (first add new states because of references in branches)
465         final Iterator<State> curStateIter = this.states.iterator();
466         final Iterator<State> newStateIter = policy.states.iterator();
467         final Map<State,State> stateDeltaMap = new HashMap<State,State>();
468         while (curStateIter.hasNext() && newStateIter.hasNext())  {
469             final State curState = curStateIter.next();
470             State newState = newStateIter.next();
471             while (!curState.name.equals(newState.name) && newStateIter.hasNext())  {
472                 cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name))
473                    .append("\" before \"").append(StringUtil_mxJPO.convertMql(curState.name)).append("\" ");
474                 _paramCache.logDebug("    - insert new state '" + newState.name + "' before '" + curState.name + "'");
475                 stateDeltaMap.put(newState, null);
476                 newState = newStateIter.next();
477             }
478             if (curState.name.equals(newState.name))  {
479                 stateDeltaMap.put(newState, curState);
480             }
481         }
482         while (newStateIter.hasNext())  {
483             final State newState = newStateIter.next();
484             cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name)).append("\" ");
485             _paramCache.logDebug("    - add new state '" + newState.name + "'");
486             stateDeltaMap.put(newState, null);
487         }
488         // check for already existing state, but not defined anymore!
489         if (curStateIter.hasNext())  {
490 throw new Exception("some states are not defined anymore!");
491         }
492 
493         // now update state information itself
494         cmd.append(';')
495            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
496         for (final Map.Entry<State, State> entry : stateDeltaMap.entrySet())  {
497             cmd.append("state \"").append(StringUtil_mxJPO.convertMql(entry.getKey().name)).append("\" ");
498             entry.getKey().calcDelta(cmd, entry.getValue());
499         }
500 
501         // set symbolic names for all policy states
502         cmd.append(';')
503            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
504         for (final State state : policy.states)  {
505             for (final String symbolicName : state.symbolicNames)  {
506                 cmd.append(" add property \"").append(StringUtil_mxJPO.convertMql(symbolicName))
507                    .append("\" value \"").append(StringUtil_mxJPO.convertMql(state.name)).append('\"');
508             }
509         }
510 
511         final boolean isMqlEscapeOn = MqlUtil_mxJPO.isEscapeOn(_paramCache);
512         try  {
513             MqlUtil_mxJPO.setEscapeOn(_paramCache);
514             MqlUtil_mxJPO.execMql(_paramCache, cmd);
515         } finally  {
516             if (!isMqlEscapeOn)  {
517                 MqlUtil_mxJPO.setEscapeOff(_paramCache);
518             }
519         }
520     }

522     /***
523      * Calculates the delta between the new and the old value. If a delta
524      * exists, the kind with the new delta is added to the string builder.
525      *
526      * @param _out      appendable instance where the delta must be append
527      * @param _kind     kind of the delta
528      * @param _newVal   new target value
529      * @param _curVal   current value in the database
530      * @throws IOException if the delta could not appended
531      */
532     protected void calcValueDelta(final Appendable _out,
533                                   final String _kind,
534                                   final String _newVal,
535                                   final String _curVal)
536             throws IOException
537     {
538         final String curVal = (_curVal == null) ? "" : _curVal;
539         final String newVal = (_newVal == null) ? "" : _newVal;
540 
541         if (!curVal.equals(newVal))  {

543         }

547      *
548      * @param _out      appendable instance where the delta must be append
549      * @param _kind     kind of the delta
550      * @param _new      new target values
551      * @param _current  current values in MX
552      * @throws IOException if the delta could not appended
553      */
554     protected void calcListDelta(final Appendable _out,
555                                  final String _kind,
556                                  final Set<String> _new,
557                                  final Set<String> _current)
558             throws IOException
559     {
560         boolean equal = (_current.size() == _new.size());
561         if (equal)  {
562             for (final String format : _current)  {
563                 if (!_new.contains(format))  {
564                     equal = false;
565                     break;
566                 }
567             }
568         }
569         if (!equal)  {
570             for (final String format : _current)  {
571                 if (!_new.contains(format))  {
572                     _out.append("remove ").append(_kind).append(" \"").append(format).append("\" ");
573                 }
574             }
575             for (final String format : _new)  {
576                 if (!_current.contains(format))  {
577                     _out.append("add ").append(_kind).append(" \"").append(format).append("\" ");
578                 }
579             }
580         }

910                 .append(StringUtil_mxJPO.joinMql(',', false, this.publicAccess, "none"))

955                 }

966                     if (!this.triggers.containsKey(trigger.name))  {
967                         _out.append("remove trigger ")

969                             .append(' ')
970                             .append(trigger.getKind())
971                             .append(' ');
972                     }

979             }
980             // signatures
981             final Set<String> newSigs = new HashSet<String>();
982             for (final Signature signature : this.signatures)  {
983                 newSigs.add(signature.name);
984             }
985             final Map<String,Signature> oldSigs = new HashMap<String,Signature>();
986             if (_oldState != null)  {
987                 for (final Signature signature : _oldState.signatures)  {
988                     if (newSigs.contains(signature.name))  {
989                         oldSigs.put(signature.name, signature);

995                 }
996             }
997             for (final Signature signature : this.signatures)  {

999                 if ((_oldState != null) && !oldSigs.containsKey(signature.name))  {
1000                     _out.append("add ");
1001                     oldSig = null;
1002                 } else  {
1003                     oldSig = oldSigs.get(signature.name);
1004                 }
1005                 _out.append("signature \"").append(StringUtil_mxJPO.convertMql(signature.name)).append("\" ");
1006                 signature.calcDelta(_out, oldSig);
1007             }
1008         }

1114         /***
1115          * Calculates the delta between the old signature and this signature.
1116          * If for the old signature a branch is defined and for the new
1117          * signature no branch, an error is thrown.
1118          *
1119          * @param _out          appendable instance where the delta must be
1120          *                      append
1121          * @param _oldSignature old signature to compare
1122          * @throws IOException if the delta could not appended
1123          */
1124         protected void calcDelta(final Appendable _out,
1125                                  final Signature _oldSignature)
1126             throws IOException
1127         {
1128             if ("".equals(this.branch))  {
1129                 if ((_oldSignature != null) && (_oldSignature.branch != null) && !"".equals(_oldSignature.branch))  {
1130 throw new Error("branch '" + _oldSignature.branch + "' exists for signature "
1131             + this.name + ", but is not defined anymore");
1132                 }
1133             } else  {
1134                 _out.append("branch \"").append(StringUtil_mxJPO.convertMql(this.branch)).append("\" ");
1135             }
1136             _out.append("filter \"").append(StringUtil_mxJPO.convertMql(this.filter)).append("\" ");
1137             // update approve users
1138             for (final String approver : this.approverUsers)  {
1139                 if ((_oldSignature == null) || !_oldSignature.approverUsers.contains(approver))  {
1140                     _out.append("add approve \"").append(StringUtil_mxJPO.convertMql(approver)).append("\" ");
1141                 }
1142             }
1143             if (_oldSignature != null)  {
1144                 for (final String approver : _oldSignature.approverUsers)  {
1145                     if (!this.approverUsers.contains(approver))  {
1146                         _out.append("remove approve \"").append(StringUtil_mxJPO.convertMql(approver)).append("\" ");
1147                     }
1148                 }
1149             }
1150             // update ignore user
1151             for (final String ignore : this.ignoreUsers)  {
1152                 if ((_oldSignature == null) || !_oldSignature.ignoreUsers.contains(ignore))  {
1153                     _out.append("add ignore \"").append(StringUtil_mxJPO.convertMql(ignore)).append("\" ");
1154                 }
1155             }
