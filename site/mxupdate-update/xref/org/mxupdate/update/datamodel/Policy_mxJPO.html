80      /***
81       * Called TCL procedure within the TCL update to parse the new policy
82       * definition. The TCL procedure calls method
83       * {@link #jpoCallExecute(ParameterCache_mxJPO, String...)} with the new
84       * policy definition. All quot's are replaced by <code>@0@0@</code> and all
85       * apostroph's are replaced by <code>@1@1@</code>.
86       *
87       * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)
88       * @see #jpoCallExecute(ParameterCache_mxJPO, String...)
89       */
90      private static final String TCL_PROCEDURE
91              = "proc updatePolicy {_sPolicy _lsArgs}  {\n"
92                  + "regsub -all {'} $_lsArgs {@0@0@} sArg\n"
93                  + "regsub -all {//\"} $sArg {@1@1@} sArg\n"
94                  + "regsub -all {//////[} $sArg {[} sArg\n"
95                  + "regsub -all {//////]} $sArg {]} sArg\n"
96                  + "mql exec prog org.mxupdate.update.util.JPOCaller policy $_sPolicy \"${sArg}\"\n"
97              + "}\n";

154      * Parses all format specific URLs.

156      * @param _url      URL to parse

158      * @see #IGNORED_URLS
159      */
160     @Override()
161     protected void parse(final String _url,
162                          final String _content)
163     {
164         if (!Policy_mxJPO.IGNORED_URLS.contains(_url))  {
165             if ("/defaultFormat/formatRef".equals(_url))  {
166                 this.defaultFormat = _content;
167             } else if ("/formatRefList/formatRef".equals(_url))  {
168                 this.formats.add(_content);
169             } else if ("/allowAllFormats".equals(_url))  {
170                 this.allFormats = true;
171 
172             } else if ("/sequence".equals(_url))  {

174 
175             } else if ("/storeRef".equals(_url))  {
176                 this.store = _content;
177 
178             } else if ("/typeRefList/typeRef".equals(_url))  {
179                 this.types.add(_content);
180             } else if ("/allowAllTypes".equals(_url))  {
181                 this.allTypes = true;
182 
183             } else if ("/stateDefList/stateDef".equals(_url))  {
184                 this.states.add(new State());
185             } else if ("/stateDefList/stateDef/name".equals(_url))  {
186                 this.states.peek().name = _content;

263     /***
264      * Calls the state prepare method. All policies properties starting with
265      * <code>state_</code> are checked if they are defining a symbolic name of
266      * a state. If yes the related symbolic name of the state is updated and
267      * removed from the property list.
268      *
269      * @param _paramCache   parameter cache
270      * @throws MatrixException if the prepare within super class failed
271      */
272     @Override()
273     protected void prepare(final ParameterCache_mxJPO _paramCache)
274         throws MatrixException
275     {
276         for (final State state : this.states)  {
277             state.prepare();
278         }
279         super.prepare(_paramCache);
280         for (final AdminProperty_mxJPO property : new HashSet<AdminProperty_mxJPO>(this.getPropertiesMap().values()))  {
281             if ((property.getName() != null) && property.getName().startsWith("state_"))  {
282                 for (final State state : this.states)  {
283                     if (state.name.equals(property.getValue()))  {
284                         state.symbolicNames.add(property.getName());
285                         this.getPropertiesMap().remove(property.getName());
286                     }
287                 }
288             }
289         }
290     }

292     /***
293      * Writes the update script for this policy.
294      *
295      * @param _paramCache   parameter cache
296      * @param _out          writer instance
297      * @throws IOException if the TCL update code could not be written
298      */
299     @Override()
300     protected void write(final ParameterCache_mxJPO _paramCache,
301                          final Appendable _out)
302         throws IOException
303     {
304         this.writeHeader(_paramCache, _out);
305         _out.append("updatePolicy \"${NAME}\" {");
306         final String suffix = this.getTypeDef().getMxAdminSuffix();
307         if (!"".equals(suffix))  {
308             _out.append(" ").append(suffix);
309         }
310         _out.append("\n  description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
311         // types
312         if (this.allTypes)  {
313             _out.append("\n  type all");
314         } else  {
315             _out.append("\n  type {")
316                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.types, null))
317                 .append("}");
318         }
319         // formats
320         if (this.allFormats)  {
321             _out.append("\n  format all");
322         } else  {
323             _out.append("\n  format {")
324                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.formats, null))
325                 .append("}");
326         }
327         _out.append("\n  defaultformat \"").append(StringUtil_mxJPO.convertTcl(this.defaultFormat)).append('\"')
328             .append("\n  sequence \"").append(StringUtil_mxJPO.convertTcl(this.sequence)).append('\"')
329             .append("\n  store \"").append(StringUtil_mxJPO.convertTcl(this.store)).append('\"')
330             .append("\n  hidden \"").append(Boolean.toString(this.isHidden())).append("\"");
331         // all states
332         for (final State state : this.states)  {
333             state.writeObject(_out);
334         }
335         _out.append("\n}");
336         this.writeProperties(_paramCache, _out);
337     }

389     /***
390      * The method is called within the update of an administration object. The
391      * method is called directly within the update.
392      * <ul>
393      * <li>All <code>@0@0@</code> are replaced by quot's and all
394      *     <code>@1@1@</code> are replaced by apostroph's.</li>
395      * <li>The new policy definition is parsed.</li>
396      * <li>A delta MQL script generated to update the policy to the new target
397      *     definition.</li>
398      * <li>All symbolic names for states are defined (as property on the
399      *     policy).</li>
400      * <li>The delta MQL script is executed.</li>
401      * </ul>
402      *
403      * @param _paramCache   parameter cache
404      * @param _args         arguments from the TCL procedure
405      * @throws Exception if a state is not defined anymore or the policy could
406      *                   not be updated
407      * @see #TCL_PROCEDURE
408      */
409     @Override()
410     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
411                                final String... _args)
412         throws Exception
413     {
414         final String code = _args[2].replaceAll("@0@0@", "'")
415                                     .replaceAll("@1@1@", "//\"");
416 
417         final PolicyDefParser_mxJPO parser = new PolicyDefParser_mxJPO(new StringReader(code));
418         final Policy_mxJPO policy = parser.policy(_paramCache, this.getTypeDef(), _args[1]);
419 
420         final StringBuilder cmd = new StringBuilder()
421                 .append("mod policy \"").append(this.getName()).append("\" ");
422 
423         // basic information
424         this.calcValueDelta(cmd, "description", policy.getDescription(), this.getDescription());
425         this.calcListDelta(cmd, "type", policy.types, this.types);
426         this.calcListDelta(cmd, "format", policy.formats, this.formats);
427         this.calcValueDelta(cmd, "defaultformat", policy.defaultFormat, this.defaultFormat);
428         this.calcValueDelta(cmd, "sequence", policy.sequence, this.sequence);
429         // hidden flag, because hidden flag must be set with special syntax
430         if (this.isHidden() != policy.isHidden())  {
431             if (!policy.isHidden())  {
432                 cmd.append('!');
433             }
434             cmd.append("hidden ");
435         }
436         // because the store of a policy could not be removed....
437         if ((policy.store != null) && !"".equals(policy.store))  {
438             this.calcValueDelta(cmd, "store", policy.store, this.store);
439         }
440 
441         // states....
442         // (first add new states because of references in branches)
443         final Iterator<State> curStateIter = this.states.iterator();
444         final Iterator<State> newStateIter = policy.states.iterator();
445         final Map<State,State> stateDeltaMap = new HashMap<State,State>();
446         while (curStateIter.hasNext() && newStateIter.hasNext())  {
447             final State curState = curStateIter.next();
448             State newState = newStateIter.next();
449             while (!curState.name.equals(newState.name) && newStateIter.hasNext())  {
450                 cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name))
451                    .append("\" before \"").append(StringUtil_mxJPO.convertMql(curState.name)).append("\" ");
452                 _paramCache.logDebug("    - insert new state '" + newState.name + "' before '" + curState.name + "'");
453                 stateDeltaMap.put(newState, null);
454                 newState = newStateIter.next();
455             }
456             if (curState.name.equals(newState.name))  {
457                 stateDeltaMap.put(newState, curState);
458             }
459         }
460         while (newStateIter.hasNext())  {
461             final State newState = newStateIter.next();
462             cmd.append("add state \"").append(StringUtil_mxJPO.convertMql(newState.name)).append("\" ");
463             _paramCache.logDebug("    - add new state '" + newState.name + "'");
464             stateDeltaMap.put(newState, null);
465         }
466         // check for already existing state, but not defined anymore!
467         if (curStateIter.hasNext())  {
468 throw new Exception("some states are not defined anymore!");
469         }
470 
471         // now update state information itself
472         cmd.append(';')
473            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
474         for (final Map.Entry<State, State> entry : stateDeltaMap.entrySet())  {
475             cmd.append("state \"").append(StringUtil_mxJPO.convertMql(entry.getKey().name)).append("\" ");
476             entry.getKey().calcDelta(cmd, entry.getValue());
477         }
478 
479         // set symbolic names for all policy states
480         cmd.append(';')
481            .append("mod policy \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ");
482         for (final State state : policy.states)  {
483             for (final String symbolicName : state.symbolicNames)  {
484                 cmd.append(" add property \"").append(StringUtil_mxJPO.convertMql(symbolicName))
485                    .append("\" value \"").append(StringUtil_mxJPO.convertMql(state.name)).append('\"');
486             }
487         }
488 
489         final boolean isMqlEscapeOn = MqlUtil_mxJPO.isEscapeOn(_paramCache);
490         try  {
491             MqlUtil_mxJPO.setEscapeOn(_paramCache);
492             MqlUtil_mxJPO.execMql(_paramCache, cmd);
493         } finally  {
494             if (!isMqlEscapeOn)  {
495                 MqlUtil_mxJPO.setEscapeOff(_paramCache);
496             }
497         }
498     }

500     /***
501      * Calculates the delta between the new and the old value. If a delta
502      * exists, the kind with the new delta is added to the string builder.
503      *
504      * @param _out      appendable instance where the delta must be append
505      * @param _kind     kind of the delta
506      * @param _newVal   new target value
507      * @param _curVal   current value in the database
508      * @throws IOException if the delta could not appended
509      */
510     protected void calcValueDelta(final Appendable _out,
511                                   final String _kind,
512                                   final String _newVal,
513                                   final String _curVal)
514             throws IOException
515     {
516         final String curVal = (_curVal == null) ? "" : _curVal;
517         final String newVal = (_newVal == null) ? "" : _newVal;
518 
519         if (!curVal.equals(newVal))  {
520             _out.append(_kind).append(" \"").append(StringUtil_mxJPO.convertMql(newVal)).append("\" ");
521         }
522     }
523 
524     /***
525      *
526      * @param _out      appendable instance where the delta must be append
527      * @param _kind     kind of the delta
528      * @param _new      new target values
529      * @param _current  current values in MX
530      * @throws IOException if the delta could not appended
531      */
532     protected void calcListDelta(final Appendable _out,
533                                  final String _kind,
534                                  final Set<String> _new,
535                                  final Set<String> _current)
536             throws IOException
537     {
538         boolean equal = (_current.size() == _new.size());
539         if (equal)  {
540             for (final String format : _current)  {
541                 if (!_new.contains(format))  {
542                     equal = false;
543                     break;
544                 }
545             }
546         }
547         if (!equal)  {
548             for (final String format : _current)  {
549                 if (!_new.contains(format))  {
550                     _out.append("remove ").append(_kind).append(" \"").append(format).append("\" ");
551                 }
552             }
553             for (final String format : _new)  {
554                 if (!_current.contains(format))  {
555                     _out.append("add ").append(_kind).append(" \"").append(format).append("\" ");
556                 }
557             }
558         }
559     }

688         /***

758         /***
759          *
760          * @param _out      appendable instance where the delta must be append
761          * @param _oldState old state to compare to (or <code>null</code> if
762          *                  this state is new)
763          * @throws IOException if the delta could not appended
764          */
765         protected void calcDelta(final Appendable _out,
766                                  final State _oldState)
767                 throws IOException
768         {
769             // basics
770             _out.append("promote ").append(String.valueOf(this.autoPromotion)).append(' ')
771                 .append("revision ").append(String.valueOf(this.revisionable)).append(' ')
772                 .append("checkouthistory ").append(String.valueOf(this.checkoutHistory)).append(' ')
773                 .append("version ").append(String.valueOf(this.versionable)).append(' ')
774                 .append("action \"").append(StringUtil_mxJPO.convertMql(this.actionProgram)).append("\" ")
775                 .append("input \"").append(StringUtil_mxJPO.convertMql(this.actionInput)).append("\" ")
776                 .append("check \"").append(StringUtil_mxJPO.convertMql(this.checkProgram)).append("\" ")
777                 .append("input \"").append(StringUtil_mxJPO.convertMql(this.checkInput)).append("\" ");
778             // route message
779             _out.append("route message \"").append(StringUtil_mxJPO.convertMql(this.routeMessage)).append("\" ");
780             for (final String routeUser : this.routeUsers)  {
781                 if ((_oldState == null) || !_oldState.routeUsers.contains(routeUser))  {
782                     _out.append("add route \"").append(StringUtil_mxJPO.convertMql(routeUser)).append("\" ");
783                 }
784             }
785             // owner access
786             _out.append("owner ")
787                 .append(StringUtil_mxJPO.joinMql(',', false, this.ownerAccess, "none"))
788                 .append(' ')
789             // public access
790                 .append("public ")
791                 .append(StringUtil_mxJPO.joinMql(',', false, this.publicAccess, "none"))
792                 .append(' ');
793             // user access
794             final Set<String> newUsers = new HashSet<String>();
795             for (final UserAccess userAccess : this.userAccess)  {
796                 newUsers.add(userAccess.userRef);
797             }
798             final Set<String> oldUser = new HashSet<String>();
799             if (_oldState != null)  {
800                 for (final UserAccess userAccess : _oldState.userAccess)  {
801                     if (newUsers.contains(userAccess.userRef))  {
802                         oldUser.add(userAccess.userRef);
803                     } else  {
804                         _out.append("remove user \"")
805                             .append(StringUtil_mxJPO.convertMql(userAccess.userRef))
806                             .append("\" all ");
807                     }
808                 }
809             }
810             for (final UserAccess userAccess : this.userAccess)  {
811                 if ((_oldState != null) && !oldUser.contains(userAccess.userRef))  {
812                     _out.append("add ");
813                 }
814                 _out.append("user \"").append(userAccess.userRef).append("\" ")
815                     .append(StringUtil_mxJPO.joinMql(',', false, userAccess.access, "none"))
816                     .append(" ")
817                     .append("filter \"")
818                     .append(StringUtil_mxJPO.convertMql(userAccess.expressionFilter))
819                     .append("\" ");
820             }
821             // triggers
822             if (_oldState != null)  {
823                 for (final Trigger trigger : _oldState.triggers.values())  {
824                     if (!this.triggers.containsKey(trigger.name))  {
825                         _out.append("remove trigger ")
826                             .append(trigger.getEventType())
827                             .append(' ')
828                             .append(trigger.getKind())
829                             .append(' ');
830                     }
831                 }
832             }
833             for (final Trigger trigger : this.triggers.values())  {
834                 _out.append("add trigger ").append(trigger.getEventType()).append(' ').append(trigger.getKind())
835                     .append(" \"").append(StringUtil_mxJPO.convertMql(trigger.program)).append("\"")
836                     .append(" input \"").append(StringUtil_mxJPO.convertMql(trigger.arguments)).append("\" ");
837             }
838             // signatures
839             final Set<String> newSigs = new HashSet<String>();
840             for (final Signature signature : this.signatures)  {
841                 newSigs.add(signature.name);
842             }
843             final Map<String,Signature> oldSigs = new HashMap<String,Signature>();
844             if (_oldState != null)  {
845                 for (final Signature signature : _oldState.signatures)  {
846                     if (newSigs.contains(signature.name))  {
847                         oldSigs.put(signature.name, signature);
848                     } else  {
849                         _out.append("remove signature \"")
850                             .append(StringUtil_mxJPO.convertMql(signature.name))
851                             .append("\" ");
852                     }
853                 }
854             }
855             for (final Signature signature : this.signatures)  {
856                 final Signature oldSig;
857                 if ((_oldState != null) && !oldSigs.containsKey(signature.name))  {
858                     _out.append("add ");
859                     oldSig = null;
860                 } else  {
861                     oldSig = oldSigs.get(signature.name);
862                 }
863                 _out.append("signature \"").append(StringUtil_mxJPO.convertMql(signature.name)).append("\" ");
864                 signature.calcDelta(_out, oldSig);
865             }
866         }

951         {
952             _out.append("\n    signature \"").append(StringUtil_mxJPO.convertTcl(this.name)).append("\" {")
953                 .append("\n      branch \"").append(StringUtil_mxJPO.convertTcl(this.branch)).append("\"")
954                 // append approver users
955                 .append("\n      approve {")
956                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.approverUsers, null))
957                 .append("}")
958                 // append ignore users
959                 .append("\n      ignore {")
960                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.ignoreUsers, null))
961                 .append("}")
962                 // append reject users
963                 .append("\n      reject {")
964                 .append(StringUtil_mxJPO.joinTcl(' ', true, this.rejectUsers, null))
965                 .append("}")
966                 // append filters
967                 .append("\n      filter \"")
968                 .append(StringUtil_mxJPO.convertTcl(this.filter)).append("\"")
969                 .append("\n    }");
970         }
971 
972         /***
973          * Calculates the delta between the old signature and this signature.
974          * If for the old signature a branch is defined and for the new
975          * signature no branch, an error is thrown.
976          *
977          * @param _out          appendable instance where the delta must be
978          *                      append
979          * @param _oldSignature old signature to compare
980          * @throws IOException if the delta could not appended
981          */
982         protected void calcDelta(final Appendable _out,
983                                  final Signature _oldSignature)
984             throws IOException
985         {
986             if ("".equals(this.branch))  {
987                 if ((_oldSignature != null) && (_oldSignature.branch != null) && !"".equals(_oldSignature.branch))  {
988 throw new Error("branch '" + _oldSignature.branch + "' exists for signature "
989             + this.name + ", but is not defined anymore");
990                 }
991             } else  {
992                 _out.append("branch \"").append(StringUtil_mxJPO.convertMql(this.branch)).append("\" ");
993             }
994             _out.append("filter \"").append(StringUtil_mxJPO.convertMql(this.filter)).append("\" ");
995             // update approve users
996             for (final String approver : this.approverUsers)  {
997                 if ((_oldSignature == null) || !_oldSignature.approverUsers.contains(approver))  {
998                     _out.append("add approve \"").append(StringUtil_mxJPO.convertMql(approver)).append("\" ");
999                 }
1000             }
1001             if (_oldSignature != null)  {
1002                 for (final String approver : _oldSignature.approverUsers)  {
1003                     if (!this.approverUsers.contains(approver))  {
1004                         _out.append("remove approve \"").append(StringUtil_mxJPO.convertMql(approver)).append("\" ");
1005                     }
1006                 }
1007             }
1008             // update ignore user
1009             for (final String ignore : this.ignoreUsers)  {
1010                 if ((_oldSignature == null) || !_oldSignature.ignoreUsers.contains(ignore))  {
1011                     _out.append("add ignore \"").append(StringUtil_mxJPO.convertMql(ignore)).append("\" ");
1012                 }
1013             }
1014             if (_oldSignature != null)  {
1015                 for (final String ignore : _oldSignature.ignoreUsers)  {
1016                     if (!this.ignoreUsers.contains(ignore))  {
1017                         _out.append("remove ignore \"").append(StringUtil_mxJPO.convertMql(ignore)).append("\" ");
1018                     }
1019                 }
1020             }
1021             // update reject users
1022             for (final String reject : this.rejectUsers)  {
1023                 if ((_oldSignature == null) || !_oldSignature.rejectUsers.contains(reject))  {
1024                     _out.append("add reject \"").append(StringUtil_mxJPO.convertMql(reject)).append("\" ");
1025                 }
1026             }
1027             if (_oldSignature != null)  {
1028                 for (final String reject : _oldSignature.rejectUsers)  {
1029                     if (!this.rejectUsers.contains(reject))  {
1030                         _out.append("remove reject \"").append(StringUtil_mxJPO.convertMql(reject)).append("\" ");
1031                     }
1032                 }
1033             }
1034         }
