170     /***
171      * Evaluates the from and to type information. Because MX does not handle
172      * the to or from side types correctly within XML exports, the from and to
173      * side types must be evaluated via a &quot;<code>print relationship ...
174      * select fromtype / totype</code>&quot; MQL statement. If this is not done
175      * in this way, there is no other possibility to evaluate the information
176      * {@link Side#typeAll} or {@link Side#relationAll}.
177      *
178      * @param _paramCache   parameter cache
179      * @throws MatrixException if the from / to side types could not be
180      *                         evaluated or the prepare from the derived class
181      *                         failed
182      */
183     @Override()
184     protected void prepare(final ParameterCache_mxJPO _paramCache)
185         throws MatrixException
186     {
187         // evaluate all from types / relationships
188         this.from.prepare(_paramCache);
189 
190         // evaluate all to types / relationships
191         this.to.prepare(_paramCache);
192 
193         super.prepare(_paramCache);
194     }

196     /***
197      * Writes all relationship specific information in the TCL update file.
198      * The relationship specific informations are:
199      * <ul>
200      * <li>hidden flag</li>
201      * <li>{@link #preventDuplicates prevent duplicates} flag</li>
202      * <li>{@link #from} and {@link #to} side informations</li>
203      * </ul>
204      *
205      * @param _paramCache   parameter cache
206      * @param _out          appendable instance to the TCL update file
207      * @throws IOException if the TCL code to the TCL update file could not be
208      *                     written
209      * @see Side#write(Appendable)
210      */
211     @Override()
212     protected void writeObject(final ParameterCache_mxJPO _paramCache,
213                                final Appendable _out)
214         throws IOException
215     {
216         _out.append(" //\n    ").append(this.isHidden() ? "" : "!").append("hidden")
217             .append(" //\n    ").append(this.preventDuplicates ? "" : "!").append("preventduplicates");
218         // rules
219         for (final String rule : this.rules)  {
220             _out.append(" //\n    add rule \"").append(StringUtil_mxJPO.convertTcl(rule)).append('\"');
221         }
222         // triggers
223         this.writeTriggers(_out);
224 
225         this.from.write(_out);
226         this.to.write(_out);
227     }

229     /***
230      * The method overwrites the original method to append the MQL statements
231      * in the <code>_preMQLCode</code> to reset this relationship. This
232      * includes:
233      * <ul>
234      * <li>reset description</li>
235      * <li>remove hidden and prevent duplicate flag</li>
236      * <li>remove all rules</li>
237      * <li>reset from and to information</li>
238      * <li>remove all from and to types</li>
239      * </ul>
240      *
241      * @param _paramCache       parameter cache
242      * @param _preMQLCode       MQL statements which must be called before the
243      *                          TCL code is executed
244      * @param _postMQLCode      MQL statements which must be called after the
245      *                          TCL code is executed
246      * @param _preTCLCode       TCL code which is defined before the source
247      *                          file is sourced
248      * @param _tclVariables     map of all TCL variables where the key is the
249      *                          name and the value is value of the TCL variable
250      *                          (the value is automatically converted to TCL
251      *                          syntax!)
252      * @param _sourceFile       souce file with the TCL code to update
253      * @throws Exception if the update from derived class failed
254      */
255     @Override()
256     protected void update(final ParameterCache_mxJPO _paramCache,
257                           final CharSequence _preMQLCode,
258                           final CharSequence _postMQLCode,
259                           final CharSequence _preTCLCode,
260                           final Map<String,String> _tclVariables,
261                           final File _sourceFile)
262         throws Exception
263     {
264         final StringBuilder preMQLCode = new StringBuilder()
265                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
266                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
267                 // remove hidden, description, prevent duplicate
268                 .append(" !hidden description \"\" !preventduplicate")
269                 // reset from information
270                 .append(" from !propagatemodify !propagateconnection ")
271                         .append("meaning \"\" cardinality one revision none clone none ")
272                 .append(" from remove type all")
273                 // reset to information
274                 .append(" to !propagatemodify !propagateconnection ")
275                         .append("meaning \"\" cardinality one revision none clone none ")
276                 .append(" to remove type all");
277         // are connections between relationships allowed?
278         // => yes, than all relationships must be removed from from / to side
279         if (_paramCache.getValueBoolean(Relationship_mxJPO.PARAM_SUPPORT_REL_CONS))  {
280             preMQLCode.append(" from remove relationship all")
281                       .append(" to remove relationship all");
282         }
283         // remove all rules
284         for (final String rule : this.rules)  {
285             preMQLCode.append(" remove rule \"").append(StringUtil_mxJPO.convertMql(rule)).append('\"');
286         }
287         // remove all from types
288         for (final String type : this.from.types)  {
289             preMQLCode.append(" from remove type \"").append(StringUtil_mxJPO.convertMql(type)).append('\"');
290         }
291         // remove all from relationships
292         for (final String rel : this.from.relations)  {
293             preMQLCode.append(" from remove relationship \"").append(StringUtil_mxJPO.convertMql(rel)).append('\"');
294         }
295         // remove all to types
296         for (final String type : this.to.types)  {
297             preMQLCode.append(" to remove type \"").append(StringUtil_mxJPO.convertMql(type)).append('\"');
298         }
299         // remove all to relationships
300         for (final String rel : this.to.relations)  {
301             preMQLCode.append(" to remove relationship \"").append(StringUtil_mxJPO.convertMql(rel)).append('\"');
302         }
303 
304         // append already existing pre MQL code
305         preMQLCode.append(";\n")
306                   .append(_preMQLCode);
307 
308         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
309     }

311     /***
312      * Stores the information for one side of a relationship.
313      */
314     private final class Side
315     {
316         /***
317          * Side string of the relationship.
318          */
319         private final String side;
320 
321         /***
322          * Side cardinality action.
323          */
324         private String cardinality = "";
325 
326         /***
327          * Side clone action.
328          */
329         private String cloneAction = "";
330 
331         /***
332          * Side meaning.
333          */
334         private String meaning = "";
335 
336         /***
337          * Side propagate connection flag.
338          */
339         private boolean propagateConnection = false;
340 
341         /***
342          * Side propagate modify flag.
343          */
344         private boolean propagateModify = false;
345 
346         /***
347          * Side revision action.
348          */
349         private String revisionAction = "";
350 
351         /***
352          * Side type list.
353          *
354          * @see #prepare(ParameterCache_mxJPO)
355          * @see #fromTypeAll
356          */
357         private final Set<String> types = new TreeSet<String>();
358 
359         /***
360          * Are all types on the side allowed?
361          *
362          * @see #prepare(ParameterCache_mxJPO)
363          * @see #types
364          */
365         private boolean typeAll = false;
366 
367         /***
368          * From side relationship list.
369          *
370          * @see #prepare(ParameterCache_mxJPO)
371          * @see #relationAll
372          */
373         private final Set<String> relations = new TreeSet<String>();
374 
375         /***
376          * Are all relationships on the from side allowed?
377          *
378          * @see #prepare(ParameterCache_mxJPO)
379          * @see #relations
380          */
381         private boolean relationAll = false;
382 
383         /***
384          * Defines the {@link #side} information.
385          *
386          * @param _side     side string
387          */
388         private Side(final String _side)
389         {
390             this.side = _side;
391         }
392 
393         /***
394          * Reads the information about the type / relationship for this side.
395          *
396          * @param _paramCache       parameter cache with the MX context
397          * @throws MatrixException if information could not be read
398          */
399         private void prepare(final ParameterCache_mxJPO _paramCache)
400             throws MatrixException
401         {
402             // evaluate all to types
403             final String[] toTypesArr = MqlUtil_mxJPO.execMql(
404                             _paramCache,
405                             new StringBuilder("escape print rel \"")
406                                     .append(StringUtil_mxJPO.convertMql(Relationship_mxJPO.this.getName()))
407                                     .append("\" select ").append(this.side).append("type dump '\n'"))
408                     .split("\n");
409             for (final String toType : toTypesArr)  {
410                 if ("all".equals(toType))  {
411                     this.typeAll = true;
412                     this.types.clear();
413                     break;
414                 } else if (!"".equals(toType)) {
415                     this.types.add(toType);
416                 }
417             }
418 
419             // are connections between relationships allowed?
420             if (_paramCache.getValueBoolean(Relationship_mxJPO.PARAM_SUPPORT_REL_CONS))  {
421                 // evaluate all from relationships
422                 final String[] fromRelsArr = MqlUtil_mxJPO.execMql(
423                                 _paramCache,
424                                 new StringBuilder("escape print rel \"")
425                                         .append(StringUtil_mxJPO.convertMql(Relationship_mxJPO.this.getName()))
426                                         .append("\" select ").append(this.side).append("rel dump '\n'"))
427                         .split("\n");
428                 for (final String fromRel : fromRelsArr)  {
429                     if ("all".equals(fromRel))  {
430                         this.relationAll = true;
431                         this.relations.clear();
432                     } else if (!"".equals(fromRel)) {
433                         this.relations.add(fromRel);
434                     }
435                 }
436             }
437         }
438 
439         /***
440          * Writes all relationship specific information in the TCL update file.
441          * The relationship specific information are:
442          * <ul>
443          * <li>{@link #propagateModify propagate modify} flag</li>
444          * <li>{@link #propagateConnection propagate connection} flag</li>
445          * <li>{@link #meaning}</li>
446          * <li>{@link #revisionAction revision behavior / action}</li>
447          * <li>{@link #cloneAction clone behavior / action}</li>
448          * </ul>
449          *
450          * @param _out          appendable instance to the TCL update file
451          * @throws IOException if the TCL code to the TCL update file could not
452          *                     be written
453          * @see Relationship_mxJPO#writeObject(ParameterCache_mxJPO, Appendable)
454          */
455         protected void write(final Appendable _out)
456             throws IOException
457         {
458             _out.append(" //\n    ").append(this.side)
459                 .append(" //\n        ").append(this.propagateModify ? "" : "!").append("propagatemodify")
460                 .append(" //\n        ").append(this.propagateConnection ? "" : "!").append("propagateconnection")
461                 .append(" //\n        meaning \"").append(StringUtil_mxJPO.convertTcl(this.meaning)).append('\"')
462                 .append(" //\n        cardinality \"")
463                         .append(StringUtil_mxJPO.convertTcl(this.cardinality)).append('\"')
464                 .append(" //\n        revision \"")
465                         .append(StringUtil_mxJPO.convertTcl(this.revisionAction)).append('\"')
466                 .append(" //\n        clone \"").append(StringUtil_mxJPO.convertTcl(this.cloneAction)).append('\"');
467             // from types
468             if (this.typeAll)  {
469                 _out.append(" //\n        add type \"all\"");
470             } else  {
471                 for (final String type : this.types)  {
472                     _out.append(" //\n        add type \"").append(StringUtil_mxJPO.convertTcl(type)).append('\"');
473                 }
474             }
475             // from relationships
476             if (this.relationAll)  {
477                 _out.append(" //\n        add relationship \"all\"");
478             } else  {
479                 for (final String relation : this.relations)  {
480                     _out.append(" //\n        add relationship \"")
481                         .append(StringUtil_mxJPO.convertTcl(relation)).append('\"');
482                 }
483             }
484         }
