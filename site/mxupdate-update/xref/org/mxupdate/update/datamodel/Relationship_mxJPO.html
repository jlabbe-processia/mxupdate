281     /***
282      * Evaluates the from and to type information. Because MX does not handle
283      * the to or from side types correctly within XML exports, the from and to
284      * side types must be evaluated via a &quot;<code>print relationship ...
285      * select fromtype / totype</code>&quot; MQL statement. If this is not done
286      * in this way, there is no other possibility to evaluate the information
287      * {@link #fromTypeAll} or {@link #toTypeAll}.
288      *
289      * @param _paramCache   parameter cache
290      * @throws MatrixException if the from / to side types could not be
291      *                         evaluated or the prepare from the derived class
292      *                         failed
293      * @todo support for from / to relationships
294      */
295     @Override()
296     protected void prepare(final ParameterCache_mxJPO _paramCache)
297         throws MatrixException
298     {
299         // evaluate all from types
300         final String[] fromTypesArr = MqlUtil_mxJPO.execMql(_paramCache,
301                                               new StringBuilder("escape print rel \"")
302                                                    .append(this.getName())
303                                                    .append("\" select fromtype dump '\n'"))
304                                      .split("\n");
305         for (final String fromType : fromTypesArr)  {
306             if ("all".equals(fromType))  {
307                 this.fromTypeAll = true;
308                 this.fromTypes.clear();
309             } else if (!"".equals(fromType)) {
310                 this.fromTypes.add(fromType);
311             }
312         }
313         // evaluate all to types
314         final String[] toTypesArr = MqlUtil_mxJPO.execMql(_paramCache,
315                                             new StringBuilder("escape print rel \"")
316                                                  .append(this.getName())
317                                                  .append("\" select totype dump '\n'"))
318                                     .split("\n");
319         for (final String toType : toTypesArr)  {
320             if ("all".equals(toType))  {
321                 this.toTypeAll = true;
322                 this.toTypes.clear();
323             } else if (!"".equals(toType)) {
324                 this.toTypes.add(toType);
325             }
326         }
327 
328         // are connections between relationships allowed?
329         if (_paramCache.getValueBoolean(Relationship_mxJPO.PARAM_SUPPORT_REL_CONS))  {
330             // evaluate all from relationships
331             final String[] fromRelsArr = MqlUtil_mxJPO.execMql(_paramCache,
332                                                  new StringBuilder("escape print rel \"")
333                                                          .append(this.getName())
334                                                          .append("\" select fromrel dump '\n'"))
335                                          .split("\n");
336             for (final String fromRel : fromRelsArr)  {
337                 if ("all".equals(fromRel))  {
338                     this.fromRelationAll = true;
339                     this.fromRelations.clear();
340                 } else if (!"".equals(fromRel)) {
341                     this.fromRelations.add(fromRel);
342                 }
343             }
344             // evaluate all to relationships
345             final String[] toRelsArr = MqlUtil_mxJPO.execMql(_paramCache,
346                                                  new StringBuilder("escape print rel \"")
347                                                          .append(this.getName())
348                                                          .append("\" select torel dump '\n'"))
349                                          .split("\n");
350             for (final String toRel : toRelsArr)  {
351                 if ("all".equals(toRel))  {
352                     this.toRelationAll = true;
353                     this.toRelations.clear();
354                 } else if (!"".equals(toRel)) {
355                     this.toRelations.add(toRel);
356                 }
357             }
358         }
359 
360         super.prepare(_paramCache);
361     }

363     /***
364      * Writes all relationship specific information in the TCL update file.
365      * The relationship specific information are:
366      * <ul>
367      * <li>hidden flag</li>
368      * <li>prevent dupplicates flag (see {@link #preventDuplicates})</li>
369      * <li>from and to side informations</li>
370      * </ul>
371      *
372      * @param _paramCache   parameter cache
373      * @param _out          appendable instance to the TCL update file
374      * @throws IOException if the TCL code to the TCL update file could not be
375      *                     written
376      */
377     @Override()
378     protected void writeObject(final ParameterCache_mxJPO _paramCache,
379                                final Appendable _out)
380         throws IOException
381     {
382         _out.append(" //\n    ").append(this.isHidden() ? "" : "!").append("hidden")
383             .append(" //\n    ").append(this.preventDuplicates ? "" : "!").append("preventduplicates");
384         // rules
385         for (final String rule : this.rules)  {
386             _out.append(" //\n    add rule \"").append(StringUtil_mxJPO.convertTcl(rule)).append('\"');
387         }
388         // triggers
389         this.writeTriggers(_out);
390         _out.append(" //\n    from")
391             .append(" //\n        ").append(this.fromPropagateModify ? "" : "!").append("propagatemodify")
392             .append(" //\n        ").append(this.fromPropagateConnection ? "" : "!").append("propagateconnection")
393             .append(" //\n        meaning \"").append(StringUtil_mxJPO.convertTcl(this.fromMeaning)).append('\"')
394             .append(" //\n        cardinality \"")
395                     .append(StringUtil_mxJPO.convertTcl(this.fromCardinality)).append('\"')
396             .append(" //\n        revision \"")
397                     .append(StringUtil_mxJPO.convertTcl(this.fromRevisionAction)).append('\"')
398             .append(" //\n        clone \"").append(StringUtil_mxJPO.convertTcl(this.fromCloneAction)).append('\"');
399         // from types
400         if (this.fromTypeAll)  {
401             _out.append(" //\n        add type \"all\"");
402         } else  {
403             for (final String type : this.fromTypes)  {
404                 _out.append(" //\n        add type \"").append(StringUtil_mxJPO.convertTcl(type)).append('\"');
405             }
406         }
407         // from relationships
408         if (this.fromRelationAll)  {
409             _out.append(" //\n        add relationship \"all\"");
410         } else  {
411             for (final String relation : this.fromRelations)  {
412                 _out.append(" //\n        add relationship \"")
413                     .append(StringUtil_mxJPO.convertTcl(relation)).append('\"');
414             }
415         }
416         _out.append(" //\n    to")
417             .append(" //\n        ").append(this.toPropagateModify ? "" : "!").append("propagatemodify")
418             .append(" //\n        ").append(this.toPropagateConnection ? "" : "!").append("propagateconnection")
419             .append(" //\n        meaning \"").append(StringUtil_mxJPO.convertTcl(this.toMeaning)).append('\"')
420             .append(" //\n        cardinality \"").append(StringUtil_mxJPO.convertTcl(this.toCardinality)).append('\"')
421             .append(" //\n        revision \"").append(StringUtil_mxJPO.convertTcl(this.toRevisionAction)).append('\"')
422             .append(" //\n        clone \"").append(StringUtil_mxJPO.convertTcl(this.toCloneAction)).append('\"');
423         // to types
424         if (this.toTypeAll)  {
425             _out.append(" //\n        add type \"all\"");
426         } else  {
427             for (final String type : this.toTypes)  {
428                 _out.append(" //\n        add type \"").append(StringUtil_mxJPO.convertTcl(type)).append('\"');
429             }
430         }
431         // to relationships
432         if (this.toRelationAll)  {
433             _out.append(" //\n        add relationship \"all\"");
434         } else  {
435             for (final String relation : this.toRelations)  {
436                 _out.append(" //\n        add relationship \"")
437                     .append(StringUtil_mxJPO.convertTcl(relation)).append('\"');
438             }
439         }
440     }

442     /***
443      * The method overwrites the original method to append the MQL statements
444      * in the <code>_preMQLCode</code> to reset this relationship. This
445      * includes:
446      * <ul>
447      * <li>reset description</li>
448      * <li>remove hidden and prevent duplicate flag</li>
449      * <li>remove all rules</li>
450      * <li>reset from and to information</li>
451      * <li>remove all from and to types</li>
452      * </ul>
453      *
454      * @param _paramCache       parameter cache
455      * @param _preMQLCode       MQL statements which must be called before the
456      *                          TCL code is executed
457      * @param _postMQLCode      MQL statements which must be called after the
458      *                          TCL code is executed
459      * @param _preTCLCode       TCL code which is defined before the source
460      *                          file is sourced
461      * @param _tclVariables     map of all TCL variables where the key is the
462      *                          name and the value is value of the TCL variable
463      *                          (the value is automatically converted to TCL
464      *                          syntax!)
465      * @param _sourceFile       souce file with the TCL code to update
466      * @throws Exception if the update from derived class failed
467      */
468     @Override()
469     protected void update(final ParameterCache_mxJPO _paramCache,
470                           final CharSequence _preMQLCode,
471                           final CharSequence _postMQLCode,
472                           final CharSequence _preTCLCode,
473                           final Map<String,String> _tclVariables,
474                           final File _sourceFile)
475         throws Exception
476     {
477         final StringBuilder preMQLCode = new StringBuilder()
478                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
479                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
480                 // remove hidden, description, prevent duplicate
481                 .append(" !hidden description \"\" !preventduplicate")
482                 // reset from information
483                 .append(" from !propagatemodify !propagateconnection ")
484                         .append("meaning \"\" cardinality one revision none clone none ")
485                 .append(" from remove type all")
486                 // reset to information
487                 .append(" to !propagatemodify !propagateconnection ")
488                         .append("meaning \"\" cardinality one revision none clone none ")
489                 .append(" to remove type all");
490         // are connections between relationships allowed?
491         // => yes, than all relationships must be removed from from / to side
492         if (_paramCache.getValueBoolean(Relationship_mxJPO.PARAM_SUPPORT_REL_CONS))  {
493             preMQLCode.append(" from remove relationship all")
494                       .append(" to remove relationship all");
495         }
496         // remove all rules
497         for (final String rule : this.rules)  {
498             preMQLCode.append(" remove rule \"").append(StringUtil_mxJPO.convertMql(rule)).append('\"');
499         }
500         // remove all from types
501         for (final String type : this.fromTypes)  {
502             preMQLCode.append(" from remove type \"").append(StringUtil_mxJPO.convertMql(type)).append('\"');
503         }
504         // remove all from relationships
505         for (final String rel : this.fromRelations)  {
506             preMQLCode.append(" from remove relationship \"").append(StringUtil_mxJPO.convertMql(rel)).append('\"');
507         }
508         // remove all to types
509         for (final String type : this.toTypes)  {
510             preMQLCode.append(" to remove type \"").append(StringUtil_mxJPO.convertMql(type)).append('\"');
511         }
512         // remove all to relationships
513         for (final String rel : this.toRelations)  {
514             preMQLCode.append(" to remove relationship \"").append(StringUtil_mxJPO.convertMql(rel)).append('\"');
515         }
516 
517         // append already existing pre MQL code
518         preMQLCode.append(";\n")
519                   .append(_preMQLCode);
520 
521         super.update(_paramCache, preMQLCode, _postMQLCode, _preTCLCode, _tclVariables, _sourceFile);
522     }
