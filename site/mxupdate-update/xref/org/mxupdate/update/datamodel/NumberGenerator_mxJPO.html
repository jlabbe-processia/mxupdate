21  package org.mxupdate.update.datamodel;
22  
23  import java.io.IOException;
24  
25  import org.mxupdate.mapping.TypeDef_mxJPO;
26  import org.mxupdate.update.BusObject_mxJPO;
27  import org.mxupdate.update.util.ParameterCache_mxJPO;
28  import org.mxupdate.update.util.StringUtil_mxJPO;
29  
30  /***
31   * @author The MxUpdate Team
32   * @version $Id: NumberGenerator_mxJPO.java 611 2009-11-02 00:20:36Z tim.moxter $
33   */
34  public class NumberGenerator_mxJPO
35      extends BusObject_mxJPO
36  {
37      /***
38       * String of the key within the parameter cache to define the name of the
39       * attribute which holds the next number of the number generator.
40       *
41       * @see #extractSymbolicNameFromCode(ParameterCache_mxJPO, StringBuilder)
42       */
43      private static final String PARAM_ATTR_NUMBERGENERATOR_NEXTNUMBER = "AttrNumberGenNextNumber";
44  
45      /***
46       * Constructor used to initialize the type definition enumeration.
47       *
48       * @param _typeDef  defines the related type definition enumeration
49       * @param _mxName   MX name of the number generator object
50       */
51      public NumberGenerator_mxJPO(final TypeDef_mxJPO _typeDef,
52                                   final String _mxName)
53      {
54          super(_typeDef, _mxName);
55      }
56  
57      /***
58       * Writes the information to update the business objects. Because for the
59       * next number attribute must be enclosing within an if clause, the
60       * original method is overwritten.
61       *
62       * @param _paramCache   parameter cache
63       * @param _out          writer instance
64       * @throws IOException if the TCL update code for the number generator
65       *                     could not be written
66       */
67      @Override()
68      protected void write(final ParameterCache_mxJPO _paramCache,
69                           final Appendable _out)
70          throws IOException
71      {
72          this.writeHeader(_paramCache, _out);
73          _out.append("mql mod bus \"${OBJECTID}\"")
74              .append(" //\n    description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
75          String nextNumber = null;
76          final String attrNextNumber = _paramCache.getValueString(NumberGenerator_mxJPO.PARAM_ATTR_NUMBERGENERATOR_NEXTNUMBER);
77          for (final AttributeValue attr : this.getAttrValuesSorted())  {
78              if (attrNextNumber.equals(attr.name))  {
79                  nextNumber = attr.value;
80              } else  {
81                  _out.append(" //\n    \"").append(StringUtil_mxJPO.convertTcl(attr.name))
82                      .append("\" \"").append(StringUtil_mxJPO.convertTcl(attr.value)).append("\"");
83              }
84          }
85          _out.append("\n")
86              .append("\n# update of the next number attribute only if not already set")
87              .append("\nset sTmp [mql print bus \"${OBJECTID}\" select attribute//[")
88                      .append(attrNextNumber).append("//] dump]")
89              .append("\nif {[string length \"${sTmp}\"]==0}  {")
90              .append("\n  mql mod bus \"${OBJECTID}\" \"")
91                      .append(attrNextNumber)
92                      .append("\" \"").append(nextNumber).append("\"")
93              .append("\n}");
94      }
95  }
