60      private static final String TCL_PROCEDURE
61              = "proc testParents {args}  {\n"
62                  + "set iIdx 0\n"
63                  + "set lsCmd [list mql exec prog org.mxupdate.update.util.JPOCaller parents]\n"
64                  + "while {$iIdx < [llength $args]}  {\n"
65                  +   "lappend lsCmd [lindex $args $iIdx]\n"
66                  +   "incr iIdx\n"

68                  + "eval $lsCmd\n"

148     /***
149      * Parses the interface specific XML export URL.
150      *
151      * @param _url      URL to parse
152      * @param _content  content of the URL to parse
153      * @see #abstractFlag
154      * @see #allTypes
155      * @see #types
156      */
157     @Override
158     protected void parse(final String _url,
159                          final String _content)
160     {
161         if (!Interface_mxJPO.IGNORED_URLS.contains(_url))  {
162             if ("/abstract".equals(_url))  {
163                 this.abstractFlag = true;
164             } else if ("/allowAllRelationships".equals(_url))  {
165                 this.allRelationships = true;
166             } else if ("/allowAllTypes".equals(_url))  {
167                 this.allTypes = true;
168             } else if ("/derivedFromInterface/interfaceTypeRefList/interfaceTypeRef".equals(_url))  {
169                 this.derived.add(_content);
170             } else if ("/relationshipDefRefList/relationshipDefRef".equals(_url))  {
171                 this.relationships.add(_content);
172             } else if ("/typeRefList/typeRef".equals(_url))  {
173                 this.types.add(_content);
174             } else  {
175                 super.parse(_url, _content);
176             }
177         }
178     }

180     /***
181      * Writes the interface specific information into the TCL update file. The
182      * interface specific information are:
183      * <ul>
184      * <li>abstract information {@link #abstractFlag}</li>
185      * <li>relationship information (all relationships defined with
186      *     {@link #allRelationships} or some relationships defined with
187      *     {@link #relationships})</li>
188      * <li>type information (all types defined with {@link #allTypes} or some
189      *     types defined with {@link #types})</li>
190      * </ul>
191      *
192      * @param _paramCache   parameter cache
193      * @param _out          appendable instance to the TCL update file
194      * @throws IOException if the interface specific information could not be
195      *                     written
196      * @see #abstractFlag
197      * @see #allRelationships
198      * @see #relationships
199      * @see #allTypes
200      * @see #types
201      */
202     @Override
203     protected void writeObject(final ParameterCache_mxJPO _paramCache,
204                                final Appendable _out)
205             throws IOException
206     {
207         // write abstract information
208         _out.append(" //\n    abstract ").append(Boolean.toString(this.abstractFlag));
209 
210         // relationship information
211         if (this.allRelationships)  {
212             _out.append(" //\n    add relationship all");
213         } else  {
214             for (final String relationship : this.relationships)  {
215                 _out.append(" //\n    add relationship \"")
216                     .append(StringUtil_mxJPO.convertTcl(relationship)).append('\"');
217             }
218         }
219 
220         // type information
221         if (this.allTypes)  {
222             _out.append(" //\n    add type all");
223         } else  {
224             for (final String type : this.types)  {
225                 _out.append(" //\n    add type \"").append(StringUtil_mxJPO.convertTcl(type)).append('\"');
226             }
227         }
228     }

244     {
245         _out.append("\n\ntestParents -").append(this.getTypeDef().getMxAdminName())
246             .append(" \"${NAME}\" -parents [list //\n");
247         for (final String parent : this.derived)  {

255     /***
256      * The method overwrites the original method to append the MQL statements
257      * in the <code>_preMQLCode</code> to reset this interface and to append
258      * the TCL procedure {@link #TCL_PROCEDURE}. This information is reset:
259      * <ul>
260      * <li>set not hidden</li>
261      * <li>reset description</li>
262      * <li>remove all types in {@link #allTypes} or {@link #types}</li>
263      * <li>remove all relationships in {@link #allRelationships} or
264      *     {@link #relationships}</li>
265      * </ul>
266      *
267      * @param _paramCache       parameter cache
268      * @param _preMQLCode       MQL statements which must be called before the
269      *                          TCL code is executed
270      * @param _postMQLCode      MQL statements which must be called after the
271      *                          TCL code is executed
272      * @param _preTCLCode       TCL code which is defined before the source
273      *                          file is sourced
274      * @param _tclVariables     map of all TCL variables where the key is the
275      *                          name and the value is value of the TCL variable
276      *                          (the value is automatically converted to TCL
277      *                          syntax!)
278      * @param _sourceFile       souce file with the TCL code to update
279      * @throws Exception if the update from derived class failed
280      * @see #TCL_PROCEDURE
281      */
282     @Override
283     protected void update(final ParameterCache_mxJPO _paramCache,
284                           final CharSequence _preMQLCode,
285                           final CharSequence _postMQLCode,
286                           final CharSequence _preTCLCode,
287                           final Map<String,String> _tclVariables,
288                           final File _sourceFile)
289             throws Exception
290     {
291         final StringBuilder preMQLCode = new StringBuilder()
292                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
293                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append('\"')
294                 .append(" !hidden description \"\"");
295 
296         // relationship information
297         if (this.allRelationships)  {
298             preMQLCode.append(" remove relationship all");
299         } else  {
300             for (final String relationship : this.relationships)  {
301                 preMQLCode.append(" remove relationship \"")
302                           .append(StringUtil_mxJPO.convertMql(relationship)).append('\"');
303             }
304         }
305 
306         // type information
307         if (this.allTypes)  {
308             preMQLCode.append(" remove type all");
309         } else  {
310             for (final String type : this.types)  {
311                 preMQLCode.append(" remove type \"").append(StringUtil_mxJPO.convertMql(type)).append('\"');
312             }
313         }
314 
315         // append already existing pre MQL code
316         preMQLCode.append(";\n")
317                   .append(_preMQLCode);
318 
319         // add TCL code for the procedure
320         final StringBuilder preTclCode = new StringBuilder()
321                 .append(Interface_mxJPO.TCL_PROCEDURE)
322                 .append(_preTCLCode);
323 
324         super.update(_paramCache, preMQLCode, _postMQLCode, preTclCode, _tclVariables, _sourceFile);
325     }

327     /***
328      * The method is called within the update of an interface object. The
329      * method is called directly within the update and checks which parent
330      * interfaces are missed in the new definition and adds missing parent
331      * interfaces to the interface object. If an interface is not defined
332      * anymore but assigned in MX, an exception is thrown.
333      * If the first argument is not &quot;parents&quot; method
334      * {@link AbstractDMWithAttributes_mxJPO#jpoCallExecute(ParameterCache_mxJPO, String...)}
335      * is called.
336      *
337      * @param _paramCache   parameter cache
338      * @param _args         arguments from the TCL procedure
339      * @throws Exception if an unknown parameter is defined, the given name of
340      *                   the administration object is not the same or an
341      *                   interface is assigned to the interface object
342      *                   within MX but not defined anymore
343      */
344     @Override
345     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
346                                final String... _args)
347             throws Exception
348     {
349         if (!"parents".equals(_args[0]))  {
350             super.jpoCallExecute(_paramCache, _args);
351         } else  {
352             // evaluate parameters
353             final String nameParam = new StringBuilder()
354                     .append('-').append(this.getTypeDef().getMxAdminName()).toString();
355             int idx = 1;
356             String name = null;
357             String parentsStr = null;
358             while (idx < _args.length)  {
359                 final String arg = _args[idx];
360                 if (nameParam.equals(arg))  {
361                     name = _args[++idx];
362                 } else if ("-parents".equals(arg))  {
363                     parentsStr = _args[++idx];
364                 } else  {
365                     throw new Exception("unknown parameter \"" + arg + "\"");
366                 }
367                 idx++;
368             }
369 
370             // check for equal administration name
371             if (!this.getName().equals(name))  {
372                 throw new Exception(this.getTypeDef().getLogging()
373                         + " '" + this.getName() + "' was called to"
374                         + " update via update script, but "
375                         + this.getTypeDef().getLogging() + " '" + name + "' was"
376                         + " called in the procedure...");
377             }
378 
379             // get all parent interfaces
380             final Pattern pattern = Pattern.compile("(//{[^//{//}]*//} )|([^ //{//}]* )");
381             final Matcher matcher = pattern.matcher(parentsStr + " ");
382             final Set<String> newParents = new TreeSet<String>();
383             while (matcher.find())  {
384                 final String parentName = matcher.group().trim().replaceAll("(^//{)|(//}$)", "");
385                 if (!"".equals(parentName))  {
386                     newParents.add(parentName);
387                 }
388             }
389 
390             // check if all current parents are within new parents
391             for (final String curParent : this.derived)  {
392                 if (!newParents.contains(curParent))  {
393                     throw new Exception("Current parent " + this.getTypeDef().getLogging()
394                             + " '" + curParent + "' must be removed from " + this.getTypeDef().getLogging()
395                             + " '" + this.getName() + "'. This is not allowed!");
396                 }
397             }
398 
399             // and append all not current derived parents
400             if (!newParents.isEmpty())  {
401                 final StringBuilder cmd = new StringBuilder()
402                         .append("mod ").append(this.getTypeDef().getMxAdminName())
403                         .append(" '").append(this.getName()).append("\' derived \"");
404                 boolean first = true;
405                 for (final String newParent : newParents)  {
406                     if (!this.derived.contains(newParent))  {
407                         _paramCache.logDebug("    - add to parent '" + newParent + "'");
408                     }
409                     if (first)  {
410                         first = false;
411                     } else  {
412                         cmd.append(',');
413                     }
414                     cmd.append(newParent);
415                 }
416                 cmd.append('\"');
417                 MqlUtil_mxJPO.execMql(_paramCache.getContext(), cmd);
418             }
419         }
