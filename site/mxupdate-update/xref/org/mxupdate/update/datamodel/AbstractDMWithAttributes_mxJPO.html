105      * Called TCL procedure within the TCL update to assign attributes to this
106      * administration object. The given values of the arguments are converted
107      * so that the values are handled correctly from TCL and MQL.
108      *
109      * @see #update(ParameterCache_mxJPO, CharSequence, CharSequence, CharSequence, Map, File)

112     private static final String TCL_PROCEDURE
113             = "proc testAttributes {args}  {\n"
114                 + "set iIdx 0\n"
115                 + "set lsCmd [list mql exec prog org.mxupdate.update.util.JPOCaller " + AbstractDMWithAttributes_mxJPO.JPO_CALLER_KEY + "]\n"
116                 + "while {$iIdx < [llength $args]}  {\n"
117                 +     "set sValue [lindex $args $iIdx]\n"
118                 +     "if {($iIdx > 0) && ([lindex $args [expr $iIdx - 1]] == \"-attributes\")}  {\n"
119                 +         "set sCmd \"\"\n"
120                 +         "foreach sOneValue $sValue  {\n"
121                 +             "regsub -all {@} \"${sOneValue}\" \"@@\" sOneValue\n"
122                 +             "regsub -all { } \"${sOneValue}\" \"@20\" sOneValue\n"
123                 +             "regsub -all {//\"} \"${sOneValue}\" \"@22\" sOneValue\n"

125                 +             "regsub -all \"//{\" \"${sOneValue}\" \"@7B\" sOneValue\n"
126                 +             "regsub -all \"//}\" \"${sOneValue}\" \"@7D\" sOneValue\n"
127                 +             "set sCmd \"${sCmd} ${sOneValue}\"\n"
128                 +         "}\n"
129                 +         "lappend lsCmd ${sCmd}\n"
130                 +     "} else  {\n"
131                 +         "regsub -all {@} \"${sValue}\" \"@@\" sValue\n"
132                 +         "regsub -all { } \"${sValue}\" \"@20\" sValue\n"

256     /***
257      * The method is called within the update of an administration object. The
258      * method is called directly within the update and checks which attributes
259      * are missed in the new definition and adds missing attributes to the
260      * administration object. If an attribute is not defined anymore but
261      * assigned in MX, an exception is thrown.
262      *
263      * @param _paramCache   parameter cache
264      * @param _args         arguments from the TCL procedure
265      * @throws Exception if an unknown parameter is defined, the given name of
266      *                   the administration object is not the same or an
267      *                   attribute is assigned to an administration object
268      *                   within MX but not defined anymore
269      */
270     @Override()
271     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
272                                final String... _args)
273         throws Exception
274     {
275         if ((_args.length == 0) || !AbstractDMWithAttributes_mxJPO.JPO_CALLER_KEY.equals(_args[0]))  {
276             super.jpoCallExecute(_paramCache, _args);
277         } else  {
278             int idx = 1;
279             String name = null;
280             String attrStr = null;
281 
282             final Set<String> ignoreAllAttrs = new HashSet<String>();
283             final Set<String> removeAllAttrs = new HashSet<String>();
284             final Set<String> ignoreAttrs = new HashSet<String>();
285             final Set<String> removeAttrs = new HashSet<String>();
286             while (idx < _args.length)  {
287                 final String arg = _args[idx];
288                 if ("-attributes".equals(arg))  {
289                     attrStr = _args[++idx];
290                 } else if ("-ignoreattr".equals(arg))  {
291                     ignoreAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
292                 } else if ("-interface".equals(arg) && this.getTypeDef().getMxAdminName().equals("interface"))  {
293                     name = this.jpoCallExecuteConvert(_args[++idx]);
294                     final Collection<String> tmp1 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_INTERFACE);
295                     if (tmp1 != null)  {
296                         ignoreAllAttrs.addAll(tmp1);
297                     }
298                     final Collection<String> tmp2 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_INTERFACE);
299                     if (tmp2 != null)  {
300                         removeAllAttrs.addAll(tmp2);
301                     }
302                 } else if ("-relationship".equals(arg) && this.getTypeDef().getMxAdminName().equals("relationship"))  {
303                     name = this.jpoCallExecuteConvert(_args[++idx]);
304                     final Collection<String> tmp1 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_RELATIONSHIP);
305                     if (tmp1 != null)  {
306                         ignoreAllAttrs.addAll(tmp1);
307                     }
308                     final Collection<String> tmp2 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_RELATIONSHIP);
309                     if (tmp2 != null)  {
310                         removeAllAttrs.addAll(tmp2);
311                     }
312                 } else if ("-removeattr".equals(arg))  {
313                     removeAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
314                 } else if ("-type".equals(arg) && this.getTypeDef().getMxAdminName().equals("type"))  {
315                     name = this.jpoCallExecuteConvert(_args[++idx]);
316                     final Collection<String> tmp1 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_TYPE);
317                     if (tmp1 != null)  {
318                         ignoreAllAttrs.addAll(tmp1);
319                     }
320                     final Collection<String> tmp2 = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_TYPE);
321                     if (tmp2 != null)  {
322                         removeAllAttrs.addAll(tmp2);
323                     }
324                 } else  {
325                     throw new UpdateException_mxJPO(
326                             UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_UKNOWN_PARAMETER,
327                             arg);
328                 }
329                 idx++;
330             }
331 
332             // check for equal administration name
333             if (!this.getName().equals(name))  {
334                 throw new UpdateException_mxJPO(
335                         UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_WRONG_OBJECT,
336                         this.getTypeDef().getLogging(),
337                         this.getName(),
338                         name);
339             }
340 
341             // extract all attributes
342             final Set<String> newAttrs = new TreeSet<String>();
343             if (attrStr != null)  {
344                 for (final String attrName : attrStr.split(" "))  {
345                     if (!"".equals(attrName))  {
346                         newAttrs.add(this.jpoCallExecuteConvert(attrName));
347                     }
348                 }
349             }
350 
351             // and update updates
352             ignoreAllAttrs.addAll(ignoreAttrs);
353             removeAllAttrs.addAll(removeAttrs);
354             this.updateAttributes(_paramCache, newAttrs, ignoreAllAttrs, removeAllAttrs);
355         }
356     }

358     /***
359      * Updates this administration object with given attributes.
360      *
361      * @param _paramCache   parameter cache
362      * @param _newAttrs     set of attributes which must be assigned
363      * @param _ignoreAttrs  set of attributes which are ignored (if assigned)
364      * @param _removeAttrs  set of attributes which could be removed (if
365      *                      assigned)
366      * @throws Exception if update failed or attribute could not be removed
367      */
368     protected void updateAttributes(final ParameterCache_mxJPO _paramCache,
369                                     final Set<String> _newAttrs,
370                                     final Set<String> _ignoreAttrs,
371                                     final Set<String> _removeAttrs)
372         throws Exception
373     {
374         // now check for not defined but existing attributes
375         for (final String attr : this.attributes)  {
376             if (!_newAttrs.contains(attr))  {
377                 boolean ignore = false;
378                 for (final String ignoreAttr : _ignoreAttrs)  {
379                     if (StringUtil_mxJPO.match(attr, ignoreAttr))  {
380                         ignore = true;
381                         _paramCache.logDebug("    - attribute '" + attr + "' is not defined but will be ignored");
382                         break;
383                     }
384                 }
385                 if (!ignore)  {
386                     boolean remove = false;
387                     for (final String removeAttr : _removeAttrs)  {
388                         if (StringUtil_mxJPO.match(attr, removeAttr))  {
389                             remove = true;
390                             _paramCache.logDebug("    - attribute '" + attr + "' is not defined and will be removed");
391                             MqlUtil_mxJPO.execMql(_paramCache,
392                                     new StringBuilder()
393                                         .append("escape mod ").append(this.getTypeDef().getMxAdminName())
394                                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
395                                         .append(" remove attribute \"")
396                                                 .append(StringUtil_mxJPO.convertMql(attr)).append("\""));
397                             break;
398                         }
399                     }
400                     if (!remove)  {
401                         throw new UpdateException_mxJPO(
402                                 UpdateException_mxJPO.Error.DM_ABSTRACTWITHATTRIBUTES_UPDATE_ATTRIBUTE_REMOVED,
403                                 attr,
404                                 this.getTypeDef().getLogging(),
405                                 this.getName());
406                     }
407                 }
408             }
409         }
410 
411         // and check for not existing attributes but needed
412         for (final String attr : _newAttrs)  {
413             if (!this.attributes.contains(attr))  {
414                 _paramCache.logDebug("    - add attribute '" + attr + "'");
415                 MqlUtil_mxJPO.execMql(_paramCache,
416                         new StringBuilder()
417                             .append("escape mod ").append(this.getTypeDef().getMxAdminName())
418                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
419                             .append(" add attribute \"").append(StringUtil_mxJPO.convertMql(attr)).append("\""));
420             }
421         }
422     }

424     /***
425      * Converts given string by replacing escaped characters.
426      *
427      * @param _value    value to convert
428      * @return replaced value from the converted value used for the JPO call
429      * @see #TCL_PROCEDURE
430      */
431     protected String jpoCallExecuteConvert(final String _value)
432     {
433         return _value.replaceAll("@20", " ")
434                      .replaceAll("@22", "//\"")
435                      .replaceAll("@27", "'")
436                      .replaceAll("@7B", "{")
437                      .replaceAll("@7D", "}")
438                      .replaceAll("@@", "@");
439     }
