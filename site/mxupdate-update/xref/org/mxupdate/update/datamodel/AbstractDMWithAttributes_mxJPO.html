95       * Called TCL procedure within the TCL update to assign attributes to this
96       * administration object. The given values of the arguments are converted

102     private static final String TCL_PROCEDURE
103             = "proc testAttributes {args}  {\n"
104                 + "set iIdx 0\n"
105                 + "set lsCmd [list mql exec prog org.mxupdate.update.util.JPOCaller attributes]\n"
106                 + "while {$iIdx < [llength $args]}  {\n"
107                 +     "set sValue [lindex $args $iIdx]\n"
108                 +     "if {($iIdx > 0) && ([lindex $args [expr $iIdx - 1]] == \"-attributes\")}  {\n"
109                 +         "set sCmd \"\"\n"
110                 +         "foreach sOneValue $sValue  {\n"
111                 +             "regsub -all {@} \"${sOneValue}\" \"@@\" sOneValue\n"
112                 +             "regsub -all { } \"${sOneValue}\" \"@20\" sOneValue\n"
113                 +             "regsub -all {//\"} \"${sOneValue}\" \"@22\" sOneValue\n"
114                 +             "regsub -all {'} \"${sOneValue}\" \"@27\" sOneValue\n"
115                 +             "regsub -all \"//{\" \"${sOneValue}\" \"@7B\" sOneValue\n"
116                 +             "regsub -all \"//}\" \"${sOneValue}\" \"@7D\" sOneValue\n"
117                 +             "set sCmd \"${sCmd} ${sOneValue}\"\n"
118                 +         "}\n"
119                 +         "lappend lsCmd ${sCmd}\n"
120                 +     "} else  {\n"
121                 +         "regsub -all {@} \"${sValue}\" \"@@\" sValue\n"
122                 +         "regsub -all { } \"${sValue}\" \"@20\" sValue\n"
123                 +         "regsub -all {//\"} \"${sValue}\" \"@22\" sValue\n"
124                 +         "regsub -all {'} \"${sValue}\" \"@27\" sValue\n"

127                 +         "lappend lsCmd ${sValue}\n"
128                 +     "}\n"
129                 +     "incr iIdx\n"
130                 + "}\n"

234     /***
235      * The method is called within the update of an administration object. The
236      * method is called directly within the update and checks which attributes
237      * are missed in the new definition and adds missing attributes to the
238      * administration object. If an attribute is not defined anymore but
239      * assigned in MX, an exception is thrown.
240      *
241      * @param _paramCache   parameter cache
242      * @param _args         arguments from the TCL procedure
243      * @throws Exception if an unknown parameter is defined, the given name of
244      *                   the administration object is not the same or an
245      *                   attribute is assigned to an administration object
246      *                   within MX but not defined anymore
247      */
248     @Override()
249     public void jpoCallExecute(final ParameterCache_mxJPO _paramCache,
250                                final String... _args)
251         throws Exception
252     {
253         int idx = 1;
254         String name = null;
255         String attrStr = null;
256 
257         final Set<String> ignoreAttrs = new HashSet<String>();
258         final Set<String> removeAttrs = new HashSet<String>();
259         while (idx < _args.length)  {
260             final String arg = _args[idx];
261             if ("-attributes".equals(arg))  {
262                 attrStr = _args[++idx];
263             } else if ("-ignoreattr".equals(arg))  {
264                 ignoreAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
265             } else if ("-interface".equals(arg))  {
266                 name = this.jpoCallExecuteConvert(_args[++idx]);
267                 final Collection<String> tmp1
268                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_INTERFACE);
269                 if (tmp1 != null)  {
270                     ignoreAttrs.addAll(tmp1);
271                 }
272                 final Collection<String> tmp2
273                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_INTERFACE);
274                 if (tmp2 != null)  {
275                     removeAttrs.addAll(tmp2);
276                 }
277             } else if ("-relationship".equals(arg))  {
278                 name = this.jpoCallExecuteConvert(_args[++idx]);
279                 final Collection<String> tmp1
280                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_RELATIONSHIP);
281                 if (tmp1 != null)  {
282                     ignoreAttrs.addAll(tmp1);
283                 }
284                 final Collection<String> tmp2
285                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_RELATIONSHIP);
286                 if (tmp2 != null)  {
287                     removeAttrs.addAll(tmp2);
288                 }
289             } else if ("-removeattr".equals(arg))  {
290                 removeAttrs.add(this.jpoCallExecuteConvert(_args[++idx]));
291             } else if ("-type".equals(arg))  {
292                 name = this.jpoCallExecuteConvert(_args[++idx]);
293                 final Collection<String> tmp1
294                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_IGNORE_TYPE);
295                 if (tmp1 != null)  {
296                     ignoreAttrs.addAll(tmp1);
297                 }
298                 final Collection<String> tmp2
299                         = _paramCache.getValueList(AbstractDMWithAttributes_mxJPO.PARAM_REMOVE_TYPE);
300                 if (tmp2 != null)  {
301                     removeAttrs.addAll(tmp2);
302                 }
303             } else  {
304                 throw new Exception("unknown parameter \"" + arg + "\"");
305             }
306             idx++;
307         }
308 
309         // check for equal administration name
310         if (!this.getName().equals(name))  {
311             throw new Exception(this.getTypeDef().getLogging()
312                     + " '" + this.getName() + "' was called to"
313                     + " update via update script, but "
314                     + this.getTypeDef().getLogging() + " '" + name + "' was"
315                     + " called in the procedure...");
316         }
317 
318         // get all attributes
319 // TODO: if a { or } is in an attribute name, the list is not created correctly
320 /*        final Pattern pattern = Pattern.compile("(//{[^//{//}]*//} )|([^ //{//}]* )");
321         final Matcher matcher = pattern.matcher(attrStr + " ");
322 */
323         final Set<String> newAttrs = new TreeSet<String>();
324         for (final String attrName : attrStr.split(" "))  {
325             if (!"".equals(attrName))  {
326                 newAttrs.add(this.jpoCallExecuteConvert(attrName));
327             }
328         }
329 
330         // now check for not defined but existing attributes
331         for (final String attr : this.attributes)  {
332             if (!newAttrs.contains(attr))  {
333                 boolean ignore = false;
334                 for (final String ignoreAttr : ignoreAttrs)  {
335                     if (StringUtil_mxJPO.match(attr, ignoreAttr))  {
336                         ignore = true;
337                         _paramCache.logDebug("    - attribute '" + attr + "' is not defined but will be ignored");
338                         break;
339                     }
340                 }
341                 if (!ignore)  {
342                     boolean remove = false;
343                     for (final String removeAttr : removeAttrs)  {
344                         if (StringUtil_mxJPO.match(attr, removeAttr))  {
345                             remove = true;
346                             _paramCache.logDebug("    - attribute '" + attr + "' is not defined and will be removed");
347                             MqlUtil_mxJPO.execMql(_paramCache,
348                                     new StringBuilder()
349                                         .append("escape mod ").append(this.getTypeDef().getMxAdminName())
350                                         .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
351                                         .append(" remove attribute \"")
352                                                 .append(StringUtil_mxJPO.convertMql(attr)).append("\""));
353                             break;
354                         }
355                     }
356                     if (!remove)  {
357                         throw new Exception("Attribute '" + attr + "' is defined to be deleted"
358                                 + " in " + this.getTypeDef().getLogging() + " '"
359                                 + this.getName() + "', but not allowed!");
360                     }
361                 }
362             }
363         }
364 
365         // and check for not existing attributes but needed
366         for (final String attr : newAttrs)  {
367             if (!this.attributes.contains(attr))  {
368                 _paramCache.logDebug("    - add attribute '" + attr + "'");
369                 MqlUtil_mxJPO.execMql(_paramCache,
370                         new StringBuilder()
371                             .append("escape mod ").append(this.getTypeDef().getMxAdminName())
372                             .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\"")
373                             .append(" add attribute \"").append(StringUtil_mxJPO.convertMql(attr)).append("\""));
374             }
375         }
376     }

378     /***
379      *
380      * @param _value    value to convert
381      * @return replaced value from the converted value used for the JPO call
382      * @see #TCL_PROCEDURE
383      */
384     protected String jpoCallExecuteConvert(final String _value)
385     {
386         return _value.replaceAll("@20", " ")
387                      .replaceAll("@22", "//\"")
388                      .replaceAll("@27", "'")
389                      .replaceAll("@7B", "{")
390                      .replaceAll("@7D", "}")
391                      .replaceAll("@@", "@");
392     }
