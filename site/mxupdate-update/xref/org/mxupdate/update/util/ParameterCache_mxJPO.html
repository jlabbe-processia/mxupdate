192     /***
193      * Creates a new instance of the parameter cache. All default values from
194      * the parameter definitions are predefined in the parameter cache.
195      *
196      * @param _context      MX context
197      * @param _stringLog    if set to <i>true</i> log via string writer,
198      *                      <i>false</i> log via <code>System.out</code>
199      * @param _paramValues  predefined parameter values to set
200      * @throws Exception if the mapping could not be initialized or a
201      *                   predefined parameter in <code>_paramValues</code> is
202      *                   not known
203      * @see #context
204      * @see #mapBoolean
205      * @see #mapList
206      * @see #mapMap
207      * @see #mapString
208      * @see #writer
209      * @see #stringWriter
210      */
211     public ParameterCache_mxJPO(final Context _context,
212                                 final boolean _stringLog,
213                                 final Map<String,String> _paramValues)
214             throws Exception
215     {
216         this.context = _context;
217 
218         this.mapping = this.initMapping(_context);
219 
220         this.mapBoolean = new HashMap<String,Boolean>();
221         this.mapInteger = new HashMap<String,Integer>();
222         this.mapList = new HashMap<String,Collection<String>>();
223         this.mapMap = new HashMap<String,Map<String,?>>();
224         this.mapString = new HashMap<String,String>();
225 
226         // evaluate default parameter values
227         for (final ParameterDef_mxJPO paramDef : this.mapping.getAllParameterDefs())  {
228             if (paramDef.getDefaultValue() != null)  {
229                 this.defineParameter(paramDef, paramDef.getDefaultValue());
230             }
231         }
232         // evaluate predefined parameter values
233         if (_paramValues != null)  {
234             for (final Map.Entry<String,String> paramValue : _paramValues.entrySet())  {
235                 final ParameterDef_mxJPO paramDef = this.mapping.getParameterDef(paramValue.getKey());
236                 if (paramDef == null)  {
237                     throw new Exception("unknown parameter definition '" + paramValue.getKey() + "'");
238                 }
239                 this.defineParameter(paramDef, paramValue.getValue());
240             }
241         }
242 
243         if (_stringLog)  {
244             this.stringWriter = new StringWriter();
245             this.writer = new PrintWriter(this.stringWriter);
246         } else  {
247             this.stringWriter = null;
248             this.writer = new PrintWriter(System.out);
249         }
250     }

252     /***
253      * Defines the parameter depending of the parameter definition
254      * <code>_paramDef</code> with related <code>_value</code>.
255      *
256      * @param _paramDef instance of the parameter definition
257      * @param _value    string value of the parameter to define
258      */
259     protected void defineParameter(final ParameterDef_mxJPO _paramDef,
260                                    final String _value)
261     {
262         switch (_paramDef.getType())  {
263             case BOOLEAN:
264                 this.mapBoolean.put(_paramDef.getName(),
265                                     Boolean.parseBoolean(_value));
266                 break;
267             case INTEGER:
268                 this.mapInteger.put(_paramDef.getName(),
269                                     Integer.parseInt(_value));
270                 break;
271             case LIST:
272                 this.mapList.put(_paramDef.getName(),
273                                  new ArrayList<String>(Arrays.asList(_value.split(","))));
274                 break;
275             case MAP:
276                 if ((_value != null) && !"".equals(_value))  {
277                     final Map<String,String> values = new HashMap<String,String>();
278                     for (final String entry : _value.split(","))  {
279                         final String[] entryArr = entry.split("=", 2);
280                         values.put(entryArr[0], (entryArr.length > 1) ? entryArr[1] : null);
281                     }
282                     this.mapMap.put(_paramDef.getName(), values);
283                 }
284                 break;
285             case STRING:
286                 this.mapString.put(_paramDef.getName(), _value);
287                 break;
288             default:
289                 throw new Error("unknown parameter type " + _paramDef.getType());
290         }
291     }

354     /***
355      * Evaluated given parameter and defines the values for given parameter
356      * depending on the list of arguments.
357      * <ul>
358      * <li><b>boolean parameter</b>: the boolean map defines for the parameter
359      *     the converted value from the default value</li>
360      * <li><b>list parameter</b>: the next argument is added to the value list
361      *     </li>
362      * <li><b>string parameter</b>: the next argument is defined as string
363      *     value</li>
364      * </ul>
365      *
366      * @param _paramDef parameter definition
367      * @param _args     list of arguments from the command line
368      * @param _index    current index within the list of arguments
369      * @return new index of the current index within the list of arguments
370      * @see #mapBoolean
371      * @see #mapList
372      * @see #mapString
373      */
374     public int evalParameter(final ParameterDef_mxJPO _paramDef,
375                              final String[] _args,
376                              final int _index)
377     {
378         int index = _index;
379 
380         if (_paramDef.getType() == ParameterDef_mxJPO.Type.BOOLEAN)  {
381             this.mapBoolean.put(_paramDef.getName(),
382                                 !Boolean.parseBoolean(_paramDef.getDefaultValue()));
383         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.INTEGER)  {
384             this.mapInteger.put(_paramDef.getName(),
385                                 Integer.parseInt(_args[++index]));
386         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.LIST)  {
387             if (!this.mapList.containsKey(_paramDef.getName()))  {
388                 this.mapList.put(_paramDef.getName(), new ArrayList<String>());
389             }
390             this.mapList.get(_paramDef.getName()).add(_args[++index]);
391         } else if (_paramDef.getType() == ParameterDef_mxJPO.Type.STRING)  {
392             this.mapString.put(_paramDef.getName(), _args[++index]);
393         }
394 
395         return index;
396     }

618     /***
619      * Appends to the log with defined <code>_logLevel</code> the
620      * <code>_text</code>.
621      *
622      * @param _logLevel     level of the log
623      * @param _text         text of the log
624      */
625     private void log(final LogLevel _logLevel,
626                      final String _text)
627     {
628         for (final String line : _text.split("\n"))  {
629             if (this.stringWriter == null)  {
630                 this.writer.append(_logLevel.getConsoleLog());
631             } else  {
632                 this.writer.append(_logLevel.getPlugInLog());
633             }
634             this.writer.append(line).append('\n');
635         }
636         this.writer.flush();
637     }

666     private enum LogLevel

686              * @return always <code>WARNING! </code>
687              */
688             @Override()
689             public String getConsoleLog()
690             {
691                 return "WARNING! ";
692             }
693         },
694         /*** Log level for information. */
695         INFO,
696         /*** Log level debug. */
697         DEBUG,
698         /*** Log level trace. */
699         TRACE;
