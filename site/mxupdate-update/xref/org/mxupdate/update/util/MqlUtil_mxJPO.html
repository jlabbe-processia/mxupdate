21  package org.mxupdate.update.util;
22  
23  import matrix.db.Context;
24  import matrix.db.MQLCommand;
25  import matrix.util.MatrixException;
26  
27  /***
28   * The JPO class holds utilities for easy MQL access.
29   *
30   * @author The MxUpdate Team
31   * @version $Id: MqlUtil_mxJPO.java 753 2010-03-07 11:49:58Z tim.moxter $
32   */
33  public final class MqlUtil_mxJPO
34  {
35      /***
36       * Dummy constructor so that the MQL utility class could not be
37       * instanciated.
38       */
39      private MqlUtil_mxJPO()
40      {
41      }
42  
43      /***
44       * Executes given MQL command and returns the trimmed result of the MQL
45       * execution.
46       *
47       * @param _paramCache   parameter cache
48       * @param _cmd          MQL command to execute
49       * @return trimmed result of the MQL execution
50       * @throws MatrixException if MQL execution failed
51       */
52      public static String execMql(final ParameterCache_mxJPO _paramCache,
53                                   final CharSequence _cmd)
54              throws MatrixException
55      {
56          return MqlUtil_mxJPO.execMql(_paramCache.getContext(), _cmd, true);
57      }
58  
59      /***
60       * Executes given MQL command and returns the trimmed result of the MQL
61       * execution.
62       *
63       * @param _context              MX context for this request
64       * @param _cmd                  MQL command to execute
65       * @param _includeMQLCommand    must be MQL command included in the
66       *                              exception?
67       * @return trimmed result of the MQL execution
68       * @throws MatrixException if MQL execution failed; includes the MQL
69       *                         command if <code>_includeMQLCommand</code> is
70       *                         set to <i>true</i>
71       */
72      public static String execMql(final Context _context,
73                                   final CharSequence _cmd,
74                                   final boolean _includeMQLCommand)
75              throws MatrixException
76      {
77          final MQLCommand mql = new MQLCommand();
78          mql.executeCommand(_context, _cmd.toString());
79          if ((mql.getError() != null) && !"".equals(mql.getError()))  {
80              throw new MatrixException(mql.getError()
81                      + (_includeMQLCommand ? ("\nMQL command was:\n" + _cmd) : ""));
82          }
83          return mql.getResult().trim();
84      }
85  
86      /***
87       * Sets the history on.
88       *
89       * @param _paramCache       parameter cache
90       * @throws MatrixException if the history could not be set to on
91       */
92      public static void setHistoryOn(final ParameterCache_mxJPO _paramCache)
93              throws MatrixException
94      {
95          MqlUtil_mxJPO.execMql(_paramCache, "history on");
96      }
97  
98      /***
99       * Sets the history off.
100      *
101      * @param _paramCache       parameter cache
102      * @throws MatrixException if the history could not be set to off
103      */
104     public static void setHistoryOff(final ParameterCache_mxJPO _paramCache)
105             throws MatrixException
106     {
107         MqlUtil_mxJPO.execMql(_paramCache, "history off");
108     }
