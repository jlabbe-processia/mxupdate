33  /***
34   * The class could be executed within the TCL update to call the original JPO
35   * update instance. To use this feature, the JPO update class must be defined
36   * from {@link AbstractPropertyObject_mxJPO}. In the derived class method
37   * {@link AbstractPropertyObject_mxJPO#jpoCallExecute(ParameterCache_mxJPO, String...)}
38   * must be overwritten.
39   *
40   * @author Tim Moxter
41   * @version $Id: JPOCaller_mxJPO.java 305 2009-05-03 16:15:44Z tim.moxter $
42   */
43  public class JPOCaller_mxJPO
44  {
45      /***
46       * Stores the current caller instance depending on the MX session context
47       * id.
48       *
49       * @see #defineInstance(ParameterCache_mxJPO, AbstractPropertyObject_mxJPO)
50       * @see #undefineInstance(ParameterCache_mxJPO)
51       * @see #mxMain(Context, String[])
52       */
53      private static final Map<String,AbstractPropertyObject_mxJPO> CALLER_INSTANCE
54              = new HashMap<String,AbstractPropertyObject_mxJPO>();
55  
56      /***
57       * Stores the current caller instance depending on the MX session context
58       * id.
59       *
60       * @see #defineInstance(ParameterCache_mxJPO, AbstractPropertyObject_mxJPO)
61       * @see #undefineInstance(ParameterCache_mxJPO)
62       * @see #mxMain(Context, String[])
63       */
64      private static final Map<String,ParameterCache_mxJPO> PARAM_CACHE
65              = new HashMap<String,ParameterCache_mxJPO>();
66  
67      /***
68       * Prefix for the name of the global environment variable to define the
69       * name of the class which must be defined to call from MQL. The global
70       * environment variable gets a prefix of the MX session id.
71       *
72       * @see #defineInstance(ParameterCache_mxJPO, AbstractPropertyObject_mxJPO)
73       * @see #undefineInstance(ParameterCache_mxJPO)
74       * @see #mxMain(Context, String[])
75       */
76      private static final String ENV_CLASS_NAME = "MXUPDATE_JPOCALLER_CLASS";
77  
78      /***
79       * The class must have a public constructor so that the class could be
80       * initialized from MX (otherwise an illegal access exception is thrown).
81       */
82      public JPOCaller_mxJPO()
83      {
84      }

86      /***
87       * Defines given <code>_instance</code> and <code>_paramCache</code> which
88       * will be called within a TCL program. The name of this JPO is also stored
89       * in a global environment variable so that the instance and parameter
90       * cache could be evaluated in all cases.
91       *
92       * @param _paramCache       parameter cache with MX context
93       * @param _instance         new instance
94       * @throws MatrixException if global environment variable which defines the
95       *                         instance could not be defined (MQL call failed)
96       * @see #PARAM_CACHE
97       * @see #CALLER_INSTANCE
98       * @see #ENV_CLASS_NAME
99       * @see #mxMain(Context, String[])
100      */
101     public static void defineInstance(final ParameterCache_mxJPO _paramCache,
102                                       final AbstractPropertyObject_mxJPO _instance)
103             throws MatrixException
104     {
105         final String sessionId = _paramCache.getContext().getSession().getSessionId();
106         JPOCaller_mxJPO.PARAM_CACHE.put(sessionId, _paramCache);
107         JPOCaller_mxJPO.CALLER_INSTANCE.put(sessionId, _instance);
108         MqlUtil_mxJPO.execMql(_paramCache.getContext(),
109                 new StringBuilder()
110                     .append("escape set env global \"")
111                     .append(JPOCaller_mxJPO.ENV_CLASS_NAME).append(StringUtil_mxJPO.convertMql(sessionId))
112                     .append("\" \"${CLASSNAME}\""));
113     }
114 
115     /***
116      * Removes from current MX session context the instance which could be
117      * called.
118      *
119      * @param _paramCache       parameter cache with MX context
120      * @throws MatrixException if the global environment variable which defines
121      *                          the instance could not be removed (MQL call
122      *                          failed)
123      * @see #PARAM_CACHE
124      * @see #CALLER_INSTANCE
125      * @see #ENV_CLASS_NAME
126      */
127     public static void undefineInstance(final ParameterCache_mxJPO _paramCache)
128             throws MatrixException
129     {
130         final String sessionId = _paramCache.getContext().getSession().getSessionId();
131         JPOCaller_mxJPO.PARAM_CACHE.remove(sessionId);
132         JPOCaller_mxJPO.CALLER_INSTANCE.remove(sessionId);
133         MqlUtil_mxJPO.execMql(_paramCache.getContext(),
134                 new StringBuilder()
135                     .append("escape unset env global \"")
136                     .append(JPOCaller_mxJPO.ENV_CLASS_NAME).append(StringUtil_mxJPO.convertMql(sessionId))
137                     .append("\""));
138     }

140     /***
141      * <p>The method calls the original instance stored depending on the matrix
142      * session id. If for the current MX session id no instance of no parameter
143      * cache defined, errors are thrown.<p/>
144      * <p>In some cases (especially if MxUpdate is installed the first time) it
145      * could be that {@link #CALLER_INSTANCE} and {@link #PARAM_CACHE} set from
146      * {@link #defineInstance(ParameterCache_mxJPO, AbstractPropertyObject_mxJPO)}
147      * within another JPO class than the class called for this method. In this
148      * case MX makes an automatically compile of the class. But this means also
149      * that the caller instance and the parameter cache could not be accessed
150      * directly. Instead the original JPO class which is used when the instance
151      * was defined is used (via Java reflection).</p>
152      *
153      * @param _context  context for this request
154      * @param _args     list of arguments
155      * @throws Exception if execution fails
156      * @see #PARAM_CACHE
157      * @see #CALLER_INSTANCE
158      * @see #ENV_CLASS_NAME
159      * @see #defineInstance(ParameterCache_mxJPO, AbstractPropertyObject_mxJPO)
160      */
161     public static void mxMain(final Context _context,
162                               final String[] _args)
163             throws Exception
164     {
165         try  {
166             final String sessionId = _context.getSession().getSessionId();
167 
168             final String callerClazzName = MqlUtil_mxJPO.execMql(_context,
169                     new StringBuilder()
170                         .append("escape get env global \"")
171                         .append(JPOCaller_mxJPO.ENV_CLASS_NAME).append(StringUtil_mxJPO.convertMql(sessionId))
172                         .append("\""));
173             // is original JPO calling class not current JPO calling class?
174             if (!"${CLASSNAME}".equals(callerClazzName))  {
175                 final Class<?> callerClazz = Class.forName("org.mxupdate.update.util." + callerClazzName);
176                 if (callerClazz == null)  {
177                     throw new Error("JPO Caller class " + callerClazzName
178                             + " does not exists in the Java VM!");
179                 }
180                 final Method method = callerClazz.getMethod("mxMain", Context.class, String[].class);
181                 if (method == null)  {
182                     throw new Error("Static method 'mxMain' of " + callerClazzName + " does not exists!");
183                 }
184                 method.invoke(null, _context, _args);
185             } else  {
186                 final AbstractPropertyObject_mxJPO instance = JPOCaller_mxJPO.CALLER_INSTANCE.get(sessionId);
187                 final ParameterCache_mxJPO paramCache = JPOCaller_mxJPO.PARAM_CACHE.get(sessionId);
188                 if (instance == null)  {
189                     throw new Error("JPO Caller instance is not defined for session "
190                             + sessionId + "!");
191                 }
192                 if (paramCache == null)  {
193                     throw new Error("Old Parameter Cache instance is not defined for session "
194                             + sessionId + "!");
195                 }
196                 instance.jpoCallExecute(paramCache.clone(_context), _args);
197             }
198         } catch (final Exception e)  {
199             e.printStackTrace(System.out);
200             throw e;
201         }
202     }
