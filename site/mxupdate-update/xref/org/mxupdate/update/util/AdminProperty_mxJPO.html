26  /***
27   * Property with name, value and referenced administration type. The kind
28   * of property (with reference, with value, ...) is stored as flag. A property
29   * is only used within administration objects (not business objects).

34  public class AdminProperty_mxJPO
35      implements Comparable<AdminProperty_mxJPO>

37      /***
38       * Set of all ignored URLs from the XML definition for admin properties.
39       *
40       * @see #parse(String, String)
41       */
42      private static final Set<String> IGNORED_URLS = new HashSet<String>(1);
43      static  {
44          AdminProperty_mxJPO.IGNORED_URLS.add("/adminRef");
45      }

74      /***
75       * <p>Parses administration property related XML elements and updates this
76       * property instance. This includes:
77       * <ul>
78       * <li>{@link #refAdminName name of referenced administration object}</li>
79       * <li>{@link #refAdminType type of referenced administration object}</li>
80       * <li>{@link #flags}</li>
81       * <li>{@link #name}</li>
82       * <li>{@link #value}</li>
83       * </ul></p>
84       * <p>If an <code>_url</code> is included in {@link #IGNORED_URLS}, this
85       * URL is ignored.</p>
86       *
87       * @param _url      URL of the XML elements
88       * @param _content  content within the <code>_url</code> of XML elements
89       * @return <i>true</i> if the <code>_url</code> could be parsed; otherwise
90       *         <i>false</i>
91       * @see #IGNORED_URLS
92       */
93      public boolean parse(final String _url,
94                           final String _content)
95      {
96          boolean parsed = true;
97          if (!AdminProperty_mxJPO.IGNORED_URLS.contains(_url))  {
98              if ("/adminRef/adminName".equals(_url))  {
99                  this.refAdminName = _content;
100             } else if ("/adminRef/adminType".equals(_url))  {
101                 this.refAdminType = _content;
102             } else if ("/flags".equals(_url))  {
103                 this.flags = _content;
104             } else if ("/name".equals(_url))  {
105                 this.name = _content;
106             } else if ("/value".equals(_url))  {
107                 this.value = _content;
108             } else  {
109                 parsed = false;
110             }
111         }
112         return parsed;
113     }

178     /***
179      * <p>Compares this administration property instance with
180      * <code>_toCompare</code> administration property. The algorithm uses
181      * <ul>
182      * <li>{@link #refAdminType}</li>
183      * <li>{@link #refAdminName}</li>
184      * <li>{@link #name}</li>
185      * <li>{@link #value}</li>
186      * </ul>
187      * in the defined order to compare this administration property against
188      * <code>_toCompare</code>. A <code>null</code> value is always lower than
189      * a non <code>null</code> value.</p>
190      *
191      * @param _toCompare    admin property instance which must be compared to
192      * @return &quot;0&quot; if both administration properties are equal;
193      *         &quot;1&quot; if greater; otherwise &quot;-1&quot;
194      */
195     public int compareTo(final AdminProperty_mxJPO _toCompare)
196     {
197         return ((this.refAdminType != null) && (this.refAdminName != null))
198                ? ((_toCompare.refAdminType != null) && (_toCompare.refAdminName != null))
199                        ? (this.refAdminType.compareTo(_toCompare.refAdminType) == 0)
200                                ? (this.refAdminName.compareTo(_toCompare.refAdminName) == 0)
201                                        ? (this.name.compareTo(_toCompare.name) == 0)
202                                                ? (this.value != null)
203                                                        ? (_toCompare.value != null)
204                                                                ? this.value.compareTo(_toCompare.value)
205                                                                : 1
206                                                        : (_toCompare.value != null)
207                                                                ? -1
208                                                                : 0
209                                                : this.name.compareTo(_toCompare.name)
210                                        : this.refAdminName.compareTo(_toCompare.refAdminName)
211                                : this.refAdminType.compareTo(_toCompare.refAdminType)
212                        : 1
213                : ((_toCompare.refAdminType != null) && (_toCompare.refAdminName != null))
214                        ? -1
215                        : (this.name.compareTo(_toCompare.name) == 0)
216                                ? (this.value != null)
217                                        ? (_toCompare.value != null)
218                                                ? this.value.compareTo(_toCompare.value)
219                                                : 1
220                                        : (_toCompare.value != null)
221                                                ? -1
222                                                : 0
223                                : this.name.compareTo(_toCompare.name);
224     }
