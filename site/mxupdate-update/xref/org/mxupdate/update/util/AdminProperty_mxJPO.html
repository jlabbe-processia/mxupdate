68      /***
69       * Parses administration property related XML elements and updates this
70       * property instance.
71       *
72       * @param _url      URL of the XML elements
73       * @param _content  content within the <code>_url</code> of XML elements
74       * @return <i>true</i> if the <code>_url</code> could be parsed; otherwise
75       *         <i>false</i>
76       */
77      public boolean parse(final String _url,
78                           final String _content)
79      {
80          boolean parsed = true;
81          if ("/adminRef".equals(_url))  {
82              // to be ignored ...
83          } else if ("/adminRef/adminName".equals(_url))  {
84              this.refAdminName = _content;
85          } else if ("/adminRef/adminType".equals(_url))  {
86              this.refAdminType = _content;
87          } else if ("/flags".equals(_url))  {
88              this.flags = _content;
89          } else if ("/name".equals(_url))  {
90              this.name = _content;
91          } else if ("/value".equals(_url))  {
92              this.value = _content;
93          } else  {
94              parsed = false;
95          }
96          return parsed;
97      }

152     /***
153      * <p>Compares this administration property instance with
154      * <code>_toCompare</code> administration property. The algorithm uses
155      * <ul>
156      * <li>{@link #refAdminType}</li>
157      * <li>{@link #refAdminName}</li>
158      * <li>{@link #name}</li>
159      * <li>{@link #value}</li>
160      * </ul>
161      * in the defined order to compare this administration property against
162      * <code>_toCompare</code>. A <code>null</code> value is always lower than
163      * a non <code>null</code> value.</p>
164      *
165      * @param _toCompare    admin property instance which must be compared to
166      * @return &quot;0&quot; if both administration properties are equal;
167      *         &quot;1&quot; if greater; otherwise &quot;-1&quot;
168      */
169     public int compareTo(final AdminProperty_mxJPO _toCompare)
170     {
171         return ((this.refAdminType != null) && (this.refAdminName != null))
172                ? ((_toCompare.refAdminType != null) && (_toCompare.refAdminName != null))
173                        ? this.refAdminType.equals(_toCompare.refAdminType)
174                                ? this.refAdminName.equals(_toCompare.refAdminName)
175                                        ? this.name.equals(_toCompare.name)
176                                                ? (this.value != null)
177                                                        ? (_toCompare.value != null)
178                                                                ? this.value.compareTo(_toCompare.value)
179                                                                : 1
180                                                        : (_toCompare.value != null)
181                                                                ? -1
182                                                                : 0
183                                                : this.name.compareTo(_toCompare.name)
184                                        : this.refAdminName.compareTo(_toCompare.refAdminName)
185                                : this.refAdminType.compareTo(_toCompare.refAdminType)
186                        : 1
187                : ((_toCompare.refAdminType != null) && (_toCompare.refAdminName != null))
188                        ? -1
189                        : this.name.equals(_toCompare.name)
190                                ? (this.value != null)
191                                        ? (_toCompare.value != null)
192                                                ? this.value.compareTo(_toCompare.value)
193                                                : 1
194                                        : (_toCompare.value != null)
195                                                ? -1
196                                                : 0
197                                : this.name.compareTo(_toCompare.name);
198     }
