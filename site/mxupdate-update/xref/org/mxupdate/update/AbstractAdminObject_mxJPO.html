119     /***
120      * Searches for all administration object of current type definition and
121      * returns all found names as set.
122      *
123      * @param _paramCache   parameter cache
124      * @return set of all administration object of current type definition
125      * @throws MatrixException if the query for current administration object
126      *                         failed
127      */
128     @Override
129     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
130             throws MatrixException
131     {
132         final StringBuilder cmd = new StringBuilder()
133                 .append("list ")
134                 .append(this.getTypeDef().getMxAdminName())
135                 .append(" ")
136                 .append(this.getTypeDef().getMxAdminSuffix());
137         final Set<String> ret = new TreeSet<String>();
138         for (final String name : MqlUtil_mxJPO.execMql(_paramCache, cmd).split("\n"))  {
139             if (!"".equals(name))  {
140                 ret.add(name);
141             }
142         }
143         return ret;
144     }

146     /***
147      * Creates a XML representation of the Object to export, parses them and
148      * executes the post preparation {@link #prepare(ParameterCache_mxJPO)}.
149      *
150      * @param _paramCache   parameter cache
151      * @throws MatrixException  if the export of the admin object failed
152      * @throws SAXException     if the exported XML document could not be
153      *                          parsed
154      * @throws IOException      should not happen; only if the input source of
155      *                          the string reader which embeds the XML document
156      *                          failed
157      * @see #getExportMQL()                 used to get the MQL command to get
158      *                                      a XML representation
159      * @see PadSaxHandler                   SAX handler to parse the XML file
160      * @see #parse(String, String)          parser called within the SAX
161      *                                      handler
162      * @see #prepare(ParameterCache_mxJPO)  called post preparation method
163      */
164     @Override
165     protected void parse(final ParameterCache_mxJPO _paramCache)
166             throws MatrixException, SAXException, IOException
167     {
168         final String xml = MqlUtil_mxJPO.execMql(_paramCache, this.getExportMQL());
169 
170         // create XML reader
171         final XMLReader reader = XMLReaderFactory.createXMLReader();
172         // register Sax Content Handler
173         final PadSaxHandler handler = new PadSaxHandler();
174         reader.setContentHandler(handler);
175         reader.setDTDHandler(handler);
176         reader.setEntityResolver(handler);
177         // parse the XML string of the export
178         final InputSource inputSource = new InputSource(new StringReader(xml));
179         inputSource.setEncoding("UTF8");
180         reader.parse(inputSource);
181         // prepare post preparation
182         this.prepare(_paramCache);
183     }

201     /***
202      * Parsed administration object related XML tags. This includes:
203      * <ul>
204      * <li>description</li>
205      * <li>is the object {@link #hidden}</li>
206      * <li>properties</li>
207      * </ul>
208      *
209      * @param _url      URL within the XML
210      * @param _content  value of the URL
211      * @see #IGNORED_URLS
212      */
213     protected void parse(final String _url,
214                          final String _content)
215     {
216         if (!AbstractAdminObject_mxJPO.IGNORED_URLS.contains(_url))  {
217             if ("/adminProperties/description".equals(_url))  {
218                 this.setDescription(_content);
219             } else if ("/adminProperties/hidden".equals(_url))  {
220                 this.hidden = true;
221 
222             } else if ("/adminProperties/propertyList/property".equals(_url))  {
223                 this.propertiesStack.add(new AdminProperty_mxJPO());
224             } else if (_url.startsWith("/adminProperties/propertyList/property"))  {
225                 if (!this.propertiesStack.peek().parse(_url.substring(38), _content))  {
226                     System.err.println("unknown parsing property url: "+_url+"("+_content+")");
227                 }
228             } else  {
229                 System.err.println("unknown parsing url: "+_url+"("+_content+")");
230             }
231         }
232     }

234     /***
235      * Sorted the properties, sets the author and version depending on the
236      * properties and reads the symbolic names.
237      *
238      * @param _paramCache   parameter cache
239      * @throws MatrixException if the symbolic names could not be extracted
240      * @see #propertiesStack
241      * @see #propertiesMap
242      */
243     protected void prepare(final ParameterCache_mxJPO _paramCache)
244             throws MatrixException
245     {
246         // sort the properties
247         for (final AdminProperty_mxJPO property : this.propertiesStack)  {
248             final StringBuilder key = new StringBuilder().append(property.getName());
249             if ((property.getRefAdminName() != null) && (property.getRefAdminType() != null))  {
250                 key.append("::").append(property.getRefAdminType())
251                    .append("::").append(property.getRefAdminName());
252             }
253             this.propertiesMap.put(key.toString(), property);
254         }
255         // set author depending on the properties
256         final AdminProperty_mxJPO author = this.propertiesMap.get(PropertyDef_mxJPO.AUTHOR.getPropName(_paramCache));
257         if (author != null)  {
258             this.setAuthor(author.getValue());
259         }
260         // set application depending on the properties
261         final AdminProperty_mxJPO appl = this.propertiesMap.get(PropertyDef_mxJPO.APPLICATION.getPropName(_paramCache));
262         if (appl != null)  {
263             this.setApplication(appl.getValue());
264         }
265         // sets the installation date depending on the properties
266         final AdminProperty_mxJPO installationDate
267                 = this.propertiesMap.get(PropertyDef_mxJPO.INSTALLEDDATE.getPropName(_paramCache));
268         if (installationDate != null)  {
269             this.setInstallationDate(installationDate.getValue());
270         }
271         // sets the installer depending on the properties
272         final AdminProperty_mxJPO installer
273                 = this.propertiesMap.get(PropertyDef_mxJPO.INSTALLER.getPropName(_paramCache));
274         if (installer != null)  {
275             this.setInstaller(installer.getValue());
276         }
277         // sets the original name depending on the properties
278         final AdminProperty_mxJPO origName
279                 = this.propertiesMap.get(PropertyDef_mxJPO.ORIGINALNAME.getPropName(_paramCache));
280         if (origName != null)  {
281             this.setOriginalName(origName.getValue());
282         }
283         // sets the version depending on the properties
284         final AdminProperty_mxJPO version = this.propertiesMap.get(PropertyDef_mxJPO.VERSION.getPropName(_paramCache));
285         if (version != null)  {
286             this.setVersion(version.getValue());
287         }
288 
289         // reads symbolic names of the administration objects
290         this.readSymbolicNames(_paramCache);
291     }

294      *
295      * @param _paramCache   parameter cache
296      * @param _out          writer instance
297      * @throws IOException  if the write of the TCL update to the writer
298      *                      instance failed
299      */
300     @Override
301     protected void write(final ParameterCache_mxJPO _paramCache,
302                          final Appendable _out)
303             throws IOException
304     {
305         this.writeHeader(_paramCache, _out);
306         _out.append("mql escape mod ")
307             .append(this.getTypeDef().getMxAdminName())
308             .append(" \"${NAME}\"");
309         if (!"".equals(this.getTypeDef().getMxAdminSuffix()))  {
310             _out.append(" ").append(this.getTypeDef().getMxAdminSuffix());
311         }
312         _out.append(" //\n    description \"").append(StringUtil_mxJPO.convertTcl(this.getDescription())).append("\"");
313         this.writeObject(_paramCache, _out);
314         this.writeProperties(_paramCache, _out);
315         this.writeEnd(_paramCache, _out);
316     }

344     /***
345      * Writes the MQL code to add all none standard
346      * {@link #propertiesMap properties} to the TCL update file.
347      *
348      * @param _paramCache   parameter cache
349      * @param _out          appendable instance to the TCL update file
350      * @throws IOException if the write to the TCL update file failed
351      */
352     protected void writeProperties(final ParameterCache_mxJPO _paramCache,
353                                    final Appendable _out)
354             throws IOException
355     {
356         for (final AdminProperty_mxJPO prop : this.propertiesMap.values())  {
357             if ((PropertyDef_mxJPO.getEnumByPropName(_paramCache, prop.getName()) == null) && !prop.isSetting())  {
358                 _out.append("\nmql escape add property \"").append(StringUtil_mxJPO.convertTcl(prop.getName())).append("\"")
359                     .append(" //\n    on ")
360                     .append(this.getTypeDef().getMxAdminName())
361                     .append(" \"${NAME}\"");
362                 if (!"".equals(this.getTypeDef().getMxAdminSuffix()))  {
363                     _out.append(' ').append(this.getTypeDef().getMxAdminSuffix());
364                 }
365                 if ((prop.getRefAdminName()) != null && (prop.getRefAdminType() != null))  {
366                     _out.append("  //\n    to ").append(prop.getRefAdminType())
367                         .append(" \"").append(StringUtil_mxJPO.convertTcl(prop.getRefAdminName())).append("\"");
368                     // if target is a table, a system is required!
369                     if ("table".equals(prop.getRefAdminType()))  {
370                         _out.append(" system");
371                     }
372                 }
373                 if (prop.getValue() != null)  {
374                     _out.append(" //\n    value \"").append(StringUtil_mxJPO.convertTcl(prop.getValue())).append("\"");
375                 }
376             }
377         }
378     }

414     /***
415      * The method overwrites the original method append MQL code to
416      * <ul>
417      * <li>remove all existing {@link #propertiesMap properties}</li>
418      * <li>define the TCL variable for the name</li>
419      * <li>define property &quot;version&quot;</li>
420      * <li>define property &quot;file date&quot;</li>
421      * <li>define &quot;installed&quot; date property if not defined</li>
422      * <li>define &quot;installer&quot; property if not defined</li>
423      * <li>define &quot;original name&quot; property if not defined</li>
424      * <li>define &quot;application&quot; property if not set or not equal to
425      *     defined value from user</li>
426      * <li>define &quot;author&quot; property if not set or not equal to value
427      *     from TCL update code (or to default value if not defined in TCL
428      *     update code)</li>
429      * <li>register administration object (if not already done) and remove all
430      *     symbolic names which are not correct</li>
431      * </ul>
432      * Then the update method of the super class is called.
433      *
434      * @param _paramCache       parameter cache
435      * @param _preMQLCode       MQL statements which must be called before the
436      *                          TCL code is executed
437      * @param _postMQLCode      MQL statements which must be called after the
438      *                          TCL code is executed
439      * @param _preTCLCode       TCL code which is defined before the source
440      *                          file is sourced
441      * @param _tclVariables     map of all TCL variables where the key is the
442      *                          name and the value is value of the TCL variable
443      *                          (the value is automatically converted to TCL
444      *                          syntax!)
445      * @param _sourceFile       souce file with the TCL code to update
446      * @throws Exception if the update from the derived class failed
447      */
448     @Override
449     protected void update(final ParameterCache_mxJPO _paramCache,
450                           final CharSequence _preMQLCode,
451                           final CharSequence _postMQLCode,
452                           final CharSequence _preTCLCode,
453                           final Map<String,String> _tclVariables,
454                           final File _sourceFile)
455             throws Exception
456     {
457         // remove all properties
458         final StringBuilder preMQLCode = new StringBuilder();
459         for (final AdminProperty_mxJPO prop : this.propertiesMap.values())  {
460             // % must be ignored because this means settings
461             if ((PropertyDef_mxJPO.getEnumByPropName(_paramCache, prop.getName()) == null) && !prop.isSetting())  {
462                 // must be done via modify because of properties without names
463                 preMQLCode.append("escape mod ").append(this.getTypeDef().getMxAdminName())
464                           .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
465                           .append(this.getTypeDef().getMxAdminSuffix())
466                           .append(" remove property \"");
467                 if (prop.getName() != null)  {
468                     preMQLCode.append(StringUtil_mxJPO.convertMql(prop.getName()));
469                 }
470                 preMQLCode.append('\"');
471                 if ((prop.getRefAdminName() != null) && (prop.getRefAdminType() != null))  {
472                     preMQLCode.append(" to ").append(prop.getRefAdminType())
473                               .append(" \"").append(StringUtil_mxJPO.convertMql(prop.getRefAdminName())).append('\"');
474                     // if target is a table, a system is required!
475                     if ("table".equals(prop.getRefAdminType()))  {
476                         preMQLCode.append(" system");
477                     }
478                 }
479                 preMQLCode.append(";\n");
480             }
481         }
482 
483         // append already existing pre MQL code
484         preMQLCode.append(_preMQLCode);
485 
486         // define version property
487         final StringBuilder postMQLCode = new StringBuilder()
488                 .append(_postMQLCode)
489                 .append("escape mod ").append(this.getTypeDef().getMxAdminName())
490                 .append(" \"").append(StringUtil_mxJPO.convertMql(this.getName())).append("\" ")
491                 .append(this.getTypeDef().getMxAdminSuffix())
492                 .append(" add property \"")
493                 .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.VERSION.getPropName(_paramCache)))
494                 .append("\" value \"")
495                 .append(StringUtil_mxJPO.convertMql(_tclVariables.get(PropertyDef_mxJPO.VERSION.name())))
496                 .append('\"');
497         // define file date property
498         postMQLCode.append(" add property \"")
499                    .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.FILEDATE.getPropName(_paramCache)))
500                    .append("\" value \"")
501                    .append(StringUtil_mxJPO.convertMql(_tclVariables.get(PropertyDef_mxJPO.FILEDATE.name())))
502                    .append('\"');
503         // is installed date property defined?
504         if ((this.getInstallationDate() == null) || "".equals(this.getInstallationDate()))  {
505             final String date = StringUtil_mxJPO.formatInstalledDate(_paramCache, new Date());
506             _paramCache.logTrace("    - define installed date '" + date + "'");
507             postMQLCode.append(" add property \"")
508                        .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.INSTALLEDDATE.getPropName(_paramCache)))
509                        .append("\" value \"")
510                        .append(StringUtil_mxJPO.convertMql(date))
511                        .append('\"');
512         }
513         // exists no installer property or installer property not equal?
514         final String instVal = _tclVariables.get(PropertyDef_mxJPO.INSTALLER.name());
515         if ((this.getInstaller() == null) || !this.getInstaller().equals(instVal))  {
516             _paramCache.logTrace("    - define installer '" + instVal + "'");
517             postMQLCode.append(" add property \"")
518                        .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.INSTALLER.getPropName(_paramCache)))
519                        .append("\" value \"")
520                        .append(StringUtil_mxJPO.convertMql(instVal))
521                        .append('\"');
522         }
523         // is original name property defined?
524         final String origNameVal = _tclVariables.get(PropertyDef_mxJPO.ORIGINALNAME.name());
525         if ((this.getOriginalName() == null) || !this.getOriginalName().equals(origNameVal))  {
526             _paramCache.logTrace("    - define original name '" + origNameVal + "'");
527             postMQLCode.append(" add property \"")
528                        .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.ORIGINALNAME.getPropName(_paramCache)))
529                        .append("\" value \"")
530                        .append(StringUtil_mxJPO.convertMql(origNameVal))
531                        .append('\"');
532         }
533         // exists no application property or application property not equal?
534         final String applVal = _tclVariables.get(PropertyDef_mxJPO.APPLICATION.name());
535         if ((this.getApplication() == null) || !this.getApplication().equals(applVal))  {
536             _paramCache.logTrace("    - define application '" + applVal + "'");
537             postMQLCode.append(" add property \"")
538                        .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.APPLICATION.getPropName(_paramCache)))
539                        .append("\" value \"")
540                        .append(StringUtil_mxJPO.convertMql(applVal))
541                        .append('\"');
542         }
543         // exists no author property or author property not equal?
544         final String authVal = _tclVariables.get(PropertyDef_mxJPO.AUTHOR.name());
545         if ((this.getAuthor() == null) || !this.getAuthor().equals(authVal))  {
546             _paramCache.logTrace("    - define author '" + authVal + "'");
547             postMQLCode.append(" add property \"")
548                        .append(StringUtil_mxJPO.convertMql(PropertyDef_mxJPO.AUTHOR.getPropName(_paramCache)))
549                        .append("\" value \"")
550                        .append(StringUtil_mxJPO.convertMql(authVal))
551                        .append('\"');
552         }
553         postMQLCode.append(";\n");
554 
555         // append registration of symbolic names
556         this.appendSymbolicNameRegistration(_paramCache,
557                                             _tclVariables.get("SYMBOLICNAME"),
558                                             postMQLCode);
559 
560         // prepare map of all TCL variables incl. name of admin object
561         final Map<String,String> tclVariables = new HashMap<String,String>();
562         tclVariables.put("NAME", this.getName());
563         tclVariables.putAll(_tclVariables);
564 
565         super.update(_paramCache, preMQLCode, postMQLCode, _preTCLCode, tclVariables, _sourceFile);
566     }
