177     /***
178      * Searches for all business object of current type definition and returns
179      * all MX names. The revision of the business object is appended to the
180      * name of the business object split by {@link #SPLIT_NAME}.
181      *
182      * @param _paramCache   parameter cache
183      * @return set of found business object names (and revision)
184      * @throws MatrixException if the query for all business objects for given
185      *                         type failed
186      */
187     @Override
188     public Set<String> getMxNames(final ParameterCache_mxJPO _paramCache)
189             throws MatrixException
190     {
191         final StringList selects = new StringList();
192         selects.addElement("name");
193         selects.addElement("revision");
194 
195         final Query query = new Query();
196         query.open(_paramCache.getContext());
197         query.setBusinessObjectType(this.getTypeDef().getMxBusType());
198         final BusinessObjectWithSelectList list = query.select(_paramCache.getContext(), selects);
199         query.close(_paramCache.getContext());
200 
201         final Set<String> ret = new TreeSet<String>();
202         for (final Object mapObj : list)  {
203             final BusinessObjectWithSelect map = (BusinessObjectWithSelect) mapObj;
204             final String busName = (String) map.getSelectDataList("name").get(0);
205             final String busRevision = (String) map.getSelectDataList("revision").get(0);
206             final StringBuilder name = new StringBuilder().append(busName);
207             if ((busRevision != null) && !"".equals(busRevision))  {
208                 name.append(BusObject_mxJPO.SPLIT_NAME).append(busRevision);
209             }
210             ret.add(name.toString());
211         }
212         return ret;
213     }

215     /***
216      * Checks if given MX name without prefix and suffix matches given match
217      * string. The MX name is split with {@link #SPLIT_NAME} to get
218      * the name and revision of the business object. A MX name matches if the
219      * business object name or revision matches.
220      *
221      * @param _paramCache   parameter cache
222      * @param _mxName       name (and revision) of the administration business
223      *                      object
224      * @param _match        string which must be matched
225      * @return <i>true</i> if the given MX name matches; otherwise <i>false</i>
226      */
227     @Override
228     public boolean matchMxName(final ParameterCache_mxJPO _paramCache,
229                                final String _mxName,
230                                final String _match)
231     {
232         final String[] nameRev = _mxName.split(BusObject_mxJPO.SPLIT_NAME);
233         return (StringUtil_mxJPO.match(nameRev[0], _match)
234                 || ((nameRev.length > 1) && StringUtil_mxJPO.match(nameRev[1], _match)));
235     }

237     /***
238      * Parses all information for given administration object.
239      * Sorts the attribute values, defines the description for the TCL
240      * update script (concatenation of the revision and description).
241      *
242      * @param _paramCache   parameter cache
243      * @throws MatrixException if the parse failed
244      * @see #attrValuesSorted
245      * @see #busCurrent
246      * @see #busDescription
247      * @see #busName
248      * @see #busOid
249      * @see #busRevision
250      * @see #busStates
251      * @see #busVault
252      */
253     @Override
254     protected void parse(final ParameterCache_mxJPO _paramCache)
255             throws MatrixException
256     {
257         final BusinessObject bus = new BusinessObject(this.getBusType(),
258                                                       this.getBusName(),
259                                                       this.getBusRevision(),
260                                                       null);
261 
262         // id, vault, description and current MX state
263         final StringList select = new StringList(4);
264         select.addElement("id");
265         select.addElement("vault");
266         select.addElement("description");
267         select.addElement("current");
268         select.addElement("policy.state");
269         final BusinessObjectWithSelect sel = bus.select(_paramCache.getContext(), select);
270         this.busOid = sel.getSelectData("id");
271         this.busVault = sel.getSelectData("vault");
272         this.busDescription = sel.getSelectData("description");
273         this.busCurrent = sel.getSelectData("current");
274         this.busStates = sel.getSelectDataList("policy.state");
275 
276         // define properties (attribute values)
277         final AttributeList attrs = bus.getAttributeValues(_paramCache.getContext(), true);
278         for (final Object attrObj : attrs)  {
279             final Attribute attr = (Attribute) attrObj;
280             if (PropertyDef_mxJPO.AUTHOR.getAttrName(_paramCache).equals(attr.getName()))  {
281                 this.setAuthor(attr.getValue());
282             } else if (PropertyDef_mxJPO.APPLICATION.getAttrName(_paramCache).equals(attr.getName()))  {
283                 this.setApplication(attr.getValue());
284             } else if (PropertyDef_mxJPO.INSTALLEDDATE.getAttrName(_paramCache).equals(attr.getName()))  {
285                 this.setInstallationDate(attr.getValue());
286             } else if (PropertyDef_mxJPO.INSTALLER.getAttrName(_paramCache).equals(attr.getName()))  {
287                 this.setInstaller(attr.getValue());
288             } else if (PropertyDef_mxJPO.VERSION.getAttrName(_paramCache).equals(attr.getName()))  {
289                 this.setVersion(attr.getValue());
290             } else if (!PropertyDef_mxJPO.FILEDATE.getAttrName(_paramCache).equals(attr.getName()))  {
291                 this.attrValuesSorted.add(new AttributeValue(attr));
292             }
293         }
294 
295         // define description
296         final StringBuilder desc = new StringBuilder();
297         if ((this.busRevision != null) && !"".equals(this.busRevision))  {
298             desc.append(this.busRevision);
299             if (this.busDescription != null)  {
300                 desc.append('\n');
301             }
302         }
303         if (this.busDescription != null)  {
304             desc.append(this.busDescription);
305         }
306         this.setDescription(desc.toString());
307 
308         // evaluate from connections
309         if (this.getTypeDef().getMxBusRelsFrom() != null)  {
310             for (final String relation : this.getTypeDef().getMxBusRelsFrom())  {
311                 this.expand4Parse(_paramCache, bus, relation, true, false);
312             }
313         }
314 
315         // evaluate to connections
316         if (this.getTypeDef().getMxBusRelsTo() != null)  {
317             for (final String relation : this.getTypeDef().getMxBusRelsTo())  {
318                 this.expand4Parse(_paramCache, bus, relation, false, true);
319             }
320         }
321 
322         // evaluate both connections
323         if (this.getTypeDef().getMxBusRelsBoth() != null)  {
324             for (final String relation : this.getTypeDef().getMxBusRelsBoth())  {
325                 // from
326                 this.expand4Parse(_paramCache, bus, relation, true, false);
327                 // to
328                 this.expand4Parse(_paramCache, bus, relation, false, true);
329             }
330         }
331     }

333     /***
334      * Expands from the business object depending on the relationship and
335      * stores the connection information. If the relationship has attributes,
336      * the related values are also evaluated.
337      *
338      * @param _paramCache   parameter cache
339      * @param _bus          the business object to expand
340      * @param _relation     name of relationship to expand
341      * @param _getFrom      must be set to <i>true</i> to expand from
342      *                      connections; otherwise <i>false</i>
343      * @param _getTo        must be set to <i>true</i> to expand to
344      *                      connections; otherwise <i>false</i>
345      * @throws MatrixException if the expand failed
346      * @see #connections
347      */
348     @SuppressWarnings("unchecked")
349     private void expand4Parse(final ParameterCache_mxJPO _paramCache,
350                               final BusinessObject _bus,
351                               final String _relation,
352                               final boolean _getFrom,
353                               final boolean _getTo)
354             throws MatrixException
355     {
356         // get attributes from relationship
357         final Map<String,Map> cache = _paramCache.defineValueMap(BusObject_mxJPO.PARAM_RELATION_ATTRS, Map.class);
358         Map<String,String> attrs = cache.get(_relation);
359         if (attrs == null)
360         {
361             attrs = new HashMap<String,String>();
362             cache.put(_relation, attrs);
363 
364             final String attrStr = MqlUtil_mxJPO.execMql(_paramCache,
365                     new StringBuilder("escape print rel \"").append(_relation)
366                             .append("\" select attribute dump '\n'"));
367             if (!"".equals(attrStr))  {
368                 for (final String attr : attrStr.split("\n"))  {
369                     attrs.put("attribute[" + attr + "]", attr);
370                 }
371             };
372         }
373 
374 
375         final StringList busSelect = new StringList(3);
376         busSelect.addElement("type");
377         busSelect.addElement("name");
378         busSelect.addElement("revision");
379         final StringList relSelect = new StringList(1 + attrs.size());
380         relSelect.addElement("id");
381         relSelect.addAll(attrs.keySet());
382         // get from objects
383         final ExpansionWithSelect expandTo = _bus.expandSelect(_paramCache.getContext(),
384                                                               _relation,
385                                                               "*",
386                                                               busSelect,
387                                                               relSelect,
388                                                               _getFrom,
389                                                               _getTo,
390                                                               (short) 1,
391                                                               null,
392                                                               null,
393                                                               (short) 0,
394                                                               true);
395 
396         for (final Object obj : expandTo.getRelationships())  {
397             this.connections.add(new Connection(_relation,
398                                                 _getFrom ? "from" : "to",
399                                                 (RelationshipWithSelect) obj,
400                                                 attrs));
401         }
402 
403     }

405     /***
406      * Writes the information to update the business objects.
407      *
408      * @param _paramCache   parameter cache
409      * @param _out          writer instance
410      * @throws IOException if the TCL update code for the business object could
411      *                     not be written
412      */
413     @Override
414     protected void write(final ParameterCache_mxJPO _paramCache,
415                          final Appendable _out)
416             throws IOException
417     {
418         this.writeHeader(_paramCache, _out);
419         _out.append("mql escape mod bus \"${OBJECTID}\"")
420             .append(" //\n    description \"")
421             .append(StringUtil_mxJPO.convertTcl(this.busDescription))
422             .append("\"");
423         for (final AttributeValue attr : this.attrValuesSorted)  {
424             attr.write(_out);
425         }
426         // write all connected objects
427         for (final Connection con : this.connections)  {
428             con.write(_out);
429         }
430         // write promotes to target states if required
431         final int target = this.busStates.indexOf(this.busCurrent);
432         for (int idx = 0; idx < target; idx++)  {
433             _out.append("\nmql promote bus \"${OBJECTID}\"");
434         }
435     }

474     /***
475      * The method overwrites the original method to
476      * <ul>
477      * <li>reset the description</li>
478      * <li>set the version and author attribute</li>
479      * <li>reset all not ignored attributes</li>
480      * <li>define the TCL variable &quot;OBJECTID&quot; with the object id of
481      *     the represented business object</li>
482      * </ul>
483      * The original method of the super class if called surrounded with a
484      * history off, because if the update itself is done the modified basic
485      * attribute and the version attribute of the business object is updated.
486      * <br/>
487      * The new generated MQL code is set in the front of the already defined
488      * MQL code in <code>_preMQLCode</code> and appended to the MQL statements
489      * in <code>_postMQLCode</code>.
490      *
491      * @param _paramCache       parameter cache
492      * @param _preMQLCode       MQL statements which must be called before the
493      *                          TCL code is executed
494      * @param _postMQLCode      MQL statements which must be called after the
495      *                          TCL code is executed
496      * @param _preTCLCode       TCL code which is defined before the source
497      *                          file is sourced
498      * @param _tclVariables     map of all TCL variables where the key is the
499      *                          name and the value is value of the TCL variable
500      *                          (the value is automatically converted to TCL
501      *                          syntax!)
502      * @param _sourceFile       souce file with the TCL code to update
503      * @throws Exception if the update executed within derived class failed
504      */
505     @Override
506     protected void update(final ParameterCache_mxJPO _paramCache,
507                           final CharSequence _preMQLCode,
508                           final CharSequence _postMQLCode,
509                           final CharSequence _preTCLCode,
510                           final Map<String,String> _tclVariables,
511                           final File _sourceFile)
512             throws Exception
513     {
514         // resets the description
515         final StringBuilder preMQLCode = new StringBuilder()
516                 .append("mod bus ").append(this.busOid).append(" description \"\"");
517 
518         // reset all attributes (if they must not be ignored...)
519         final Set<String> ignoreAttrs = (this.getTypeDef().getMxBusIgnoredAttributes() == null)
520                                         ? new HashSet<String>(0)
521                                         : new HashSet<String>(this.getTypeDef().getMxBusIgnoredAttributes());
522         final Map<String,String> defaultAttrValues
523                 = _paramCache.defineValueMap(BusObject_mxJPO.PARAM_DEFAULT_ATTRS, String.class);
524         for (final AttributeValue attr : this.attrValuesSorted)  {
525             if (!ignoreAttrs.contains(attr.name))  {
526                 if (!defaultAttrValues.containsKey(attr.name))  {
527                     final String def = MqlUtil_mxJPO.execMql(_paramCache,
528                                                new StringBuilder().append("print attr '")
529                                                        .append(attr.name).append("' select default dump"));
530                     defaultAttrValues.put(attr.name, def);
531                 }
532                 preMQLCode.append(" \"").append(attr.name).append("\" \"")
533                         .append(defaultAttrValues.get(attr.name)).append('\"');
534             }
535         }
536         preMQLCode.append(";\n");
537 
538         // disconnect all objects
539         for (final Connection con : this.connections)  {
540             preMQLCode.append("disconnect connection ").append(con.conId).append(";\n");
541         }
542 
543         // append demotes if required
544         final int target = this.busStates.indexOf(this.busCurrent);
545         for (int idx = 0; idx < target; idx++)  {
546             preMQLCode.append("\ndemote bus ").append(this.busOid).append(";\n");
547         }
548 
549         // append other pre MQL code
550         preMQLCode.append(_preMQLCode);
551 
552         // post update MQL statements
553         final StringBuilder postMQLCode = new StringBuilder()
554                 .append(_postMQLCode)
555                 .append("mod bus ").append(this.busOid)
556         // define version
557                 .append(" \"").append(PropertyDef_mxJPO.VERSION.getAttrName(_paramCache))
558                         .append("\" \"").append(_tclVariables.get(PropertyDef_mxJPO.VERSION.name())).append('\"')
559         // define file date
560                 .append(" \"").append(PropertyDef_mxJPO.FILEDATE.getAttrName(_paramCache))
561                         .append("\" \"").append(_tclVariables.get(PropertyDef_mxJPO.FILEDATE.name())).append('\"');
562         // is installed date property defined?
563         if ((this.getInstallationDate() == null) || "".equals(this.getInstallationDate()))  {
564             final String date = StringUtil_mxJPO.formatInstalledDate(_paramCache, new Date());
565             _paramCache.logTrace("    - define installed date '" + date + "'");
566             postMQLCode.append(" \"").append(PropertyDef_mxJPO.INSTALLEDDATE.getAttrName(_paramCache))
567                     .append("\" \"").append(date).append('\"');
568         }
569         // exists no installer property or installer property not equal?
570         final String instVal = _tclVariables.get(PropertyDef_mxJPO.INSTALLER.name());
571         if ((this.getInstaller() == null) || !this.getInstaller().equals(instVal))  {
572             _paramCache.logTrace("    - define installer '" + instVal + "'");
573             postMQLCode.append(" \"").append(PropertyDef_mxJPO.INSTALLER.getAttrName(_paramCache))
574                     .append("\" \"").append(instVal).append('\"');
575         }
576         // exists no application property or application property not equal?
577         final String applVal = _tclVariables.get(PropertyDef_mxJPO.APPLICATION.name());
578         if ((this.getApplication() == null) || !this.getApplication().equals(applVal))  {
579             _paramCache.logTrace("    - define application '" + applVal + "'");
580             postMQLCode.append(" \"").append(PropertyDef_mxJPO.APPLICATION.getAttrName(_paramCache))
581                     .append("\" \"").append(applVal).append('\"');
582         }
583         // exists no author property or author property not equal?
584         final String authVal = _tclVariables.get(PropertyDef_mxJPO.AUTHOR.name());
585         if ((this.getAuthor() == null) || !this.getAuthor().equals(authVal))  {
586             _paramCache.logTrace("    - define author '" + authVal + "'");
587             postMQLCode.append(" \"").append(PropertyDef_mxJPO.AUTHOR.getAttrName(_paramCache))
588                     .append("\" \"").append(authVal).append('\"');
589         }
590         postMQLCode.append(";\n");
591 
592         // prepare map of all TCL variables incl. id of business object
593         final Map<String,String> tclVariables = new HashMap<String,String>();
594         tclVariables.put("OBJECTID", this.busOid);
595         tclVariables.putAll(_tclVariables);
596 
597         // update must be done with history off (because not required...)
598         try  {
599             MqlUtil_mxJPO.setHistoryOff(_paramCache);
600             super.update(_paramCache, preMQLCode, postMQLCode, _preTCLCode, tclVariables, _sourceFile);
601         } finally  {
602             MqlUtil_mxJPO.setHistoryOn(_paramCache);
603         }
604     }

740         /***
741          * Compares this attribute name with given attribute name. The related
742          * attribute values are not used to compare. If the attribute names
743          * includes numbers and the attribute names has the same prefix (before
744          * the numbers), the numbers are compared as integer.<br/>
745          * <b>Examples:</b><br/>
746          * Temp &gt; Name<br/>
747          * Temp = Temp<br/>
748          * Temp 5 &lt Temp 10
749          *

755          *         attribute name is lexicographically greater than the
756          *         compared attribute name
757          */
758         public int compareTo(final AttributeValue _attribute)
759         {
760             final String name1 = this.name.replaceAll(" [0-9]*$", "");
761             final String name2 = _attribute.name.replaceAll(" [0-9]*$", "");
762             int ret = 0;
763             if (name1.equals(name2) && !name1.equals(this.name))  {

765                 final Integer num1 = Integer.parseInt(this.name.substring(index).trim());
766                 final Integer num2 = Integer.parseInt(_attribute.name.substring(index).trim());

930             final Iterator<BusObject_mxJPO.AttributeValue> compIter = _compare.values.iterator();
931             int ret = 0;
