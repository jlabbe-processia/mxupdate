21  package org.mxupdate.mapping;
22  
23  import java.util.Arrays;
24  import java.util.Collection;
25  
26  /***
27   * Groups the type definition enumeration used for the parameter definitions.
28   *
29   * @author The MxUpdate Team
30   * @version $Id: TypeDefGroup_mxJPO.java 358 2009-05-30 12:00:49Z tim.moxter $
31   */
32  public final class TypeDefGroup_mxJPO
33          extends AbstractValue_mxJPO
34  {
35      /***
36       * Name list of type definition instances for which this type definition
37       * group is defined.
38       *
39       * @see #getTypeDefList()
40       * @see #defineValue(Mapping_mxJPO, String, String)
41       */
42      private Collection<String> typeDefList;
43  
44      /***
45       * The constructor is defined private so that a new instance could only
46       * created within this class.
47       *
48       * @param _name     name of the type definition group
49       */
50      private TypeDefGroup_mxJPO(final String _name)
51      {
52          super(_name);
53      }
54  
55      /***
56       * Defines the values of the type definition group.
57       *
58       * @param _mapping  cache for all mapping
59       * @param _key      key of the type definition group (including the name of
60       *                  type definition group and the kind of the type
61       *                  definition separated by a point)
62       * @param _value    value of the related value
63       * @throws Exception if key is not found
64       */
65      protected static void defineValue(final Mapping_mxJPO _mapping,
66                                        final String _key,
67                                        final String _value)
68              throws Exception
69      {
70          final String enumName = _key.replaceAll("//..*", "");
71          final String key = _key.substring(enumName.length() + 1);
72  
73          TypeDefGroup_mxJPO group = _mapping.getTypeDefGroupMap().get(enumName);
74          if (group == null)  {
75              group = new TypeDefGroup_mxJPO(enumName);
76              _mapping.getTypeDefGroupMap().put(enumName, group);
77          }
78  
79          if ("TypeDefList".endsWith(key))  {
80              group.typeDefList = Arrays.asList(_value.split(","));
81          } else  {
82              group.defineValues(key, _value);
83          }
84      }
85  
86      /***
87       * Returns the name list of type definitions.
88       *
89       * @return name list of type definitions
90       * @see #typeDefList
91       */
92      public Collection<String> getTypeDefList()
93      {
94          return this.typeDefList;
95      }
96  }
