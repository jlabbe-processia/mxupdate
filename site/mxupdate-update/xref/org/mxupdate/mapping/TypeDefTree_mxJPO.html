26  /***
27   * Tree hierarchy of the type definitions used for the eclipse plug-in.

29   * @author The MxUpdate Team

31   */
32  public final class TypeDefTree_mxJPO
33  {
34      /***
35       * Sub key property for label of type definition trees.
36       */
37      private static final String SUBKEY_LABEL = "Label";
38  
39      /***
40       * Sub key property for the list of sub type definition trees.
41       */
42      private static final String SUBKEY_SUBTREES = "SubTypeDefTreeList";
43  
44      /***
45       * Sub key property for the list of type definitions.
46       */
47      private static final String SUBKEY_TYPEDEFS = "TypeDefList";

62      private String label;

66       * tree is defined.

71      private Collection<String> subTypeDefTreeList;

73      /***
74       * Name list of type definition instances for which this type definition
75       * tree is defined.
76       *

80      private Collection<String> typeDefList;

83       * The constructor is defined private so that a new instance could only
84       * created within this class.

86       * @param _name     name of the type definition group
87       */
88      private TypeDefTree_mxJPO(final String _name)
89      {
90          this.name = _name;
91      }
92  
93      /***
94       * Defines the values of the type definition tree.
95       *
96       * @param _mapping  cache for all mapping
97       * @param _key      key of the type definition tree (including the name of
98       *                  type definition tree and the kind of the type
99       *                  definition separated by a point)
100      * @param _value    value of the related value
101      * @throws Exception if key is not found
102      */
103     protected static void defineValue(final Mapping_mxJPO _mapping,
104                                       final String _key,
105                                       final String _value)
106             throws Exception
107     {
108         final String enumName = _key.replaceAll("//..*", "");
109         final String key = _key.substring(enumName.length() + 1);
110 
111         TypeDefTree_mxJPO tree = _mapping.getTypeDefTreeMap().get(enumName);
112         if (tree == null)  {
113             tree = new TypeDefTree_mxJPO(enumName);
114             _mapping.getTypeDefTreeMap().put(enumName, tree);
115         }
116 
117         if (TypeDefTree_mxJPO.SUBKEY_LABEL.endsWith(key))  {
118             tree.label = _value;
119         } else if (TypeDefTree_mxJPO.SUBKEY_SUBTREES.endsWith(key))  {
120             tree.subTypeDefTreeList = Arrays.asList(_value.split(","));
121         } else if (TypeDefTree_mxJPO.SUBKEY_TYPEDEFS.endsWith(key))  {
122             tree.typeDefList = Arrays.asList(_value.split(","));
123         } else  {
124             throw new Exception("unknown key " + _key + " with value '" + _value + "' defined!");
125         }
126     }
