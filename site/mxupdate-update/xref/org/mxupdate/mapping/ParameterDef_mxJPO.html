24  /***
25   * Holds the definition of all parameters.
26   *
27   * @author Tim Moxter
28   * @version $Id: ParameterDef_mxJPO.java 308 2009-05-03 19:22:11Z tim.moxter $
29   */
30  public final class ParameterDef_mxJPO
31          extends AbstractValue_mxJPO
32  {
33      /***
34       * Enumeration to define the value type.
35       */
36      public enum Type
37      {
38          /***
39           * If the parameter is boolean, that means the parameter is defined
40           * (true) or not (false).
41           */
42          BOOLEAN,
43  
44          /***
45           * If the parameter is defined the value type is integer.
46           */
47          INTEGER,
48  
49          /***
50           * A list of strings is used for this type. That means that every
51           * argument of a parameter is append as string to the list.
52           */
53          LIST,
54  
55          /***
56           * A map of strings is used.
57           */
58          MAP,
59  
60          /***
61           * If the parameter is defined, the value is set to a string. If e.g.
62           * the parameter is defined twice, the second definition of the
63           * parameter overwrites the first definition.
64           */
65          STRING;
66      }
67  
68      /***
69       * Default value of the parameter.
70       *
71       * @see #getDefaultValue()
72       */
73      private String defaultValue;
74  
75      /***
76       * Type of the parameter.
77       *
78       * @see Type
79       * @see #getType()
80       */
81      private Type type;
82  
83      /***
84       * The constructor is defined private so that a new instance could only
85       * created within this class.
86       *
87       * @param _name     name of the parameter definition
88       */
89      private ParameterDef_mxJPO(final String _name)
90      {
91          super(_name);
92      }
93  
94      /***
95       * Defines the values of the parameter definition.
96       *
97       * @param _mapping  cache for all mapping
98       * @param _key      key to define
99       * @param _value    related value to define
100      * @throws Exception if key is not found
101      */
102     protected static void defineValue(final Mapping_mxJPO _mapping,
103                                       final String _key,
104                                       final String _value)
105             throws Exception
106     {
107         final String enumName = _key.replaceAll("//..*", "");
108         final String key = _key.substring(enumName.length() + 1);
109 
110         ParameterDef_mxJPO param = _mapping.getParameterDefJPOsMap().get(enumName);
111         if (param == null)  {
112             param = new ParameterDef_mxJPO(enumName);
113             _mapping.getParameterDefJPOsMap().put(enumName, param);
114         }
115 
116         if ("Default".equals(key))  {
117             param.defaultValue = _value;
118         } else if ("Type".equals(key))  {
119             param.type = ParameterDef_mxJPO.Type.valueOf(_value.toUpperCase());
120         } else  {
121             param.defineValues(key, _value);
122         }
123     }
124 
125     /***
126      * Getter method for instance variable {@link #defaultValue} returning the
127      * default value.
128      *
129      * @return default value of the parameter
130      * @see #defaultValue
131      */
132     public String getDefaultValue()
133     {
134         return this.defaultValue;
135     }
136 
137     /***
138      * Getter method for instance variable {@link #type} returning the
139      * parameter type.
140      *
141      * @return parameter type
142      * @see #type
143      */
144     public Type getType()
145     {
146         return this.type;
147     }
148 
149     /***
150      * Prepares the string representation of the parameter definition as
151      * concatenation of the name and type.
152      *
153      * @return string representation of the parameter definition
154      */
155     @Override
156     public String toString()
157     {
158         return new StringBuilder()
159             .append("[Parameter Definition ")
160                 .append("'").append(this.getName()).append("', ")
161                 .append("type = ").append(this.type)
162             .append("]")
163             .toString();
164     }
