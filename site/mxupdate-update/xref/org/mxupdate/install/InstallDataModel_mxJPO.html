159     /***
160      * Method used as entry from the MQL interface to install / update the data
161      * model for MxUpdate. The data model is installed / updated in this order:
162      * <ul>
163      * <li>install trigger group type</li>
164      * <li>install trigger group policy</li>
165      * <li>install trigger group relationship</li>
166      * <li>update attributes in
167      *     {@link #updateAttributes(ParameterCache_mxJPO, String, String, String, String, String, String)}</li>
168      * <li>update used business types by appending required MxUpdate attributes
169      *     in {@link #updateBusTypes(ParameterCache_mxJPO)}</li>
170      * <li>define plug-in properties in
171      *     {@link #makePluginProperty(ParameterCache_mxJPO, File)}</li>
172      * <li>register the MxUdpate Update programs with their symbolic names in
173      *     {@link #registerPrograms(ParameterCache_mxJPO, String, String, String, String, String)}</li>
174      * <li>register MxUpdate as application in
175      *     {@link #registerMxUpdate(ParameterCache_mxJPO, String, String)}</li>
176      * </ul>
177      *
178      * @param _context      MX context for this request
179      * @param _args         first value defines the source installation path
180      *                      (in development it is the resources sub directory),
181      *                      second value the version of MxUpdate which must be
182      *                      installed
183      * @throws Exception if installation failed
184      * @see #installTriggerGroupType(ParameterCache_mxJPO, File, String, String, String, String, String, String)
185      * @see #installTriggerGroupPolicy(ParameterCache_mxJPO, String, String, String, String, String, String)
186      * @see #installTriggerGroupRelation(ParameterCache_mxJPO, String, String, String, String, String, String)
187      * @see #updateAttributes(ParameterCache_mxJPO, String, String, String, String, String, String)
188      * @see #updateBusTypes(ParameterCache_mxJPO)
189      * @see #registerPrograms(ParameterCache_mxJPO, String, String, String, String, String)
190      * @see #registerMxUpdate(ParameterCache_mxJPO, String, String)
191      * @see #makePluginProperty(ParameterCache_mxJPO, File)
192      */
193     public void mxMain(final Context _context,
194                        final String... _args)
195             throws Exception
196     {
197         final File path = new File(_args[0]);
198         final String applVersion = _args[1];
199 
200         // initialize mapping
201         final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, false);
202 
203         final String applName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_APPLNAME);
204         final String authorName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_AUTHOR);
205         final String installerName = paramCache.getValueString(InstallDataModel_mxJPO.PARAM_INSTALLER);
206         final String fileDate = StringUtil_mxJPO.formatFileDate(paramCache, new Date());
207         final String installedDate = StringUtil_mxJPO.formatInstalledDate(paramCache, new Date());
208 
209         this.installTriggerGroupType(paramCache, path,
210                 applName, applVersion,
211                 authorName, installerName,
212                 fileDate, installedDate);
213         this.installTriggerGroupPolicy(paramCache,
214                 applName, applVersion,
215                 authorName, installerName,
216                 fileDate, installedDate);
217         this.installTriggerGroupRelation(paramCache,
218                 applName, applVersion,
219                 authorName, installerName,
220                 fileDate, installedDate);
221         this.updateAttributes(paramCache,
222                 applName, applVersion,
223                 authorName, installerName,
224                 fileDate, installedDate);
225         this.updateBusTypes(paramCache);
226         this.makePluginProperty(paramCache, path);
227         this.registerPrograms(paramCache,
228                 applName, applVersion,
229                 authorName, installerName,
230                 installedDate);
231         this.registerMxUpdate(paramCache, applName, applVersion);
232     }

234     /***
235      * All programs evaluated with {@link #LIST_MXUPDATE_PROGRAMS} are checked
236      * if they must be registered with a symbolic name and all other specific
237      * properties are set. If they are not already registered, the registration
238      * of the symbolic names of the programs are done. And also all other
239      * properties are set only if they are not already set.
240      *
241      * @param _paramCache       parameter cache
242      * @param _applName         used application name of the MxUpdate Update
243      *                          deployment tool
244      * @param _applVersion      Mx Update version
245      * @param _authorName       used author name
246      * @param _installerName    used installer name
247      * @param _installedDate    used installed date
248      * @throws Exception if registration of the symbolic names for all MxUpdate
249      *                   Update programs failed
250      * @see #LIST_MXUPDATE_PROGRAMS
251      */
252     protected void registerPrograms(final ParameterCache_mxJPO _paramCache,
253                                     final String _applName,
254                                     final String _applVersion,
255                                     final String _authorName,
256                                     final String _installerName,
257                                     final String _installedDate)
258         throws Exception
259     {
260         final TypeDef_mxJPO jpoTypeDef
261                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_JPO);
262         final TypeDef_mxJPO mqlTypeDef
263                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_MQL);
264 
265         final String progs = MqlUtil_mxJPO.execMql(_paramCache,
266                                                    InstallDataModel_mxJPO.LIST_MXUPDATE_PROGRAMS);
267         for (final String progLine : new TreeSet<String>(Arrays.asList(progs.split("\n"))))  {
268             final String[] progLineArr = progLine.split("@");
269             final String progName = progLineArr[0];
270             // do we have a JPO?
271             final AbstractObject_mxJPO prog;
272             if ("true".equalsIgnoreCase(progLineArr[1]))  {
273                 prog = jpoTypeDef.newTypeInstance(progName);
274             } else  {
275                 prog = mqlTypeDef.newTypeInstance(progName);
276             }
277             _paramCache.logInfo("check program '" + progName + "'");
278             this.registerObject(_paramCache, prog);
279             final StringBuilder cmd = new StringBuilder()
280                     .append("escape mod program \"").append(StringUtil_mxJPO.convertMql(progName)).append("\" ");
281             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.APPLICATION, _applName, cmd, false);
282             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.AUTHOR, _authorName, cmd, false);
283             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.INSTALLER, _installerName, cmd, false);
284             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.VERSION, _applVersion, cmd, false);
285             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.ORIGINALNAME, progName, cmd, false);
286             this.checkProperty(_paramCache, prog, PropertyDef_mxJPO.INSTALLEDDATE, _installedDate, cmd, true);
287             MqlUtil_mxJPO.execMql(_paramCache, cmd);
288         }
289     }

291     /***
292      * Makes the registration of MxUpdate as application.
293      *
294      * @param _paramCache   parameter cache with MX context
295      * @param _applName     used application name of the MxUpdate Update
296      *                      deployment tool
297      * @param _version      MxUpdate version
298      * @throws MatrixException if registration of MxUpdate failed
299      * @see #PARAM_PROGAPPL
300      */
301     protected void registerMxUpdate(final ParameterCache_mxJPO _paramCache,
302                                     final String _applName,
303                                     final String _version)
304             throws MatrixException
305     {
306         final String progName = _paramCache.getValueString(InstallDataModel_mxJPO.PARAM_PROGAPPL);
307 
308         _paramCache.logInfo("register MxUpdate " + _version);
309         MqlUtil_mxJPO.execMql(_paramCache, new StringBuilder()
310                 .append("escape mod prog \"").append(StringUtil_mxJPO.convertMql(progName)).append("\" ")
311                 .append("add property \"appVersion").append(StringUtil_mxJPO.convertMql(_applName)).append("\" ")
312                 .append("value \"").append(StringUtil_mxJPO.convertMql(_version)).append("\""));
313     }

315     /***
316      * The trigger group type is installed.
317      *
318      * @param _paramCache       parameter cache with MX context
319      * @param _path             installation path (needed to fetch the icons)
320      * @param _applName         used application name of the MxUpdate Update
321      *                          deployment tool
322      * @param _applVersion      Mx Update version
323      * @param _authorName       used author name
324      * @param _installerName    used installer name
325      * @param _fileDate         used file date
326      * @param _installedDate    used installed date
327      * @throws Exception if the trigger group type could not be installed
328      */
329     protected void installTriggerGroupType(final ParameterCache_mxJPO _paramCache,
330                                            final File _path,
331                                            final String _applName,
332                                            final String _applVersion,
333                                            final String _authorName,
334                                            final String _installerName,
335                                            final String _fileDate,
336                                            final String _installedDate)
337             throws Exception
338     {
339 
340         final TypeDef_mxJPO trigGrpTypeDef
341                 = _paramCache.getMapping().getTypeDef(InstallDataModel_mxJPO.TYPEDEF_TRIGGER_GROUP);
342         _paramCache.logInfo("check type '" + trigGrpTypeDef.getMxBusType() + "'");
343         final String installed = MqlUtil_mxJPO.execMql(_paramCache, new StringBuilder()
344                 .append("escape list type \"")
345                         .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType())).append('\"'));
346         if ("".equals(installed))  {
347             _paramCache.logDebug("    - create");
348             MqlUtil_mxJPO.execMql(_paramCache,
349                     new StringBuilder()
350                         .append("escape add type \"")
351                         .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType()))
352                         .append("\""));
353         }
354         _paramCache.logDebug("    - update description");
355         final StringBuilder cmd = new StringBuilder()
356                 .append("escape mod type \"")
357                 .append(StringUtil_mxJPO.convertMql(trigGrpTypeDef.getMxBusType())).append("\" ")
358                 .append("description \"Type used to group trigger definitions.\" ");
359 
360         // update type icon
361         final File file = new File(_path, trigGrpTypeDef.getIconPath());
362         if (file.exists())  {
363             _paramCache.logDebug("    - update type icon");
364             cmd.append("icon \"").append(StringUtil_mxJPO.convertMql(file.toString())).append("\" ");
365         }
366 
367         // append property settings
368         final AbstractObject_mxJPO instance = _paramCache.getMapping()
369                                             .getTypeDef(InstallDataModel_mxJPO.TYPEDEF_TYPE)
370                                             .newTypeInstance(trigGrpTypeDef.getMxBusType());
371         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.APPLICATION, _applName, cmd, false);
372         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.AUTHOR, _authorName, cmd, false);
373         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.INSTALLER, _installerName, cmd, false);
374         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.VERSION, _applVersion, cmd, false);
375         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.ORIGINALNAME,
376                            trigGrpTypeDef.getMxBusType(), cmd, false);
377         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.FILEDATE, _fileDate, cmd, true);
378         this.checkProperty(_paramCache, instance, PropertyDef_mxJPO.INSTALLEDDATE, _installedDate, cmd, true);
379         MqlUtil_mxJPO.execMql(_paramCache, cmd);
380 
381         this.registerObject(_paramCache, instance);
382     }

737                                   final AbstractObject_mxJPO _instance)
738             throws Exception
739     {
