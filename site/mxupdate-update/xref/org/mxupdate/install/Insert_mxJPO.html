195     /***
196      *
197      * @param _context  context for this request
198      * @param _args     arguements from the MX console including as first
199      *                  parameter the root path and as second parameter the
200      *                  MxUpdate version
201      * @throws Exception
202      */
203     public void mxMain(final Context _context,
204                        final String[] _args)
205             throws Exception
206     {
207         final File rootPath = new File(_args[0]);
208         final String version = _args[1];
209 
210 
211         // get installed JPOs
212         final Map<String,Date> installedProgs = this.evaluteInstalledJPOs(_context);
213 
214         final Map<String,Map<String,ClassFile>> mapPckFiles = new TreeMap<String,Map<String,ClassFile>>();
215         final Map<String,ClassFile> jpoMap = new TreeMap<String,ClassFile>();
216         this.evaluateFiles(mapPckFiles, jpoMap, rootPath, new File(""));
217 
218         // delete obsolete JPOs
219         for (final String progName : installedProgs.keySet())  {
220             if (!jpoMap.containsKey(progName))  {
221 System.out.println("delete jpo '" + progName + "'");
222                 execMql(_context,
223                         new StringBuilder()
224                                 .append("delete program '").append(progName).append('\''));
225             }
226         }
227 
228         // create new / update JPOs
229         final boolean isMqlEscapeOn = isEscapeOn(_context);
230         try  {
231             setEscapeOn(_context);
232 
233             for (final Map.Entry<String,Map<String,ClassFile>> newPckFilesEntry : mapPckFiles.entrySet())  {
234                 // evaluate JPOs from current package
235                 final Map<String,String> class2Pck = new HashMap<String,String>();
236                 for (final ClassFile classFile : newPckFilesEntry.getValue().values())  {
237                     class2Pck.put(classFile.className, classFile.completeName);
238                 }
239                 // install all JPOs from current package
240                 for (final ClassFile classFile : newPckFilesEntry.getValue().values())  {
241                     if (!installedProgs.containsKey(classFile.jpoName))  {
242 System.out.println("install jpo '" + classFile.jpoName + "'");
243                         classFile.create(_context, class2Pck, version);
244                     } else  {
245                         final Date mxDate = installedProgs.get(classFile.jpoName);
246                         if ((mxDate == null) || !mxDate.equals(classFile.getLastModified()))  {
247 System.out.println("update jpo '" + classFile.jpoName + "'");
248                             classFile.update(_context, class2Pck, version);
249                         }
250                     }
251                 }
252             }
253         }
254         finally
255         {
256             if (!isMqlEscapeOn)  {
257                 setEscapeOff(_context);
258             }
259         }
260     }

264     /***
265      * Searches for the installed JPOs and returns them including the
266      * information about the last modified date of the installed file. If a
267      * program is found which is not a JPO and does not have the extension
268      * <code>.properties</code>, an information is printed that the program is
269      * ignored.
270      *
271      * @param _context  context for this request
272      * @return map of already installed JPOs and the last modified date of the
273      *         installed file
274      * @throws MatrixException if the installed JPOs could not be evaluated
275      * @see #CMD_LISTJPOS
276      */
277     protected Map<String,Date> evaluteInstalledJPOs(final Context _context)
278             throws MatrixException
279     {
280         final String jpos = execMql(_context, CMD_LISTJPOS);
281         final Map<String,Date> installedProgs = new TreeMap<String,Date>();
282         for (final String oneJPO : jpos.split("\n"))  {
283             final String[] oneJPOArr = oneJPO.split("\t");
284             final String name = oneJPOArr[0];
285             final String modDate = (oneJPOArr.length > 1) ? oneJPOArr[1] : "";
286             final String isJava = (oneJPOArr.length > 2) ? oneJPOArr[2].trim() : "";
287             if ("TRUE".equalsIgnoreCase(isJava))  {
288                 Date mxDate;
289                 try  {
290                     mxDate = DATE_FORMAT.parse(modDate);
291                 } catch (final ParseException e)  {
292                     mxDate = null;
293                 }
294                 installedProgs.put(name, mxDate);
295             } else if (!name.endsWith(".properties")) {
296                 System.out.println("program '" + name + "' is ignored because not a JPO");
297             }
298         }
299         return installedProgs;
300     }

302     /***
303      *
304      * @param _mapPckFiles  map of packages and their JPOs
305      * @param _jpoMap       map of all JPO programs
306      * @param _rootPath     path of the root (where the JPO code is located)
307      * @param _packagePath  path of the package depending on the root path
308      */
309     protected void evaluateFiles(final Map<String,Map<String,ClassFile>> _mapPckFiles,
310                                  final Map<String,ClassFile> _jpoMap,
311                                  final File _rootPath,
312                                  final File _packagePath)
313     {
314         final File path = new File(_rootPath, _packagePath.toString());
315         for (final File file : path.listFiles())  {
316             if (file.isDirectory())  {
317                 this.evaluateFiles(_mapPckFiles, _jpoMap, _rootPath, new File(_packagePath, file.getName()));
318             } else if (file.getName().endsWith(JPO_FILE_EXTENSION))  {
319                 final ClassFile classFile = new ClassFile(_packagePath, file);
320                 Map<String,ClassFile> pckFiles = _mapPckFiles.get(classFile.pckName);
321                 if (pckFiles == null)  {
322                     pckFiles = new TreeMap<String,ClassFile>();
323                     _mapPckFiles.put(classFile.pckName, pckFiles);
324                 }
325                 pckFiles.put(classFile.className, classFile);
326                 _jpoMap.put(classFile.jpoName, classFile);
327             }
328         }

332     private class ClassFile
333     {
334         /***
335          * Link to the JPO file.
336          */
337         final File jpoFile;
338 
339         /***
340          * Name of the package in which the class is defined.
341          */
342         final String pckName;
343 
344         /***
345          * Name of the class without package but with &quot;mxJPO&quot;
346          * extension).
347          */
348         final String className;
349 
350         /***
351          * Internal MX used name of the JPO (excl. the &quot;mxJPO&quot;
352          * extension).
353          */
354         final String jpoName;
355 
356         /***
357          * Holds the complete name of the class (package name and class name).
358          */
359         final String completeName;
360 
361         ClassFile(final File _packagePath,
362                   final File _jpoFile)
363         {
364             this.jpoFile = _jpoFile;
365             this.pckName = _packagePath.getPath()
366                                        .replace(File.separatorChar, '.')
367                                        .replaceAll("^//.", "");
368             this.className = _jpoFile.getName().replaceAll(JAVA_FILE_EXTENSION + "$", "");
369             this.completeName = "".equals(this.pckName)
370                                 ? this.className
371                                 : this.pckName + "." + this.className;
372             this.jpoName = this.completeName.replaceAll(JPO_EXTENSION + "$", "");
373         }

401                 throws IOException, MatrixException
402         {
403             execMql(_context,
404                     new StringBuilder()
405                     .append("add program '").append(this.jpoName).append("' java ")
406                     .append("property \"").append(Insert_mxJPO.PROP_AUTHOR).append("\" value \"")
407                             .append(Insert_mxJPO.VALUE_AUTHOR).append("\" ")
408                     .append("property \"").append(Insert_mxJPO.PROP_INSTALLER).append("\" value \"")

411                             .append(Insert_mxJPO.DATE_INSTALLED.format(new Date())).append("\" ")
412                     .append("property \"").append(Insert_mxJPO.PROP_APPLICATION).append("\" value \"")
413                             .append(Insert_mxJPO.VALUE_APPLICATION).append("\" ")
414                     .append("property \"").append(Insert_mxJPO.PROP_ORIGINAL_NAME).append("\" value \"")
415                             .append(this.jpoName).append("\" "));
416             this.update(_context, _class2Pck, _version);
417         }

419         /***
420          * Because there is a different behavior for backslashes between MX
421          * versions, the JPO program must be included. So the source code is
422          * only updated so that the source code includes for all used classes
423          * the depending package names.
424          *
425          * @param _context      context for this request
426          * @param _class2Pck    used JPO name within the code and the related
427          *                      class name
428          * @param _version      application version
429          * @throws IOException if the JPO file could not be read
430          * @throws MatrixException if the JPO could not updated
431          * @see #getCode(Map)
432          */
433         public void update(final Context _context,
434                            final Map<String,String> _class2Pck,
435                            final String _version)
436                 throws IOException, MatrixException
437         {
438             final CharSequence code = this.getCode(_class2Pck);
439 
440             final File tmpInqFile = File.createTempFile(this.jpoFile.getName(), "");
441             try  {
442                 tmpInqFile.delete();
443                 tmpInqFile.mkdir();
444 
445                 final File file = new File(tmpInqFile, this.jpoFile.getName());
446 
447                 try  {
448                     final Writer outTCL = new FileWriter(file);
449                     outTCL.append(code.toString().trim());
450                     outTCL.flush();
451                     outTCL.close();
452 
453                     execMql(_context,
454                             new StringBuilder()
455                                 .append("insert program '").append(file.toString().replaceAll("////", "/")).append("';")
456                                 .append("mod program '").append(this.jpoName).append("'")
457                                         .append(" add property \"").append(Insert_mxJPO.PROP_FILEDATE).append("\" value \"")
458                                                 .append(Insert_mxJPO.DATE_FORMAT.format(getLastModified())).append('\"')
459                                         .append(" add property \"").append(Insert_mxJPO.PROP_VERSION).append("\" value \"")
460                                                 .append(_version).append("\";"));
461                 } finally  {
462                     file.delete();
463                 }
464             } finally  {
465                 tmpInqFile.delete();
466             }
467         }

469         /***
470          *
471          * @param _class2Pck    used JPO name within the code and the related
472          *                      class name
473          * @return code in the MQL syntax
474          * @throws IOException if the file could not be read
475          */
476         private CharSequence getCode(final Map<String,String> _class2Pck)
477                 throws IOException
478         {
479             final StringBuilder code = new StringBuilder();
480 
481             final Map<String,String> class2Pck = new HashMap<String,String>(_class2Pck);
482 
483             final BufferedReader reader = new BufferedReader(new FileReader(this.jpoFile));
484             try  {
485                 String line = reader.readLine();
486                 while (line != null)  {
487                     final Matcher pckMatch = Insert_mxJPO.PATTERN_PACKAGE.matcher(line);
488                     if (pckMatch.find())  {
489                         code.append(pckMatch.group()).append('\n');
490                     } else  {
491                         final Matcher impMatch = Insert_mxJPO.PATTERN_IMPORT.matcher(line);
492                         if (impMatch.find())  {
493                             final String impClass= impMatch.group();
494                             // extract class name from imported name
495                             final Matcher classWithoutPckMatch = Insert_mxJPO.PATTERN_JPOWITHOUTPCK.matcher(impClass);
496                             classWithoutPckMatch.find();
497                             class2Pck.put(classWithoutPckMatch.group(), impClass);
498                             code.append('\n');
499                         } else  {
500                             final Matcher matcher = Insert_mxJPO.PATTERN_JPO.matcher(line);
501                             int start = 0;
502                             final StringBuilder newLine = new StringBuilder();
503                             while (matcher.find())  {
504                                 if (this.className.equals(matcher.group()))  {
505                                     newLine.append(line.substring(start, matcher.start()))
506                                            .append(this.className);
507                                     start = matcher.start() + matcher.group().length();
508                                 } else  {
509                                     final String clazzName = class2Pck.containsKey(matcher.group())
510                                                              ? class2Pck.get(matcher.group())
511                                                              : matcher.group();
512                                     newLine.append(line.substring(start, matcher.start()))
513                                            .append(clazzName);
514                                     start = matcher.start() + matcher.group().length();
515                                 }
516                             }
517                             newLine.append(line.substring(start, line.length()));
518 
519                             code.append(newLine.toString()/*.replaceAll("////", "////////////////").replaceAll("\"", "////\"")*/).append('\n');
520                         }
521                     }
522                     line = reader.readLine();
523                 }
524             } finally  {
525                 reader.close();
526             }
527 
528             return code;
529         }
