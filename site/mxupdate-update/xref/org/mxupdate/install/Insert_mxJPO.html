41  /***
42   * The JPO is used to install the Mx Update JPO file package. Because the Mx
43   * update JPO classes are using imports and Matrix could not handle imports of
44   * JPO classes, the complete MQL insert is rewritten.
45   *
46   * @author The MxUpdate Team
47   * @version $Id: Insert_mxJPO.java 465 2009-07-12 19:46:26Z tim.moxter $
48   */
49  public class Insert_mxJPO
50  {
51      /***
52       * Name of the JPO property holding the last modified date of the file.
53       */
54      private static final String PROP_FILEDATE = "file date";

56      /***
57       * MQL command line to list the installed MxUpdate JPOs and the depending
58       * last modified date of the installed file.
59       *
60       * @see #evaluteInstalledJPOs(Context)
61       */
62      private static final String CMD_LISTJPOS
63              = "list prog \"MxUpdate,org.mxupdate.*,net.sourceforge.mxupdate.*\" "
64                      + "select name isjavaprogram property[" + Insert_mxJPO.PROP_FILEDATE + "].value "
65                      + "dump \"\t\"";

83      /***
84       *
85       * @param _context  context for this request
86       * @param _args     arguements from the MX console including as first
87       *                  parameter the root path and as second parameter the
88       *                  MxUpdate version
89       * @throws Exception if installation of the MxUpdate JPOs failed
90       */
91      public void mxMain(final Context _context,
92                         final String[] _args)
93              throws Exception
94      {
95          final File rootPath = new File(_args[0]);
96          final String version = _args[1];
97  
98          // get installed JPOs
99          final Map<String,Date> installedProgs = this.evaluteInstalledJPOs(_context);
100 
101         final org.mxupdate.util.JPOHandler_mxJPO jpoHandler
102                 = new org.mxupdate.util.JPOHandler_mxJPO(rootPath);
103 
104         // delete obsolete JPOs
105         final Set<String> newJPOs = jpoHandler.getJpoNames();
106         for (final String progName : installedProgs.keySet())  {
107             if (!newJPOs.contains(progName))  {
108 System.out.println("delete jpo '" + progName + "'");
109                 this.execMql(_context,
110                         new StringBuilder()
111                                 .append("delete program \"").append(this.convertMql(progName)).append('\"'));
112             }
113         }
114 
115         // create new / update JPOs
116         for (final ClassFile classFile : jpoHandler.getClassFiles())  {
117             // install all JPOs from current package
118             if (!installedProgs.containsKey(classFile.getJpoName()))  {
119 System.out.println("install jpo '" + classFile.getJpoName() + "'");
120                 this.update(_context, classFile, version);
121             } else  {
122                 final Date mxDate = installedProgs.get(classFile.getJpoName());
123                 if ((mxDate == null) || !mxDate.equals(classFile.getLastModified()))  {
124 System.out.println("update jpo '" + classFile.getJpoName() + "'");
125                     this.update(_context, classFile, version);
126                 }
127             }
128         }
129     }

131     /***
132      * Searches for the installed JPOs and returns them including the
133      * information about the last modified date of the installed file. If a
134      * program is found which is not a JPO and does not have the extension
135      * <code>.properties</code>, an information is printed that the program is
136      * ignored.
137      *
138      * @param _context  context for this request
139      * @return map of already installed JPOs and the last modified date of the
140      *         installed file
141      * @throws MatrixException if the installed JPOs could not be evaluated
142      * @see #CMD_LISTJPOS
143      */
144     protected Map<String,Date> evaluteInstalledJPOs(final Context _context)
145             throws MatrixException
146     {
147         final String jpos = this.execMql(_context, Insert_mxJPO.CMD_LISTJPOS);
148         final Map<String,Date> installedProgs = new TreeMap<String,Date>();
149         for (final String oneJPO : jpos.split("\n"))  {
150             final String[] oneJPOArr = oneJPO.split("\t");
151             final String name = oneJPOArr[0];
152             final String isJava = (oneJPOArr.length > 1) ? oneJPOArr[1].trim() : "";
153             final String modDate = (oneJPOArr.length > 2) ? oneJPOArr[2] : "";
154             if ("TRUE".equalsIgnoreCase(isJava))  {
155                 Date mxDate;
156                 try  {
157                     mxDate = Insert_mxJPO.DATE_FORMAT.parse(modDate);
158                 } catch (final ParseException e)  {
159                     mxDate = null;
160                 }
161                 installedProgs.put(name, mxDate);
162             } else if (!name.endsWith(".properties")) {
163 System.out.println("program '" + name + "' is ignored because not a JPO");
164             }
165         }
166         return installedProgs;
167     }

169     /***
170      * Because there is a different behavior for backslashes between MX
171      * versions, the JPO program must be included. So the source code is
172      * only updated so that the source code includes for all used classes
173      * the depending package names.
174      *
175      * @param _context      context for this request
176      * @param _classFile    class file instance (represents the JPO which must
177      *                      be updates)
178      * @param _version      application version
179      * @throws IOException if the JPO file could not be read
180      * @throws MatrixException if the JPO could not updated
181      * @see #getCode(Map)
182      */
183     public void update(final Context _context,
184                        final ClassFile _classFile,
185                        final String _version)
186             throws IOException, MatrixException
187     {
188         final CharSequence code = _classFile.getCode();
189 
190         final File tmpInqFile = File.createTempFile(_classFile.getJpoFileName(), "");
191         try  {
192             tmpInqFile.delete();
193             tmpInqFile.mkdir();
194 
195             final File file = new File(tmpInqFile, _classFile.getJpoFileName());
196 
197             try  {
198                 final Writer outTCL = new FileWriter(file);
199                 outTCL.append(code.toString().trim());
200                 outTCL.flush();
201                 outTCL.close();
202 
203                 this.execMql(_context,
204                         new StringBuilder()
205                             .append("escape insert program \"")
206                                             .append(this.convertMql(file.toString().replaceAll("////", "/")))
207                                             .append("\";")
208                             .append("escape mod program \"")
209                                             .append(this.convertMql(_classFile.getJpoName())).append('\"')
210                                     .append(" add property \"")
211                                             .append(this.convertMql(Insert_mxJPO.PROP_FILEDATE)).append("\" value \"")
212                                             .append(this.convertMql(Insert_mxJPO.DATE_FORMAT
213                                                     .format(_classFile.getLastModified())))
214                                             .append('\"'));
215             } finally  {
216                 file.delete();
217             }
218         } finally  {
219             tmpInqFile.delete();
220         }
221     }

223     /***
224      * Executes given MQL command and returns the trimmed result of the MQL
225      * execution.
226      *
227      * @param _context              MX context for this request
228      * @param _cmd                  MQL command to execute
229      * @return trimmed result of the MQL execution
230      * @throws MatrixException if MQL execution failed; includes the MQL
231      *                         command if <code>_includeMQLCommand</code> is
232      *                         set to <i>true</i>
233      */
234     protected String execMql(final Context _context,
235                              final CharSequence _cmd)
236             throws MatrixException
237     {
238         final MQLCommand mql = new MQLCommand();
239         mql.executeCommand(_context, _cmd.toString());
240         if ((mql.getError() != null) && !"".equals(mql.getError()))  {
241             throw new MatrixException(mql.getError()
242                     + "\nMQL command was:\n" + _cmd);
243         }
244         return mql.getResult().trim();
245     }

247     /***
248      * Converts given string by escaping the &quot; so that in escape mode on
249      * string could be handled with &quot; and '.
250      *
251      * @param _text     character stream to convert
252      * @return converted string
253      */
254     public String convertMql(final CharSequence _text)
255     {
256         return (_text != null)
257                ? _text.toString().replaceAll("////", "////////")
258                                  .replaceAll("//\"", "////\"")
259                : "";
260     }
