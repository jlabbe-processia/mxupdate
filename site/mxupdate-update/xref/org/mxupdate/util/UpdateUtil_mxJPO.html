39  /***
40   * Utility class for updating configuration item.

45  public final class UpdateUtil_mxJPO
46  {
47      /***
48       * String of the key within the parameter cache that administration objects
49       * compiled after they are updated.
50       */
51      private static final String PARAM_COMPILE = "Compile";
52  
53      /***
54       * The constructor is defined so that no instance of the update utility
55       * could be created.
56       */
57      private UpdateUtil_mxJPO()
58      {
59      }

61      /***
62       *
63       * @param _paramCache       parameter cache
64       * @param _clazz2names      depending on the type definition the related
65       *                          files with MX name which must be updated
66       * @param _versionInfo      defines how it is checked if an update is
67       *                          required (see {@link UpdateCheck_mxJPO})
68       * @throws Exception if update failed
69       * @see UpdateCheck_mxJPO
70       */
71      public static void update(final ParameterCache_mxJPO _paramCache,
72                                final Map<TypeDef_mxJPO,Map<File,String>> _clazz2names,
73                                final UpdateCheck_mxJPO _versionInfo)
74          throws Exception
75      {
76          // create if needed (and not in the list of existing objects)
77          UpdateUtil_mxJPO.create(_paramCache, _clazz2names);
78  
79          // update
80          final List<AbstractObject_mxJPO> compiles = new ArrayList<AbstractObject_mxJPO>();
81          final boolean compile = _paramCache.getValueBoolean(UpdateUtil_mxJPO.PARAM_COMPILE);
82          for (final TypeDef_mxJPO clazz : _paramCache.getMapping().getAllTypeDefs())  {
83              final Map<File,String> clazzMap = _clazz2names.get(clazz);
84              if (clazzMap != null)  {
85                  for (final Map.Entry<File, String> fileEntry : clazzMap.entrySet())  {
86                      final AbstractObject_mxJPO instance = clazz.newTypeInstance(fileEntry.getValue());
87                      _paramCache.logInfo("check "+instance.getTypeDef().getLogging()
88                             + " '" + fileEntry.getValue() + "'");
89  
90                      final boolean update;
91                      final String version = _paramCache.getValueBoolean(ParameterCache_mxJPO.KEY_FILEDATE2VERSION)
92                                      ? Long.toString(fileEntry.getKey().lastModified() / 1000)
93                                      : _paramCache.getValueString(ParameterCache_mxJPO.KEY_VERSION);
94                      if (_versionInfo == UpdateCheck_mxJPO.FILEDATE)  {
95                          final Date fileDate = new Date(fileEntry.getKey().lastModified());
96                          final String instDateString = instance.getPropValue(_paramCache,
97                                                                              PropertyDef_mxJPO.FILEDATE);
98                          Date instDate;
99                          if ((instDateString == null) || "".equals(instDateString))  {
100                             instDate = null;
101                         } else  {
102                             try {
103                                 instDate = StringUtil_mxJPO.parseFileDate(_paramCache, instDateString);
104                             } catch (final ParseException e) {
105                                 instDate = null;
106                             }
107                         }
108                         if (fileDate.equals(instDate))  {
109                             update = false;
110                         } else  {
111                             update = true;
112                             _paramCache.logDebug("    - update to version from " + fileDate);
113                         }
114                     } else if (_versionInfo == UpdateCheck_mxJPO.VERSION)  {
115                         final String instVersion = instance.getPropValue(_paramCache,
116                                                                         PropertyDef_mxJPO.VERSION);
117                         if (instVersion.equals(version))  {
118                             update = false;
119                         } else  {
120                             update = true;
121                             if (_paramCache.getValueBoolean(ParameterCache_mxJPO.KEY_FILEDATE2VERSION))  {
122                                 _paramCache.logDebug("    - update to version from "
123                                        + new Date(fileEntry.getKey().lastModified()));
124                             } else  {
125                                 _paramCache.logDebug("    - update to version " + version);
126                             }
127                         }
128                     } else  {
129                         update = true;
130                         _paramCache.logDebug("    - update");
131                     }
132                     // execute update
133                     if (update && UpdateUtil_mxJPO.updateOne(_paramCache, instance, fileEntry.getKey(), version) && compile)  {
134                         compiles.add(instance);
135                     }
136                 }
137             }
138         }
139 
140         // compile
141         UpdateUtil_mxJPO.compile(_paramCache, compiles);
142     }

144     /***
145      * Searches for given <code>_typeDefs</code> related existing CI's.
146      *
147      * @param _paramCache   parameter cache (used to get the MX context)
148      * @param _typeDefs     set of type definitions for which existing CI's are
149      *                      searched
150      * @return found existing CI's
151      * @throws Exception if existing CI's could not be found
152      */
153     protected static Map<TypeDef_mxJPO,Set<String>> getExistingCIs(final ParameterCache_mxJPO _paramCache,
154                                                                    final Set<TypeDef_mxJPO> _typeDefs)
155         throws Exception
156     {
157         final Map<TypeDef_mxJPO,Set<String>> existingNames = new HashMap<TypeDef_mxJPO,Set<String>>();
158         for (final TypeDef_mxJPO clazz : _typeDefs)  {
159             if (!existingNames.containsKey(clazz))  {
160                 final AbstractObject_mxJPO instance = clazz.newTypeInstance(null);
161                 existingNames.put(clazz, instance.getMxNames(_paramCache));
162             }
163         }
164         return existingNames;
165     }

167     /***
168      * Creates not existing CI's.
169      *
170      * @param _paramCache       parameter cache (used to get the MX context)
171      * @param _clazz2names      depending on the type definition the related
172      *                          files with MX name which must be updated
173      * @throws Exception if create failed
174      */
175     protected static void create(final ParameterCache_mxJPO _paramCache,
176                                  final Map<TypeDef_mxJPO,Map<File,String>> _clazz2names)
177         throws Exception
178     {
179         // evaluate for existing administration objects
180         final Map<TypeDef_mxJPO,Set<String>> existingNames = UpdateUtil_mxJPO.getExistingCIs(_paramCache, _clazz2names.keySet());
181 
182         // create if needed (and not in the list of existing objects)
183         for (final TypeDef_mxJPO clazz : _paramCache.getMapping().getAllTypeDefs())  {
184             final Map<File,String> clazzMap = _clazz2names.get(clazz);
185             if (clazzMap != null)  {
186                 for (final Map.Entry<File, String> fileEntry : clazzMap.entrySet())  {
187                     final Set<String> existings = existingNames.get(clazz);
188                     if (!existings.contains(fileEntry.getValue()))  {
189                         final AbstractObject_mxJPO instance = clazz.newTypeInstance(fileEntry.getValue());
190                         _paramCache.logInfo("create "+instance.getTypeDef().getLogging() + " '" + fileEntry.getValue() + "'");
191                         instance.create(_paramCache);
192                     }
193                 }
194             }
195         }
196     }

198     /***
199      * Updates on <code>_file</code> for <code>_instance</code> within a
200      * transaction (if a transaction was not already started).
201      *
202      * @param _paramCache       parameter cache (used to get the MX context)
203      * @param _instance         instance to update
204      * @param _file             file with target definition
205      * @param _version          version to update
206      * @return <i>true</i> if update was done; otherwise <i>false</i>
207      * @throws Exception if update of the instance failed
208      */
209     protected static boolean updateOne(final ParameterCache_mxJPO _paramCache,
210                                        final AbstractObject_mxJPO _instance,
211                                        final File _file,
212                                        final String _version)
213         throws Exception
214     {
215         boolean commit = false;
216         final boolean transActive = _paramCache.getContext().isTransactionActive();
217         try  {
218             if (!transActive)  {
219                 _paramCache.getContext().start(true);
220             }
221             _instance.update(_paramCache, _file, _version);
222             if (!transActive)  {
223                 _paramCache.getContext().commit();
224             }
225             commit = true;
226         } finally  {
227             if (!commit && !transActive && _paramCache.getContext().isTransactionActive())  {
228                 _paramCache.getContext().abort();
229             }
230         }
231         return commit;
232     }

234     /***
235      * Compiles given <code>_compiles</code> files.
236      *
237      * @param _paramCache   parameter cache (used for logging purposes)
238      * @param _compiles     list of files to compile
239      */
240     protected static void compile(final ParameterCache_mxJPO _paramCache,
241                                   final List<AbstractObject_mxJPO> _compiles)
242     {
243         for (final AbstractObject_mxJPO instance : _compiles)  {
244             try  {
245                 if (instance.compile(_paramCache))  {
246                     _paramCache.logInfo("compile " + instance.getTypeDef().getLogging()
247                            + " '" + instance.getName() + "'");
248                 }
249             } catch (final Exception e)  {
250                 _paramCache.logError("compile of " + instance.getTypeDef().getLogging()
251                        + " '" + instance.getName() + "' failed:\n" + e.toString());
252             }
253         }
254     }
