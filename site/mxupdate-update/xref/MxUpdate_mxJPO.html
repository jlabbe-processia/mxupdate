175     /***
176      *
177      * @param _paramCache   parameter cache
178      * @throws MatrixException
179      */
180     private void prepareParams(final ParameterCache_mxJPO _paramCache)
181             throws MatrixException
182     {
183         this.description.clear();
184         this.allParams.clear();
185         this.paramsModes.clear();
186         this.paramsParameters.clear();
187         this.paramsTypeDefs.clear();
188         this.paramsUpdateChecks.clear();
189 
190         ////////////////////////////////////////////////////////////////////////
191         // parameters
192 
193         for (final ParameterDef_mxJPO parameter : _paramCache.getMapping().getAllParameterDefs())  {
194             if (parameter.getParameterList() != null)  {
195                 for (final String param : parameter.getParameterList())  {
196                     final String paramStr = (param.length() > 1)
197                                             ? "--" + param
198                                             : "-" + param;
199                     this.paramsParameters.put(paramStr, parameter);
200                 }
201                 final StringBuilder desc = new StringBuilder().append(parameter.getParameterDesc());
202                 if ((parameter.getDefaultValue() != null)
203                         && (parameter.getType() != ParameterDef_mxJPO.Type.BOOLEAN))  {
204                     desc.append('\n').append("(Default '");
205                     if (parameter.getType() == ParameterDef_mxJPO.Type.LIST)  {
206                         desc.append(parameter.getDefaultValue().replaceAll(",", ", "));
207                     } else  {
208                         desc.append(parameter.getDefaultValue());
209                     }
210                     desc.append("')");
211                 }
212                 this.appendDescription(desc,
213                                        parameter.getParameterList(),
214                                        parameter.getParameterArgs());
215             }
216         }
217 
218         ////////////////////////////////////////////////////////////////////////
219         // modes
220 
221         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
222             for (final String param : mode.getParameterList(_paramCache))  {
223                 final String paramStr = (param.length() > 1)
224                                         ? "--" + param
225                                         : "-" + param;
226                 this.paramsModes.put(paramStr, mode);
227             }
228             this.appendDescription(mode.getParameterDesc(_paramCache),
229                                    mode.getParameterList(_paramCache),
230                                    null);
231         }
232 
233         ////////////////////////////////////////////////////////////////////////
234         // update checks
235 
236         for (final UpdateCheck_mxJPO updateCheck : UpdateCheck_mxJPO.values())  {
237             for (final String param : updateCheck.getParameterList(_paramCache))  {
238                 if (param.length() == 1)  {
239                     this.paramsUpdateChecks.put("-" + param, updateCheck);
240                 } else  {
241                     this.paramsUpdateChecks.put("--" + param, updateCheck);
242                 }
243             }
244             this.appendDescription(updateCheck.getParameterDesc(_paramCache),
245                                    updateCheck.getParameterList(_paramCache),
246                                    null);
247         }
248 
249         ////////////////////////////////////////////////////////////////////////
250         // type definitions
251 
252         // first create list all type definitions which could be used...
253         final Set<TypeDef_mxJPO> all = new HashSet<TypeDef_mxJPO>();
254         for (final TypeDef_mxJPO typeDef : _paramCache.getMapping().getAllTypeDefs())  {
255             if (!typeDef.isBusCheckExists() || typeDef.existsBusType(_paramCache))  {
256                 all.add(typeDef);
257                 if (typeDef.getParameterList() != null)  {
258                     this.defineParameter(null,
259                             typeDef,
260                             typeDef.getParameterDesc(),
261                             typeDef.getParameterList());
262                 }
263                 if (typeDef.getParameterListOpp() != null)  {
264                     for (final String param : typeDef.getParameterListOpp())  {
265                         final String paramStr = (param.length() == 1)
266                                                 ? "-" + param
267                                                 : "--" + param;
268                         this.paramsTypeDefsOpp.put(paramStr, typeDef);
269                     }
270                     // store description
271                     this.appendDescription(typeDef.getParameterDescOpp(),
272                             typeDef.getParameterListOpp(),
273                             Arrays.asList(new String[]{"MATCH"}));
274                 }
275             }
276         }
277 
278         // define type definition group parameters depending on existing type
279         // definitions (and only if at minimum one type definition of the
280         // groups exists...)
281         for (final TypeDefGroup_mxJPO group : _paramCache.getMapping().getAllTypeDefGroups())  {
282             final Set<TypeDef_mxJPO> curTypeDefs = new HashSet<TypeDef_mxJPO>();
283             for (final String typeDefName : group.getTypeDefList())  {
284                 final TypeDef_mxJPO typeDef = _paramCache.getMapping().getTypeDef(typeDefName);
285                 if (all.contains(typeDef))  {
286                     curTypeDefs.add(typeDef);
287                 }
288             }
289             if (!curTypeDefs.isEmpty())  {
290                 this.defineParameter(curTypeDefs,
291                                      null,
292                                      group.getParameterDesc(),
293                                      group.getParameterList());
294             }
295         }
296    }

298     /***
299      *
300      * @param _paramsList       to which parameter list must the parameter
301      *                          appended
302      * @param _clazz            class implementing the import / export (or
303      *                          <code>null</code> if the complete parameter
304      *                          list must be used)
305      * @param _description      description of the parameter
306      * @param _longParams       list of long parameters strings
307      */
308     private void defineParameter(final Collection<TypeDef_mxJPO> _paramsList,
309                                  final TypeDef_mxJPO _clazz,
310                                  final String _description,
311                                  final Collection<String> _longParams)
312     {
313         final Collection<TypeDef_mxJPO> tmp;
314         if (_clazz == null)  {
315             tmp = _paramsList;
316         } else  {
317             tmp = new HashSet<TypeDef_mxJPO>();
318             // add to given set of parameters
319             if (_paramsList != null)  {
320                 _paramsList.add(_clazz);
321             }
322             tmp.add(_clazz);
323         }
324 
325         // check for long parameters and test for double definition
326         for (final String param : _longParams)  {
327             final String paramStr = (param.length() == 1)
328                                     ? "-" + param
329                                     : "--" + param;
330             this.paramsTypeDefs.put(paramStr, tmp);
331         }
332 
333         // store description
334         this.appendDescription(_description,
335                                _longParams,
336                                Arrays.asList(new String[]{"MATCH"}));
337     }

339     /***
340      * Appends a description for a defined list of parameters. If a parameter
341      * is defined twice, an error is thrown.
342      *
343      * @param _description  description to append
344      * @param _params       related parameters
345      * @param _args         text of the arguments for the list of parameters
346      *                      (or <code>null</code> if not defined)
347      * @see #allParams
348      * @see #description
349      */
350     private void appendDescription(final CharSequence _description,
351                                    final Collection<String> _params,
352                                    final Collection<String> _args)
353     {
354         // check for double parameter definitions
355         for (final String param : _params)  {
356             if (this.allParams.contains(param))  {
357                 throw new Error("double definition of parameter '" + param
358                         + "' with description '" + _description + "'");
359             }
360             this.allParams.add(param);
361         }
362 
363         // check if first parameter is not a short parameter
364         final String firstParam = _params.iterator().next();
365         final StringBuilder param = new StringBuilder();
366         if (firstParam.length() > 1)  {
367             param.append('-').append(firstParam.charAt(0)).append('_');
368         }
369 
370 
371         // append all parameters to the description text
372         boolean first = true;
373         for (final String paramString : _params)  {
374             if (first)  {
375                 first = false;
376             } else  {
377                 param.append(',');
378             }
379             if (paramString.length() > 1)  {
380                 param.append('-');
381             }
382             param.append('-').append(paramString);
383         }
384 
385         // append arguments
386         if (_args != null)  {
387             for (final String arg : _args)  {
388                 param.append(" <").append(arg).append('>');
389             }
390         }
391 
392         if (_description == null)  {
393             throw new Error("descriptions for parameter " + param + " not defined!");
394         }
395 
396         this.description.put(param.toString(), _description.toString());
397     }

399     public void mxMain(final Context _context,
400                        final String... _args)
401             throws Exception
402     {
403         try {
404             final ParameterCache_mxJPO paramCache = new ParameterCache_mxJPO(_context, false);
405 
406             this.prepareParams(paramCache);
407 
408             // to be sure....
409             MqlUtil_mxJPO.execMql(_context, "verbose off", false);
410 
411             Mode_mxJPO mode = null;
412 
413             final Map<Collection<TypeDef_mxJPO>,List<String>> clazz2matches
414                     = new HashMap<Collection<TypeDef_mxJPO>,List<String>>();
415             final Map<TypeDef_mxJPO,List<String>> clazz2matchesOpp
416                     = new HashMap<TypeDef_mxJPO,List<String>>();
417 
418             boolean unknown = false;
419 
420             UpdateCheck_mxJPO versionInfo = null;
421 
422 
423             for (int idx = 0; idx < _args.length; idx++)  {
424                 final String arg = _args[idx];
425                 if (this.paramsTypeDefs.containsKey(arg))  {
426                     final Collection<TypeDef_mxJPO> clazzes = this.paramsTypeDefs.get(arg);
427                     final String name = _args[++idx];
428                     List<String> names = clazz2matches.get(clazzes);
429                     if (names == null)  {
430                         names = new ArrayList<String>();
431                         clazz2matches.put(clazzes, names);
432                     }
433                     names.add(name);
434                 } else if (this.paramsTypeDefsOpp.containsKey(arg))  {
435                     final TypeDef_mxJPO typeDef = this.paramsTypeDefsOpp.get(arg);
436                     final String name = _args[++idx];
437                     List<String> names = clazz2matchesOpp.get(typeDef);
438                     if (names == null)  {
439                         names = new ArrayList<String>();
440                         clazz2matchesOpp.put(typeDef, names);
441                     }
442                     names.add(name);
443                 } else if (this.paramsModes.containsKey(arg)) {
444                     final Mode_mxJPO tmpMode = this.paramsModes.get(arg);
445                     if (mode != null)  {
446                         if ((mode == Mode_mxJPO.HELP) || (tmpMode == Mode_mxJPO.HELP))  {
447                             mode = Mode_mxJPO.HELP;
448                         } else  {
449                             throw new Error("A mode is already defined and could not be defined twice!");
450                         }
451                     } else  {
452                         mode = tmpMode;
453                     }
454                 } else if (this.paramsParameters.containsKey(arg))  {
455                     idx = paramCache.evalParameter(this.paramsParameters.get(arg),
456                                                    _args,
457                                                    idx);
458                 } else if (this.paramsUpdateChecks.containsKey(arg))  {
459                     versionInfo = this.paramsUpdateChecks.get(arg);
460                 } else  {
461                     unknown = true;
462                     paramCache.logError("unknown pararameter "  + arg);
463                 }
464             }
465 
466             if (unknown || (Mode_mxJPO.HELP == mode) || (mode == null))  {
467                 this.printHelp(paramCache);
468             } else if (Mode_mxJPO.EXPORT == mode)  {
469                 this.export(paramCache, clazz2matches, clazz2matchesOpp);
470             } else if (Mode_mxJPO.IMPORT == mode)  {
471                 this.update(paramCache, clazz2matches, clazz2matchesOpp, versionInfo);
472             } else if (Mode_mxJPO.DELETE == mode)  {
473                 this.delete(paramCache, clazz2matches, clazz2matchesOpp);
474             }
475 
476         } catch (final Exception e)  {
477             e.printStackTrace(System.out);
478             throw e;
479         }
480 
481     }

483     /***
484      * Prints the help for the MxUpdate functionality.
485      *
486      * @param _paramCache   parameter cache
487      * @throws IOException if the help could not be printed
488      * @see #description
489      */
490     private void printHelp(final ParameterCache_mxJPO _paramCache)
491             throws IOException
492     {
493         final String prefix = _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_PREFIX);
494         final int lengthLine = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHLINE);
495         final int lengthParam = _paramCache.getValueInteger(MxUpdate_mxJPO.PARAM_HELP_LENGTHPARAMS);
496         final Appendable out = System.out;
497 
498         // first print the usage text with the mode parameters
499         out.append('\n')
500            .append(prefix)
501            .append(_paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_USAGE).trim())
502            .append(' ');
503         boolean first = true;
504         for (final Mode_mxJPO mode : Mode_mxJPO.values())  {
505             final String param = mode.getParameterList(_paramCache).iterator().next();
506             if (first)  {
507                 first = false;
508             } else  {
509                 out.append(" | ");
510             }
511             if (param.length() > 1)  {
512                 out.append('-');
513             }
514             out.append('-');
515             out.append(param);
516         }
517         out.append(" | ... \n\n");
518 
519         // print all parameters with description
520         for (final Map.Entry<String,String> descLine : this.description.entrySet())  {
521 
522             StringBuilder line = new StringBuilder().append(prefix);
523 
524             // append list of parameters
525             if ((descLine.getKey().length() > 3) && (descLine.getKey().charAt(2) == '_'))  {
526                 line.append("   ").append(descLine.getKey().substring(3));
527             } else  {
528                 line.append(descLine.getKey());
529             }
530 
531             // append spaces behind parameter list
532             if (line.length() > lengthParam)  {
533                 out.append(line).append('\n');
534                 line = new StringBuilder().append(prefix);
535             }
536             for (int i = line.length(); i < lengthParam; i++)  {
537                 line.append(' ');
538             }
539 
540             // append parameter description
541             first = true;
542             for (final String partDesc : descLine.getValue().toString().split("\n"))  {
543                 if (first == true)  {
544                     first = false;
545                 } else  {
546                     out.append(line).append('\n');
547                     line = new StringBuilder().append(prefix);
548                     for (int i = 0; i < lengthParam; i++)  {
549                         line.append(' ');
550                     }
551                 }
552                 for (final String desc : partDesc.split(" "))  {
553                     if (!"".equals(desc))  {
554                         if (line.length() > lengthLine)  {
555                             out.append(line).append('\n');
556                             line = new StringBuilder().append(prefix);
557                             for (int i = 0; i < lengthParam; i++)  {
558                                 line.append(' ');
559                             }
560                         }
561                         line.append(' ').append(desc);
562                     }
563                 }
564             }
565             out.append(line).append('\n');
566         }
567 
568         // print copyright
569         for (final String copyright : _paramCache.getValueString(MxUpdate_mxJPO.PARAM_HELP_COPYRIGHT).split("\n"))  {
570             out.append(prefix).append(copyright).append('\n');
571         }
572         out.append('\n');
573     }

575     /***
576      * Exports matching administration objects to given path.
577      *
578      * @param _paramCache       parameter cache
579      * @param _clazz2matches    map of update classes and the depending
580      *                          list of string which must match to export
581      * @param _clazz2matchesOpp map of update classes and the depending list of
582      *                          match string of MX names which must be excluded
583      *                          from export
584      * @throws Exception if none path or more than one path is defined or if
585      *                   the export failed
586      */
587     protected void export(final ParameterCache_mxJPO _paramCache,
588                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
589                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp)
590             throws Exception
591     {
592         final Collection<String> paths = _paramCache.getValueList(MxUpdate_mxJPO.PARAM_PATH);
593 
594         // check for definition of min. / max. one path
595         if (paths.isEmpty())  {
596             throw new Exception("no path is defined, but required for the export!");
597         }
598         if (paths.size() > 1)  {
599             throw new Exception("more than one path is defined, but maximum is allowed for the export!");
600         }
601         final String pathStr = paths.iterator().next();
602 
603         // evaluate all matching administration objects
604         final Map<TypeDef_mxJPO,Set<String>> clazz2names
605                 = this.getMatching(_paramCache, _clazz2matches, _clazz2matchesOpp);
606 
607         // export
608         for (final Map.Entry<TypeDef_mxJPO,Set<String>> entry : clazz2names.entrySet())  {
609             for (final String name : entry.getValue())  {
610                 final AbstractObject_mxJPO instance = entry.getKey().newTypeInstance(name);
611                 final File path = new File(pathStr + File.separator + instance.getPath());
612                 _paramCache.logInfo("export "+instance.getTypeDef().getLogging() + " '" + name + "'");
613                 instance.export(_paramCache, path);
614             }
615         }
616     }

646      *
647      * @param _paramCache       parameter cache
648      * @param _clazz2matches    map of update classes and the depending

655     protected void delete(final ParameterCache_mxJPO _paramCache,
656                           final Map<Collection<TypeDef_mxJPO>,List<String>> _clazz2matches,
657                           final Map<TypeDef_mxJPO,List<String>> _clazz2matchesOpp)
658             throws Exception
659     {
660         // check for definition of min. / max. one path
661         if (_paramCache.getValueList(MxUpdate_mxJPO.PARAM_PATH).isEmpty())  {

828                             allowed = false;
829                             break;
830                         }
831                     }
832                     if (allowed)  {
