21  package org.mxupdate.eclipse;
22  
23  import java.text.MessageFormat;
24  import java.util.MissingResourceException;
25  import java.util.ResourceBundle;
26  
27  /***
28   * Utility class to handle message translation for the MxUpdate eclipse
29   * plug-in.
30   *
31   * @author The MxUpdate Team
32   * @version $Id: Messages.java 689 2010-01-31 14:51:02Z tim.moxter $
33   */
34  public final class Messages
35  {
36      /***
37       * Name of the property bundle holding all messages.
38       */
39      private static final String BUNDLE_NAME = "src/main/resources/plugin"; //$NON-NLS-1$
40  
41      /***
42       * Resource bundle to the MxUpdate eclipse plug-in properties.
43       */
44      private static final ResourceBundle RESOURCE_BUNDLE
45              = ResourceBundle.getBundle(Messages.BUNDLE_NAME);
46  
47      /***
48       * Constructor is defined to avoid creating an instance of this message
49       * utility class.
50       */
51      private Messages()
52      {
53      }
54  
55      /***
56       * Translated given key by using the resource bundle
57       * {@link #RESOURCE_BUNDLE}. The translated key is then formated (with the
58       * message formatter {@link MessageFormat}). If the key could not be
59       * translated, the key itself surrounded with <code>!!!</code> is returned.
60       *
61       * @param _key          key to translate
62       * @param _arguments    arguments to used to fill into the translated
63       *                      string
64       * @return translated string for key; or if key could not be translated,
65       *         original key surrounded with &quot;!!!&quot;.
66       * @see #RESOURCE_BUNDLE
67       */
68      public static String getString(final String _key,
69                                     final Object... _arguments)
70      {
71          String ret;
72          try {
73              ret = Messages.RESOURCE_BUNDLE.getString(_key);
74          } catch (final MissingResourceException e) {
75              ret = "!!!" + _key + "!!!";
76          }
77          return MessageFormat.format(ret, _arguments);
78      }
79  }
