113     /***
114      * If this page is shown the search for configuration items is performed
115      * depending on the defined values from the {@link #step1 first step} of
116      * the wizard.
117      *
118      * @param _visible  must be <i>true</i> that the search is executed
119      */
120     @Override()
121     public void setVisible(final boolean _visible)
122     {
123         if (_visible)  {
124             final List<ISearchItem> items = Activator.getDefault().getAdapter().search(this.step1.getTypeDefs(), this.step1.getMatch());
125 
126             this.table.removeAll();
127             for (final ISearchItem item : items)  {
128                 final TableItem tableItem = new TableItem(this.table, SWT.NONE);
129                 tableItem.setText(new String[]{item.getName(), item.getFileName(), item.getFilePath()});
130                 tableItem.setData(item);
131 
132                 final String typeDef = item.getTypeDef();
133                 if (!this.images.containsKey(typeDef))  {
134                     final ImageDescriptor imageDescr = Activator.getDefault().getAdapter().getImageDescriptor(typeDef);
135                     if (imageDescr != null)  {
136                         this.images.put(typeDef, imageDescr.createImage());
137                     } else  {
138                         this.images.put(typeDef, null);
139                     }
140                 }
141                 tableItem.setImage(this.images.get(typeDef));
142             }
143         }
144         super.setVisible(_visible);
145     }

260     /***
261      * Checks if at minimum one configuration item from {@link #table} is
262      * selected. If not an error message is shown.
263      *
264      * @return <i>true</i> if at minimum on configuration item is selected;
265      *         otherwise <i>false</i>
266      */
267     @Override()
268     protected boolean validateSourceGroup()
269     {
270         final boolean ret;
271         if ((this.table != null) && this.table.getSelectionCount() == 0)  {
272             this.setErrorMessage(Messages.getString("ImportWizard.Wizard.Step2.TableError")); //$NON-NLS-1$
273             ret = false;
274         } else  {
275             ret = true;
276         }
277         return ret;
278     }
