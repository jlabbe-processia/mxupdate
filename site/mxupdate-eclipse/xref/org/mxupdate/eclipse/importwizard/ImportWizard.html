44  /***
45   * Import wizard for configuration items directly from MX.
46   *
47   * @author The MxUpdate Team
48   * @version $Id: ImportWizard.java 714 2010-02-07 11:46:06Z tim.moxter $
49   */
50  public class ImportWizard
51      extends Wizard
52      implements IImportWizard
53  {
54      /***
55       * Step 1 page of the import wizard.
56       */
57      private Step1TypeNamePage step1;
58  
59      /***
60       * Step 2 page of the import wizard.
61       */
62      private Step2ConfigurationItemsPage step2;
63  
64      /***
65       * Performs the finish of the import for configuration items from MX. The
66       * configuration items are exported from MX and new created in Eclipse
67       * workspace.
68       *
69       * @return always <i>true</i>
70       */
71      @Override()
72      public boolean performFinish()
73      {
74          final Set<ISearchItem> selectedItems = this.step2.getSelectedItems();
75          final IPath targetPath = this.step2.getTargetPath();
76  
77          final IRunnableWithProgress op = new IRunnableWithProgress() {
78              public void run(final IProgressMonitor _monitor)
79              {
80                  for (final ISearchItem selected : selectedItems)  {
81  
82                      Activator.getDefault().getConsole().logInfo(Messages.getString("ImportWizard.WizardPerformFinishLog", //$NON-NLS-1$
83                                                                                     selected.getFileName()));
84  
85                      // export file from MX
86                      final IExportItem exportItem = Activator.getDefault().getAdapter().export(selected.getTypeDef(), selected.getName());
87                      // and create file in Eclipse
88                      if (exportItem != null)  {
89                          final IPath newPath = targetPath.append(exportItem.getFilePath()).append(exportItem.getFileName());
90                          final IFile newFileHandle = ResourcesPlugin.getWorkspace().getRoot().getFile(newPath);
91  
92                          final CreateFileOperation op = new CreateFileOperation(
93                                  newFileHandle,
94                                  null,
95                                  new ByteArrayInputStream(exportItem.getContent().getBytes()),
96                                  Messages.getString("ImportWizard.WizardPerformFinishTitle")); //$NON-NLS-1$
97                          try
98                          {
99                              op.execute(_monitor, WorkspaceUndoUtil.getUIInfoAdapter(ImportWizard.this.getContainer().getShell()));
100                         }
101                         catch(final ExecutionException ex)
102                         {
103                             Activator.getDefault().getConsole().logError(Messages.getString("ImportWizard.WizardPerformFinishException"), ex); //$NON-NLS-1$
104                         }
105                     }
106                 }
107             }
108         };
109         try
110         {
111             this.getContainer().run(true, true, op);
112         }
113         catch(final InterruptedException ex)
114         {
115             Activator.getDefault().getConsole().logError(Messages.getString("ImportWizard.WizardPerformFinishInterrupted")); //$NON-NLS-1$
116         }
117         catch(final InvocationTargetException ex)
118         {
119             Activator.getDefault().getConsole().logError(Messages.getString("ImportWizard.WizardPerformFinishException"), ex); //$NON-NLS-1$
120         }
121 
122         return true;
123     }
124 
125     /***
126      * Initializes the two different steps for this wizard.
127      *
128      * @param _workbench    related workbench
129      * @param _selection    structured selection
130      * @see #step1

132      */
133     public void init(final IWorkbench _workbench,

136         this.setWindowTitle(Messages.getString("ImportWizard.WizardHeader")); //$NON-NLS-1$
137         this.setNeedsProgressMonitor(true);
138         this.step1 = new Step1TypeNamePage();
139         this.step2 = new Step2ConfigurationItemsPage(this.step1, _selection);
140     }
141 
142     /***
143      * Appends {@link #step1 first} and {@link #step2 second step} page to this
144      * wizard.
145      *
146      * @see #step1
147      * @see #step2
148      */
149     @Override()

151         super.addPages();
152         this.addPage(this.step1);
153         this.addPage(this.step2);
154     }
