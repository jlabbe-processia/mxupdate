211     /***
212      * Executes current string in the input text field {@link #textInput}. The
213      * string is added to the history entries of the last entry of the history
214      * is not the same or if the string is not an empty string. This string is
215      * added to the output text field together with the current amount of
216      * input strings defined with {@link #count}. The returned values from the
217      * data base are added to the output text field {@link #textOutput} with
218      * color {@link #colorOuput} or, if an error was thrown, with color
219      * {@link #colorError}.
220      *
221      * @see #textInput
222      * @see #history
223      * @see #textOutput
224      * @see #colorOuput
225      * @see #colorError
226      */
227     private void execute()
228     {
229         final String inputText = this.textInput.getText();
230 
231         // store new shell command only if not equal and not already stored
232         // (as last entry)
233         if (!"".equals(inputText) //$NON-NLS-1$
234                 && ((this.history.size() == 0) || !this.history.get(this.history.size() - 1).equals(inputText)))  {
235             this.history.add(inputText);
236         }
237         this.historyPos = this.history.size();
238 
239         // remove all history entries if more than maximum stored
240         while (this.historyPos > this.historyMax)  {
241             this.history.remove(0);
242             this.historyPos = this.history.size();
243         }
244 
245         // reset input text field
246         this.textInput.setText(""); //$NON-NLS-1$
247 
248         // write input command into output text field
249         this.count++;
250         this.textOutput.append("\n" + this.count + "> " + inputText ); //$NON-NLS-1$ //$NON-NLS-2$
251         this.textOutput.append("\n"); //$NON-NLS-1$
252 
253         final StyleRange txtStyleRange = new StyleRange();
254         txtStyleRange.start = this.textOutput.getCharCount();
255         try {
256             this.textOutput.append(Activator.getDefault().getAdapter().execute(inputText));
257             txtStyleRange.foreground = this.colorOuput;
258         } catch (final Exception e1)  {
259             this.textOutput.append(e1.toString());
260             txtStyleRange.foreground = this.colorError;
261         }
262         final int charCount = this.textOutput.getCharCount();
263         txtStyleRange.length = charCount - txtStyleRange.start;
264         this.textOutput.setStyleRange(txtStyleRange);
265         this.textOutput.append("\n"); //$NON-NLS-1$
266 
267         // go to last line
268         this.textOutput.setSelection(charCount);
269     }

271     /***
272      * Shows previous entry of the history in the input text field.
273      *
274      * @see #history
275      * @see #historyPos
276      * @see #textInput
277      */
278     private void prevHistoryEntry()
279     {
280         if (this.historyPos > 0)  {
281             this.historyPos--;
282             this.textInput.setText(this.history.get(this.historyPos));
283             this.textInput.setSelection(this.history.get(this.historyPos).length());
284         }
285     }
286 
287     /***
288      * Shows next entry of the history in the input text field.
289      *
290      * @see #history
291      * @see #historyPos
292      * @see #textInput
293      */
294     private void nextHistoryEntry()
295     {
296         // get next history entry (into input text field)
297         if (this.historyPos < this.history.size())  {
298             this.historyPos++;
299             this.textInput.setText(this.history.get(this.historyPos));
300             this.textInput.setSelection(this.history.get(this.historyPos).length());
301         // if last index => input text must be cleaned
302         } else if (this.historyPos == this.history.size())  {
303             this.textInput.setText(""); //$NON-NLS-1$
304             this.textInput.setSelection(0);
305         }
306     }
