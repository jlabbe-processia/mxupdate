21  package org.mxupdate.eclipse.console;
22  
23  import java.util.Collection;
24  
25  import org.eclipse.swt.graphics.RGB;
26  import org.mxupdate.eclipse.Messages;
27  import org.mxupdate.eclipse.preferences.AbstractWorkspacePreference;
28  
29  /***
30   * Class with all console specific preferences to define the colors depending
31   * on the logging level.
32   *
33   * @author The MxUpdate Team
34   * @version $Id: ConsolePreference.java 688 2010-01-31 13:33:16Z tim.moxter $
35   */
36  public final class ConsolePreference
37          extends AbstractWorkspacePreference<ConsolePreference>
38  {
39      /***
40       * Debug log level with blue color.
41       */
42      public static final ConsolePreference DEBUG;
43  
44      /***
45       * Error log level with dark red color.
46       */
47      public static final ConsolePreference ERROR;
48  
49      /***
50       * Info log level with black color.
51       */
52      public static final ConsolePreference INFO;
53  
54      /***
55       * Trace log level with gray color.
56       */
57      public static final ConsolePreference TRACE;
58  
59      /***
60       * Warning log level with dark magenta color.
61       */
62      public static final ConsolePreference WARN;
63  
64      static
65      {
66          ERROR = new ConsolePreference("CONSOLE_COLOR_ERROR", RGB.class, new RGB(128, 0, 0));
67          WARN = new ConsolePreference("CONSOLE_COLOR_WARN", RGB.class, new RGB(255, 0, 255));
68          INFO = new ConsolePreference("CONSOLE_COLOR_INFO", RGB.class, new RGB(0, 0, 0));
69          DEBUG = new ConsolePreference("CONSOLE_COLOR_DEBUG", RGB.class, new RGB(0, 0, 255));
70          TRACE = new ConsolePreference("CONSOLE_COLOR_TRACE", RGB.class, new RGB(192, 192, 192));
71      }
72  
73  
74      /***
75       *
76       * @param _name             name of the shell preference
77       * @param _defaultClazz     related class of the default value
78       * @param _defaultValue     value of the default value
79       */
80      private ConsolePreference(final String _name,
81                                final Class<?> _defaultClazz,
82                                final Object _defaultValue)
83      {
84          super(_name, _defaultClazz, _defaultValue, ConsolePreference.class);
85      }
86  
87      /***
88       * Returns the string used in the console as prefix for each line to show
89       * the logging level. The console text is translated by using the message
90       * key
91       * &quot;ConsolePreference.CONSOLE_TEXT_&lt;LogLevel&gt;.ConsoleText&quot;.
92       *
93       * @return console text string
94       */
95      public String getConsoleText()
96      {
97          return Messages.getString(new StringBuilder()
98                  .append(this.getClass().getSimpleName())
99                  .append('.').append(this.name())
100                 .append('.').append("ConsoleText")
101                 .toString());
102     }
103 
104     /***
105      *
106      * @return collection of all console preferences
107      */
108     public static Collection<ConsolePreference> values()
109     {
110         return AbstractWorkspacePreference.values(ConsolePreference.class);
111     }
112 }
