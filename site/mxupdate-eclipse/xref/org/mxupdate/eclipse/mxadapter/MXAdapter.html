182     /***
183      * Initializes the image descriptors read from the plug in properties.
184      *
185      * @see #imageMap
186      * @see #typeDef2Image
187      */
188     protected void initImageDescriptors()
189     {
190         final Properties imageConfig = new Properties();
191         if (this.properties.getImageConfig() != null)  {
192             final InputStream is = new ByteArrayInputStream(this.properties.getImageConfig().getBytes());
193             try {
194                 imageConfig.load(is);
195             } catch (final IOException e) {
196                 this.console.logError("MXAdapter.ExceptionInitImageDescriptorsLoadPropertiesFailed", e);
197             }
198         }
199 
200         // extract all admin type names
201         final Set<String> admins = new HashSet<String>();
202         for (final Object keyObj : imageConfig.keySet())  {
203             final String key = keyObj.toString().replaceAll("//..*", ""); //$NON-NLS-1$ //$NON-NLS-2$
204             admins.add(key);
205         }
206 
207         // remove already stored images
208         this.typeDef2Image.clear();
209         this.imageMap.clear();
210 
211         // prepare image cache
212         for (final String admin : admins)  {
213             final String prefix = imageConfig.getProperty(admin + ".FilePrefix"); //$NON-NLS-1$
214             final String suffix = imageConfig.getProperty(admin + ".FileSuffix"); //$NON-NLS-1$
215             final String iconStr = imageConfig.getProperty(admin + ".Icon"); //$NON-NLS-1$
216 
217             final byte[] bin = Base64.decodeBase64(iconStr.getBytes());
218             final InputStream in = new ByteArrayInputStream(bin);
219 
220             final ImageDescriptor imageDesriptor = ImageDescriptor.createFromImageData(new ImageData(in));
221 
222             // mapping between file prefix / extension and image
223             Map<String,ImageDescriptor> mapPrefix = this.imageMap.get(suffix);
224             if (mapPrefix == null)  {
225                 mapPrefix = new HashMap<String,ImageDescriptor>();
226                 this.imageMap.put(suffix, mapPrefix);
227             }
228             mapPrefix.put(prefix, imageDesriptor);
229 
230             // mapping between type definition and image
231             this.typeDef2Image.put(admin, imageDesriptor);
232         }
233     }

247     /***
248      * Connects to the MX database.
249      *
250      * @throws Exception if connect failed
251      * @see #connected
252      * @see #mxContext
253      * @see #checkVersions()
254      */
255     public void connect()
256         throws Exception
257     {
258         if (this.connector != null)  {
259             this.console.logInfo(Messages.getString("MXAdapter.AlreadyConnected")); //$NON-NLS-1$
260         } else  {
261             this.connector = this.properties.getMode().initConnector(this.project, this.console);
262 
263             // check versions
264             this.checkVersions();
265 
266             try {
267                 // read properties
268                 final String newProps = (String) this.executeEncoded(null, "GetProperty", null).get(MXAdapter.RETURN_KEY_VALUES);
269                 final String curProps = this.properties.getImageConfig();
270 
271                 // update if required
272                 if (!newProps.equals(curProps))  {
273                     this.properties.storeImageConfig(newProps);
274                     this.console.logInfo(Messages.getString("MXAdapter.PluginPropertiesChanged")); //$NON-NLS-1$
275 
276                     // reload image descriptors
277                     this.initImageDescriptors();
278 
279                     // and refresh project
280                     this.project.touch(new NullProgressMonitor());
281                 }
282 
283             } catch (final Exception e) {
284                 this.console.logError(Messages.getString("MXAdapter.ConnectFailed"), e); //$NON-NLS-1$
285             }
286 
287         }
288     }

290     /***
291      * Checks that the version of the MxUpdate Eclipse Plug-In and the version
292      * of the MxUpdate Update Deployment Tool have the same major and minor
293      * number within their versions. If not equal the plug-in automatically
294      * disconnects from MX.
295      *
296      * @see #disconnect()
297      * @see #getPlugInVersion()
298      * @see #connect()
299      */
300     protected void checkVersions()
301     {
302         String pluginVersion = null;
303         try {
304             pluginVersion = this.getPlugInVersion();
305         } catch (final IOException e) {
306             this.console.logError(Messages.getString("MXAdapter.ExceptionGetPlugInVersion"), e); //$NON-NLS-1$
307         }
308 
309         String updateVersionOrg = null;
310         try {
311             updateVersionOrg = (String) this.executeEncoded(null, "GetVersion", null).get(MXAdapter.RETURN_KEY_VALUES);
312         } catch (final Exception e) {
313             this.console.logError(Messages.getString("MXAdapter.ExceptionGetUpdateVersion"), e); //$NON-NLS-1$
314         }
315         final String updateVersion = (updateVersionOrg != null) ? updateVersionOrg.replace('-', '.') : null;
316 
317         final String[] pluginVersions = (pluginVersion != null) ? pluginVersion.split("//.") : null; //$NON-NLS-1$
318         final String[] updateVersions = (updateVersion != null) ? updateVersion.split("//.") : null; //$NON-NLS-1$
319         if ((pluginVersion == null) || (pluginVersions.length < 2)
320                 || (updateVersion == null) || (updateVersions.length < 2)
321                 || !pluginVersions[0].equals(updateVersions[0]) || !pluginVersions[1].equals(updateVersions[1]))  {
322             this.console.logError(Messages.getString("MXAdapter.CheckVersionsNoConnectAllowed", //$NON-NLS-1$
323                                                      pluginVersion,
324                                                      updateVersion));
325         }
326     }

328     /***
329      * Returns for this plug-in the version stored within manifest file.
330      *
331      * @return found plug-in version within manifest file
332      * @throws IOException if the manifest file could not found or not opened
333      * @see #MANIFEST_FILE
334      */
335     protected String getPlugInVersion()
336         throws IOException
337     {
338         final InputStream in = this.getClass().getClassLoader().getResourceAsStream(MXAdapter.MANIFEST_FILE);
339         final byte[] buffer = new byte[in.available()];
340         in.read(buffer);
341         in.close();
342 
343         final String manifest = new String(buffer);
344         String version = null;
345         for (final String line : manifest.split("\n"))  {
346             if (line.startsWith(MXAdapter.TEXT_BUNDLE_VERSION))  {
347                 version = line.substring(MXAdapter.LENGTH_BUNDLE_VERSION).trim();
348                 break;
349             }
350         }
351 
352         return version;
353     }

379     /***
380      * Updates given MX update files in the MX database. If
381      * {@link #PREF_UPDATE_FILE_CONTENT} is set, also the file content is
382      * transfered within the update (e.g. if an update on another server is
383      * done).
384      *
385      * @param _files    MxUpdate file which must be updated
386      * @param _compile  if <i>true</i> all JPOs are compiled; if <i>false</i>
387      *                  no JPOs are compiled, only an update is done
388      * @throws Exception if update failed (or included connect)
389      * @see #execMql(CharSequence)
390      */
391     public void update(final List<IFile> _files,
392                        final boolean _compile)
393         throws Exception
394     {
395         if (this.connector == null)  {
396             this.connect();
397         }
398 
399         // update by file content
400         if (this.connector.isUpdateByFileContent())  {
401             final Map<String,String> files = new HashMap<String,String>();
402             for (final IFile file: _files)  {
403                 try  {
404                     final InputStream in = new FileInputStream(file.getLocation().toFile());
405                     final byte[] bytes = new byte[in.available()];
406                     in.read(bytes);
407                     in.close();
408                     files.put(file.getLocation().toString(),
409                               new String(bytes, file.getCharset()));
410                 } catch (final UnsupportedEncodingException e)  {
411                     this.console.logError(Messages.getString("MXAdapter.ExceptionConvertFileContent", //$NON-NLS-1$
412                                                              file.getLocation().toString()),
413                                           e);
414                 } catch (final CoreException e) {
415                     this.console.logError(Messages.getString("MXAdapter.ExceptionFileCharSet", //$NON-NLS-1$
416                                                              file.getLocation().toString()),
417                                           e);
418                 } catch (final IOException e) {
419                     this.console.logError(Messages.getString("MXAdapter.ExceptionReadFileContentFailed", //$NON-NLS-1$
420                                                              file.getLocation().toString()),
421                                           e);
422                 }
423             }
424             try {
425                 final Map<?,?> bck = this.executeEncoded(new String[]{"Compile", String.valueOf(_compile)},
426                                                          "Update",
427                                                          new Object[]{"FileContents", files});
428                 this.console.appendLog((String) bck.get(MXAdapter.RETURN_KEY_LOG));
429                 final Exception ex = (Exception) bck.get(MXAdapter.RETURN_KEY_EXCEPTION);
430                 if (ex != null)  {
431                     this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed",  //$NON-NLS-1$
432                                                              files.keySet().toString()),
433                                           ex);
434                 }
435             } catch (final Exception e)  {
436                 this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed",  //$NON-NLS-1$
437                                                          files.keySet().toString()),
438                                       e);
439             }
440         // update by file names
441         } else  {
442             final Set<String> fileNames = new HashSet<String>();
443             for (final IFile file: _files)  {
444                 fileNames.add(file.getLocation().toString());
445             }
446             try {
447                 final Map<?,?> bck = this.executeEncoded(new String[]{"Compile", String.valueOf(_compile)},
448                                                          "Update",
449                                                          new Object[]{"FileNames", fileNames});
450                 this.console.appendLog((String) bck.get(MXAdapter.RETURN_KEY_LOG));
451                 final Exception ex = (Exception) bck.get(MXAdapter.RETURN_KEY_EXCEPTION);
452                 if (ex != null)  {
453                     this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed",  //$NON-NLS-1$
454                                                              fileNames.toString()),
455                                           ex);
456                 }
457             } catch (final Exception e)  {
458                 this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed", //$NON-NLS-1$
459                                                          fileNames.toString()),
460                                       e);
461             }
462         }
463     }

465     /***
466      * {@inheritDoc}
467      * The type definition root is evaluated directly in MX and only converted
468      * in the required format.
469      */
470     public ITypeDefRoot getTypeDefRoot()
471         throws Exception
472     {
473         if (this.connector == null)  {
474             this.connect();
475         }
476 
477         Map<?,?> bck = null;
478         try {
479             bck = this.executeEncoded(null, "TypeDefTreeList", null);
480         } catch (final Exception e) {
481             this.console.logError(Messages.getString("MXAdapter.ExceptionRootTypeDefFailed"), e); //$NON-NLS-1$
482         }
483 
484         final ITypeDefRoot ret;
485         if (bck == null)  {
486             ret = null;
487         } else if (bck.get(MXAdapter.RETURN_KEY_EXCEPTION) != null)  {
488             ret = null;
489             this.console.logError(Messages.getString("MXAdapter.ExceptionRootTypeDefFailed"), //$NON-NLS-1$
490                                   (Exception) bck.get(MXAdapter.RETURN_KEY_EXCEPTION));
491         } else  {
492             final Map<?,?> treeMap = (Map<?,?>) bck.get(MXAdapter.RETURN_KEY_VALUES);
493             ret = new ITypeDefRoot()  {
494                 private final Collection<ITypeDefNode> typeDefNodes = new ArrayList<ITypeDefNode>();
495                 public Collection<ITypeDefNode> getSubTypeDef()
496                 {
497                     return this.typeDefNodes;
498                 }
499             };
500             this.appendSubNode(ret, (Map<?,?>) treeMap.get("All"), treeMap);
501         }
502         return ret;
503     }

505     /***
506      * Appends to the <code>_node</code> all required / defined sub nodes.
507      *
508      * @param _node         node where sub nodes must be appended
509      * @param _treeNode     current tree node which must be appended
510      * @param _treeMap      map with all tree definitions from MX
511      */
512     protected void appendSubNode(final ITypeDefRoot _node,
513                                  final Map<?,?> _treeNode,
514                                  final Map<?,?> _treeMap)
515     {
516         final Collection<?> typeDefTreeList = (Collection<?>) _treeNode.get("TypeDefTreeList");
517         if (typeDefTreeList != null)  {
518             for (final Object subTreeNameObj : typeDefTreeList)  {
519                 if ((subTreeNameObj != null) && !"".equals(subTreeNameObj))  {
520                     final Map<?,?> subTreeNode = (Map<?,?>) _treeMap.get(subTreeNameObj);
521                     final String subLabel = (String) subTreeNode.get("Label");
522                     final Collection<?> subTypeDefs = (Collection<?>) subTreeNode.get("TypeDefList");
523                     final String[] typeDefArr;
524                     if ((subTypeDefs != null) && !subTypeDefs.isEmpty())  {
525                         typeDefArr = new String[subTypeDefs.size()];
526                         int idx = 0;
527                         for (final Object typeDefObj : subTypeDefs)  {
528                             typeDefArr[idx++] = (String) typeDefObj;
529                         }
530                     } else  {
531                         typeDefArr = new String[0];
532                     }
533                     final ITypeDefNode subNode = new ITypeDefNode() {
534                         private final Collection<ITypeDefNode> typeDefNodes = new ArrayList<ITypeDefNode>();
535                         public String getLabel()
536                         {
537                             return subLabel;
538                         }
539                         public Set<String> getTypeDefs()
540                         {
541                             return new HashSet<String>(Arrays.asList(typeDefArr));
542                         }
543                         public Collection<ITypeDefNode> getSubTypeDef()
544                         {
545                             return this.typeDefNodes;
546                         }
547                     };
548                     this.appendSubNode(subNode, subTreeNode, _treeMap);
549                     _node.getSubTypeDef().add(subNode);
550                 }
551             }
552         }
553     }

560      * @return found search items

562     public List<ISearchItem> search(final Set<String> _typeDefList,
563                                     final String _match)
564     {
565         Map<?,?> bck = null;
566         try {
567             bck = this.executeEncoded(null,
568                                       "Search",
569                                       new Object[]{"TypeDefList", _typeDefList,
570                                                    "Match", _match});
571         } catch (final Exception e) {
572             this.console.logError(Messages.getString("MXAdapter.ExceptionExportFailed"), e); //$NON-NLS-1$
573         }
574 
575         final List<ISearchItem> ret = new ArrayList<ISearchItem>();
576         if (bck.get(MXAdapter.RETURN_KEY_EXCEPTION) != null)  {
577             this.console.logError(Messages.getString("MXAdapter.ExceptionSearchFailed"), //$NON-NLS-1$
578                                   (Exception) bck.get(MXAdapter.RETURN_KEY_EXCEPTION));
579         } else  {
580             final List<?> values = (List<?>) bck.get(MXAdapter.RETURN_KEY_VALUES);
581             for (final Object valueObj : values)  {
582                 final Map<?,?> value = (Map<?,?>) valueObj;
583                 ret.add(new ISearchItem() {
584                     public String getFileName()
585                     {
586                         return (String) value.get("FileName");
587                     }
588                     public String getFilePath()
589                     {
590                         return (String) value.get("FilePath");
591                     }
592                     public String getName()
593                     {
594                         return (String) value.get("Name");
595                     }
596                     public String getTypeDef()
597                     {
598                         return (String) value.get("TypeDef");
599                     }
600                 });
601             }
602         }
603         return ret;
604     }

606     /***
607      * Extract the TCL update code for given <code>_file</code> from MX.
608      *
609      * @param _file     name of the update file for which the TCL update code
610      *                  within MX must be extracted
611      * @return configuration item update code for given <code>_file</code>
612      * @throws Exception if export failed
613      */
614     public IExportItem export(final IFile _file)
615         throws Exception
616     {
617         final IExportItem ret;
618 
619         if (this.connector == null)  {
620             this.connect();
621         }
622 
623         // if the file is a JPO, the package name is included in the MX name
624         if (_file.getName().endsWith(MXAdapter.END_JPO_FILE))  {
625             // hard coded for the workaround that the package of a JPO must read
626             ret = this.export("JPO", this.extractMxName(_file));
627         } else  {
628             final Map<?,?> bck = this.executeEncoded(null, "Export", new Object[]{"FileName", _file.getName()});
629 
630             if (bck.get(MXAdapter.RETURN_KEY_EXCEPTION) != null)  {
631                 throw (Exception) bck.get(MXAdapter.RETURN_KEY_EXCEPTION);
632             }
633 
634             final Map<?,?> value = (Map<?,?>) bck.get(MXAdapter.RETURN_KEY_VALUES);
635             Activator.getDefault().getConsole().logInfo(Messages.getString("MXAdapter.ExportLog", //$NON-NLS-1$
636                                                                            (String) value.get("FileName"))); //$NON-NLS-1$
637             Activator.getDefault().getConsole().appendLog((String) bck.get(MXAdapter.RETURN_KEY_LOG));
638             ret = new IExportItem() {
639                 public String getFileName()
640                 {
641                     return (String) value.get("FileName");
642                 }
643                 public String getFilePath()
644                 {
645                     return (String) value.get("FilePath");
646                 }
647                 public String getName()
648                 {
649                     return (String) value.get("Name");
650                 }
651                 public String getTypeDef()
652                 {
653                     return (String) value.get("TypeDef");
654                 }
655                 public String getContent()
656                 {
657                     return (String) value.get("Code");
658                 }
659             };
660         }
661 
662         return ret;
663     }
