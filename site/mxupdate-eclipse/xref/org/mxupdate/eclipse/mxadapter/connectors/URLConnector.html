140     /***
141      * Initializes the connection to the MX server.
142      *
143      * @param _projectPath          path to the project temporary folder
144      * @param _bundle               bundle of the plug-in to access the class
145      *                              and required libraries for the server
146      * @param _javaPath             path for the Java executable
147      * @param _mxJarPath            path of the MX Jar library
148      * @param _url                  URL of the MX server
149      * @param _user                 name of the user on the MX server
150      * @param _passwd               password of the user on the MX server
151      * @param _updateByFileContent  <i>true</i> if update is done by
152      *                              transmitting the file content; otherwise
153      *                              <i>false</i>
154      * @throws Exception if connection to the MX server could not be started
155      */
156     public URLConnector(final File _projectPath,
157                         final Bundle _bundle,
158                         final String _javaPath,
159                         final String _mxJarPath,
160                         final String _url,
161                         final String _user,
162                         final String _passwd,
163                         final boolean _updateByFileContent)
164         throws Exception
165     {
166         super(_updateByFileContent);
167 
168         // copy the required classes and JAR library to temporary project dir.
169         if (!_projectPath.exists())  {
170             _projectPath.mkdirs();
171         }
172         for (final Class<?> clazz : URLConnector.SERVER_CLASSES)  {
173             // the URL must use slashes to be a valid URL (and
174             // File.separatorChar delivers backslashes in Windows)
175             final String clazzFileName = "/" + clazz.getName().replace('.', '/') + ".class";
176             this.copy(
177                     this.getClass().getClassLoader().getResource(clazzFileName),
178                     new File(_projectPath, "/bin" + clazzFileName));
179         }
180 
181         // prepare class path (and always slashes instead of backslashes)
182         final StringBuilder classPath = new StringBuilder()
183                 .append("bin").append('/').append('.')
184                 .append(File.pathSeparatorChar)
185                 .append(_mxJarPath.replace('//', '/'));
186 
187         // start process
188         final ProcessBuilder pb = new ProcessBuilder(
189                 _javaPath,
190                 "-classpath", classPath.toString(),
191                 URLConnectorServer.class.getName(),
192                 _url,
193                 _user,
194                 _passwd);
195         pb.directory(_projectPath);
196 
197         this.process = pb.start();
198 
199         this.out = new OutputStreamWriter(this.process.getOutputStream());
200         this.inHandler = new InputStreamHandler(this.process.getInputStream());
201         this.errHandler = new ErrorStreamHandler(this.process.getErrorStream());
202         new Thread(this.inHandler).start();
203         new Thread(this.errHandler).start();
204 
205         try  {
206             this.testConnection();
207             this.connected = true;
208         } finally  {
209             if (!this.connected)  {
210                 this.disconnect();
211             }
212         }
213     }

246     /***
247      *
248      * @param _method   name of the method on the server to call
249      * @param _arg1     first argument
250      * @param _arg2     second argument
251      * @param _arg3     third argument
252      * @return returns string from the server
253      * @throws Exception {@link IOException} if the {@link #out} stream could
254      *                   not be written,
255      *                   {@link ClassNotFoundException} if returned value from
256      *                   the server could not be decoded,
257      *                   {@link InterruptedException} if the {@link #process}
258      *                   is interrupted
259      */
260     protected String execute(final String _method,
261                              final String _arg1,
262                              final String _arg2,
263                              final String _arg3)
264         throws Exception
265     {
266         // prepare parameters
267         final Map<Integer,String> args = new HashMap<Integer,String>();
268         args.put(0, _method);
269         args.put(1, _arg1);
270         args.put(2, _arg2);
271         args.put(3, _arg3);
272         final String encoded = CommunicationUtil.encode(args);
273 
274         // write to output stream
275         this.out.write(String.valueOf(encoded.length()));
276         this.out.write(' ');
277         this.out.write(encoded);
278         this.out.write(' ');
279         this.out.flush();
280 
281         // wait for return
282         final int length = Integer.valueOf(this.readOneWord());
283         final String value = this.readOneWord();
284         if (value.length() != length)  {
285             throw new Exception("wrong parameters");
286         }
287         final Map<String,?> bck = CommunicationUtil.decode(value);
288 
289         // return found values
290         return (String) bck.get("ret");
291     }

293     /***
294      * Reads next word from the input stream.
295      *
296      * @return read word
297      * @throws Exception    if an error has occurred while reading next word
298      *                      this error will be thrown
299      */
300     private String readOneWord()
301         throws Exception
302     {
303         while (!this.inHandler.wordExists() && this.errHandler.isEmpty())  {
304             Thread.sleep(1000);
305         }
306         if (!this.errHandler.isEmpty())  {
307             throw new Exception(this.errHandler.read());
308         }
309         return this.inHandler.readWord();
310     }

313     /***
314      * {@inheritDoc}
315      *
316      * @throws IOException if the {@link #out stream} to the MX server could
317      *                     not be written
318      * @throws InterruptedException if the thread sleep failed (for the loops
319      *                     while waiting for the exit of the {@link #process})
320      */
321     public void disconnect()
322         throws IOException, InterruptedException
323     {
324         final Map<Integer,String> args = new HashMap<Integer,String>();
325         args.put(0, "exit");
326 
327         final String encoded = CommunicationUtil.encode(args);
328 
329         try  {
330             if (this.connected)  {
331                 this.out.write(String.valueOf(encoded.length()));
332                 this.out.write(' ');
333                 this.out.write(encoded);
334                 this.out.write(' ');
335                 this.out.flush();
336             }
337         } finally  {
338             // destroy process if "normal" exit does not work...
339             IllegalThreadStateException ex = null;
340             int count = 0;
341             do {
342                 ex = null;
343                 try  {
344                     this.process.exitValue();
345                 } catch (final IllegalThreadStateException e)  {
346                     ex = e;
347                     Thread.sleep(1000);
348                     count++;
349                 }
350             } while ((ex != null) && (count < 10));
351             if (ex != null)  {
352                 this.process.destroy();
353             }
354         }
355     }

357     /***
358      * Copy the content of the <code>_from</code> URL to the target
359      * <code>_to</code> file.
360      *
361      * @param _from     URL of the file which must be copied
362      * @param _to       target file to which must be copied
363      * @throws IOException if the copy failed
364      */
365     private void copy(final URL _from,
366                       final File _to)
367         throws IOException
368     {
369         _to.getParentFile().mkdirs();
370         if (!_to.exists())  {
371             _to.createNewFile();
372         }
373 
374         final InputStream in = _from.openStream();
375         try  {
376             final OutputStream fileOut = new FileOutputStream(_to);
377             try {
378                 IOUtils.copy(in, fileOut);
379             } catch (final Throwable e)  {
380                 e.printStackTrace();
381             } finally {
382                 fileOut.close();
383             }
384         } finally  {
385             in.close();
386         }
387     }

389     /***
390      * Handler for the error stream. The error stream is read within a separate
391      * thread into {@link #buffer}.
392      */
393     private static final class ErrorStreamHandler
394         extends Thread
395     {
396         /***
397          * Error input stream.
398          */
399         private final InputStream in;
400 
401         /***
402          * Used to store already read characters from the
403          * {@link #in error input stream}.
404          */
405         private final StringBuilder buffer = new StringBuilder();
406 
407         /***
408          * Initializes the {@in error input stream}.
409          *
410          * @param _in new input stream
411          */
412         public ErrorStreamHandler(final InputStream _in)
413         {
414             this.in = _in;
415         }
416 
417         /***
418          * <p>Reads from the {@link #in error input stream} current existing
419          * characters and stores them in {@link #buffer}. If no characters
420          * exists, this thread waits for 1s and starts again.</p>
421          * <p>In the case of an thrown exception (e.g. {@link #in} is closed,
422          * ...), an error will be thrown.</p>
423          */
424         @Override
425         public void run()
426         {
427             try {
428                 while (true)  {
429                     synchronized (this.in) {
430                         synchronized(this.buffer)  {
431                             while (this.in.available() > 0)  {
432                                 this.buffer.append((char) this.in.read());
433                             }
434                         }
435                     }
436                     Thread.sleep(1000);
437                 }
438             } catch (final InterruptedException e) {
439                 throw new Error(e);
440             } catch (final IOException e) {
441                 throw new Error(e);
442             }
443         }
444 
445         /***
446          * Checks if {@link #buffer} is empty.
447          *
448          * @return <i>true</i> if {@link #buffer} contains no character;
449          *          otherwise <i>false</i>
450          *
451          */
452         public boolean isEmpty()
453         {
454             synchronized (this.buffer)  {
455                 return this.buffer.length() == 0;
456             }
457         }
458 
459         /***
460          * Reads current read buffer. A synchronize on {@link #buffer} is done
461          * so that the method waits until no characters are read anymore by
462          * {@link #run()}.
463          *
464          * @return string
465          * @see #buffer
466          */
467         public String read()
468         {
469             synchronized (this.buffer)  {
470                 final String ret = this.buffer.toString();
471                 this.buffer.delete(0, this.buffer.length());
472                 return ret.toString();
473             }
474         }
475     }

477     /***
478      * Handler for the input stream of the process. The input stream is read
479      * within a thread and stored in a buffer.
480      */
481     private static final class InputStreamHandler
482         extends Thread
483     {
484         /***
485          * Input stream of the process.
486          */
487         private final InputStream in;
488 
489         /***
490          * Buffer to store read characters.
491          */
492         private final Vector<Integer> buffer = new Vector<Integer>();
493 
494         /***
495          * Initializes the input stream handler.
496          *
497          * @param _in       input stream of the process
498          */
499         public InputStreamHandler(final InputStream _in)
500         {
501             this.in = _in;
502         }
503 
504         /***
505          * <p>Reads from the {@link #in input stream} current existing
506          * characters and stores them in {@link #buffer}. If no characters
507          * exists, this thread waits for 1s and starts again.</p>
508          * <p>In the case of an thrown exception (e.g. {@link #in} is closed,
509          * ...), an error will be thrown.</p>
510          */
511         @Override()
512         public void run()
513         {
514             try {
515                 while (true)  {
516                     while (this.in.available() > 0)  {
517                         this.buffer.add(this.in.read());
518                     }
519                     Thread.sleep(1000);
520                 }
521             } catch (final InterruptedException e) {
522                 throw new Error(e);
523             } catch (final IOException e) {
524                 throw new Error(e);
525             }
526         }
527 
528         /***
529          * Checks if a word exists.
530          *
531          * @return <i>true</i> if {@link #buffer} contains a space (and a
532          *         complete word was returned); otherwise <i>false</i>
533          * @see #buffer
534          */
535         public boolean wordExists()
536         {
537             return this.buffer.contains(32);
538         }
539 
540         /***
541          * Reads from {@link #buffer} the next word until a space
542          * (character 32) is defined.
543          *
544          * @return read word
545          */
546         public String readWord()
547         {
548             final StringBuilder ret = new StringBuilder();
549             int bck;
550             while ((bck = this.buffer.remove(0)) != 32)  {
551                 ret.append((char) bck);
552             }
553             return ret.toString();
554         }
555     }
