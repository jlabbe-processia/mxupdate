21  package org.mxupdate.eclipse;
22  
23  import java.util.List;
24  
25  import org.eclipse.core.resources.IFile;
26  
27  /***
28   * Interface defining a common deployment adapter within eclipse needed to
29   * reuse the UI implementation.
30   *
31   * @author The MxUpdate Team
32   * @version $Id: IDeploymentAdapter.java 681 2010-01-31 12:31:28Z tim.moxter $
33   */
34  public interface IDeploymentAdapter
35  {
36      /***
37       * Connects to the database.
38       *
39       * @return <i>true</i> if connected; otherwise <i>false</i>
40       */
41      boolean connect();
42  
43      /***
44       * Disconnects from the database.
45       *
46       * @return <i>true</i> if disconnected; otherwise <i>false</i>
47       */
48      boolean disconnect();
49  
50      /***
51       * Updates given update <code>_files</code> in the database.
52       *
53       * @param _files    update files to update in the database
54       * @param _compile  if <i>true</i> all program files are compiled; if
55       *                  <i>false</i> no files are compiled
56       */
57      void update(final List<IFile> _files,
58                  final boolean _compile);
59  
60      /***
61       * Extracts for given <code>_file</code> name the update code from the
62       * database.
63       *
64       * @param _file update file for which the update code must be extracted
65       * @return extracted update code
66       * @throws Exception if the update code from the database could not be
67       *                   extracted
68       * @todo exceptions!
69       */
70      String extractCode(final IFile _file) throws Exception;
71  
72      /***
73       * Executes given <code>_command</code> within the console.
74       *
75       * @param _command  command to execute
76       * @return value from executed command
77       * @todo exceptions!
78       */
79      String execute(final CharSequence _command) throws Exception;
80  }
