41   * The activator class controls the MxUpdate plug-in life cycle.

50      /***
51       * Map used to hold the images for MxUpdate files. The first key is the
52       * file extension, the second key the file prefix.
53       */
54      public static final Map<String,Map<String,ImageDescriptor>> IMAGEMAP
55              = new HashMap<String,Map<String,ImageDescriptor>>();
56  
57      /***
58       * ID of the MxUpdate plug-in.
59       */
60      public static final String PLUGIN_ID = "MxUpdateEclipsePlugIn"; //$NON-NLS-1$

65      private static Activator plugIn;

68       * MxUpdate plug-in console.
69       *
70       * @see #showConsole()
71       */
72      private Console console = null;

74      private IDeploymentAdapter adapter;

77  	/*
78  	 * (non-Javadoc)
79  	 * @see org.eclipse.ui.plugin.AbstractUIPlugin#start(org.osgi.framework.BundleContext)
80  	 */
81      @Override
82      public void start(final BundleContext _context)
83              throws Exception
84      {
85  try  {
86          super.start(_context);
87          Activator.plugIn = this;
88  
89          this.console = new Console();
90          this.console.activate();
91          ConsolePlugin.getDefault().getConsoleManager().addConsoles(new IConsole[]{this.console});
92  
93  final Properties properties = new Properties();
94  final String propStr = this.getPreferenceStore().getString("pluginProperties"); //$NON-NLS-1$
95  if (propStr != null)  {
96      final InputStream is = new ByteArrayInputStream(propStr.getBytes());
97      properties.load(is);
98  }
99  
100 // extract all admin type names
101 final Set<String> admins = new HashSet<String>();
102 for (final Object keyObj : properties.keySet())  {
103     final String key = keyObj.toString().replaceAll("//..*", ""); //$NON-NLS-1$ //$NON-NLS-2$
104     admins.add(key);
105 }
106 
107 // prepare image cache
108 for (final String admin : admins)  {
109     final String prefix = properties.getProperty(admin + ".FilePrefix"); //$NON-NLS-1$
110     final String suffix = properties.getProperty(admin + ".FileSuffix"); //$NON-NLS-1$
111     final String iconStr = properties.getProperty(admin + ".Icon"); //$NON-NLS-1$
112 
113     final byte[] bin = Base64.decodeBase64(iconStr.getBytes());
114     final InputStream in = new ByteArrayInputStream(bin);
115 
116     final ImageData ret = new ImageData(in);
117 
118 
119     Map<String,ImageDescriptor> mapPrefix = Activator.IMAGEMAP.get(suffix);
120     if (mapPrefix == null)  {
121         mapPrefix = new HashMap<String,ImageDescriptor>();
122         Activator.IMAGEMAP.put(suffix, mapPrefix);
123     }
124     mapPrefix.put(prefix, ImageDescriptor.createFromImageData(ret));
125 }
126 } catch (final Throwable e)  {
127     e.printStackTrace(System.out);
128     this.console.logError("ERROR", e); //$NON-NLS-1$
129 }
130 this.adapter = new MXAdapter(this.getPreferenceStore(), this.console);
131 
132     }
133

138     @Override
139     public void stop(final BundleContext _context)
140             throws Exception
141     {
142         ConsolePlugin.getDefault().getConsoleManager().removeConsoles(new IConsole[]{ this.console });
143         this.console = null;
144         Activator.plugIn = null;
145         super.stop(_context);
146     }
