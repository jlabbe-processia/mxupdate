21  package org.mxupdate.eclipse;
22  
23  import org.eclipse.ui.console.ConsolePlugin;
24  import org.eclipse.ui.console.IConsole;
25  import org.eclipse.ui.plugin.AbstractUIPlugin;
26  import org.mxupdate.eclipse.adapter.IDeploymentAdapter;
27  import org.mxupdate.eclipse.console.Console;
28  import org.mxupdate.eclipse.mxadapter.MXAdapter;
29  import org.osgi.framework.BundleContext;
30  
31  /***
32   * The activator class controls the MxUpdate plug-in life cycle.
33   *
34   * @author The MxUpdate Team
35   * @version $Id: Activator.java 715 2010-02-07 15:43:18Z tim.moxter $
36   */
37  public class Activator
38      extends AbstractUIPlugin
39  {
40      /***
41       * ID of the MxUpdate plug-in.
42       */
43      public static final String PLUGIN_ID = "MxUpdateEclipsePlugIn"; //$NON-NLS-1$
44  
45      /***
46       * Shared instance of the MxUpdate plug-in.
47       */
48      private static Activator plugIn;
49  
50      /***
51       * MxUpdate plug-in console.
52       *
53       * @see #showConsole()
54       */
55      private Console console = null;
56  
57      /***
58       * Adapter implementing the interface to MX.
59       */
60      private IDeploymentAdapter adapter;
61  
62  
63  	/*
64  	 * (non-Javadoc)
65  	 * @see org.eclipse.ui.plugin.AbstractUIPlugin#start(org.osgi.framework.BundleContext)
66  	 */
67      @Override
68      public void start(final BundleContext _context)
69              throws Exception
70      {
71  try  {
72          super.start(_context);
73          Activator.plugIn = this;
74  
75          this.console = new Console();
76          this.console.activate();
77          ConsolePlugin.getDefault().getConsoleManager().addConsoles(new IConsole[]{this.console});
78  
79  } catch (final Throwable e)  {
80      e.printStackTrace(System.out);
81      this.console.logError("ERROR", e); //$NON-NLS-1$
82  }
83          this.adapter = new MXAdapter(this.getPreferenceStore(), this.console);
84      }
85  
86      /*
87       * (non-Javadoc)
88       * @see org.eclipse.ui.plugin.AbstractUIPlugin#stop(org.osgi.framework.BundleContext)
89       */
90      @Override
91      public void stop(final BundleContext _context)
92              throws Exception
93      {
94          ConsolePlugin.getDefault().getConsoleManager().removeConsoles(new IConsole[]{ this.console });
95          this.console = null;
96          Activator.plugIn = null;
97          super.stop(_context);
98      }
99  
100     /***
101      * Returns the default shared instance of the MxUpdate eclipse plug-in.
102      *
103      * @return shared instance
104      * @see #plugIn
105      */
106     public static Activator getDefault()
107     {
108         return Activator.plugIn;
109     }
110 
111     /***
112      * Returns the console for the plug-in. The method is the getter method for
113      * instance variable {@link #console}.
114      *
115      * @return console of the plug-in
116      * @see #console
117      */
118     public Console getConsole()
119     {
120         return this.console;
121     }
122 
123     /***
124      * Shows the MxUpdate plug-in console.
125      *
126      * @see #console    MxUpdate plug-in console
127      */
128     public void showConsole()
129     {
130         ConsolePlugin.getDefault().getConsoleManager().showConsoleView(this.console);
131     }
132 
133     public IDeploymentAdapter getAdapter()
134     {
135         return this.adapter;
136     }
137 }
