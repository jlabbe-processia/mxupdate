196         /***
197          * {@inheritDoc}
198          * The {@link URLConnector} is initialized. Depending on the
199          * {@link #propSavePassword} the MX user name / password is asked from
200          * the user.
201          *
202          * @see SSHConnector
203          */
204         @Override()
205         public IConnector initConnector(final IProject _project,
206                                               final ProjectProperties _properties,
207                                               final Console _console)
208             throws Exception
209         {
210             final String mxURL = _properties.getString(this.propUrl, "");
211 
212             // MX authentication
213             final String mxUser;
214             final String mxPasswd;
215             if (_properties.getBoolean(this.propSavePassword, this.valSavePassword))  {
216                 mxUser = _properties.getString(this.propUser, null);
217                 mxPasswd = _properties.getPassword(this.propPassword);
218             } else  {
219                 final AuthenticationDialog loginPage = new AuthenticationDialog(
220                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("MxGroup")),
221                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.propUser)),
222                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.propPassword)),
223                         _properties.getString(this.propUser, null));
224                 loginPage.open();
225                 if (!loginPage.isOkPressed())  {
226                     throw new Exception(
227                             Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterNotMXAuthenticated")));
228                 }
229                 mxUser = loginPage.getUserName();
230                 mxPasswd = loginPage.getPassword();
231             }
232 
233             _console.logInfo(
234                     Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterConnectTo"), //$NON-NLS-1$
235                                        mxURL));
236 
237             return new URLConnector(
238                     _project,
239                     _properties.getString(this.propJavaPath, this.valJavaPath),
240                     _properties.getString(this.propMxJarLibraryPath, ""), //$NON-NLS-1$
241                     mxURL, mxUser, mxPasswd,
242                     _properties.getBoolean(this.propUpdateByFileContent, true));
243         }

248          * @return <code>null</code> if property keys {@link #propJavaPath},

264                 ret = null;
265             }

352         /***
353          * {@inheritDoc}
354          * The connection settings are read from external file and the
355          * connection is initialized.
356          */
357         @Override()
358         public IConnector initConnector(final IProject _project,
359                                         final ProjectProperties _properties,
360                                         final Console _console)
361             throws Exception
362         {
363             final String propFile       = _properties.getString(this.propPropPath, ""); //$NON-NLS-1$
364             final String propKeyJava    = _properties.getString(this.propJavaPath, ""); //$NON-NLS-1$
365             final String propKeyMXJar   = _properties.getString(this.propMxJarLibraryPath, ""); //$NON-NLS-1$
366             final String propKeyUrl     = _properties.getString(this.propUrl, ""); //$NON-NLS-1$
367             final String propKeyName    = _properties.getString(this.propName, ""); //$NON-NLS-1$
368             final String propKeyPass    = _properties.getString(this.propPassword, ""); //$NON-NLS-1$
369             final String propKeyFileCnt = _properties.getString(this.propUpdateByFileContent, ""); //$NON-NLS-1$
370 
371             // read file
372             _console.logInfo(Messages.getString(
373                     new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterReadExternalFile"), propFile)); //$NON-NLS-1$
374             final Properties extProps = new Properties();
375             try {
376                 extProps.load(new FileInputStream(propFile));
377             } catch (final FileNotFoundException e) {
378                 final String message = Messages.getString(
379                         new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterFileOpenFailed"), propFile); //$NON-NLS-1$
380                 _console.logError(message, e); //$NON-NLS-1$
381                 throw new Exception(message, e);
382             } catch (final IOException e) {
383                 final String message = Messages.getString(
384                         new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterFileOpenFailed"), propFile); //$NON-NLS-1$
385                 _console.logError(message, e); //$NON-NLS-1$
386                 throw new Exception(message, e);
387             }
388 
389             // Java path
390             final String javaPath;
391             if (propKeyJava.isEmpty())  {
392                 javaPath = this.valJavaPath;
393             } else  {
394                 javaPath = extProps.getProperty(propKeyJava);
395             }
396 
397             // MX Jar Library path
398             final String mxJarLibraryPath = extProps.getProperty(propKeyMXJar);
399 
400             // URL
401             final String mxURL;
402             if (propKeyUrl.isEmpty())  {
403                 mxURL = "";
404             } else  {
405                 mxURL = extProps.getProperty(propKeyUrl);
406             }
407 
408             // user and password
409             final String mxUser;
410             final String mxPasswd;
411             if (propKeyName.isEmpty() || propKeyPass.isEmpty())  {
412                 final AuthenticationDialog loginPage = new AuthenticationDialog(
413                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("MXGroupLogin")),
414                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.propName)),
415                         Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.propPassword)),
416                         extProps.getProperty(propKeyName, null));
417                 loginPage.open();
418                 if (!loginPage.isOkPressed())  {
419                     throw new Exception(
420                             Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterNotMXAuthenticated")));
421                 }
422                 mxUser = loginPage.getUserName();
423                 mxPasswd = loginPage.getPassword();
424             } else  {
425                 mxUser = extProps.getProperty(propKeyName);
426                 mxPasswd = extProps.getProperty(propKeyPass);
427             }
428 
429             // update by file content flag
430             final boolean flagByFileContent;
431             if (propKeyFileCnt.isEmpty())  {
432                 flagByFileContent = true;
433             } else  {
434                 flagByFileContent = Boolean.valueOf(extProps.getProperty(propKeyFileCnt));
435             }
436 
437             _console.logInfo(Messages.getString(
438                     new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("InitAdapterConnectTo"), //$NON-NLS-1$
439                     mxURL));
440 
441             return new URLConnector(_project, javaPath, mxJarLibraryPath, mxURL, mxUser, mxPasswd, flagByFileContent);
442         }
443

567         /***
568          * {@inheritDoc}
569          *
570          * @return <code>null</code> if property keys {@link #propSSHServer},
571          *         {@link #propSSHUser}, {@link #propMXMQLPath} and
572          *         {@link #propMXUserName} are not empty and
573          *         {@link #propSSHPort} is between 0 and 65535;
574          *         otherwise message with error text
575          */
576         @Override()
577         public String isValid(final ProjectProperties _properties)
578         {
579             final String ret;
580             if (_properties.getString(this.propSSHServer, "").isEmpty())  { //$NON-NLS-1$
581                 ret = Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("MissingSSHServer")); //$NON-NLS-1$
582             } else if (_properties.isWrong(this.propSSHPort))  {
583                 ret = Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("WrongSSHPort")); //$NON-NLS-1$
584             } else if (_properties.getInteger(this.propSSHPort, this.valSSHPort) < 0)  {
585                 ret = Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("SSHPortToLow"), //$NON-NLS-1$
586                                          _properties.getInteger(this.propSSHPort, this.valSSHPort));
587             } else if (_properties.getInteger(this.propSSHPort, this.valSSHPort) > 65535)  {
588                 ret = Messages.getString(new StringBuilder(ProjectProperties.MSG_PREFIX).append(this.prefix).append("SSHPortToBig"), //$NON-NLS-1$
589                                          _properties.getInteger(this.propSSHPort, this.valSSHPort));
