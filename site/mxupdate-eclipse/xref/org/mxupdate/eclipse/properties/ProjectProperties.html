455     /***
456      * Stores current values of the {@link #properties} in given
457      * <code>_file</code>. If the <code>_file</code> does not exists, the file
458      * is created.
459      *
460      * @throws CoreException    if file could not be created or updated
461      * @throws IOException      if properties could not be written
462      * @see #PROP_FILE_NAME
463      */
464     public void store()
465         throws CoreException, IOException
466     {
467         final IFile file = this.project.getFile(ProjectProperties.PROP_FILE_NAME);
468         file.refreshLocal(IResource.DEPTH_ZERO, null);
469         if (!file.exists())  {
470             file.create(new ByteArrayInputStream(new byte[]{}), true, null);
471         }
472         final StringWriter writer = new StringWriter();
473         this.properties.store(writer, null);
474         final String content = writer.getBuffer().toString();
475         writer.close();
476 
477         file.setContents(new ByteArrayInputStream(content.getBytes(ProjectProperties.ENCODING)), true, false, null);
478     }

481     /***
482      * Returns encrypted/decrypted by salt password. Uses SHA-1 Message Digest
483      * Algorithm as defined in NIST's FIPS 180-1. The output of this algorithm
484      * is a 160-bit digest.
485      *
486      * @param _password     password to encrypt / decrypt
487      * @param _decrypt      <i>true</i> to decrypt or <i>false</i> to encrypt
488      * @return decrypted / encrypted by salt password
489      * @see #PDE_ALGORITHM
490      * @see #PDE_PASSWORD
491      * @see #PDE_SALT
492      * @see #PDE_ITERATION
493      */
494     private String decryptEncrypt(final String _password,
495                                   final boolean _decrypt)
496     {
497         String ret = null;
498         if (_password != null) {
499             try {
500                 // create PBE parameter set
501                 final PBEParameterSpec pbeParamSpec = new PBEParameterSpec(ProjectProperties.PDE_SALT, ProjectProperties.PDE_ITERATION);
502                 final PBEKeySpec pbeKeySpec = new PBEKeySpec(ProjectProperties.PDE_PASSWORD, ProjectProperties.PDE_SALT, ProjectProperties.PDE_ITERATION);
503                 final SecretKeyFactory keyFac = SecretKeyFactory.getInstance(ProjectProperties.PDE_ALGORITHM);
504                 final SecretKey pbeKey = keyFac.generateSecret(pbeKeySpec);
505 
506                 final Cipher cipher = Cipher.getInstance(pbeKey.getAlgorithm());
507 
508                 if (_decrypt) {
509                     cipher.init(Cipher.DECRYPT_MODE, pbeKey, pbeParamSpec);
510                     // decode base64 to get bytes
511                     final byte[] dec = Base64.decodeBase64(_password.getBytes(ProjectProperties.ENCODING));
512                     // decrypt
513                     final byte[] ciphertext = cipher.doFinal(dec);
514 
515                     ret = new String(ciphertext, ProjectProperties.ENCODING);
516 
517                 } else {
518                     cipher.init(Cipher.ENCRYPT_MODE, pbeKey, pbeParamSpec);
519                     final byte[] pwdText = _password.getBytes(ProjectProperties.ENCODING);
520                     // encrypt the cleartext
521                     final byte[] ciphertext = cipher.doFinal(pwdText);
522 
523                     ret = new String(Base64.encodeBase64(ciphertext), ProjectProperties.ENCODING);
524                 }
525             } catch (final Exception e) {
526                 throw new Error(e);
527             }
528         }
529         return ret;
530     }
