178     /***
179      * Connects to the MX database.
180      *
181      * @return <i>true</i> if already connected or connect to MX database was
182      *         successfully; otherwise <i>false</i> is returned
183      * @see #connected
184      * @see #mxContext
185      * @see #checkVersions()
186      */
187     public boolean connect()
188     {
189         boolean connect = false;
190         if (this.connected)  {
191             connect = true;
192             this.console.logInfo(Messages.getString("MXAdapter.AlreadyConnected")); //$NON-NLS-1$
193         } else  {
194             final boolean connectAllowed;
195             final String user;
196             final String passwd;
197             final String host;
198 
199             // is external configured
200             if (this.preferences.getBoolean(MXAdapter.PREF_EXTERNAL_CONFIGURED))  {
201                 final String propFile = this.preferences.getString(MXAdapter.PREF_PROP_FILE);
202                 final String propHost = this.preferences.getString(MXAdapter.PREF_PROP_KEY_URL);
203                 final String propName = this.preferences.getString(MXAdapter.PREF_PROP_KEY_NAME);
204                 final String propPass = this.preferences.getString(MXAdapter.PREF_PROP_KEY_PASSWORD);
205 
206                 this.console.logInfo(Messages.getString("MXAdapter.ReadExternalFile", propFile)); //$NON-NLS-1$
207                 final Properties extProps = new Properties();
208                 boolean read = false;
209                 try {
210                     extProps.load(new FileInputStream(propFile));
211                     read = true;
212                 } catch (final FileNotFoundException e) {
213                     this.console.logError(Messages.getString("MXAdapter.ExceptionExternalFileOpenFailed"), e); //$NON-NLS-1$
214                 } catch (final IOException e) {
215                     this.console.logError(Messages.getString("MXAdapter.ExceptionExternalFileOpenFailed"), e); //$NON-NLS-1$
216                 }
217 
218                 if (!read)  {
219                     connectAllowed = false;
220                     host = null;
221                     user = null;
222                     passwd = null;
223                 } else if ((propName == null) || "".equals(propName) || (propPass == null) || "".equals(propPass))  {
224                     host = extProps.getProperty(propHost);
225                     final MXLoginPage loginPage = new MXLoginPage(host, this.preferences.getString(MXAdapter.PREF_NAME));
226                     loginPage.open();
227                     connectAllowed = loginPage.isOkPressed();
228                     user = loginPage.getUserName();
229                     passwd = loginPage.getPassword();
230                 } else  {
231                     connectAllowed = true;
232                     host = extProps.getProperty(propHost);
233                     user = extProps.getProperty(propName);
234                     passwd = extProps.getProperty(propPass);
235                 }
236 
237             // internal configured
238             } else {
239                 host =  this.preferences.getString(MXAdapter.PREF_URL);
240                 if (!this.preferences.getBoolean(MXAdapter.PREF_STORE_PASSWORD))  {
241                     final MXLoginPage loginPage = new MXLoginPage(host, this.preferences.getString(MXAdapter.PREF_NAME));
242                     loginPage.open();
243                     connectAllowed = loginPage.isOkPressed();
244                     user = loginPage.getUserName();
245                     passwd = loginPage.getPassword();
246                 } else  {
247                     connectAllowed = true;
248                     user =  this.preferences.getString(MXAdapter.PREF_NAME);
249                     passwd =  this.preferences.getString(MXAdapter.PREF_PASSWORD);
250                 }
251             }
252 
253             if (connectAllowed)  {
254 
255                 try {
256                     this.mxContext = new Context(host);
257                     this.mxContext.resetContext(user, passwd, null);
258                     this.mxContext.connect();
259                     this.connected = this.mxContext.isConnected();
260                     connect = true;
261                     this.console.logInfo(Messages.getString("MXAdapter.ConnectedTo", host)); //$NON-NLS-1$
262 
263                     // read properties
264                     final String newProps = this.execute("exec prog org.mxupdate.plugin.GetProperties"); //$NON-NLS-1$
265                     final String curProps = this.preferences.getString(MXAdapter.PREF_PROPERTIES);
266                     if (!newProps.equals(curProps))  {
267                         this.preferences.setValue(MXAdapter.PREF_PROPERTIES, newProps);
268                         this.console.logInfo(Messages.getString("MXAdapter.PluginPropertiesChanged")); //$NON-NLS-1$
269                     }
270                 } catch (final MatrixException e) {
271                     this.console.logError(Messages.getString("MXAdapter.ConnectFailed"), e); //$NON-NLS-1$
272                 }
273 
274                 // check versions
275                 if (this.connected)  {
276                     this.checkVersions();
277                 }
278             }
279         }
280         return connect;
281     }

283     /***
284      * Checks that the version of the MxUpdate Eclipse Plug-In and the version
285      * of the MxUpdate Update Deployment Tool have the same major and minor
286      * number within their versions. If not equal the plug-in automatically
287      * disconnects from MX.
288      *
289      * @see #disconnect()
290      * @see #getPlugInVersion()
291      * @see #connect()
292      */
293     protected void checkVersions()
294     {
295         String pluginVersion = null;
296         try {
297             pluginVersion = this.getPlugInVersion();
298         } catch (final IOException e) {
299             this.console.logError(Messages.getString("MXAdapter.ExceptionGetPlugInVersion"), e); //$NON-NLS-1$
300         }
301 
302         String updateVersion = null;
303         try {
304             updateVersion = this.execute("exec prog org.mxupdate.plugin.GetVersion").replaceAll("-", ".");
305         } catch (final MatrixException e) {
306             this.console.logError(Messages.getString("MXAdapter.ExceptionGetUpdateVersion"), e); //$NON-NLS-1$
307         }
308 
309         final String[] pluginVersions = (pluginVersion != null) ? pluginVersion.split("//.") : null;
310         final String[] updateVersions = (updateVersion != null) ? updateVersion.split("//.") : null;
311         if ((pluginVersion == null) || (pluginVersions.length < 2)
312                 || (updateVersion == null) || (updateVersions.length < 2)
313                 || !pluginVersions[0].equals(updateVersions[0]) || !pluginVersions[1].equals(updateVersions[1]))  {
314             this.console.logError(Messages.getString("MXAdapter.CheckVersionsNoConnectAllowed", //$NON-NLS-1$
315                                                      pluginVersion,
316                                                      updateVersion));
317         }
318     }

320     /***
321      * Returns for this plug-in the version stored within manifest file.
322      *
323      * @return found plug-in version within manifest file
324      * @throws IOException if the manifest file could not found or not opened
325      * @see #MANIFEST_FILE
326      */
327     protected String getPlugInVersion()
328             throws IOException
329     {
330         final InputStream in = this.getClass().getClassLoader().getResourceAsStream(MXAdapter.MANIFEST_FILE);
331         final byte[] buffer = new byte[in.available()];
332         in.read(buffer);
333         in.close();
334 
335         final String manifest = new String(buffer);
336         String version = null;
337         for (final String line : manifest.split("\n"))  {
338             if (line.startsWith(MXAdapter.TEXT_BUNDLE_VERSION))  {
339                 version = line.substring(MXAdapter.LENGTH_BUNDLE_VERSION).trim();
340                 break;
341             }
342         }
343 
344         return version;
345     }

375     /***
376      * Updates given MX update files in the MX database. If
377      * {@link #PREF_UPDATE_FILE_CONTENT} is set, also the file content is
378      * transfered within the update (e.g. if an update on another server is
379      * done).
380      *
381      * @param _files    MxUpdate file which must be updated
382      * @param _compile  if <i>true</i> all JPOs are compiled; if <i>false</i>
383      *                  no JPOs are compiled, only an update is done
384      * @see #execMql(CharSequence)
385      */
386     public void update(final List<IFile> _files,
387                        final boolean _compile)
388     {
389         // update by file content
390         if (this.preferences.getBoolean(MXAdapter.PREF_UPDATE_FILE_CONTENT))  {
391             final Map<String,String> files = new HashMap<String,String>();
392             for (final IFile file: _files)  {
393                 try  {
394                     final InputStream in = new FileInputStream(file.getLocation().toFile());
395                     final byte[] bytes = new byte[in.available()];
396                     in.read(bytes);
397                     in.close();
398                     files.put(file.getLocation().toString(),
399                               new String(bytes, file.getCharset()));
400                 } catch (final UnsupportedEncodingException e)  {
401                     this.console.logError(Messages.getString("MXAdapter.ExceptionConvertFileContent", //$NON-NLS-1$
402                                                              file.getLocation().toString()),
403                                           e);
404                 } catch (final CoreException e) {
405                     this.console.logError(Messages.getString("MXAdapter.ExceptionFileCharSet", //$NON-NLS-1$
406                                                              file.getLocation().toString()),
407                                           e);
408                 } catch (final IOException e) {
409                     this.console.logError(Messages.getString("MXAdapter.ExceptionReadFileContentFailed", //$NON-NLS-1$
410                                                              file.getLocation().toString()),
411                                           e);
412                 }
413             }
414             try {
415                 this.console.logInfo(this.jpoInvoke("org.mxupdate.plugin.Update",
416                                                     "updateByContent",
417                                                     files,
418                                                     _compile));
419             } catch (final Exception e)  {
420                 this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed",  //$NON-NLS-1$
421                                                          files.keySet().toString()),
422                                       e);
423             }
424         // update by file names
425         } else  {
426             final Set<String> fileNames = new HashSet<String>();
427             for (final IFile file: _files)  {
428                 fileNames.add(file.getLocation().toString());
429             }
430             try {
431                 this.console.logInfo(this.jpoInvoke("org.mxupdate.plugin.Update",
432                                                     "updateByName",
433                                                     fileNames,
434                                                     _compile));
435             } catch (final Exception e)  {
436                 this.console.logError(Messages.getString("MXAdapter.ExceptionUpdateFailed", //$NON-NLS-1$
437                                                          fileNames.toString()),
438                                       e);
439             }
440         }
441     }

459     /***
460      * Executes given MQL command and returns the trimmed result of the MQL
461      * execution. The MX context {@link #mxContext} is connected to the data
462      * base if not already done.
463      *
464      * @param _command  MQL command to execute
465      * @return trimmed result of the MQL execution
466      * @throws MatrixException if MQL execution fails
467      * @see #mxContext
468      * @see #connect()
469      */
470     public String execute(final CharSequence _command)
471             throws MatrixException
472     {
473         if (!this.connected)  {
474             this.connect();
475         }
476 
477         final MQLCommand mql = new MQLCommand();
478         mql.executeCommand(this.mxContext, _command.toString());
479         if ((mql.getError() != null) && !"".equals(mql.getError()))  { //$NON-NLS-1$
480             throw new MatrixException(mql.getError() + "\nMQL command was:\n" + _command);
481         }
482         return mql.getResult().trim();
483     }

485     /***
486      * Calls given <code>_method</code> in <code>_jpo</code>. The MX context
487      * {@link #mxContext} is connected to the database if not already done.
488      *
489      * @param _jpo          name of JPO to call
490      * @param _method       method of the called <code>_jpo</code>
491      * @param _parameters   list of all parameters for the <code>_jpo</code>
492      *                      which are automatically encoded encoded
493      * @return returned value from the called <code>_jpo</code>
494      * @throws IOException      if the parameter could not be encoded
495      * @throws MatrixException  if the called <code>_jpo</code> throws an
496      *                          exception
497      * @see #mxContext
498      * @see #connect()
499      */
500     protected String jpoInvoke(final String _jpo,
501                                final String _method,
502                                final Object... _parameters)
503         throws IOException, MatrixException
504     {
505         if (!this.connected)  {
506             this.connect();
507         }
508 
509         // encode parameters
510         final String[] paramStrings = new String[_parameters.length];
511         int idx = 0;
512         for (final Object parameter : _parameters)  {
513             final ByteArrayOutputStream out = new ByteArrayOutputStream();
514             final ObjectOutputStream oos = new ObjectOutputStream(out);
515             oos.writeObject(parameter);
516             oos.close();
517             paramStrings[idx++] = new String(Base64.encodeBase64(out.toByteArray()));
518         }
519 
520         return (String) JPO.invoke(this.mxContext, _jpo, null, _method, paramStrings, String.class);
521     }
